# Start Screen Messages
# Parameters:
#   trigger, a trigger with country which decides if this is a valid part. Empty will be treated as always=yes.
#   localization, a key to a localized string.
#	location, decides where this chunk of text is located in the description, cannot be negative.

### FIRST PARTS

# Autobot Cybertron
part = {
	location = 0
	localization = "START_SCREEN_PART_ONE_AUTOBOT"
	
	trigger = {
		has_origin = origin_cybertron
		OR = {
			has_civic = civic_machine_autobots
			has_civic = civic_autobots
		}
		NOR = {
			has_civic = civic_machine_decepticons
			has_country_flag = custom_start_screen 
		}
	}		
}

part = {
	location = 0
	localization = "START_SCREEN_PART_ONE_AUTOBOT_NO_CYBERTRON"
	
	trigger = {
		OR = {
			has_civic = civic_machine_autobots
			has_civic = civic_autobots
		}
		NOR = {
			has_origin = origin_cybertron
			has_civic = civic_machine_decepticons
			has_civic = civic_decepticons
			has_country_flag = custom_start_screen 
		}
	}		
}
