district_ocean_housing = {
	base_buildtime = 480
	is_capped_by_modifier = yes
	
	show_on_uncolonized = { has_district_ocean_housing = yes
				has_ocean_housing_deposits = yes }
	potential = { has_district_ocean_housing = yes
			has_ocean_housing_deposits = yes }
	
	allow = { housing_district_allowed = yes }
		
	conversion_ratio = 1
	convert_to = {
		district_city
		district_hive
		district_nexus
		district_suburban_city
	}
	
	resources = {
		category = planet_districts_cities
		cost = { minerals = 500 }
		upkeep = { energy = 4 }		
	}
	
	planet_modifier = { 
		planet_housing_add = 10 
		planet_max_districts_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = { planet_housing_add = -1 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_housing_1 }
		}
		modifier = { planet_housing_add = 1 }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_housing_2 }
		}
		modifier = { planet_housing_add = 1 }
	}
	
	ai_weight = {
		weight = 1.5
		# Does the planet need housing?
		modifier = {
			factor = 25
			need_housing = yes
		}
		
		# No Housing needed
		modifier = {
			factor = 0
			free_housing > 2
		}
	}
	ai_resource_production = {
		energy = 1.5
		trigger = {
			always = yes
		}
	}
}
