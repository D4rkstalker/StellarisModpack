district_arcology_ringworld_housing = {
	base_buildtime = 1200
	is_capped_by_modifier = no
	icon = district_arcology_housing
	
	show_on_uncolonized = {
		or = {
			is_planet_class = pc_ringworld_city
			is_planet_class = pc_ringworld_eco_city
		}
	}
	
	potential = {
		or = {
			is_planet_class = pc_ringworld_city
			is_planet_class = pc_ringworld_eco_city
		}
	}
	
	conversion_ratio = 1
	convert_to = {
		district_rw_city
		district_rw_nexus
		district_rw_hive
	}	
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 2000
		}
		upkeep = {
			energy = 10
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 60
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -20
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_1 
			}
		}
		modifier = {
			planet_housing_add = 10
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 10
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 5
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 20
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_extended_hives }
		}
		modifier = {
			planet_housing_add = 20
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_optimized_nexus }
		}
		modifier = {
			planet_housing_add = 20
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 60
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		modifier = {
			job_clerk_add = 60
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		text = job_clerk_effect_desc
	}	
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}	
	
	ai_weight = {
		weight = 1
		modifier = 
		{
			factor = 2
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 0
				}
			}
		}
		modifier = {
			factor = 5
			free_housing < 0
		}

		# Let's not go full spam on this.
		modifier = {
			factor = 0
			free_housing > 2
		}
	}

	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
