	is_capped_by_modifier = no

	conversion_ratio = 1

		district_nexus

	}	
	
	resources = {

		cost = {
			minerals = 20000
		}
		upkeep = {
			energy = 5
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 125
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 15
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		modifier = {
			job_clerk_add = 30
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		text = job_clerk_effect_desc
	}	
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}	
	
	ai_weight = {
		weight = 1
		modifier = 
		{
			factor = 2
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 0
				}
			}
		}
		modifier = {
			factor = 5
			free_housing < 0
		}

		# Let's not go full spam on this.
		modifier = {
			factor = 0
			free_housing > 2
		}
	}

	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
