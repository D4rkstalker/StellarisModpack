	}

	}
	
	resources = {
		category = planet_districts
		cost = {
			alloys = 200
		}
		upkeep = {
			energy = 2
		}
	}
	
	planet_modifier = {
		planet_housing_add = 4
		planet_amenities_add = 3
		planet_max_buildings_add = 2
		job_shadowgate_researcher_add = 1
	}
	
	ai_weight = {
		weight = 0

			weight = 5

		}
		modifier = {
			weight = 5

		}
		modifier = {
			factor = 0
			num_districts = { type = district_shadowgate_staging_area value >= 3 }
		}
	}
}
