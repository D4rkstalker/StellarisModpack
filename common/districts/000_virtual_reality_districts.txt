district_virtual_housing = {
	base_buildtime = 360
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			is_planet_class = pc_virtual_reality
			is_planet_class = pc_virtual_stellar_industry
			is_planet_class = pc_virtual_industry
		}
	}
	
	potential = {
		OR = {
			is_planet_class = pc_virtual_reality
			is_planet_class = pc_virtual_stellar_industry
			is_planet_class = pc_virtual_industry
		}
	}
	
	conversion_ratio = 1
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 500
			energy = 500
		}
		upkeep = {
			energy = 5
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 50
	}	
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			OR = {
				owner = { is_machine_empire = yes }
				owner = { is_hive_empire = yes }
			}
		}
		modifier = {
			job_maintenance_drone_add = 10
		}
	}
	
	ai_weight = {
		weight = 1

		# There are already unworked jobs
		# Blocks districts unworkable by remaining pops (slaves/robots etc)
		# Will be improved... at some point
		modifier = {
			factor = 0
			free_jobs > 0 
		}
	}
}
