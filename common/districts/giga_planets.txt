	}

	}

		category = planet_districts
		cost = { minerals = @giga_planet_cost }
		upkeep = { energy = @giga_planet_maintenance }
	}

	# base stats

		modifier = {
			planet_housing_add = 2
		}
	}

		potential = {
			exists = owner

			}
		}

	}
	# jobs : fallen empires
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire = yes }
		}

	}
	# jobs : gestalts
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		job_technician_drone_add = 2 
	}

	# descriptions
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}

	}

		trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }

	}

	planet_modifier = { planet_housing_add = 2 }

		potential = {
			exists = owner

				is_megacorp = no
				is_death_cult_empire = no
			}
		}
		# Megacorp NO, Death Cult NO.
		job_priest_add = 2 
	}

		potential = {
			exists = owner
			owner = {
				is_megacorp = yes
				is_death_cult_empire = no
			}
		}
		# Megacorp YES, Death Cult NO.
		job_priest_add = 1
		job_manager_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = no
				is_death_cult_empire = yes
			}
		}
		# Megacorp NO, Death Cult YES.
		job_death_priest_add = 2
		job_mortal_initiate_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = yes
				is_death_cult_empire = yes
			}
		}
		# Megacorp YES, Death Cult YES.
		job_death_priest_add = 1
		job_mortal_initiate_add = 1
		job_manager_add = 1
	}

	# Priests desc?
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_death_cult_empire = no
			}
		}
		text = job_priest_effect_desc
	}

	# Death Priests and Mortal Initiates desc?

			}
		}
