district_ult_hab_utopia_arcology = {
	base_buildtime = 480
	icon = district_ult_utopia_arcology
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			has_modifier = ult_utopia_infrastructure_hab_economic
			has_modifier = ult_utopia_infrastructure_hab_research
		}
	}
	
	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		is_planet_class = pc_habitat
		OR = {
			has_modifier = ult_utopia_infrastructure_hab_economic
			has_modifier = ult_utopia_infrastructure_hab_research
		}
	}
	
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 10
		planet_max_buildings_add = 2
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -1
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_public_works }
		}
		modifier = {
			job_healthcare_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_1 
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_culture_worker_add = 1
		}
	}
}
