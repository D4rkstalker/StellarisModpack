@rw_cost = 2000
@rw_cost_sr = 100
@rw_maintenance = 10
@rw_maintenance_sr = 2
@rw_district_buildtime = 600
machinedlcmod_district_rw_commercial = {
	base_buildtime = @rw_district_buildtime
	is_capped_by_modifier = no
	icon = district_rw_commercial
	
	show_on_uncolonized = {
		is_planet_class = pc_ringworld_habitable
		exists = from
		from = { has_valid_civic = civic_machine_trader }
	}
	
	potential = {
		is_planet_class = pc_ringworld_habitable
		exists = owner
		owner = { has_valid_civic = civic_machine_trader }
	}
	
	convert_to = {
		district_rw_generator
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = @rw_cost
			rare_crystals = @rw_cost_sr
		}
		upkeep = {
			energy = @rw_maintenance
			rare_crystals = @rw_maintenance_sr
		}		
	}
		
	triggered_planet_modifier = {
		planet_housing_add = 20
		job_machinedlcmod_merchant_add = 5
		job_machinedlcmod_clerk_add = 15
	}

	triggered_desc = {
		text = job_machinedlcmod_merchant_effect_desc
	}	

	triggered_desc = {
		text = job_machinedlcmod_clerk_effect_desc
	}	
		
	ai_weight = {
		weight = 0
		
		modifier = {
			weight = 20000
			owner = {
				has_monthly_income = {
					resource = energy
					value < 200
				}
				has_monthly_income = {
					resource = rare_crystals
					value > 2
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				has_monthly_income = {
					resource = rare_crystals
					value < 1
				}
			}
		}
		modifier = {
			factor = 0.01
			OR = { 
				planet = { has_district = district_rw_farming }
				planet = { has_district = district_rw_science } 
				planet = { has_district = district_rw_generator } 
			}
		}
	}

	ai_resource_production = {
		energy = 10
		trigger = {
			always = yes
		}
	}
}
