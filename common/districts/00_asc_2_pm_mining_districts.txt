district_ringworld_mining_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			is_planet_class = pc_ringworld_mining
		}
	}
	
	potential = {
		OR = {
			is_planet_class = pc_ringworld_mining
		}

	allow = {
		NOT = { has_modifier = resort_colony }

	conversion_ratio = 0.5

	resources = {

	planet_modifier = {

	}

			}
		}

			job_miner_add = 10
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = yes
			}
		}
		modifier = {
			job_mining_drone_add = 10
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		text = job_miner_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = yes
			}
		}
		text = job_mining_drone_effect_desc
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -5
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_1 
			}
		}
		modifier = {
			planet_housing_add = 5
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 5
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 2.5
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 10
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_extended_hives }
		}
		modifier = {
			planet_housing_add = 10
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_optimized_nexus }
		}
		modifier = {
			planet_housing_add = 10
		}
	}
	
	prerequisites = {
	}
	
	ai_weight = {
		weight = 1
		modifier = 
		{
			factor = 3
			owner = { 
				has_monthly_income = {
					resource = minerals
					value < 0
				}
			}
		}

		modifier = {
			factor = 3
			free_housing < 0
		}

		modifier = {
			factor = 3
			any_owned_pop = { 
				OR = { 
					is_unemployed = yes 
					has_job = servant 
				}
			}
		}

		# No one to work it anyway
		modifier = {
			factor = 0
			free_housing > 0
			NOT = {
				any_owned_pop = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
			}
		}
	}

	ai_resource_production = {
		minerals = 1
		trigger = {
			always = yes
		}
	}
}
