district_training_soldier = {
	base_buildtime = 240
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		from = { is_regular_empire = yes }
		OR = {
			is_planet_class = pc_training
			is_planet_class = pc_training_shielded
		}
	}
	
	potential = {
		OR = {
			is_planet_class = pc_training
			is_planet_class = pc_training_shielded
		}
		owner = { is_regular_empire = yes }
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
		num_districts = { type = district_training_soldier value < 20 }
	}	
	
	conversion_ratio = 1
	convert_to = {
		district_ruined_system
	}	

	resources = {
		category = planet_districts
		cost = {
			minerals = @base_cost
		}
		upkeep = {
			energy = 1
		}
	}
	
	planet_modifier = {
		planet_housing_add = 4
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				NOT = {
					has_ascension_perk = ap_mind_over_matter
				}
			}
		}
		modifier = {
			job_enforcer_per_pop = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_ascension_perk = ap_mind_over_matter
			}
		}
		modifier = {
			job_telepath_per_pop = 0.05
		}
	}
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_soldier_add = 3
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
		}
		text = district_pm_20
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		text = job_soldier_effect_desc
	}
	
	prerequisites = {
	}
	
	ai_weight = {
		weight = 1.5
	}

	ai_resource_production = {
		minerals = 2
		trigger = {
			always = yes
		}
	}
}
