@hab_cost = 500
@hab_maintenance = 2

district_hab_infrastructure = {
	base_buildtime = 720
	is_capped_by_modifier = no

	show_on_uncolonized = {
		is_planet_class = pc_habitat
	}
	
	potential = {
		is_planet_class = pc_habitat
	}
	
	allow = {
		not = {
			check_variable = {
				which = queued_infrastructure
				value > 5
			}
		}
	}
	
	on_queued = {
		planet = { 
			change_variable = {
				which = "queued_infrastructure" 
				value = 1
			}
		}	
	}
	
	on_unqueued = {
		planet = { 
			change_variable = {
				which = "queued_infrastructure" 
				value = -1
			}
		}	
	}
	
	on_built = {
		planet = { 
			change_variable = {
				which = "built_infrastructure" 
				value = 1
			}
			remove_modifier = hab_infrastructure_1
			remove_modifier = hab_infrastructure_2
			remove_modifier = hab_infrastructure_3
			remove_modifier = hab_infrastructure_4
			remove_modifier = hab_infrastructure_5
			remove_modifier = hab_infrastructure_6
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 1
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_1
					days = -1
				}
			}
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 2
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_2
					days = -1
				}
			}
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 3
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_3
					days = -1
				}
			}
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 4
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_4
					days = -1
				}
			}
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 5
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_5
					days = -1
				}
			}
			if = {
				limit = {
					check_variable = {
						which = built_infrastructure
						value = 6
					}
				}
				add_modifier = {
					modifier = hab_infrastructure_6
					days = -1
				}
			}
			remove_district = district_hab_infrastructure
		}
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = @hab_cost
		}
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 1000
			free_district_slots < 3
		}
		modifier = {
			weight = 99000
			free_district_slots < 2
		}
	}
}

district_hab_administration = {
	base_buildtime = 240
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		is_planet_class = pc_habitat
		nor = {
			has_planet_flag = mining_habitat
			has_planet_flag = research_habitat
			has_planet_flag = energy_habitat
		}
		not = { has_planet_flag = pf_hide_hab_administration }
		
		exists = from
		from = { 
			is_gestalt = no 
		}
	}
	
	potential = {
		is_planet_class = pc_habitat
		nor = {
			has_planet_flag = mining_habitat
			has_planet_flag = research_habitat
			has_planet_flag = energy_habitat
		}
		not = { has_planet_flag = pf_hide_hab_administration }
		
		exists = owner
		owner = { 
			is_gestalt = no 
		}
	}
	
	resources = {
		category = planet_districts_hab_leisure
		cost = {
			minerals = @hab_cost
		}
		upkeep = {
			energy = @hab_maintenance
		}		
	}
		
	planet_modifier = {
		planet_housing_add = 6
		job_service_add = 3
		job_bureaucrat_add = 2
		job_administrator_add = 1
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_regular_empire = yes
			}
		}
		text = job_administrator_effect_desc
		text = job_service_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 98000
			free_district_slots < 2
			num_districts = { type = district_hab_administration value < 1 }
		}
	}

	ai_resource_production = {
		unity = 1
		trigger = {
			always = yes
		}
	}
}
