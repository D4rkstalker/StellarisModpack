district_ringworld_training_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		from = { is_regular_empire = yes }
		OR = {
			is_planet_class = pc_ringworld_training
			is_planet_class = pc_ringworld_training_shielded
		}
	}
	
	potential = {
		OR = {
			is_planet_class = pc_ringworld_training
			is_planet_class = pc_ringworld_training_shielded
		}
		owner = { is_regular_empire = yes }
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
	}	
	
	conversion_ratio = 0.5
	convert_to = {
		district_ringworld_ruined_system
		district_rw_city
		district_rw_nexus
		district_rw_hive
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 2000
			volatile_motes = 25
			rare_crystals = 25
			exotic_gases = 25
		}
		upkeep = {
			energy = 10
			volatile_motes = 1
			rare_crystals = 1
			exotic_gases = 1
		}
	}
	
	planet_modifier = {
		planet_housing_add = 20
		planet_orbital_bombardment_damage = -0.025
	}
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_enforcer_add = 5
			job_soldier_add = 5
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		text = job_soldier_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		text = job_enforcer_desc
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -5
		}
	}	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_1 
			}
		}
		modifier = {
			planet_housing_add = 5
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 5
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				has_technology = tech_housing_2 
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 2.5
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 10
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_extended_hives }
		}
		modifier = {
			planet_housing_add = 10
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_swapped_tradition = tr_prosperity_optimized_nexus }
		}
		modifier = {
			planet_housing_add = 10
		}
	}
	
	prerequisites = {
	}
	
	ai_weight = {
		weight = 1
		
		modifier = {
			factor = 3
			free_housing < 0
		}

		# No one to work it anyway
		modifier = {
			factor = 0
			free_housing > 0
			NOT = {
				any_owned_pop = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
			}
		}
	}
}
