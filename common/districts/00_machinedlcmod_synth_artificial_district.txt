synth_segment_city = {
	base_buildtime = 480
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		from = { NOT = { is_country_type = fallen_empire } }
		OR = {
			from = { has_ascension_perk = ap_synth_artificial_worlds }
			has_planet_flag = planet_synth_artificial
		}
		is_planet_class = pc_synth_artificial
	}
	
	potential = {
		exists = owner
		owner = { NOT = { is_country_type = fallen_empire } }
		OR = {
			owner = { has_ascension_perk = ap_synth_artificial_worlds }
			has_planet_flag = planet_synth_artificial
		}
		is_planet_class = pc_synth_artificial
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 1
	convert_to = {
		district_arcology_housing
		district_nexus
		district_hive
		district_city
	}
	
	resources = {
		category = planet_districts
		cost = {
			alloys = 250
		}	
		upkeep = {
			energy = 3
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 10
	}

	triggered_planet_modifier = {
		potential = { 
			owner = { NOT = { has_authority = auth_machine_intelligence } }
		}
		job_clerk_add = 3
	}
	
	triggered_planet_modifier = {
		potential = {
			owner = { has_authority = auth_machine_intelligence } 
		}
		job_maintenance_drone_add = 3
	}
	
	triggered_desc = {
		trigger = {
			owner = { NOT = { has_authority = auth_machine_intelligence } }
		}
		text = job_clerk_effect_desc
	}
	
	triggered_desc = {
		trigger = {
			owner = { has_authority = auth_machine_intelligence } 
		}
		text = job_maintenance_drone_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 900
			planet = {
				planet_devastation < 1
				years_passed > 10
				free_housing < 4
			}
		}
		modifier = {
			weight = 1
			num_free_districts = { type = synth_segment_mining value = 0 }
			num_free_districts = { type = synth_segment_generator value = 0 }
			num_free_districts = { type = synth_segment_farming value = 0 }
		}
		modifier = {
			factor = 0
			NOT = {
				num_free_districts = { type = synth_segment_mining value = 0 }
				num_free_districts = { type = synth_segment_generator value = 0 }
				num_free_districts = { type = synth_segment_farming value = 0 }
			}
			OR = {
				num_free_districts = { type = synth_segment_mining value > 0 }
				num_free_districts = { type = synth_segment_generator value > 0 }
				num_free_districts = { type = synth_segment_farming value > 0 }
			}
			count_pops = {
				limit = {
					is_unemployed = yes
					NOR = {
						is_pop_category = specialist
						is_pop_category = ruler
					}
				}
				count >= 1
			}
		}
		modifier = {
			factor = 0
			NOT = {
				num_free_districts = { type = synth_segment_mining value = 0 }
				num_free_districts = { type = synth_segment_generator value = 0 }
				num_free_districts = { type = synth_segment_farming value = 0 }
			}
			owner = {
				has_resource = { 
					type = minerals 
					amount < 400 
				}
			}
		}
		modifier = {
			factor = 0
			NOT = {
				num_free_districts = { type = synth_segment_mining value = 0 }
				num_free_districts = { type = synth_segment_generator value = 0 }
				num_free_districts = { type = synth_segment_farming value = 0 }
			}
			free_jobs > 1
			free_housing > 1
		}
	}

	ai_resource_production = {
		consumer_goods = 2
		trigger = {
			always = yes
		}
	}
}
