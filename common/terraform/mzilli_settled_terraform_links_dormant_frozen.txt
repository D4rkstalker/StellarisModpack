terraform_link = {
	from = "pc_dormant_frozen"
	to = "pc_continental"
	
	energy = 3000
	duration = 2700
	
	condition = {
		has_technology = "tech_terrestrial_sculpting"
	}

	effect = {
		from = {
			if = {
				limit = { solar_system = { has_star_flag = sol } }
				set_planet_entity = { entity = "continental_planet_03_entity" }
			}
		}
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 30
			ideal_planet_class = pc_continental
		}
		modifier = {
			weight = 15
			OR = {
				ideal_planet_class = pc_ocean
				ideal_planet_class = pc_tropical
			}
		}
	}
}

# Machine Worlds
terraform_link = {
	to = "pc_machine"
	from = "pc_dormant_frozen"
	
	energy = 10000
	duration = 7200
	
	potential = {
		has_ascension_perk = ap_machine_worlds
	}
	
	ai_weight = {
		weight = 20
		modifier = {
			factor = 0
			from = {
				any_owned_pop = {
					NOR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				is_colony = no
			}
		}
	}
}
