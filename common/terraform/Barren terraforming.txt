terraform_link = {
	from = "pc_barren"
	to = "pc_alpine"
	
	energy = @habbarrenenergycost
	duration = @habbarrentime
	
	potential = {
		FROM = { has_modifier = colonized_planet }
		OR = {
			NOT = { has_authority = auth_machine_intelligence }
			has_civic = civic_machine_servitor
		}
	}
	
	condition = {
		FROM = { has_modifier = terraforming_candidate }
		has_resource = { type = food amount > @habbarrenfoodcost }
	}
	
	effect = {
		add_resource = { food = @habbarrenfoodcostnegative }
	}
	
	ai_weight = {
		weight = 5
		modifier = {
			factor = 0
			OR = {
				has_trait = trait_mechanical
				AND = {
					has_trait = trait_machine_unit
					NOT = { owner = { has_valid_civic = civic_machine_servitor } }
					NOT = { owner = { has_valid_civic = civic_machine_assimilator } }
				}
			}
		}
		modifier = {
			factor = 1.5
			any_owned_pop = {
				has_trait = trait_pc_alpine_preference
			}
		}
		modifier = {
			factor = 1.5
			has_trait = trait_pc_alpine_preference
		}
	}
}
