asteroid = {
	trigger = {
		owner = {
			OR = {
				has_technology = tech_asteroid_strike
				AND = {
					owner = { is_country_type = enclave_mercenary }
					root.controller = {
						has_technology = tech_asteroid_strike
					}
				}
			}
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 2.0
	army_damage = 1.5

	kill_pop_chance = {
		base = 0.5
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
	}
}

library_organism = {
	trigger = {
		owner = {
			OR = {
				has_country_flag = the_library_organism
				AND = {
					owner = { is_country_type = enclave_mercenary }
					root.controller = {
						has_country_flag = the_library_organism
					}
				}
			}
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 0.5
	army_damage = 1.5

	kill_pop_chance = {
		base = 1.0
		modifier = {
			factor = 0.5
			OR = {
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
	}
}

ferrophage = {
	trigger = {
		owner = {
			OR = {
				has_relic = r_the_defragmentor
				AND = {
					owner = { is_country_type = enclave_mercenary }
					root.controller = {
						has_relic = r_the_defragmentor
					}
				}
			}
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 2.0
	army_damage = 0.75

	kill_pop_chance = { # Should only target robots
		base = 2.0
		modifier = {
			factor = 0
			NOR = {
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = {
					NOR = {
						is_robot_empire = yes
						is_hostile = root.owner
					}
				}
			}
		}
	}
}
