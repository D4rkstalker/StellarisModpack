punishment = {
	trigger = {
		NOT = { has_global_flag = acot_exterminatus_forbidden }
		OR = {
			owner = { is_country_type = ascended_empire }
			owner = { is_country_type = original_empire }		
			owner = { is_country_type = original_empire_active }	
			owner = { is_country_type = awakened_fallen_empire }
			owner = { is_country_type = fallen_empire }		
			owner = { is_country_type = swarm }
			owner = { is_country_type = ai_empire }		
			owner = { is_country_type = extradimensional }	
			owner = { is_country_type = extradimensional_2 }	
			owner = { is_country_type = extradimensional_3 }	
			owner = { is_country_type = cybrex_empire }	
			owner = { is_country_type = sentinels }	
			owner = { has_dm_tier_munitions = yes }
		}
	  }
	icon_frame = 3
	default = yes
	
	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 2.5
	army_damage = 3 
	
	kill_pop_chance = {
		base = 1.0
	}
	min_pops_to_kill_pop = 0
	
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 100
		modifier = {
			factor = 100
			OR = {
				owner = { has_ethic = ethic_fanatic_xenophile }
				owner = { has_ethic = ethic_fanatic_pacifist }
				owner = { has_valid_civic = civic_machine_servitor }				
				owner = { has_valid_civic = civic_machine_assimilator }			
				owner = { is_country_type = original_empire }		
				owner = { is_country_type = original_empire_active }				
			}
		}
	}
}
