
ag_ancient_bombardment = {
	trigger = {
		OR = {
			owner = { ag_is_ancient_country = yes }
			AND = {
				owner = { NOT = { has_policy_flag = orbital_bombardment_selective } }
				has_fleet_flag = ag_has_ancient_ship_in_fleet
			}
		}
	}
	
	icon_frame = 3
	default = yes
	stop_when_armies_dead = no
	abduct_pops = no
	planet_damage = 5.0
	army_damage = 7.5	
	
	kill_pop_chance = { base = 1.0 }
	min_pops_to_kill_pop = 0
	
	ai_weight = {
		weight = 1001
	}
}

ag_zeta_crisis_bombardment = {
	trigger = { owner = { is_country_type = ag_zeta_crisis_country } }
	
	icon_frame = 3
	default = yes
	stop_when_armies_dead = no
	abduct_pops = no
	planet_damage = 8.0
	army_damage = 25.0
	
	kill_pop_chance = {
		base = 1.0
		modifier = {
			factor = 3.0
			OR = {
				pop_has_trait = trait_mechanical
				pop_has_trait = trait_machine_unit
			}
		}
		modifier = {
			factor = 2.0
			NOR = {
				pop_has_trait = trait_mechanical
				pop_has_trait = trait_machine_unit
			}
			OR = {
				pop_has_trait = trait_cybernetic
				pop_has_trait = trait_self_modified
			}
		}
		modifier = {
			factor = 0.5
			NOR = {
				pop_has_trait = trait_mechanical
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_cybernetic
				pop_has_trait = trait_self_modified
			}
			pop_has_trait = trait_psionic
		}
		modifier = {
			factor = 0.75
			NOR = {
				pop_has_trait = trait_mechanical
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_cybernetic
				pop_has_trait = trait_self_modified
				pop_has_trait = trait_psionic
			}
			pop_has_trait = trait_latent_psionic
		}
	}
	min_pops_to_kill_pop = 0

	ai_weight = {
		weight = 1001
	}
}
