#From Unicron Head Relic - Unlocked by Relic or Tech
unicron = {
	trigger = {
		owner = {
			OR = {
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
			OR = {
				has_modifier = unicron_technology 
				has_technology = tech_dark_energon_bombardment 
			}
		}
	}
	default = no
	
	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 2.0
	army_damage = 1.5
	
	kill_pop_chance = { # Organics will die, but machines will be adversely affected by the effects of dark energon
		base = 2.0
		modifier = {
			factor = 5.0
			OR = {
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0
	
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0.01
			exists = from
			from = { 
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
	}
}
