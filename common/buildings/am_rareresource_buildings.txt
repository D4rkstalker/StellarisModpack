building_luxury_market = {
	base_buildtime = 360
	base_cap_amount = 1

	category = trade
	
	potential = {
		NOT = { is_planet_class = pc_machine }
		exists = owner
		owner = { is_regular_empire = yes }
		owner = { not = { has_policy_flag = alp_private_businesses_prohibited }}
		owner = { resource_stockpile_compare = { resource = sr_rare_metals value >= 1 }}
		owner = { resource_stockpile_compare = { resource = sr_precious_stones value >= 1 }}
	}
	
	allow = {
		exists = owner
		owner = {
			has_monthly_income = { resource = sr_precious_stones value >= 1 }
			has_monthly_income = { resource = sr_rare_metals value >= 1 }
		}
	}
	
	destroy_trigger = {
		exists = owner
		owner = {
			is_ai = yes
			or = {
				resource_stockpile_compare = { resource = sr_rare_metals value < 1 }
				resource_stockpile_compare = { resource = sr_precious_stones value < 1 }
			}
		}
	}
	
	planet_modifier = {
		job_merchant_add = 1
		trade_value_mult = 0.05
		planet_amenities_add = 4
	}
	
	country_modifier = {
		planet_amenities_add = 1
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			sr_precious_stones = 1
			sr_rare_metals = 1
		}
		produces = {
			consumer_goods = 1
		}
		produces = {
			trigger = { 
				planet = {
					exists = owner
					owner = { resource_stockpile_compare = { resource = sr_rare_metals value >= 1 }}
				}
			}
			sr_wealth = 1
			consumer_goods = 1
		}
		produces = {
			trigger = { 
				planet = {
					exists = owner
					owner = { resource_stockpile_compare = { resource = sr_precious_stones value >= 1 }}
				}
			}
			sr_wealth = 1
			consumer_goods = 1
		}
	}
	
	ai_weight = {
		weight = 5
		modifier = {
			factor = 2.0
			free_jobs < 5
			need_more_wealth = yes
		}
		modifier = {
			weight = 0
			free_jobs > 4
			need_more_wealth = no
		}
		modifier = {
			weight = 0
			exists = owner
			owner = {
				or = {
					has_monthly_income = { resource = sr_precious_stones value < 2 }
					has_monthly_income = { resource = sr_rare_metals value < 2 }
				}
			}
		}
	}

	ai_resource_production = {
		sr_wealth = 1
		trigger = {
			always = yes
		}
	}
}
