building_robot_assembly_plant = {
	base_buildtime = 360
	base_cap_amount = 1

	category = pop_assembly
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes 
			NOT = { has_policy_flag = robots_outlawed }
		}
		NOT = { has_modifier = slave_colony }
	}
	
	convert_to = {
		building_machine_assembly_plant
	}
	
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				OR = {
					is_regular_empire = no 
					has_policy_flag = robots_outlawed
				}
			}
			has_modifier = slave_colony
			AND = {
				owner = { is_ai = yes }
				free_jobs = 0
				OR = {
					num_buildings = {
						type = any
						value = 16
					}
					is_planet_class = pc_habitat
				}
			}
		}
	}
	
	planet_modifier = {
		job_roboticist_add = 2
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 5
		}
	}	
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_mechanical_empire = no }		
		}
		text = job_roboticist_effect_desc
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_mechanical_empire = yes }		
		}
		text = job_roboticist_effect2_desc
	}
	
	prerequisites = {
		"tech_robotic_workers"
	}
	
	upgrades = {
		building_synthetic_megaplex
	}
	
	ai_weight = {
		weight = 50000
		modifier = {
			factor = 0
			OR = {
				is_planet_class = pc_habitat
				is_planet_class = pc_city
				num_buildings = { type = any value >= 14 }
				owner = { 
					has_monthly_income = {
						resource = energy
						value < 5
					}
				}
			}
		}
	}
}
