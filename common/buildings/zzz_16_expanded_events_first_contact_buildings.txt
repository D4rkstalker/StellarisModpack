building_lost_sanctuary = {
	icon = building_organic_sanctuary
	can_build = no
	can_demolish = yes
	can_be_ruined = yes

	category = unity

	empire_limit = {
		base = 1
	}

	base_buildtime = @b0_time

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b0_minerals
		}
		upkeep = {
			energy = @b0_upkeep
		}
	}

	potential = {
		exists = owner
		owner = {
			has_origin = origin_paradise_lost
		}
		NOT = { has_planet_flag = building_lost_sanctuary_upgraded }
	}

	destroy_trigger = {
		exists = owner
		owner = {
			NOT = { has_origin = origin_paradise_lost }
		}
	}

	planet_modifier = {
		job_lost_trophy_add = 2
	}

	triggered_desc = {
		text = job_lost_trophy_effect_desc
	}
}
