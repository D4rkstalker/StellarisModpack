building_ie_institute = {
	base_buildtime = 5280
	base_cap_amount = 1

	category = research

	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = { is_regular_empire = yes }
		is_planet_class = pc_ie_gateway
	}

	destroy_trigger = {
		exists = owner
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			owner = {
				is_gestalt = no
				is_regular_empire = no
			}
			buildings_no_exotic_gases = yes
		}
	}

	convert_to = {
		building_ie_supercomputer
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 6600
			exotic_gases = 550
		}
		upkeep = {
			energy = 55
			exotic_gases = 11
		}
	}

	planet_modifier = {
		planet_researchers_produces_mult = 0.15
		job_head_researcher_add = 11
	}

	prerequisites = {
		"tech_global_research_initiative"
	}

	triggered_desc = {
		text = job_head_researcher_effect_desc
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	ai_resource_production = {
		trigger = { always = yes }
		physics_research = 33
		society_research = 33
		engineering_research = 33
	}
	ai_resource_production = {
		trigger = {
			planet_resource_compare = {
				resource = physics_research
				value >= 20
			}
		}
		physics_research = 33
		society_research = 33
		engineering_research = 33
	}
	ai_resource_production = {
		trigger = {
			planet_resource_compare = {
				resource = physics_research
				value >= 40
			}
		}
		physics_research = 33
		society_research = 33
		engineering_research = 33
	}
}
