building_commercial_zone = {
	base_buildtime = @b1_time

	exempt_from_ai_planet_specialization = no
	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
		NOT = { has_building = building_low_tech_capital }
	}

    destroy_trigger = {
        exists = owner
        OR = {
            owner = { is_regular_empire = no }
            has_modifier = slave_colony
        }
    }

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	planet_modifier = {
		job_clerk_add = 3
        job_bugged_trader_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_merchant_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		text = job_merchant_effect_desc
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	upgrades = {
		"building_commercial_megaplex"
	}

    triggered_desc = {
        text = job_clerk_effect_desc
    }
    triggered_desc = {
        text = job_bugged_trader_effect_desc
    }
}

building_commercial_megaplex = {

	category = trade

		exists = owner

		}
	}

	resources = {
		category = planet_buildings

			minerals = @b2_minerals

		}

			energy = @b2_upkeep
			rare_crystals = @b2_rare_upkeep
		}
	}

    planet_modifier = {
        job_merchant_add = 1
		job_bugged_trader_add = 2
        job_clerk_add = 4
    }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_merchant_add = 1
		}
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_desc = {
        text = job_bugged_trader_effect_desc
    }
	triggered_desc = {
		text = job_merchant_effect_desc
	}

	prerequisites = {
		"tech_interstellar_economics"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	ai_resource_production = {
		energy = 6
		trigger = {
			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
				job = servant
					value >= 10
				}
			}
		}
	}
}

building_galactic_stock_exchange = {
	base_buildtime = @b2_time

	category = trade
	can_build = no
   	potential = {
        NOR = {
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
   	}

    allow = {
		OR = {
			has_global_flag = galactic_market_founded
			owner = {
				is_bugged_globalism = yes
			}
		}
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
	}


	planet_modifier = {
		job_merchant_add = 2
		trade_value_mult = 0.1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_civic = civic_merchant_guilds
					has_civic = civic_bugged_state_monopoly
				}
			}
		}
		modifier = {
			job_merchant_add = 2
			trade_value_mult = 0.1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			rare_crystals = @b2_rare_upkeep
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	triggered_desc = {
		text = job_merchant_effect_desc
	}

	ai_resource_production = {
		energy = 5
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}

	ai_resource_production = {
		energy = 3
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
