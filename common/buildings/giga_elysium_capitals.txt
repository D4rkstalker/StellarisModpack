		planet_housing_add = 5

		job_clerk_add = 2

		planet_max_branch_office_buildings_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_tradition = tr_psionics_psi_corps
			}
		}
		modifier = {
			job_enforcer_add = -1
			job_telepath_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_synthetics_finish
			}
		}

			job_clerk_add = -1

		}
	}

	triggered_desc = {
		text = job_enforcer_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_tradition = tr_psionics_psi_corps
			}
		}
		text = job_telepath_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				or = {
					has_active_tradition = tr_synthetics_finish
					and = {
						not = { has_tradition = tr_genetics_adopt }

	triggered_desc = {

			}
		}

	}
	triggered_planet_modifier = {
