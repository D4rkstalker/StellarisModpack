building_luxury_residence = {
	base_buildtime = @b1_time

	category = amenity

	potential = {
		NOT = { is_planet_class = pc_habitat }
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = {
			is_regular_empire = yes
			is_ai = no
			NOT = { has_valid_civic = civic_shared_burden }
		}
	}


	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				free_building_slots = 0
				free_jobs < 1
				free_housing > 10
				owner = {
					is_ai = yes
				}
			}
			owner = { is_regular_empire = no }
			owner = { has_valid_civic = civic_shared_burden }
			has_modifier = slave_colony
		}
	}

	convert_to = {
		building_drone_storage
		building_hive_warren
		building_communal_housing
	}

	planet_modifier = {
		planet_housing_add = 3
		planet_amenities_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_non_swapped_tradition = tr_domination_imperious_architecture
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	upgrades = {
		"building_paradise_dome"
	}

	ai_weight = {
		weight = 0.5
			modifier = {
				factor = 0
				unsupervised_destruction = yes
			}
	}
}

building_holo_theatres = {
	base_buildtime = @b1_time

	category = amenity

	potential = {
		NOT = {	has_modifier = slave_colony	}
		owner = { is_regular_empire = yes }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				free_building_slots = 0
				owner = { is_ai = yes }
				OR = {
					free_amenities > 30
					AND = {
						count_pops = { limit = { is_sapient = yes } count < 75 }
					}
					AND = {
						owner = {
							is_ai = yes
							NOT = { has_policy_flag = robots_outlawed }
							has_technology = tech_robotic_workers
						}
						NOT = { has_building = building_robot_assembly_plant }
					}
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				NOT = { has_valid_civic = civic_warrior_culture }
			}
		}
		modifier = {
			job_entertainer_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_warrior_culture
			}
		}
		modifier = {
			job_duelist_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	upgrades = {
		"building_hyper_entertainment_forum"
	}

	prerequisites = {
		"tech_holo_entertainment"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				NOT = { has_valid_civic = civic_warrior_culture }
			}
		}
		text = job_entertainer_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_valid_civic = civic_warrior_culture
			}
		}
		text = job_duelist_effect_desc
	}

	ai_weight = {
		weight = 0.7
		
		modifier = {
			factor = 5
			free_amenities < 0	
			free_building_slots > 2
			not = {
				has_planet_flag = just_built_theater
			}
		}

		modifier ={
			factor = 0.1
			num_pops < 15 
		}
		modifier ={
			factor = 0.01
			free_amenities > 30
		}
		modifier ={
			factor = 0.2
			free_amenities > 15
		}
		modifier ={
			factor = 0.3
			free_amenities > 10
		}
		modifier = {
			factor = 20
			free_amenities < -6
			num_pops > 14
		}
		modifier = {
			factor = 5
			free_amenities < -2
			num_pops > 14
		}
		modifier = {
			factor = 10
			free_amenities < -8
			num_pops > 14
		}
		modifier = {
			factor = 100
			free_amenities < -10
			num_pops > 14
		}
		modifier = {
			factor = 1.5
			has_planet_flag = just_destroyed_theater
			free_amenities < 10
		}
	}

	on_destroy = {
		set_timed_planet_flag = {
			flag = just_destroyed_theater
			days = 35
		}
	}
	on_built = {
		set_timed_planet_flag = {
			flag = just_built_theater
			days = 35
		}
		if = {
			limit = { 
				free_housing < 0
				free_district_slots = 0 
			}
			city_district_check = yes
		}	
	}
}
