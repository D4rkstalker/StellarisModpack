building_mining_consortium = {
	base_buildtime = 360

	category = trade
	
	potential = {
		or = {
			habitable_structure = no
			and = {
				is_planet_class = pc_habitat
				has_planet_flag = mining_habitat
			}
		}
		NOT = { is_planet_class = pc_machine }
		NOT = { has_modifier = resort_colony }
		exists = owner
		owner = { is_regular_empire = yes }
	}
	
	planet_modifier = {
		job_miner_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {  not = { has_policy_flag = alp_private_businesses_prohibited }}
		}
		modifier = {
			trade_value_add = 2
			job_retailer_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_prohibited }
		}
		modifier = {
			job_labourer_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_freedom }
		}
		modifier = {
			trade_value_add = 2
			job_retailer_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { not = { has_policy_flag = alp_private_businesses_freedom } }
		}
		modifier = {
			job_miner_add = 1
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			minerals = 1
		}
		upkeep = {
			trigger = { owner = { not = { has_policy_flag = alp_private_businesses_prohibited }}}
			minerals = 1
		}
		produces = {
			sr_waste = 2
		}
		produces = {
			trigger = { owner = { not = { has_policy_flag = alp_private_businesses_prohibited }}}
			sr_wealth = 1
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_mining_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_miner_effect_desc
	}
	
	prerequisites = {
		"tech_mechanized_mining"
	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			has_building = building_mining_consortium
		}
		modifier = {
			factor = 5
			owner = { 
				has_monthly_income = {
					resource = minerals
					value < 5
				}
			}
		}
		modifier = {
			factor = 0
			owner = { 
				has_monthly_income = {
					resource = minerals
					value > 50
				}
			}
		}
	}

	ai_resource_production = {
		minerals = 1
		trigger = {
			always = yes
		}
	}
}
