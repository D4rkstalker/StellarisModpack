building_ie_luxury_residence = {
	base_buildtime = 3960

	category = amenity

	potential = {
		OR = {
			NOT = { is_planet_class = pc_habitat }
			has_planet_flag = advanced_habitat
		}
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = {
			is_regular_empire = yes
			NOT = { has_valid_civic = civic_shared_burden }
		}
		is_planet_class = pc_ie_gateway
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				free_building_slots = 0
				free_jobs < 1
				free_housing > 10
				owner = {
					is_ai = yes
				}
			}
			owner = { is_regular_empire = no }
			owner = { has_valid_civic = civic_shared_burden }
			has_modifier = slave_colony
		}
	}

	convert_to = {
		building_ie_drone_storage
		building_ie_hive_warren
		building_ie_communal_housing
	}

	planet_modifier = {
		planet_housing_add = 33
		planet_amenities_add = 55
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_non_swapped_tradition = tr_domination_imperious_architecture
			}
		}
		modifier = {
			planet_housing_add = 22
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 4400
		}
		upkeep = {
			energy = 22
		}
	}

	upgrades = {
		"building_ie_paradise_dome"
	}
}

	}

				free_building_slots = 0

						count_owned_pop = { limit = { is_sapient = yes } count < 75 }

						}

							has_building = building_ie_clinic
							has_building = building_ie_hospital
						}

						owner = {
							NOT = { has_policy_flag = robots_outlawed }
