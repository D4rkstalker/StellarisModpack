	}

		potential = {
			exists = owner

				is_gestalt = no

		}
	}

			exists = owner
			owner = {
				has_ethic = ethic_fanatic_militarist
			}
		}
		modifier = {
			job_soldier_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
