building_ie_commercial_zone = {
	base_buildtime = 3960

	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
		is_planet_class = pc_ie_gateway
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}

	planet_modifier = {
		job_clerk_add = 33
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 4400
		}
		upkeep = {
			energy = 22
		}
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	upgrades = {
		"building_ie_commercial_megaplex"
	}

	ai_resource_production = {
		energy = 22
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 33
		trigger = {
			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
					job = servant
					value >= 10
				}
			}
		}
	}
}

building_ie_commercial_megaplex = {
	base_buildtime = 5280
	can_build = no

	category = trade

	potential = {
		exists = owner

		text = job_merchant_effect_desc

	ai_resource_production = {

}

building_ie_galactic_stock_exchange = {
	base_buildtime = 5280
	base_cap_amount = 1

	category = trade

	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		is_planet_class = pc_ie_gateway
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
			buildings_no_crystals = yes
		}
	}


	planet_modifier = {
		job_merchant_add = 22
		trade_value_mult = 0.20
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 6600
			rare_crystals = 550
		}
		upkeep = {
			energy = 55
			rare_crystals = 11
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	triggered_desc = {
		text = job_merchant_effect_desc
	}

	ai_resource_production = {
		energy = 33
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 55
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}
	ai_resource_production = {
		energy = 33
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
