apsr8_building_part1 = {
	base_buildtime = @apsr8part1cost_time
	icon = apsr8_building_part1
	base_cap_amount = 1
	
	is_capped_by_modifier = no

	category = resource
	
	can_demolish = no
#	can_be_ruined = no
#	can_be_disabled = no
	position_priority = 49

	allow = {
		apsr_buildings_allow = yes
	}
	
	potential = {
		AND = {
			has_modifier = "apsr8_planet_modifier"
			NOT = {
				has_planet_flag = apsr8_cataclysm
			}
		}
	}
	
	# Настройка производства ДЗ
	triggered_planet_modifier = { # 50%
		potential = {
			has_global_flag = apsr_advanced_knowledge_50
		}
		modifier = {
			planet_apsr_buildings_produces_mult = -0.50
		}
	}
	triggered_planet_modifier = { # 150%
		potential = {
			has_global_flag = apsr_advanced_knowledge_150
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 0.50
		}
	}
	triggered_planet_modifier = { # 200%
		potential = {
			has_global_flag = apsr_advanced_knowledge_200
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 1.00
		}
	}
	triggered_planet_modifier = { # 300%
		potential = {
			has_global_flag = apsr_advanced_knowledge_300
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 2.00
		}
	}
	triggered_planet_modifier = { # 500%
		potential = {
			has_global_flag = apsr_advanced_knowledge_500
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 4.00
		}
	}

			
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_hive_empire = yes 
				has_technology = tech_apsr8_building_part1
			}
		}
		modifier = {
			job_apsr8_technician_hive_add = @apsr8part1job
			job_apsr8_power_engineer_hive_add = @apsr8part1powjob
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_machine_empire = yes 
				has_technology = tech_apsr8_building_part1
			}
		}
		modifier = {
			job_apsr8_technician_machine_add = @apsr8part1job
			job_apsr8_power_engineer_machine_add = @apsr8part1powjob
		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
				has_technology = tech_apsr8_building_part1
			}
		}
		modifier = {
			job_apsr8_technician_normal_add = @apsr8part1job
			job_apsr8_power_engineer_normal_add = @apsr8part1powjob
		}
	}
	
	resources = {
		category = planet_apsr_buildings
		cost = {
			minerals = @apsr8part1cost_minerals
			apsr_knowledge = @apsr8part1cost_knowledges
		}
		upkeep = {
			trigger = { owner = { has_technology = tech_apsr8_building_part1 } }
			energy = @apsr8part1upkeep_energy
			minerals = @apsr8part1upkeep_minerals
		}
		upkeep = {
			trigger = { owner = { NOT = { has_technology = tech_apsr8_building_part1 } } }
			energy = @apsr8part1upkeep_fake
		}
		produces = {
			trigger = { 
				exists = owner
				owner = { 
					has_technology = tech_apsr8_building_part1
				}
			}
			apsr_knowledge = 0.3
		}
	}
	
	upgrades = {
		"apsr8_building_part2"
	}	
	
	prerequisites = {
		"tech_apsr8_building_part1"
	}
	show_tech_unlock_if = {
		always = no
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				NOT = {
					has_technology = tech_apsr8_building_part1
				}
			}
		}
		text = apsr_building_fake_desc
	}
}
