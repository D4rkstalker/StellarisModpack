building_commercial_zone_alt = {
	base_buildtime = @b1_time

	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}

	planet_modifier = {
		job_clerk_add = 1
		job_merchant_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_authority = auth_corporate }
		}
		modifier = {
			job_wageslave_add = 1
			job_retailer_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { not = { has_authority = auth_corporate }}
		}
		modifier = {
			job_clerk_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_prohibited }
		}
		modifier = {
			job_clerk_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_restricted }
		}
		modifier = {
			job_clerk_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_regulated }
		}
		modifier = {
			job_retailer_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_policy_flag = alp_private_businesses_freedom }
		}
		modifier = {
			job_wageslave_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
		produces = {
			sr_wealth = 0.25
		}
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	upgrades = {
		"building_commercial_megaplex_alt"
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			OR = {
				has_building = building_holo_theatres
				has_building = building_hyper_entertainment_forum
			}
			OR = {
				num_buildings = { type = building_research_lab_1 value = 3 } #Prevents AI falling behind on research to all but the most research dedicated players
				has_building = building_research_lab_2
				has_building = building_research_lab_3
			}
			NOR = {
				has_building = building_commercial_zone_alt
				has_building = building_commercial_megaplex_alt
			}
		}
		modifier = {
			factor = 10
			exists = owner
			owner = {
				is_megacorp = yes
			}
			num_buildings = { type = building_commercial_zone_alt value < 1 }
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}
