b_psi_council = {
	base_buildtime = 240
	
	can_be_disabled = no
	can_be_ruined = no
	can_demolish = no
	
	allow = {
		NOR = {
			owner = { has_country_flag = psi_testing_enabled }
			owner = { has_country_flag = building_psi }
		}
	}

	resources = {
            category = planet_buildings
            cost = {
		alloys = 100
		energy = 100
            }
            upkeep = {
                energy = 3
            }
	}

	prerequisites = {
		"tech_telepathy"
	}

	destroy_trigger = {
		owner = {
			NOT = { has_technology = "tech_telepathy" }
		}
	}

	abort_trigger = {
		has_country_flag = psi_testing_enabled
	}

	on_queued = {
		owner = { set_country_flag = building_psi }
	}
    
	on_unqueued = {
		owner = { remove_country_flag = building_psi }
	}

	on_built = {
		set_planet_flag = psicouncil
		owner = { set_country_flag = psi_testing_enabled }
		owner = { remove_country_flag = building_psi }
	}
    
	on_destroy = {
		remove_planet_flag = psicouncil
		owner = { remove_country_flag = psi_testing_enabled }
		owner = { remove_country_flag = psi_testing_ongoing }
	}
}
