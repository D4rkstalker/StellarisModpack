# Mote Harvesting Traps Upgrades

building_mote_harvesting_traps_2 = { 
	base_buildtime = @b2_time 
	can_build = no
	icon = building_mote_harvesting_traps_2
	category = resource 
	potential = {
		NOT = { has_modifier = resort_colony }
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }

			exists = owner

		}
		modifier = {
			job_mote_harvesting_drone_add = @b1_jobs
		}

			minerals = @b1_minerals
			alloys = 50
		}

			energy = @b1_upkeep

		}
	}

	upgrades = { 
		"building_mote_harvesting_traps_3"
	}

	prerequisites = {
		"tech_mote_harvesting_traps_2"

	}
