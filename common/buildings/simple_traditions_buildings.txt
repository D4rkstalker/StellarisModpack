simple_traditions_galactic_university = { 
	base_buildtime = 600
	category = government
	base_cap_amount = 1
	icon = simple_traditions_university

	planet_modifier = {
		job_bureaucrat_add = 1
		job_head_researcher_add = 1
		job_culture_worker_add = 2
	}
	on_built = {
		owner = {
			country_event = {
				id = simple_tradition.8
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 5
			consumer_goods = 2
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			NOT = {
				owner = { has_tradition = tr_leadership_adopt }
			}
			is_capital = no
			owner = {
				OR = {
					is_megacorp = yes
					is_hive_empire = yes
					is_machine_empire = yes
				}
			}
		}
	}
	convert_to = {
		simple_traditions_galactic_university_megacorp
		simple_traditions_galactic_university_hive
		simple_traditions_galactic_university_machine
	}

	potential = {
		exists = owner
		is_capital = yes
		owner = {
			has_tradition = tr_leadership_adopt
			is_megacorp = no
			is_hive_empire = no
			is_machine_empire = no
		}
	}
}
