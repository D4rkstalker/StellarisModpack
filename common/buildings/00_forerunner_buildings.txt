# SENTINEL FACTORY
building_sentinel_factory = {
	base_buildtime = @b2_time
	icon = "building_sentinel_factory"
	base_cap_amount = 1
	
	category = pop_assembly
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = {
			OR = {
				has_trait = trait_forerunner_species
				has_trait = trait_robot_sentinel_ai
				has_trait = trait_robot_promethean_ai
				has_trait = trait_robot_created_ai
			}
		}
		NOT = { has_modifier = slave_colony }
	}	
	
	convert_to = {
	}
	allow = {
		has_upgraded_capital = yes
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				OR = {
					is_regular_empire = no 
					has_policy_flag = robots_outlawed
				}
			}
			has_modifier = slave_colony
			AND = {
				owner = { is_ai = yes }
				free_jobs < 1
				OR = {
					num_buildings = {
						type = any
						value = 16
					}
					is_planet_class = pc_habitat
				}
			}
		}
	}
	planet_modifier = {
		pop_assembly_speed = 0.10
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_roboticist_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_replicator_add = 1
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_replicator_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_roboticist_effect_desc
	}
	country_modifier = {
	}
	
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
	}
	prerequisites = {
		"tech_robotic_workers"
	}
	upgrades = {
		"building_promethean_factory"
	}
	ai_weight = {
		weight = 50000
		modifier = {
			factor = 0
			OR = {
				is_planet_class = pc_habitat
				is_planet_class = pc_city
				num_buildings = { type = any value >= 14 }
			}
		}
	}
	
}
