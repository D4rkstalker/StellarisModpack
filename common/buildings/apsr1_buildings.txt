### UPKEEP
@apsr1part1upkeep_energy = 3
@apsr1part1upkeep_minerals = 1

# CUSTODIAN JOBS
@apsr1part1job = 1
@apsr1part2job = 4
@apsr1part3job = 10




apsr1_building_part1 = {
	base_buildtime = @apsr1part1cost_time
	icon = apsr1_building_part1
	base_cap_amount = 1
	
	is_capped_by_modifier = no

	category = government
	
	can_demolish = no
#	can_be_ruined = no
#	can_be_disabled = no
	position_priority = 49
	
	allow = {
		apsr_buildings_allow = yes
	}
	
	potential = {
		has_modifier = "apsr1_planet_modifier"
	}
	
	# Настройка производства ДЗ
	triggered_planet_modifier = { # 50%
		potential = {
			has_global_flag = apsr_advanced_knowledge_50
		}
		modifier = {
			planet_apsr_buildings_produces_mult = -0.50
		}
	}
	triggered_planet_modifier = { # 150%
		potential = {
			has_global_flag = apsr_advanced_knowledge_150
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 0.50
		}
	}
	triggered_planet_modifier = { # 200%
		potential = {
			has_global_flag = apsr_advanced_knowledge_200
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 1.00
		}
	}
	triggered_planet_modifier = { # 300%
		potential = {
			has_global_flag = apsr_advanced_knowledge_300
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 2.00
		}
	}
	triggered_planet_modifier = { # 500%
		potential = {
			has_global_flag = apsr_advanced_knowledge_500
		}
		modifier = {
			planet_apsr_buildings_produces_mult = 4.00
		}
	}

			
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_hive_empire = yes 
				has_technology = tech_apsr1_building_part1
			}
		}
		modifier = {
			job_apsr1_custodian_hive_add = @apsr1part1job
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_machine_empire = yes 
				has_technology = tech_apsr1_building_part1
			}
		}
		modifier = {
			job_apsr1_custodian_machine_add = @apsr1part1job
		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
				has_technology = tech_apsr1_building_part1
			}
		}
		modifier = {
			job_apsr1_custodian_normal_add = @apsr1part1job
		}
	}
	
	resources = {
		category = planet_apsr_buildings
		cost = {
			minerals = @apsr1part1cost_minerals
			apsr_knowledge = @apsr1part1cost_knowledges
		}
		upkeep = {
			trigger = { owner = { has_technology = tech_apsr1_building_part1 } }
			energy = @apsr1part1upkeep_energy
			minerals = @apsr1part1upkeep_minerals
		}
		upkeep = {
			trigger = { owner = { NOT = { has_technology = tech_apsr1_building_part1 } } }
			energy = @apsr1part1upkeep_fake
		}
		produces = {
			trigger = { 
				exists = owner
				owner = { 
					has_technology = tech_apsr1_building_part1
				}
			}
			apsr_knowledge = 0.3
		}
	}
	
	upgrades = {
		"apsr1_building_part2"
	}	
	
	prerequisites = {
		"tech_apsr1_building_part1"
	}
	show_tech_unlock_if = {
		always = no
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				NOT = {
					has_technology = tech_apsr1_building_part1
				}
			}
		}
		text = apsr_building_fake_desc
	}
}
