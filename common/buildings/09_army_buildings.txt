building_stronghold = {
	base_buildtime = 240
	add_to_first_building_slot = yes

	category = army
	
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		AND = {
			OR = {
				num_buildings = { type = building_stronghold value > 1 }
				has_building = building_fortress
			}
			free_building_slots = 0
			owner = { 
				used_naval_capacity_percent < 0.6
				is_at_war = no # don't destroy while at war
			}
		}
	}

		}

	}

	}
	triggered_desc = {

		}
		text = job_soldier_effect_desc

	ai_weight = {

			factor = 200

			has_technology = tech_battleships 
			has_technology = tech_energy_torpedoes_1 
			used_naval_capacity_integer > 300
			resource_stockpile_compare = {
				resource = alloys
				value > 4000
			}
		}
		}

		modifier = {
			factor = 20
			free_building_slots > 2
			years_passed > 50
			owner = {
			used_naval_capacity_integer > 300
			used_naval_capacity_percent > 0.7
			resource_stockpile_compare = {
				resource = alloys
				value > 4000
			} }
		}

		modifier = {
			factor = 0.5
			free_amenities < 5
		}

			factor = 0.5
			owner = { needs_goods = yes }
		}
		modifier = {
			factor = 5
			lategame = yes
		}
		modifier ={
			factor = 3
			lategame = yes
			num_buildings = {
				type = building_stronghold
				value < 3
			}
		}
		modifier ={
			factor = 1.5
			owner = {
				used_naval_capacity_percent > 0.9
			}
		}
		modifier ={
			factor = 0.05
			owner = {
				used_naval_capacity_percent < 0.6
			}
		}
		modifier = {

			factor = 4

			factor = 10

			}
		}
		modifier = {
			factor = 0.1
			owner = {
				or = {
				resource_stockpile_compare = {
					resource = alloys
					value < 2000
				}
				used_naval_capacity_percent < 0.9
			}
			}
		}
	}

	on_built = {
		if = {
			limit = {
				free_housing < 0
				free_district_slots = 0
			}
		}
	}

building_fortress = {
	base_buildtime = @b1_time
	can_build = no
	planetary_ftl_inhibitor = yes
	add_to_first_building_slot = yes

	category = army
	
	allow = {
		has_upgraded_capital = yes
	}
	
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		OR = {
			AND = {
				num_buildings = { type = building_fortress value > 1 }
				free_building_slots = 0
				owner = { used_naval_capacity_percent < 0.4	}
			}
			buildings_no_motes = yes
		}
	}
	
	planet_modifier = {
		planet_housing_add = 3
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 3
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 3
		}
	}

	resources = {
		category = planet_buildings_strongholds
		cost = {
			minerals = @b2_minerals
			volatile_motes = @b2_rare_cost
		}			
		upkeep = {
			energy = 1
			volatile_motes = @b2_rare_upkeep
		}
	}
	
	prerequisites = {
		"tech_global_defense_grid"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
	
	ai_weight = {
		weight = 50
		modifier = {
			factor = 0
			unsupervised_destruction = yes
			has_planet_flag = ecumen
			not = {
				has_building = building_ministry_production
			}
			not = {
				has_building = building_stronghold
			}
			owner = {
				has_technology = tech_global_production_strategy
			}
		}
		modifier = {
			factor = 0.01
			owner = {
				has_country_flag = adv_foundry_built
			}
		}
		modifier = {
			factor = 0
			unsupervised_destruction = yes
			owner = { resource_income_compare = {
				resource = volatile_motes
				value < 0 } 
			not = {
				has_country_flag = adv_foundry_destroyed
			}
			}
		}
		modifier = {
			factor = 0.01
			owner = {
				resource_income_compare = {
					resource = volatile_motes
					value < 2
				}
			}
		}
	}


	on_built = {
		owner = { set_timed_country_flag = {
			flag = adv_foundry_built
			days = 1000 }
		}
	}

}
