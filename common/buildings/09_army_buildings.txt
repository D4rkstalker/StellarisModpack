building_stronghold = {
	base_buildtime = 240
	add_to_first_building_slot = yes

	category = army
	
	potential = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		AND = {
			OR = {
				num_buildings = { type = building_stronghold value > 1 }
				has_building = building_fortress
			}
			free_building_slots = 0
			owner = { 
				used_naval_capacity_percent < 0.6
				is_at_war = no # don't destroy while at war
			}
		}
	}
	
	planet_modifier = {
		planet_housing_add = 1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_feudal_realm }
		}
		modifier = {
			planet_housing_add = 1

		}
	}

	resources = {
		category = planet_buildings_strongholds

			minerals = @b1_minerals
		}			
		upkeep = {

		}
		produces = {
			trigger = {

		}

		"tech_planetary_defenses"
	}
	
	upgrades = {
		"building_fortress"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}	
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			free_jobs < 1
			free_building_slots > 1
			owner = { used_naval_capacity_percent > 0.9	}
		}
		modifier = {
			weight = 100
			is_planet_class = pc_habitat
			owner = { used_naval_capacity_percent > 0.8	}
		}
		modifier = {
			weight = 25
			num_pops > 39
			NOR = {	
				has_building = building_stronghold 
				has_building = building_fortress
			}
		}
	}
}
