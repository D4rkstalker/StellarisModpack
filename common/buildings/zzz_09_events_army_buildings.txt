building_planetary_shield_generator = {
	base_buildtime = @b3_time
	base_cap_amount = 1
	position_priority = 100

	category = army
	
	potential = {
		NOT = { has_global_flag = mem_more_events_mod }
	}

	allow = {
		has_upgraded_capital = yes
		#hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}

	#destroy_trigger = {
	#	exists = owner
	#	owner = { is_ai = yes }
	#}

	planet_modifier = {
		planet_orbital_bombardment_damage = -0.5
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_country_flag = moon_bump_shield_completed }
		}
		modifier = {
			planet_orbital_bombardment_damage = -0.25
		}
	}

	resources = {
		category = planet_buildings
		produces = {
			physics_research = 5
			engineering_research = 5
		}
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
	}

	prerequisites = {
		"tech_planetary_shield_generator"
	}
	
	show_tech_unlock_if = {
		NOT = { has_global_flag = mem_more_events_mod }
	}
	
	ai_resource_production = {
		trigger = { always = yes }
		physics_research = 1
		society_research = 1
		engineering_research = 1
	}
	# ai should build these to protect against locusts
	ai_resource_production = {
		alloys = 8
		minerals = 8
		food = 8
		society_research = 10
		engineering_research = 10
		physics_research = 10
		trigger = {
			solar_system = {
				any_neighbor_system = { has_star_flag = space_locusts_current_location }
			}
		}
	}
	ai_resource_production = {
		alloys = 16
		minerals = 16
		food = 16
		society_research = 20
		engineering_research = 20
		physics_research = 20
		trigger = {
			solar_system = { has_star_flag = space_locusts_current_location }
		}
	}
}
