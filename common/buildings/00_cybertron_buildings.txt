#ENERGON MINES
building_energon_mine = {
	base_buildtime = @b1_time
	icon = "building_energon_mine"
	
	is_capped_by_modifier = yes
	
	category = resource
	
	potential = {
		NOT = { has_modifier = resort_colony }
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
			}
		}
		modifier = {
			planet_jobs_energon_crystals_produces_mult = 0.10
			job_energon_miner_add = 4
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = yes
			}
		}
		modifier = {
			planet_jobs_energon_crystals_produces_mult = 0.10
			job_energon_mining_drone_add = 4
		}
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}
	prerequisites = {
		"tech_energon_mining"
	}
	ai_weight = {
		weight = 200
		modifier = {
			weight = 30
			owner = {
				has_monthly_income = {
					resource = energon_crystals
					value < 3
				}#5 is too high since that means 2 extra buildings. Better to just go with 1 extra building
			}
		}
		modifier = {
			weight = 200
			owner = {
				has_monthly_income = {
					resource = energon_crystals
					value < 0
				}#5 is too high since that means 2 extra buildings. Better to just go with 1 extra building
			}
		}
		modifier = {
			weight = 1000
			owner = {
				OR = {
					has_trait = trait_cybertronian
					has_trait = trait_bw_cybertronian
					has_trait = trait_transtech_cybertronian
				}
			}
		}
	}
	ai_resource_production = {
		trigger = { always = yes }
		energon_crystals = 10
	}
}
