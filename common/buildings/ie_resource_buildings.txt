	base_buildtime = 3960

	category = resource

	potential = {
		exists = owner

			OR = {
				country_uses_food = yes

			}
		}
		NOT = { is_planet_class = pc_machine }

		exists = owner

				NOT = { is_planet_class = pc_habitat }

		}
		modifier = {
			job_fe_acolyte_farm_add = 22
		}
