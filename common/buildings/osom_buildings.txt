osom_building_speleology_facility_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = research

	potential = {
		NOR = {
			uses_district_set = habitat
			uses_district_set = ring_world
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
		building_relaxed_basic_income_check = yes
	}

	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
	}

	allow = {
		has_upgraded_capital = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_osom_speleologist_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_osom_mapping_drone_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_osom_mapping_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}
	
	upgrades = {
		"osom_building_speleology_facility_2"
	}

	prerequisites = {
		"osom_tech_speleology_1"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_osom_speleologist_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_osom_mapping_drone_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_osom_mapping_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 75
			num_pops >= 15
			free_amenities > 2 #Prevents running out of amenities
			owner = {
				OR = {
					has_monthly_income = {
						resource = consumer_goods
						value > 0
					}
					country_uses_consumer_goods = no
				}
			} # so that we don't run out of consumer goods
		}
		modifier = {
			weight = 50
			free_amenities > 2 #Prevents running out of amenities
			owner = {
				OR = {
					has_monthly_income = {
						resource = consumer_goods
						value > 0
					}
					country_uses_consumer_goods = no
				}
			} # so that we don't run out of consumer goods

			# allow additional research buildings if we are behind in tech, but not if we already have several have upgraded ones
			num_buildings = { type = building_research_lab_1 value < 4 }
			NOR = {
				num_buildings = { type = building_research_lab_2 value > 1 }
				num_buildings = { type = building_research_lab_3 value > 1 }
			}
			OR = {
				num_pops >= 40
				owner = {
					any_neighbor_country = {
						is_country_type = default
						relative_power = {
							who = prev
							category = technology
							value > equivalent
						}
						relative_power = {
							who = prev
							category = all
							value < overwhelming
						}
					}
				}
			}
			OR = {
				owner = {
					is_gestalt = yes
					has_monthly_income = {
						resource = alloys
						value > 6
					}
					has_monthly_income = {
						resource = minerals
						value > 50
					}
				}
				AND = {
					owner = {
						is_regular_empire = yes
						has_monthly_income = {
							resource = minerals
							value > 80
						}
					}
					jobs_work_minerals_goods = yes
				}
				owner = {
					is_regular_empire = yes
					AND = {
						has_monthly_income = {
							resource = consumer_goods
							value > 25
						}
						has_monthly_income = {
							resource = minerals
							value > 50
						}
					}
				}
			}
		}
		modifier = {
			factor = 0
			free_jobs > 3
		}
		modifier = {
			weight = 100
			free_building_slots = 0
			has_building_construction = no
			has_building = osom_building_speleology_facility_1
		}#Repairs
	}
}

osom_building_speleology_facility_2 = {
	base_buildtime = @b2_time
	can_build = no

	category = research

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
			has_modifier = resort_colony
			buildings_no_exotic_gases = yes
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_head_researcher_add = 1
			job_osom_speleologist_add = 4
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_brain_drone_add = 1
			job_osom_mapping_drone_add = 4
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_calculator_add = 1
			job_osom_mapping_drone_add = 4
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			exotic_gases = @b2_rare_cost
		}
		upkeep = {
			consumer_goods = @b1_upkeep
			energy = @b2_upkeep
			exotic_gases = @b2_rare_upkeep
		}
	}

	upgrades = {
		"osom_building_speleology_facility_3"
	}

	prerequisites = {
		"osom_tech_speleology_2"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_head_researcher_effect_desc
		text = job_osom_speleologist_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_brain_drone_effect_desc
		text = job_osom_mapping_drone_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_calculator_effect_desc
		text = job_osom_mapping_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
		weight = 10
			planet = {
				owner = {
					has_monthly_income = {
						resource = exotic_gases
						value > 0
					}
					OR = {
						AND = {
							is_regular_empire = yes
							has_monthly_income = {
								resource = consumer_goods
								value > 5
							}
						}
						AND = {
							is_regular_empire = no
							has_monthly_income = {
								resource = minerals
								value > 75
							}
						}
					}
				}
			}
		}
		modifier = {
			factor = 0
			free_jobs > 3 #Extremely high priority building
		}
	}
}
