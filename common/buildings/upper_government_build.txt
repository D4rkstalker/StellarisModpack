building_super_computers_station = {
	
	base_buildtime = @building_build_time1
	can_demolish = yes
	base_cap_amount = 2

		category = government
		
		potential = {
			NOT = { has_modifier = resort_colony }
			NOT = { has_modifier = slave_colony }
		}

		allow = {
			has_upgraded_capital = yes
		}

		destroy_trigger = {
			exists = owner
            owner = {
                is_ai = yes
            }
            AND = {
				free_building_slots = 0
				owner = {
					has_monthly_income = {
						resource = energy
						value < -2
					}
				}
			}
			owner = {
				NOT = {
					has_country_flag = freed_energy
				}
			}
		}

		on_destroy = {
			owner = {
				set_timed_country_flag = {
					flag = freed_energy
					days = 120
				}
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = { is_regular_empire = yes }
			}
			modifier = {
				job_computer_technician_add = 3
			}
		}
	
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = { is_gestalt = yes }
			}
			modifier = {
				job_computer_drone_add = 3
			}
		}
		resources = {
			category = planet_buildings
			cost = {
				minerals = @building_build_cost1
			}
			upkeep = {
				energy = 1
			}
		}

		upgrades = {
			building_hyper_computers_station
		}

		triggered_desc = {
			trigger = {
				exists = owner
				owner = { is_gestalt = yes }
			}
			text = job_computer_drone_effect_desc
		}
		triggered_desc = {
			trigger = {
				exists = owner
				owner = { is_gestalt = no }
			}
			text = job_computer_technician_effect_desc
		}

		ai_weight = {
			weight = 0
			modifier = {
				weight = 0
				owner = {
					has_monthly_income = {
						resource = energy
						value > 8
					}
				}
				num_buildings = {
					type = any
					value >= 12
				}
				AND = {
					free_district_slots >= 2
					free_building_slots >= 2
				}
				count_pops = {
					limit = {
						has_job = no
					}
					count >= 3
				}
			}

			modifier = {
				weight = 0
				has_building = building_super_computers_station
			}

			modifier = {
				factor = 0
				owner = { has_country_flag = buildings_new_super_computers_station }
			}
		}

		on_built = {
			owner = { set_timed_country_flag = { flag = buildings_new_super_computers_station days = 480 } }
		}

		prerequisites = {
			"tech_basic_science_lab_1"
	}
}
