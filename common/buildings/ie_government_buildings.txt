building_ie_precinct_house = {
	base_buildtime = 3960

	category = government

	potential = {
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
		is_planet_class = pc_ie_gateway
	}

	allow = {
		hidden_trigger = {
			OR = {
				owner = { is_ai = no }
				AND = {
					planet_crime > 15
					NOR = {
						AND = {
							has_building = building_ie_precinct_house
							owner = { has_technology = tech_colonial_centralization }
						}
						has_building = building_ie_hall_judgment
						num_buildings = { type = building_ie_precinct_house value = 3 }
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			NOT = {	owner = { is_regular_empire = yes }	}
			AND = {
				owner = { is_ai = yes } #Players dont like random buildings blowing up
				planet = { has_building = building_ie_hall_judgment }
			}#This way the AI gets to reclaim its building slots that it sunk into this expense
		}
	}

	planet_modifier = {
		job_enforcer_add = 22
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 4400
		}
		upkeep = {
			energy = 22
		}
	}

	prerequisites = {
		"tech_planetary_government"
	}

	triggered_desc = {
		text = job_enforcer_effect_desc
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	upgrades = {
		building_ie_hall_judgment
	}
}
