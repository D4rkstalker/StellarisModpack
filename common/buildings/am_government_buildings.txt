building_state_agency = {
	base_buildtime = 90
	category = government
	
	potential = {
		owner = { 
			is_regular_empire = yes
			is_ai = yes
		}
	}
	
	allow = {
		num_pops > 70
		free_jobs < 1
	}
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		free_jobs > 8
	}

	planet_modifier = {
		job_service_add = 1
		job_labourer_add = 1
		job_wageslave_add = 1
		planet_amenities_add = -5
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
			sr_wealth = 200
		}
		upkeep = { sr_wealth = 1 }
	}
	
	ai_weight = {
		weight = 1
		modifier = { weight = 100 has_modifier = resolve_employment_crisis }
		modifier = {
			weight = 0
			num_pops < 70
			not = { has_modifier = resolve_employment_crisis }
			or = {
				free_jobs > 0
				not_enough_wealth = yes
				free_amenities < 15
			}
		}
		modifier = {
			factor = 5
			count_owned_pops = { limit = { is_unemployed = yes } count > 3 }
		}
	}
}
