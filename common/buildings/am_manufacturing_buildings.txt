building_biofuels_processing_1 = {
	base_buildtime = 360

	category = manufacturing
	
	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		owner = {
			has_monthly_income = {
				resource = sr_biomass
				value >= 4
			}
			has_monthly_income = {
				resource = sr_natural_fuels
				value >= 1
			}			
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_biofuels_engineer_add = 3
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		produces = {
			sr_plastics = 4
			energy = 2
			sr_waste = 5
		}
		produces = {
			trigger = {
				owner = {
					is_gestalt = yes
				}
			}
			sr_plastics = 8
		}
		upkeep = {
			trigger = {
				owner = {
					and = {
						resource_stockpile_compare = {
							resource = sr_natural_fuels
							value >= 1
						}
						resource_stockpile_compare = {
							resource = sr_biomass
							value >= 4
						}
					}
				}			
			}
			sr_biomass = 4
			sr_natural_fuels = 1
		}
		upkeep = {
			trigger = {
				owner = {
					country_uses_food = yes
					or = {
						resource_stockpile_compare = {
							resource = sr_natural_fuels
							value < 1
						}
						resource_stockpile_compare = {
							resource = sr_biomass
							value < 4
						}
					}
				}			
			}
			minerals = 1
			food = 2
		}
		upkeep = {
			trigger = {
				owner = {
					country_uses_food = no
					or = {
						resource_stockpile_compare = {
							resource = sr_natural_fuels
							value < 1
						}
						resource_stockpile_compare = {
							resource = sr_biomass
							value < 4
						}
					}
				}			
			}
			minerals = 3
		}
	}

	prerequisites = {
		tech_basic_plastics
	}	
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			basics_are_met = yes
			free_jobs < 3
			not = { has_building = building_biofuels_processing_1 }
			exists = owner
			owner = {
				has_monthly_income = {
					resource = sr_biomass
					value >= 5
				}
				has_monthly_income = {
					resource = sr_natural_fuels
					value >= 5
				}
				has_monthly_income = {
					resource = sr_plastics
					value < 2
				}				
			}
		}
	}
	
	upgrades = {
		building_biofuels_processing_2
	}

	ai_resource_production = {
		sr_plastics = 1
		energy = 1
		trigger = {
			always = yes
		}
	}
}
