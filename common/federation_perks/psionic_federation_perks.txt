covenant_federation_passive = {
	members_modifier = {
		description = covenant_federation_passive_effect
		pop_growth_speed = -0.1
	}
	leader_modifier = {
		description = covenant_federation_passive_leader_effect
		country_base_sr_zro_produces_add = 1
		pop_growth_speed = 0.1
	}
	on_activate = {
		#	set_diplomacy_action_setting = {
		#		action = action_ask_to_leave_federation
		#		settings = {
		#			acceptance_type = leader
		#		}
		#	}
		every_member = { 
	limit = { 
	is_federation_leader = yes 
	} 
		add_resource = { sr_zro = 100	}
		}
		
	every_member = { 
	limit = { 
	#is_federation_leader = no 
	NOT = { 
	has_authority = auth_hive_mind
	}
	ruler = {  is_organic_species = yes }
	} 
		if = {
			limit = { ruler = { not = { has_trait = leader_trait_ruler_chosen	} } }
			ruler = { add_ruler_trait = leader_trait_ruler_chosen
			set_leader_flag = covenant_chosen } 
		}
	}
		
	}
	on_deactivate = {
		#set_diplomacy_action_setting = {
		#	action = action_ask_to_leave_federation 
		#	settings = {
		#		acceptance_type = default
		#	}
		#}
		
		every_member = { 
			limit = { 
			any_owned_leader = { has_leader_flag = covenant_chosen }
			}
			every_owned_leader = {
			limit = { has_leader_flag = covenant_chosen }
				remove_ruler_trait = leader_trait_ruler_chosen
				remove_leader_flag = covenant_chosen
			}
			}
			}
}

covenant_federation_boon_2 = {
	members_modifier = {
		description = covenant_enable_shroud_entity
	}
	
	on_activate = {
		every_member = { 
			create_fleet = {
							name = "NAME_Psionic_Avatar"
							settings = { 
								can_upgrade = no 
								can_change_composition = no 
								can_change_leader = no 
								uses_naval_capacity = no
								spawn_debris = no
							}			
							effect = {
								set_fleet_flag = covenant_fleet
								set_owner = prev
								create_ship = {
									name = "NAME_Avatar"
									design = "NAME_Shroud_Avatar"
									prefix = no
									upgradable = no
								}
								set_location = prev.capital_scope
							}
						}
		}
	}
	on_deactivate = {
		every_member = { 
		if = { 
			limit = {any_owned_fleet = { has_fleet_flag = covenant_fleet } } 
			every_owned_fleet = { 
				limit = {  has_fleet_flag = covenant_fleet }
				destroy_fleet = this
			}
		}
		
		}
	
	}

}
