@activation_cost = 3000
@triumph_duration = 3600
@triumph_duration_short = 1800
@triumph_duration_veryshort = 900

r_memorex = {
	activation_duration = @triumph_duration
	portrait = "GFX_relic_orb_insight"
	sound = "relic_activation_the_omnicodex"

	resources = {
		category = relics
		# Activation cost
		cost = {
			unity = 6000
		}
	}
	
	ai_weight = {
		weight = 100
		modifier = {
			factor = 0
			NOT = {
				any_owned_planet = {
					num_pops > 19
					NOT = { has_modifier = the_memorex }
				}
			}
		}
	}
	
	triggered_country_modifier = {
		potential = {
			always = yes
		}
		species_leader_exp_gain = 0.5
		councilor_skill_add = 1
	}

	score = 2000

	active_effect = {
		custom_tooltip = relic_memorex
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown" 
				days = @triumph_duration
			}
			if = {
				limit = {
					NOT = { has_country_flag = relic_memorex_activated }
				}
				set_timed_country_flag = { flag = relic_memorex_activated days = @triumph_duration }
			}
		}
	}

	# Possible check for activation
	possible = {
		custom_tooltip = {
			fail_text = "requires_relic_no_cooldown"
			NOT = { has_modifier = relic_activation_cooldown }
		}
	}
}

r_three_londor = {
	activation_duration = @triumph_duration
	portrait = "GFX_relic_londor_core"
	sound = "relic_activation_isolated_contingency_core"

	resources = {
		category = relics
		# Activation cost
		cost = {
			unity = @activation_cost
		}
	}
	
	ai_weight = {
		weight = 2
	}
	
	triggered_country_modifier = {
		potential = {
			is_hive_empire = no
			has_global_flag = expanded_pops_active
		}
		custom_tooltip = r_three_londor_passive_effect
	}
	triggered_country_modifier = {
		potential = {
			is_hive_empire = no
			NOT = { has_global_flag = expanded_pops_active }
		}
		planet_pops_robotics_upkeep_mult = -0.10
	}
	triggered_country_modifier = {
		potential = {
			is_hive_empire = yes
		}
		planet_crime_no_happiness_add = -10
	}

	score = 1000

	active_effect = {
		if = {
			limit = {
				OR = {
					is_hive_empire = yes
					AND = {
						is_gestalt = no
						has_policy_flag = robots_outlawed
					}
				}
			}
			add_modifier = {
				modifier = "three_londor_triumph_hive" 
				days = @triumph_duration
			}
		}
		else = {
			add_modifier = {
				modifier = "three_londor_triumph" 
				days = @triumph_duration
			}
		}
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown" 
				days = @triumph_duration
			}
		}
	}

	# Possible check for activation
	possible = {
		custom_tooltip = {
			fail_text = "requires_relic_no_cooldown"
			NOT = { has_modifier = relic_activation_cooldown }
		}
	}
}
