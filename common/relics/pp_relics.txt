	portrait = "GFX_relic_paper_pope"

		category = relics
		cost = {
			influence = @activation_cost_pp
			energy = @activation_cost_pp
		}

		potential = {
			always = yes
		}

	}

	active_effect = {

	}

	possible = {
		custom_tooltip = {

		}
	}
