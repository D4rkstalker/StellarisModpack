triggered_opinion_reso_damage_rune = {
	trigger = {
		is_country_type = default
		is_active_resolution = "resolution_power_runes_ban_mighty"
		FROM = {
			has_relic = r_damage_rune
			has_country_flag = r_damage_rune_active
		}
	}

	opinion = {
		base = -75
	}
}
