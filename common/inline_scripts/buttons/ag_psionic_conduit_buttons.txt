
ag_psionic_conduit_minus_10_$ag_index$ = {
    potential = { hidden_trigger = { always = yes } }
    allow = { hidden_trigger = {
        exists = event_target:ag_psionic_conduit_megastructure
        event_target:ag_psionic_conduit_megastructure = { NOT = { has_megastructure_flag = ag_psionic_conduit_no_produce_$ag_index$ } NAND = {
            is_variable_set = ag_alpha_psionic_conduit_$ag_index$_percent
            check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value <= 0 }
        } }
    } }
    effect = { hidden_effect = { event_target:ag_psionic_conduit_megastructure = {
		change_variable = { which = ag_alpha_psionic_conduit_stability value = value:ag_alpha_psionic_conduit_unstability|ag_resource|$ag_index$| }
        if = { limit = { NOT = { is_variable_set = ag_alpha_psionic_conduit_$ag_index$_percent } } set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 0 } }
        else = {
			if = {
				limit = { check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value < 10 } }
				subtract_variable = { which = ag_alpha_psionic_conduit_used_capacity value = ag_alpha_psionic_conduit_$ag_index$_percent }
				set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 0 }
			}
			else = {
				subtract_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 10 }
				subtract_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 10 }
			}
        }
        ag_alpha_psionic_conduit_refresh = yes
		subtract_variable = { which = ag_alpha_psionic_conduit_stability value = value:ag_alpha_psionic_conduit_unstability|ag_resource|$ag_index$| }
    } } }
}

ag_psionic_conduit_plus_10_$ag_index$ = {
    potential = { hidden_trigger = { always = yes } }
    allow = { hidden_trigger = {
        exists = event_target:ag_psionic_conduit_megastructure
        event_target:ag_psionic_conduit_megastructure = { NOT = { has_megastructure_flag = ag_psionic_conduit_no_produce_$ag_index$ } NOR = {
			AND = {
				is_variable_set = ag_alpha_psionic_conduit_$ag_index$_percent
				check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value >= 100 }
			}
			AND = {
				is_variable_set = ag_alpha_psionic_conduit_used_capacity
				check_variable = { which = ag_alpha_psionic_conduit_used_capacity value >= 100 }
			}
        } }
    } }
    effect = { hidden_effect = { event_target:ag_psionic_conduit_megastructure = {
		change_variable = { which = ag_alpha_psionic_conduit_stability value = value:ag_alpha_psionic_conduit_unstability|ag_resource|$ag_index$| }
        if = {
			limit = { NOT = { is_variable_set = ag_alpha_psionic_conduit_$ag_index$_percent } }
			if = {
				limit = { check_variable = { which = ag_alpha_psionic_conduit_used_capacity value <= 90 } }
				set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 10 }
				change_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 10 }
			}
			else = {
				set_variable = { which = ag_alpha_psionic_conduit_tmp value = 100 }
				subtract_variable = { which = ag_alpha_psionic_conduit_tmp value = ag_alpha_psionic_conduit_used_capacity }
				set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = ag_alpha_psionic_conduit_tmp }
				set_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 100 }
				clear_variable = ag_alpha_psionic_conduit_tmp
			}
		}
        else = {
			if = {
				limit = {
					check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value <= 90 }
					check_variable = { which = ag_alpha_psionic_conduit_used_capacity value <= 90 }
				}
				change_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 10 }
				change_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 10 }
			}
			else = {
				set_variable = { which = ag_alpha_psionic_conduit_tmp value = 100 }
				if = {
					limit = { check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value < ag_alpha_psionic_conduit_used_capacity } }
					subtract_variable = { which = ag_alpha_psionic_conduit_tmp value = ag_alpha_psionic_conduit_used_capacity }
					change_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = ag_alpha_psionic_conduit_tmp }
					set_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 100 }
				}
				else_if = {
					limit = { check_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value > ag_alpha_psionic_conduit_used_capacity } }
					subtract_variable = { which = ag_alpha_psionic_conduit_tmp value = ag_alpha_psionic_conduit_$ag_index$_percent }
					change_variable = { which = ag_alpha_psionic_conduit_used_capacity value = ag_alpha_psionic_conduit_tmp }
					set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 100 }
				}
				else_if = {
					limit = { always = yes }
					set_variable = { which = ag_alpha_psionic_conduit_used_capacity value = 100 }
					set_variable = { which = ag_alpha_psionic_conduit_$ag_index$_percent value = 100 }
				}
				clear_variable = ag_alpha_psionic_conduit_tmp
			}
        }
        ag_alpha_psionic_conduit_refresh = yes
		subtract_variable = { which = ag_alpha_psionic_conduit_stability value = value:ag_alpha_psionic_conduit_unstability|ag_resource|$ag_index$| }
    } } }
}
