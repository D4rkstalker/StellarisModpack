
special_project = {
	key = ag_$ag_index$_project
	cost = {
		base = @ag_$ag_index$_project_base_cost
		scaled_modifier = { limit = { is_capital = yes owner = { check_variable = { which = ag_research_project_counter value > 0 } } } scope = this add = @ag_$ag_index$_project_extra_cost calc = planets_in_country }
		scaled_modifier = { limit = { is_capital = yes owner = { check_variable = { which = ag_research_project_counter value > 1 } } } scope = this add = @ag_$ag_index$_project_extra_cost calc = planets_in_country }
		scaled_modifier = { limit = { is_capital = yes owner = { check_variable = { which = ag_research_project_counter value > 2 } } } scope = this add = @ag_$ag_index$_project_extra_cost calc = planets_in_country }
		scaled_modifier = { limit = { is_capital = yes owner = { check_variable = { which = ag_research_project_counter value > 3 } } } scope = this add = @ag_$ag_index$_project_extra_cost calc = planets_in_country }
		scaled_modifier = { limit = { is_capital = yes owner = { check_variable = { which = ag_research_project_counter value > 4 } } } scope = this add = @ag_$ag_index$_project_extra_cost calc = planets_in_country }
	}
	tech_department = $ag_primary_area$_technology
	picture = $ag_picture$
	icon = "gfx/interface/icons/situation_log/situation_log_ag_$ag_icon$.dds"
	
	event_scope = ship_event
	event_chain = ag_$ag_index$_project_chain
	
	requirements = { shipclass_science_ship = 1 leader = scientist }
	
	on_success = {
		owner = {
			set_country_flag = ag_rp_round_area_$ag_primary_area$
			set_country_flag = ag_rp_round_ag_$ag_index$_project
		}
		fire_on_action = { on_action = ag_on_research_project_round_completed scopes = { from = root.from } }
	}
	on_fail = { }
}
