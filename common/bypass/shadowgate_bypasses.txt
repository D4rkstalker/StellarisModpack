	ftl_multiplier = 0
	icon_frame = 12
	remember_instances = no
	animation_time = 1
	galactic_map_tooltip_header=SHADOWGATE_GALACTIC
	galactic_map_tooltip_desc=SHADOWGATE_DESC
	connection_type = scripted_connection
	extends_sensors = no
	ftl_ship_effect = "shadowrift_ftl_ship_effect_entity"
	ftl_bypass_effect = "wormhole_ftl_bypass_effect_entity"
	prerequisites = { }
	
	potential = {
		always = yes
	}

	country_can_use = {
		if = { limit = { is_ai = yes OR = { shadowgate_is_fe = yes shadowgate_is_ae = yes } } OR = { event_target:shadowgate_planet_a = { exists = owner owner = { is_at_war_with = prevprev } } NOR = { has_ethic = ethic_fanatic_spiritualist has_ethic = ethic_fanatic_materialist } } }
	}

	scripted_connection = {
		node = { has_star_flag = has_shadowrift NOT = { has_star_flag = shadowrift_origin } }
		nexus = { has_star_flag = shadowrift_origin }
	}

	on_action = "on_entering_shadowrift"
}
