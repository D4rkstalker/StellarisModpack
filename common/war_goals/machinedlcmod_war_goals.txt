#for exterminator only
wg_mass_disposal = {
    casus_belli = cb_mass_disposal
    
	hide_if_no_cb = yes
	set_defender_wargoal = wg_end_threat
    threat_multiplier = 8.0
	
	surrender_acceptance = -200

	potential = {
        always = yes
	}
	possible = {
        always = yes
    }
    
	on_accept = {
        custom_tooltip = artificial_extinction
        hidden_effect = {
                every_country = {
                limit = { 
                    is_ai = no
                    NOT = { is_same_value = root }
                    has_communications = root
                }
                add_opinion_modifier = { modifier = opinion_genocidal who = root }
                add_threat = {
                    who = root
                    amount = 2
                    }
                }
                from = {
                every_owned_starbase = { set_owner = root }		  		
                every_owned_planet = {
                destroy_colony = yes
                change_pc = pc_nuked
                    every_owned_pop = {
                    kill_pop = yes
                    }			
                }
                every_subject = {
                    every_owned_starbase = { set_owner = root }
                    every_owned_planet = {
                    change_pc = pc_nuked
                    destroy_colony = yes
                        every_owned_pop = {
                        kill_pop = yes
                        }			
                    }
                }
            }	  
        }	
    }

    on_status_quo = {
        custom_tooltip = massacre
        hidden_effect = {
            every_country = {
            limit = { 
                is_ai = no
                NOT = { is_same_value = root }
                has_communications = root
            }
            add_opinion_modifier = { modifier = opinion_genocidal who = root }
                add_threat = {
                who = root
                amount = 1.5
                }
            }
            from = {
            every_owned_planet = {
                limit = { is_controlled_by = root }
                    change_pc = pc_nuked
                    destroy_colony = yes
                    every_owned_pop = {
                        kill_pop = yes
                    }
                }
            }
        }
    }

    ai_weight = {
        weight = 100
        modifier = {
            factor = 100
            root = {
                OR = {
                    has_valid_civic = civic_machine_terminator
                    is_berserk_fallen_machine_empire = yes
                }
            }
        }		
    }
}

wg_mass_assimilation = {
    casus_belli = cb_mass_assimilation

	hide_if_no_cb = yes
	set_defender_wargoal = wg_end_threat
    threat_multiplier = 2.0
	
	surrender_acceptance = -200

	potential = {
        always = yes
	}
	possible = {
		always = yes
    }
    
	on_accept = {
        custom_tooltip = instant_assimilation
        hidden_effect = {
                every_country = {
                limit = {
                    NOT = { is_same_value = root }
                    has_communications = root
                }
                add_opinion_modifier = { modifier = opinion_assimilated_a_world who = root }
                add_threat = {
                    who = root
                    amount = 2
                    }
                }
            from = {
                every_owned_starbase = { set_owner = root }		  		
                every_owned_planet = {
                    set_owner = root         
                    every_owned_pop = { change_species = owner_species }
                }
                every_subject = {
                    every_owned_starbase = { set_owner = root }
                    every_owned_planet = {
                        set_owner = root
                        every_owned_pop = { change_species = owner_species }
                    }
                }
            }	  
        }	
    }

    on_status_quo = {
        custom_tooltip = partial_assimilation
        hidden_effect = {
            every_country = {
            limit = { 
                NOT = { is_same_value = root }
                has_communications = root
            }
            add_opinion_modifier = { modifier = opinion_assimilated_a_world who = root }
                add_threat = {
                who = root
                amount = 1.5
                }
            }
            from = {
                every_owned_planet = {
                    limit = { is_controlled_by = root }
                    set_owner = root
                    every_owned_pop = { change_species = owner_species }
                    solar_system = {
                        every_fleet_in_system = {
                            limit = {
                                is_ship_class = shipclass_starbase
                            }
                            set_owner = root
                        }
                    }
                }
            }
        }
    }

    ai_weight = {
        weight = 100
        modifier = {
            factor = 100
            root = {
            has_valid_civic = civic_machine_assimilator
          }
      }		
    }
}
