acot_wg_mountain_king = {
	casus_belli = acot_cb_mountain_king

	total_war = no

	hide_if_no_cb = yes
	defender_default = no
	
	cede_claims = occupied_only

	surrender_acceptance = -25
	war_exhaustion = 0.50

	set_defender_wargoal = acot_wg_mountain_king

	potential = {
		is_country_type = awakened_fallen_empire
		NOT = { has_ethic = ethic_gestalt_consciousness }
		from = { has_country_flag = prepare_to_die_by_fe }		
	}
	possible = {
		always = yes
	}

	on_accept = {
		hidden_effect = {
			save_event_target_as = mountain_king_victor
			from = { save_event_target_as = mountain_king_loser }	
			if = {
				limit = { 
					OR = {
						is_country_type = default
						is_country_type = ascended_empire
						is_country_type = lost_empire
					}
				}
				country_event = { id = acot_diplo.16 days = 1 }
			}
			if = {
				limit = { 
					is_country_type = awakened_fallen_empire
				}
				from = {
					country_event = { id = acot_diplo.17 days = 1 }
				}
			}
		}
	}
	
	on_status_quo = {
		set_timed_country_flag = {
			flag = acot_robbed_of_victory
			years = 5
		}
	}
	
	ai_weight = {
		weight = 10000
	}
}
