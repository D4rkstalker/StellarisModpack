
wg_ccrebel_independence = {
	casus_belli = cb_event
	
	hide_if_no_cb = yes
	set_defender_wargoal = wg_conquest
	
	potential = {
		always = no # event only
	}
	possible = {
		always = yes
	}
	
	surrender_acceptance = -100
	war_exhaustion = 0.75
	
	on_status_quo = {
		custom_tooltip = wg_ccrebel_independence_status_quo
		hidden_effect = {
            if = {
                limit = { is_subject = yes }
                set_subject_of = {
                    who = none
                }
            }
            if = {
                limit = {
                    any_country = {
                        has_relation_flag = {
                            who = ROOT
                            flag = ccrebel_supporting_independence_war
                        }
                    }
                }
                random_country = {
                    limit = {
                        has_relation_flag = {
                            who = ROOT
                            flag = ccrebel_supporting_independence_war
                        }
                    }
                    country_event = { id = ccrebel_wg_script.3 days = 5 }
                }
            }				
		}	
	}
	
	on_accept = {
		custom_tooltip = wg_ccrebel_independence_accept
		hidden_effect = {
            if = {
                limit = { is_subject = yes }
                set_subject_of = {
                    who = none
                }
            }
			every_system_within_border = {
				limit = {
					from = { has_claim = prev }
				}
				remove_claims = { who = from }
            }
            if = {
                limit = {
                    any_country = {
                        has_relation_flag = {
                            who = ROOT
                            flag = ccrebel_supporting_independence_war
                        }
                    }
                }
                random_country = {
                    limit = {
                        has_relation_flag = {
                            who = ROOT
                            flag = ccrebel_supporting_independence_war
                        }
                    }
                    country_event = { id = ccrebel_wg_script.3 days = 5 }
                }
            }					
		}
	}
	
	on_wargoal_set = {
        hidden_effect = {
            every_system_within_border = { 
                closest_system = {
                    min_steps = 1
                    max_steps = 1
                    limit = {
                        is_within_borders_of = from
                        starbase = { has_starbase_size = starbase_outpost }
                        NOT = { any_system_planet = { is_colony = yes } }
                    }                    
                    add_claims = {
                        who = root
                        num_of_claims = 1
                        show_notification = no
                    }
                }
                closest_system = {
                    min_steps = 1
                    max_steps = 1
                    limit = {
                        is_within_borders_of = from
                        starbase = { has_starbase_size = starbase_outpost }
                        NOT = { root = { has_claim = prev } }
                        NOT = { any_system_planet = { is_colony = yes } }
                    }                    
                    add_claims = {
                        who = root
                        num_of_claims = 1
                        show_notification = no
                    }
                }
            }	
            from = {
                every_owned_planet = {
                    limit = {
                        any_owned_pop = { is_same_species = root }
                    }
                    solar_system = {
                        add_claims = {
                            who = root
                            num_of_claims = 1
                            show_notification = no
                        }
                        closest_system = {
                            min_steps = 1
                            max_steps = 1
                            limit = {
                                is_within_borders_of = from
                                starbase = { has_starbase_size = starbase_outpost }
                                NOT = { root = { has_claim = prev } }
                                NOT = { any_system_planet = { is_colony = yes } }
                            }
                            add_claims = {
                                who = root
                                num_of_claims = 1
                                show_notification = no
                            }
                        }
                        closest_system = {
                            min_steps = 1
                            max_steps = 1
                            limit = {
                                is_within_borders_of = from
                                starbase = { has_starbase_size = starbase_outpost }
                                NOT = { root = { has_claim = prev } }
                                NOT = { any_system_planet = { is_colony = yes } }
                            }
                            add_claims = {
                                who = root
                                num_of_claims = 1
                                show_notification = no
                            }
                        }
                    }
                }
            }
        }
	}
	
	ai_weight = {
		weight = 100
	}
}
