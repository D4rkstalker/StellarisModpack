wg_abduct = {
	casus_belli = cb_abduct
	
	hide_if_no_cb = no
	total_war = no
	release_occupied_systems_on_status_quo = yes
	surrender_acceptance = -100

	potential = {
		has_total_war_cb = no
		has_valid_civic = civic_hive_parasitic
	}
	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
	}

		
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}			


		on_accept = {
		#custom_tooltip = wg_abduct_accept	
		hidden_effect = {
			from = {
				add_threat = {
					who = root
					amount = 1
				}				
			}	
		}		
		from = {
			set_subject_of = {
				who = root
				subject_type = pop_tributary
			}		
		}
		}

			

	
	on_status_quo = {
		if = {
			limit = {
				last_created_country = { 
					has_country_flag = released_empire 
				}
			}

			last_created_country = { 
				set_subject_of = {
					who = root
					subject_type = pop_tributary
				}
			}
		}	
	}
	
	ai_weight = {
		weight = 1000
	}
}



wg_restraint = {
	casus_belli = cb_restraint

	hide_if_no_cb = yes
	cede_claims  = yes

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -50

	potential = {
		NOT = { has_authority = auth_corporate }
		has_total_war_cb = no
	}
	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
	}
	on_accept = {
		hidden_effect = {
			from = {
				add_threat = {
					who = root
					amount = 0.25
				}
			}
		}
		from = {
			every_subject = {
				limit = {
					is_subject_type = pop_tributary
					NOT = { is_at_war_with = overlord }					
				}
				## release
				set_subject_of = {
					who = none
				}
				root = { add_opinion_modifier = { modifier = opinion_we_liberated who = prev } }
				add_opinion_modifier = { modifier = opinion_liberated_us who = root }
			}

			set_subject_of = {
				who = root
				subject_type = tributary
			}				
		}
	}

	on_status_quo = {
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 0
			NOT = { has_ai_personality_behaviour = wants_tribute }
		}
	}
}
