# industry regular empire decisions
decision_simple_planet_mining = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = minerals_designation

	resources = {
		category = decisions
		cost = {
			unity = 250
		}
	}

	effect = {
		if = {
			limit = {
				owner = {
					is_hive_empire = no
					is_machine_empire = no
				}
			}
			custom_tooltip = decision_simple_planet_mining_effect_desc
			hidden_effect = {
				add_modifier = { modifier = "simple_planet_mining" }
			}
		} else = {
			custom_tooltip = decision_simple_planet_mining_gestalt_effect_desc
			hidden_effect = {
				add_modifier = { modifier = "simple_planet_mining_gestalt" }
			}
		}
	}

	potential = {
		owner = {
			has_tradition = tr_industry_finish
		}
		NOT = { has_modifier = simple_planet_mining }
		NOT = { has_modifier = simple_planet_energy }
		NOT = { has_modifier = simple_planet_farmer }
		NOT = { has_modifier = simple_planet_production }
		NOT = { has_modifier = simple_planet_refinery }
		NOT = { has_modifier = simple_planet_mining_gestalt }
		NOT = { has_modifier = simple_planet_energy_gestalt }
		NOT = { has_modifier = simple_planet_farmer_gestalt }
		NOT = { has_modifier = simple_planet_production_gestalt }
		NOT = { has_modifier = simple_planet_refinery_gestalt }
	}

	allow = {
		NOT = { has_modifier = simple_planet_mining }
		NOT = { has_modifier = simple_planet_energy }
		NOT = { has_modifier = simple_planet_farmer }
		NOT = { has_modifier = simple_planet_production }
		NOT = { has_modifier = simple_planet_refinery }
		NOT = { has_modifier = simple_planet_mining_gestalt }
		NOT = { has_modifier = simple_planet_energy_gestalt }
		NOT = { has_modifier = simple_planet_farmer_gestalt }
		NOT = { has_modifier = simple_planet_production_gestalt }
		NOT = { has_modifier = simple_planet_refinery_gestalt }
	}

	ai_weight = {
		weight = 5
		modifier = {
			factor = 2
			OR = {
				has_designation = col_mining
				has_designation = col_habitat_mining
			}
		}
		modifier = {
			factor = 2
			num_minerals > 20
		}
		modifier = {
			factor = 2
			num_minerals > 40
		}
		modifier = {
			factor = 2
			num_minerals > 80
		}
		modifier = {
			factor = 2
			num_minerals > 200
		}
		modifier = {
			factor = 0
			OR = {
				NOT = {
					OR = {
						has_district = district_mining
						has_district = district_mining_uncapped
						has_district = district_hab_mining
					}
				num_pops < 30
				}
			}
		}
	}
}
