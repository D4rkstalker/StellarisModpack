# Science decision
decision_simple_planet_space_station = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_satellite_in_orbit
	enactment_time = 720

	resources = {
		category = decisions
		cost = {
			energy = 1000
			alloys = 50
		}
	}

	potential = {
		owner = {
			NOT = {
				OR = {
					is_hive_empire = yes
					is_machine_empire = yes
				}
			}
			has_tradition = tr_research_adopt
		}
		NOT = { 
			OR = {
				has_modifier = simple_planet_space_station
				has_modifier = simple_planet_space_station_machine
				has_modifier = simple_planet_space_station_hive
			}
		}
	}

	allow = {
		NOT = {
			OR = {
				is_hive_empire = yes
				is_machine_empire = yes
			}
		}
		NOT = { 
			OR = {
				has_modifier = simple_planet_space_station
				has_modifier = simple_planet_space_station_machine
				has_modifier = simple_planet_space_station_hive
			}
		}
	}

	effect = {
		add_modifier = {
			modifier = simple_planet_space_station
		}
	}

	ai_weight = {
		weight = 5
	}
}
