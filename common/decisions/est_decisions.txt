#Raider 2: Raid Planet.
decision_est_piracy_2 = {
	sound = event_ground_battle
	#icon = decision_est_cooperativity_5_activate
	
	resources = {
		category = decisions
		cost = {
			influence = 10
		}
	}
	
	potential = {
		is_colony = yes
		from = {
			has_non_swapped_tradition = tr_est_piracy_2
		}
	}

	allow = {
		custom_tooltip = {
			Not = {
				has_planet_flag = flag_est_piracy_2_recently_raided
			}
			fail_text = consecrated_worlds_too_many
		}
		custom_tooltip = {
			NOT = { exists = owner }
			fail_text = consecrated_worlds_planet_owned
		}
	}
	
	effect = {
		custom_tooltip = decision_est_cooperativity_5_machines_activate_tooltip
		hidden_effect = {
			set_timed_planet_flag = { 
				flag = flag_est_piracy_2_recently_raided 
				years = 10 
			}
		}
		add_planet_devastation = 25
		from = {
			add_resource = {
				minerals = 25
				energy = 25
			}
		}
		add_threat = { who = from amount = 2 }
		if = {
			limit = { num_districts = { type = any value > 4 } }
			from = {
				add_resource = {
					minerals = 25
					energy = 25
				}
			}
			add_threat = { who = from amount = 2 }
		}
		if = {
			limit = { num_districts = { type = any value > 8 } }
			from = {
				add_resource = {
					minerals = 25
					energy = 25
				}
			}
			add_threat = { who = from amount = 2 }
		}
		if = {
			limit = { num_districts = { type = any value > 12 } }
			from = {
				add_resource = {
					minerals = 25
					energy = 25
				}
			}
			add_threat = { who = from amount = 2 }
		}
		if = {
			limit = { num_districts = { type = any value > 16 } }
			from = {
				add_resource = {
					minerals = 25
					energy = 25
				}
			}
			add_threat = { who = from amount = 2 }
		}
		if = {
			limit = { num_districts = { type = any value > 20 } }
