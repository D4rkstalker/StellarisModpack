#Distribution: Food
pw_decision_tls_infrastructure_distribution_1 = {
	owned_planets_only = yes
	sound = event_ringworld_build_start

	enactment_time = @pw_tls_time_1
	resources = {
		category = decisions
		cost = {
			minerals = @pw_tls_mineral_cost_1
			food = @pw_tls_resource_cost
		}
	}

	prerequisites = {
		pw_tech_egalitarian_distribution_services
	}

	potential = {
		exists = owner
		owner = {
			is_egalitarian = yes
		}
		NOR = {
			has_deposit = pw_d_infrastructure_food_distribution
			has_deposit = pw_d_infrastructure_energy_distribution
			has_deposit = pw_d_infrastructure_goods_distribution
			has_deposit = pw_d_infrastructure_luxury_distribution
		}
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}

	allow = {
		OR = {
			has_active_building = pw_building_transplanetary_logistics_network
			has_active_building = pw_building_transhabitat_logistics_network
		}
	}

	effect = {
		add_deposit = pw_d_infrastructure_food_distribution
		hidden_effect = {
			planet_event = { id = pw_wonder.1705 }
		}
	}

	ai_weight = {
		weight = 0.75
		modifier = {
			factor = 0
			owner = {
				OR = {
					has_monthly_income = {
						resource = consumer_goods
						value < 5
					}
					has_monthly_income = {
						resource = energy
						value < 10
					}
				}
			}
		}
		modifier = {
			factor = 0.1
			free_building_slots > 1
		}
		modifier = {
			factor = 1.25
			free_building_slots = 0
		}
		modifier = {
			factor = 0.1
			free_district_slots > 5
		}
		modifier = {
			factor = 1.25
			free_district_slots < 2
		}
		modifier = {
			factor = 1.25
			owner = { has_ethic = ethic_fanatic_egalitarian }
		}
		modifier = {
			factor = 1.25
			owner = { has_valid_civic = civic_shared_burden }
		}
		modifier = {
			factor = 0.75
			owner = { is_megacorp = yes }
		}
		modifier = {
			factor = 1.5
			num_pops > 50
		}
		modifier = {
			factor = 1.5
			num_pops > 75
		}
		modifier = {
			factor = 1.5
			num_pops > 100
		}
		modifier = {
			factor = 2
			num_pops > 150
		}
	}
}
