decision_prep_terraforming = {
	icon = decision_planet_beam
	
	potential = {
		exists = space_owner
		space_owner = { 
			is_same_value = from
			or = {
				has_technology = "tech_terrestrial_sculpting"
				has_technology = "tech_starbase_3"			
			}
		}
		from = {
			is_machine_empire = no
			or = {
				is_ai = no
				and = {
					is_ai = yes
					resource_stockpile_compare = {
						resource = energy
						value > 12000
					}
					resource_stockpile_compare = {
						resource = influence
						value > 850
					}						
				}
			}
		}

		not = { is_colony = yes }
		is_hostile_candidate = yes
		not = { has_modifier = terraforming_candidate }
	}
	
	allow = {
		or = {
			and = {
				is_moon = yes
				planet_size < 15
				planet_size > 3
			}
			and = {
				planet_size < 13
				planet_size > 4
			}
			and = {
				space_owner = {
					has_ascension_perk = ap_all_these_worlds
				}
				planet_size < 19
				planet_size > 3				
			}
			is_planet_class = pc_asteroid
			is_planet_class = pc_ice_asteroid
		}
		not = { has_anomaly = yes }
		or = {
			solar_system = {
				not = {
					any_system_planet = {
						has_planet_flag = pf_limited_terraform
					}
				}
			}
			space_owner = {
				has_ascension_perk = ap_all_these_worlds
			}
		}
	}
	
	resources = {
		category = decisions
		cost = {
			energy = 500
			influence = 25	
		}
		cost = {
			trigger = {
				nor = {
					is_planet_class = pc_asteroid
					is_planet_class = pc_ice_asteroid
				}
			}
			energy = 500
			influence = 75
		}
		cost = {
			trigger = {
				planet_size > 7
			}
			energy = 1000
			influence = 50
		}
		cost = {
			trigger = {
				planet_size > 9
				exists = space_owner
				space_owner = {
					not = { has_ascension_perk = ap_all_these_worlds }
				}
			}
			energy = 1000
			influence = 100
		}
		cost = {
			trigger = {
				planet_size > 12
				exists = space_owner
				space_owner = {
					not = { has_ascension_perk = ap_all_these_worlds }
				}
			}
			energy = 1000
			influence = 100
		}
		cost = {
			trigger = {
				solar_system = {
					any_system_planet = {
						has_planet_flag = pf_limited_terraform
					}
				}
				exists = space_owner
				space_owner = {
					has_ascension_perk = ap_all_these_worlds
				}
			}
			energy = 1000
			influence = 100
		}		
	}
	
	effect = {
		custom_tooltip = decision_prep_terraforming_custom
		set_planet_flag = pf_limited_terraform
		if = {
			limit = {
				not = { has_modifier = terraforming_candidate }
			}
			add_modifier = {
				modifier = terraforming_candidate
				days = -1
			}
		}

		set_planet_flag = pf_hide_city
		set_planet_flag = pf_hide_generator
		set_planet_flag = pf_hide_mining
		set_planet_flag = pf_hide_farming
	}
	
	ai_weight = {
		weight = 1
	}
}
