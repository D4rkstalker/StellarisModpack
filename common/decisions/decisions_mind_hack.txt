##########################################################################
# Planet edicts
##########################################################################

# Sets the edict-cost for all edicts with this variable
@planetSDecisionCost = 100
@planetSDecisionDuration = 180
@planetSDecisionEndCost = 25

# Authoritarian Mindset
decision_planet_authoritarian_mindset = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_mindset
	
	resources = {
		category = decisions
		cost = {
                influence = @planetSDecisionCost
		}
    }
        
	potential = {
		owner = { is_regular_empire = yes }
                NOT = { has_modifier = authoritarian_mindset }
                NOT = { has_modifier = egalitarian_mindset }
                NOT = { has_modifier = xenophobe_mindset }
                NOT = { has_modifier = xenophile_mindset }
                NOT = { has_modifier = militarist_mindset }
                NOT = { has_modifier = pacifist_mindset }
                NOT = { has_modifier = spiritualist_mindset }
                NOT = { has_modifier = materialist_mindset }
	}

	effect = {
			add_modifier = { modifier = "authoritarian_mindset" days = @planetSDecisionDuration }
	}
            
	prerequisites = {
		        "tech_technology_implantation"
	}
	ai_weight = {
		weight = 1
		
	}
}

decision_end_planet_authoritarian_mindset = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_mindset

	resources = {
                category = decisions
                cost = {
                        influence = @planetSDecisionEndCost
		}
	}
	
	potential = {
		owner = { is_regular_empire = yes }
		has_modifier = authoritarian_mindset
	}
	
	effect = {
			remove_modifier = authoritarian_mindset
	}
	
	ai_weight = {
		weight = 0
	}
}
