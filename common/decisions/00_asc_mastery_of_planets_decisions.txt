
# Mining Project
decision_mining_ring_project = {
	owned_planets_only = yes
	icon = decision_mining_project

	enactment_time = 3600
	resources = {
		category = decisions
		cost = {
			minerals = 20000
			volatile_motes = 500
			influence = 300
		}
	}
	
	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = penal_colony
			has_modifier = slave_colony
			has_modifier = recently_changed_ringworld
		}
		is_planet_class = pc_ringworld_habitable
		exists = owner
		owner = { 
			OR = {
				has_ascension_perk = ap_planet_master 
				has_ascension_perk = ap_planet_master_gestalt
			}
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = decision_mining_ring_project_requirements
			OR = {
				AND = {
					num_districts = { type = district_rw_farming value < 1 }
					num_districts = { type = district_rw_science value < 1 }
					num_districts = { type = district_rw_generator value < 1 }
					num_districts = { type = district_rw_commercial value < 1 }
				}
				check_variable = { which = mop_count_mining_district value > 4 }
			}
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = decision_mining_ring_project_requirements
			OR = {
				AND = {
					num_buildings = { type = building_mineral_purification_plant value > 0 }
					num_buildings = { type = building_resource_silo value > 3 }
				}
				AND = {
					num_buildings = { type = building_mineral_purification_hub value > 0 }
					num_buildings = { type = building_resource_silo value > 3 }
				}
			}
			num_districts = { type = district_rw_farming value < 1 }
			num_districts = { type = district_rw_science value < 1 }
			num_districts = { type = district_rw_generator value < 1 }
			num_districts = { type = district_rw_commercial value < 1 }
			num_districts = { type = district_rw_city value < 1 }
			num_districts = { type = district_rw_nexus value < 1 }
			num_districts = { type = district_rw_hive value < 1 }
		}
	}
	
	effect = {
		# Make mining-planet
		custom_tooltip = decision_mining_project_effects
		hidden_effect = {
			clear_deposits = yes
			add_deposit = d_for_bad_ai
			change_pc = pc_ringworld_mining
			set_planet_entity = {
				entity = "ringworld_mining_entity"
				graphical_culture = owner
			}
			owner = {
				country_event = { id = distar.201 } # port for living planet distar.197, triggers in event
				country_event = { id = asc_pm.10 }
				set_country_flag = high_planet_mining
				country_event = { id = asc_pm.5002 }
				change_variable = { which = high_planet_mining value = 1 }
				switch = {
					trigger = graphical_culture
					humanoid_01		= { prev = { set_planet_entity = { entity = humanoid_01_ringworld_mining_entity_01_entity	} } }
					plantoid_01		= { prev = { set_planet_entity = { entity = plantoid_01_ringworld_mining_entity_01_entity	} } }
					mammalian_01	= { prev = { set_planet_entity = { entity = mammalian_01_ringworld_mining_entity_01_entity	} } }
					reptilian_01	= { prev = { set_planet_entity = { entity = reptilian_01_ringworld_mining_entity_01_entity 	} } }
					avian_01		= { prev = { set_planet_entity = { entity = avian_01_ringworld_mining_entity_01_entity		} } }
					molluscoid_01	= { prev = { set_planet_entity = { entity = molluscoid_01_ringworld_mining_entity_01_entity 	} } }
					fungoid_01		= { prev = { set_planet_entity = { entity = fungoid_01_ringworld_mining_entity_01_entity 		} } }
					arthropoid_01	= { prev = { set_planet_entity = { entity = arthropoid_01_ringworld_mining_entity_01_entity 	} } }
