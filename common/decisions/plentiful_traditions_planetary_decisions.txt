	}

		planet_event = {
			id = plentiful_traditions_plentiful_deeds_pedict.1
		}
	}

	allow = { } #kann ja ruhig mehrfach genutzt werden...

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			NOT = {
				owner = {
					any_country = {
						is_primitive = no
						NOT = {
							is_same_value = prev
							is_same_species = root.owner
							opinion = {
								who = root.owner
								value = 0
							}
						}
						any_owned_planet = {
							distance = {
								source = root
								max_distance <= 350
							}
						}
					}
				}
			}
		}
	}

	effect = {
		every_planet = {
			limit = {
				exists = owner
				owner = {
					is_primitive = no
					has_communications = root.owner
					NOR = {
						is_same_value = root.owner
						is_same_species = root.owner
					}
				}
				distance = {
					source = ROOT
					max_distance <= 350
				}
			}
			hidden_effect = {
				owner = {
					add_opinion_modifier = {
						who = root.owner
						modifier = opinion_plentiful_deeds
					}
				}
			}
		}
	}
}
