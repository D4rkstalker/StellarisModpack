AAR_DEC_torchlight_GESTALT = {
	owned_planets_only = yes
	sound = event_administrative_work
#	enactment_time = 3600

	resources = {
		category = decisions
		cost = {
			energy = 1000
			unity = 100
		}
	}
	
	potential = {
		# 帝国条件
		owner = {
			has_ascension_perk = AAR_AP_torchlight
			is_gestalt = yes
		}
	}
	allow = {
		custom_tooltip = {
			fail_text = "AAR_failtips_NOT_enough_pop"
			OR = {
				num_pops > 24
				check_variable = { which = PR_var_plnt_HC value > 24 }
			}
		}
		NOT = { has_modifier = AAR_PM_torchlight_GESTALT }	# 未添加火炬mod
	}

	effect = {
		custom_tooltip = AAR_TIPS_dec_torchlight_GESTALT
		hidden_effect = {
			set_update_modifiers_batch = begin	#_|修正更新|暂停

			## 火炬mod添加
			add_modifier = {
				modifier = AAR_PM_torchlight_GESTALT
				days = 3600
			}
		#	remove_modifier = AAR_PM_torchlight_GESTALT
			remove_modifier = AAR_PM_torchlight_PSONIC
			remove_modifier = AAR_PM_torchlight_REGULAR
			remove_modifier = AAR_PM_torchlight_BACKFIRE_1
			remove_modifier = AAR_PM_torchlight_BACKFIRE_2
			remove_modifier = AAR_PM_torchlight_BACKFIRE_3
			# 极限效率
			remove_modifier = AAR_PM_torchlight_Produces
			export_modifier_to_variable = { modifier = planet_jobs_produces_mult variable = TEMP_var_plnt_P }
			multiply_variable = { which = TEMP_var_plnt_P value = 0.1 }	# 10%总产率增幅
			if = { limit = { check_variable = { which = TEMP_var_plnt_P value < 0.25 } }	# 最少25%
				set_variable = { which = TEMP_var_plnt_P value = 0.25 }
			}
			add_modifier = { modifier = AAR_PM_torchlight_Produces days = 3600 multiplier = TEMP_var_plnt_P }
			# clear
			clear_variable = TEMP_var_plnt_P

			## 节点融合
			# 获得融合人数
			set_variable = { which = TEMP_var_plnt_pop value = trigger:num_pops }
			multiply_variable = { which = TEMP_var_plnt_pop value = 0.04 }
			round_variable_to_closest = { which = TEMP_var_plnt_pop value = 1 }
			# 执行融合
			if = { limit = { check_variable = { which = TEMP_var_plnt_pop value > 0 } }
				# AI	|优化|=不清人,仅清人力
				if = { limit = { owner = { is_ai = yes } }
					while = { count = TEMP_var_plnt_pop
						# 消耗人力
						if = { limit = { check_variable = { which = PR_var_plnt_MAN value > 0 } }
							PR_prmt_eft_plnt_VAR_MAN_DB = { MAN = -1 ALL = -1 }
						}
					#	else = {
					#		random_owned_pop = { limit = { is_unemployed = yes }		kill_pop = yes }
					#	}
					}
				}
				# PC
				else = {
					while = { count = TEMP_var_plnt_pop
						# 消耗人力
						if = { limit = { check_variable = { which = PR_var_plnt_MAN value > 0 } }
							PR_prmt_eft_plnt_VAR_MAN_DB = { MAN = -1 ALL = -1 }
						}
						# 消耗人口
						else = {
							random_owned_pop = {
								weights = { base = 1
									modifier = { factor = 8		is_unemployed = yes }	# 无业
									modifier = { factor = 6		OR = { is_pop_category = slave		is_pop_category = purge } }	# 贱民
									modifier = { factor = 4		OR = { is_pop_category = worker		is_pop_category = simple_drone } }	# 低阶层
									modifier = { factor = 2		OR = { is_pop_category = specialist } }	# 中阶层
								}
								kill_pop = yes
							}
						}
					}
				}
			}
			# clear
			clear_variable = TEMP_var_plnt_pop
		
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}
	
	ai_weight = {
		weight = 5
	}
}

AAR_DEC_webway_EXPLORE = {
	owned_planets_only = yes

#	enactment_time = 180
	resources = {
		category = decisions
		cost = {
			unity = 500
			energy = 3000
		}
	}
	
	potential = {
		is_occupied_flag = no
		owner = { has_ascension_perk = AAR_AP_webway }
	}
	
	effect = {
		custom_tooltip = AAR_tips_DEC_webway_EXPLORE
		hidden_effect = {
			solar_system = {
				### 存在网道设施
				if = {
					limit = {
						OR = {
							has_megastructure = AAR_MEGA_webway_BASE
							has_megastructure = AAR_MEGA_webway_NODE
							has_megastructure = AAR_MEGA_webway_NEXUS
						}
					}
					prev = { planet_event = { id = AAR_evt_WEBWAY.11 } }	# 网道勘探失败
				}
				### 没有网道设施
				else = {
					prev.owner = {
						AAR_eft_cntr_WEBWAY_cap_DB = yes	# 网道capDB
						
						##	|网道容量充足|
						if = { limit = { check_variable = { which = AAR_var_cntr_WEBWAY_used value < AAR_var_cntr_WEBWAY_cap } }
						#	prevprev = { planet_event = { id = AAR_evt_WEBWAY.10 } }	# 网道勘探成功
							##	直接部署基址	参考[网道勘探成功]
							prev = {
								#	部署基址
								spawn_megastructure = {
									name = "NAME_AAR_mega_WEBWAY_BASE"
									type = AAR_MEGA_webway_BASE
									orbit_angle = 0
									orbit_distance = 25
									owner = root.owner
								#	graphical_culture = fromfrom
									planet = star
								}
								#	直接开始枢纽升级
								random_system_megastructure = { limit = { is_megastructure_type = AAR_MEGA_webway_BASE }
									upgrade_megastructure_to = AAR_MEGA_webway_NEXUS
								}
								#	网道基址腐化	|code保留|
								set_timed_star_flag = { flag = AAR_flg_sstm_WEBWAY_base days = 185 }	#_|90|因为EFT升级不触发mega的[on_build_start],因此直接延长腐化FLG
								#	网道sstm循环
								if = { limit = { NOT = { has_star_flag = AAR_flg_sstm_WEBWAY_debug } }
									set_star_flag = AAR_flg_sstm_WEBWAY_debug
									system_event = { id = AAR_evt_WEBWAY.25 days = 1 random = 7 }	# 初始触发
								}
							}
						}
						##	|网道容量不足|
						else = {
							prevprev = { planet_event = { id = AAR_evt_WEBWAY.11 } }	# 网道勘探失败
						}
					}
				}
			}
		}
	}
	
	ai_weight = {
		weight = 0
	}
}

AAR_DEC_prerogative_clean = {
	owned_planets_only = yes

	enactment_time = 180
	resources = {
		category = decisions
		cost = {
			influence = 25
			unity = 100
		}
	}
	
	potential = {
		is_occupied_flag = no
		owner = { has_ascension_perk = eap_imperial_prerogative }
	}
	allow = {
		custom_tooltip = {
			fail_text = AAR_failtips_CLEAN_claim
			hidden_trigger = {
				owner = { is_ai = no	# PC
					NOT = { has_country_flag = AAR_flg_cntr_CLEAN_claim }
				}
			}
		}
	}
	
	effect = {
		custom_tooltip = AAR_tips_DEC_prerogative_clean
		hidden_effect = {
			## INIT
			set_variable = { which = TEMP_var_SSTM_claimed value = 0 }	# |被宣称星系数量|
			
			## 被宣称星系DB
			owner = {
				every_system_within_border = {
					limit = {
						any_country = { has_claim = prev }
						distance = {#几何距离=75
							source = prevprev
							type = euclidean
							max_distance < 75
						}
					}
					prevprev = {#星球
						change_variable = { which = TEMP_var_SSTM_claimed value = 1 }
					}
				}
			}
			
			## 消耗DB
			set_variable = { which = TEMP_var_influence value = TEMP_var_SSTM_claimed }
			multiply_variable = { which = TEMP_var_influence value = 75 }
			set_variable = { which = TEMP_var_influence_cost value = TEMP_var_SSTM_claimed }
			multiply_variable = { which = TEMP_var_influence_cost value = -75 }
			set_variable = { which = TEMP_var_unity value = TEMP_var_SSTM_claimed }
			multiply_variable = { which = TEMP_var_unity value = 350 }
			set_variable = { which = TEMP_var_unity_cost value = TEMP_var_SSTM_claimed }
			multiply_variable = { which = TEMP_var_unity_cost value = -350 }
			planet_event = { id = AAR_evt_PREROGATIVE.31 }
		}
	}
	
	ai_weight = {
		weight = 0
	}
}

							has_defensive_pact = prevprev.owner
							is_guaranteeing = prevprev.owner
							is_in_federation_with = prevprev.owner
							has_overlord = prevprev.owner
							is_forced_friendly = prevprev.owner
						#	is_forced_neutral = prevprev.owner
						}
					}
					distance = {#135超空间通信
						source = prev
						type = hyperlane
						max_distance < 135
					}
				}
				# |星球数量|统计
				prev.owner = {
					change_variable = { which = TEMP_var_enemy_colony value = 1 }
				}
				# 添加态度
				owner = {
					random_list = {
						3 = {}
						3 = { add_opinion_modifier = { modifier = AAR_opinion_hostile_propaganda who = prevprev.owner } }
						2 = { while = { count = 2		add_opinion_modifier = { modifier = AAR_opinion_hostile_propaganda who = prevprev.owner } } }
						1 = { while = { count = 3		add_opinion_modifier = { modifier = AAR_opinion_hostile_propaganda who = prevprev.owner } } }
					}
				}
				# 添加PM
				remove_modifier = AAR_PM_prerogative_propaganda
				remove_modifier = AAR_PM_prerogative_propaganda_GESTALT
				if = { limit = { owner = { is_gestalt = no } }
					add_modifier = { modifier = AAR_PM_prerogative_propaganda days = 3600 }
				}
				else = {
					add_modifier = { modifier = AAR_PM_prerogative_propaganda_GESTALT days = 3600 }
				}
				remove_planet_flag = AAR_flg_plnt_PREROGATIVE_clean_READY	# |AAR权力之巅星球清理READY|清理	防止修正意外清除
			}
			
			## aar权巅宣传战note
			owner = {
				if = { limit = { is_ai = no }		country_event = { id = AAR_evt_PREROGATIVE.32 } }	# PC
				else = { clear_variable = TEMP_var_enemy_colony }
			}
			
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}

		weight = 5
