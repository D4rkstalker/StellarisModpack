decision_run_jobs_evaluation = {
	sound = event_administrative_work
	owned_planets_only = yes
	resources = {
		category = decisions
		cost = {
		}
	}
	
	potential = {
		not = { has_planet_flag = pf_filter_vanilla }
		not = { has_planet_flag = pf_did_job_eval }
		exists = owner
		owner = { is_ai = no }
	}
	effect = {
		every_owned_pop = {
			limit = {
				not = { has_job = criminal }
				not = { has_job = destitute }
				not = { is_unemployed = yes }
				not = { is_pop_category = ruler }
			}
			unemploy_pop = yes
			clear_pop_category = yes
			# unemploy_pop = yes
		}
		check_planet_employment = yes
		# check_planet_employment = yes
		
		set_timed_planet_flag = {
			flag = pf_did_job_eval
			days = 5
		}
	}
	
	ai_weight = {
		weight = 0
	}
}

decision_conscription = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_victorious_army

	resources = {
		category = decisions
		cost = { sr_wealth = 1000 }
	}
	
	potential = {
		not = { has_planet_flag = pf_filter_vanilla }
		not = { has_planet_flag = pf_filter_districts }
		num_pops > 80
		exists = owner
		owner = { 
			is_regular_empire = yes
			or = {
				has_ethic = ethic_militarist
				has_ethic = ethic_fanatic_militarist
			}
		}
	}
	
	allow = {
		NOT = { has_modifier = planet_employee_subsidies }
		NOT = { has_modifier = planet_worship_as_work }
		NOT = { has_modifier = planet_conscription }
		NOT = { has_modifier = planet_state_labour }
		exists = owner
		or = {
			free_jobs < 1
			owner = { is_at_war = yes }
		}
	}
	
	effect = {
		add_modifier = {
			modifier = planet_conscription
			days = -1
		}
	}
	
	ai_weight = {
		weight = 2
		modifier = {
			factor = 0
			owner = {
				resource_stockpile_compare = {
					resource = sr_wealth
					value < 2000
				}
			}
		}
	}
}
