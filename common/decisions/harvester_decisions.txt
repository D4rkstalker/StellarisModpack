decision_harvester_stripmine_world = {
	owned_planets_only = no
	sound = event_administrative_work
	icon = decision_strip_mining
	enactment_time = 0
	resources = {
		category = decisions
		cost = {
			influence = 25
		}
	}

	potential = {
		# This kind of chained scoping is highly illegal and has been banned in over 100 jurisdictions, including the one I reside in.
		solar_system = { 
			exists = owner
			owner = { has_country_flag = harvest_relic_decision } 
		}
		not = { has_modifier = harvested_stage_4 }
		OR = { 
			is_content_world = no 
			# Can't touch Zarqlan's house, but worlds in ur home system are fair game
			solar_system = {
				is_capital_system = yes
			}
		}
		nor = {
			is_star = yes
			is_asteroid = yes
			is_planet_class = pc_gas_giant
			is_planet_class = pc_shattered
			
			is_colony = yes
			is_colonizable = yes
		}
	}

	allow = {
		planet_size >= 10
	}

	effect = {
		custom_tooltip = decision_harvester_stripmine_world_effects_short
		
		# Trimmed down version of stripmining event, reduces error log clutter
		if = {
			limit = {
				OR = {
					is_planet_class = pc_frozen
					is_planet_class = pc_barren_cold
					is_planet_class = pc_tundra
					is_planet_class = pc_arctic
				}
			}
			change_pc = pc_barren_cold
		}
		else = {
			change_pc = pc_barren
		}
		
		add_deposit = harvester_deposit_stage_4
		add_modifier = { modifier = "harvested_stage_4" days = -1 }
		solar_system = { owner = { remove_country_flag = harvest_relic_decision } }
	}
}
