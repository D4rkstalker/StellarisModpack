# Sanctify Singularity
decision_sanctify_singularity = {
	icon = decision_surreal_visions
	sound = event_administrative_work
	
	resources = {
		category = decisions
		cost = {
			influence = 75
		}
	}

	potential = {
		exists = space_owner
		space_owner = { is_same_value = from }
		from = {
			has_valid_civic = civic_physics_cult
			has_ascension_perk = ap_consecrated_worlds_nihilistic
		}
		is_planet_class = pc_black_hole
		NOR = {
			has_planet_flag = sanctified@from
			has_modifier = sanctify_1
		}
	}

	allow = {
		custom_tooltip = {
			from = {
				check_variable = {
					which = num_sanctified_worlds
					value < 3
				}
			}
			fail_text = sanctified_worlds_too_many
		}
	}

	effect = {
		add_modifier = {
			modifier = "sanctify_1"
			days = -1
		}
		hidden_effect = { #make similar to cw ap
			set_planet_flag = sanctified@from
			from = {
				if = {
					limit = { check_variable = { which = num_sanctified_worlds value < 1 }	}
					set_variable = {
						which = num_sanctified_worlds
						value = 0
					}
				}
				change_variable = {
					which = num_sanctified_worlds
					value = 1
				}
				if = {
					limit = {
						check_variable = {
							which = num_sanctified_worlds
							value < 1
						}
					}
					add_modifier = {
						modifier = "sanctify_1_empire"
					}
				}
				else_if = {
					limit = {
						check_variable = {
							which = num_sanctified_worlds
							value = 1
						}
					}
					add_modifier = {
						modifier = "sanctify_2_empire"
					}
				}
				else_if = {
					limit = {
						check_variable = {
							which = num_sanctified_worlds
							value = 2
						}
					}
					add_modifier = {
						modifier = "sanctify_3_empire"
					}
				}
				else = {
					add_modifier = {
						modifier = "sanctify_1_empire"
					}
				}
			}
		}
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 10
			habitable_planet = yes
			NOT = {
				space_owner = {
					any_owned_species = {
						prevprev = { habitability = { who = prev value > 0.35 }	}
					}
				}
			}
		}
	}
}

# TMA Intelligence Boost
decision_tma_intelligence = {
	owned_planets_only = yes
	icon = decision_synth_sabotage
	sound = event_administrative_work

	enactment_time = 0
	resources = {
		category = decisions
		cost = {
			influence = 50
		}
	}
	
	potential = {
		exists = owner
		owner = {
			has_country_flag = maai_creator
		}
		has_global_flag = master_ai_project_complete
		NOT = { has_global_flag = tma_hostile }
	}
	
	allow = {
		NOT = {
			has_modifier = tma_boost_planet
		}
		num_pops > 5
	}
	
	effect = {
		custom_tooltip = tma_intelligence_planet_tt
		add_modifier = { modifier = "tma_boost_planet" days = 720 }
		hidden_effect = {
			owner = {
				if = {
					limit = {
						num_pops < 10
					}
					change_variable = {
						which = tma_intelligence
						value = 2
					}
				}
				else_if = {
					limit = {
						num_pops >= 10
						num_pops < 20
						
					}
					change_variable = {
						which = tma_intelligence
						value = 3
					}
				}
				else_if = {
					limit = {
						num_pops >= 20
						num_pops < 30
						
					}
					change_variable = {
						which = tma_intelligence
						value = 5
					}
				}
				else_if = {
					limit = {
						num_pops >= 30
						num_pops < 40
						
					}
					change_variable = {
						which = tma_intelligence
						value = 8
					}
				}
				else_if = {
					limit = {
						num_pops >= 40
						num_pops < 50
						
					}
					change_variable = {
						which = tma_intelligence
						value = 9
					}
				}
				else_if = {
					limit = {
						num_pops >= 50
						
					}
					change_variable = {
						which = tma_intelligence
						value = 10
					}
				}
				else = {
					random_list = {
						10 = {
							change_variable = {
								which = tma_intelligence
								value = 5
							}
						}
						10 = {
							change_variable = {
								which = tma_intelligence
								value = 10
							}
						}
					}
				}
			}
		}
	}
	
	ai_weight = {
		weight = 10
	}
}
