# Wide Range Shield Array
utility_component_template = {
	key = "SHIP_AURA_SHIELDS"
	size = medium
	icon = "GFX_ship_part_wide_range_shield_array"
	icon_frame = 1
	power = 0

	size_restriction = { juggernaut harvest_rcu }
	component_set = "ship_aura_components"

	
	prerequisites = { "tech_harvester_weapons_2" }

	friendly_aura = {
		name = "aura_shields"
		system_wide = yes

		stack_info = {			#A ship can only be affected by one aura of a certain id. If multiple auras have the same ID the one with the highest priority will be applied.
			id = friendly_shield
			priority = 1
		}

		modifier = {
			ship_shield_add = 1500
			ship_shield_regen_add_static = 15
		}

		graphics = {
			area_effect = {
				entity = "circle_area_entity"
				dynamic_scale = yes
			}
			ship_effect = {
				entity = "ship_aura_negative_entity"
				dynamic_scale = no
			}
		}
	}

	ai_weight = {
		weight = 100
	}
}
