##### DEPOSIT BLOCKERS #####

# Стартовый блокиратор

apsr_unique_deposit_blocker = {
	time = 360

	is_for_colonizable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	planet_modifier = {
		planet_max_districts_add = -2
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 500
		}
	}

	on_cleared = {
		custom_tooltip = apsr_unique_deposit_blocker_cleared_tooltip
		hidden_effect = {
			apsr_unique_deposit_blocker_cleared = yes
		}
	}

	ai_weight = {
		weight = 5
	}
}
