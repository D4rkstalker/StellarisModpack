			energy = 500

		}
	}

	on_cleared = {
		if = {
			limit = {

				50 = {
					owner = { add_resource = { minor_artifacts = 1 } }
				}
				50 = {
					owner = { add_resource = { minor_artifacts = 2 } }
				}
			}
		}
	}
