## The Infestation Deposits by Annatar ##

d_exofungus_contamination = {
	icon = d_infestation
	time = 180
	
	is_for_colonizable = yes
	category = deposit_cat_blockers
	
	resources = {
		category = deposit_blockers
		cost = {
			energy = 300
		}
	}
	
	potential = { 
		always = no
	}
	
	planet_modifier = {
		planet_max_districts_add = -1
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_exomycosis }
		}
		modifier = {
			pop_growth_speed = 0.05
		}
	}
	
	on_cleared = {
		if = {
			limit = {
				exists = owner
				owner = { has_origin = origin_exomycosis }
			}
			create_pop = {
				species = owner
			}
		}
		else = {
			add_planet_devastation = 20
		}
	}

	drop_weight = {
		weight = 0
	}
	
	ai_weight = {
		weight = 2
	}
}
