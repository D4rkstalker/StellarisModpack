d_malfunctioning_caretaker_blocker = {
	icon = d_glitchy_matrix
	time = 360

	is_for_colonizable = yes
	category = deposit_cat_rare_blocker

	can_be_cleared = {
		always = yes
	}

	resources = {
		category = deposit_blockers
		cost = {
			alloys = 1000
		}
	}

	potential = {
		always = yes
	}

	planet_modifier = {
		planet_max_districts_add = -1
		pop_amenities_usage_mult = 0.25
	}

	on_cleared = {
		hidden_effect = {
			if = {
				limit = {
					exists = owner
					owner = {
						NOT = { has_ethic = ethic_gestalt_consciousness }
					}
				}
				while = {
					count = 2
					create_pop = {
						species = event_target:harvester_species
						ethos = random
					}
				}
			}
			else_if = {
				limit = {
					exists = owner
					owner = {
						has_ethic = ethic_gestalt_consciousness
					}
				}
				while = {
					count = 2
					create_pop = {
						species = event_target:harvester_species_machine
						ethos = random
					}
				}			
			}
			planet_event = { id = harvester.600 }
		}
	}
	
	drop_weight = {
		weight = 0
	}
}
