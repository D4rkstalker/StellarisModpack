d_collapsed_burrows_remote = {
	icon = d_deep_sinkhole
	time = 120

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	planet_modifier = {
		planet_max_districts_add = -1
	}

	on_cleared = {
	custom_tooltip = creates_remote_drone
	random_owned_pop = {
			limit = { species = { has_species_flag = overminded_flag } } 
			species = { save_event_target_as = overminded_species }
			}	
			
		create_pop = {
			species = event_target:overminded_species 
		}
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 300
		}
	}
}


# Hibernating Lithoids are generated on lithoid homeworlds at game-start
d_hibernating_lithoids_remote = {
	icon = d_buried_lithoids
	time = 360

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	on_cleared = {
		if = {
			limit = {
				is_homeworld = no
			}
			owner = { add_resource = { minerals = 300 } }
		}

		if = {
			limit = {
				is_homeworld = yes
			}
			custom_tooltip = creates_remote_drone
			random_owned_pop = {
			limit = { species = { has_species_flag = overminded_flag } } 
			species = { save_event_target_as = overminded_species }
			}	
			
			create_pop = {
				species = event_target:overminded_species 
			}
		}
	}

	resources = {
		category = deposit_blockers
		cost = {
			trigger = {
				planet = {
					is_homeworld = no
				}
			}
			energy = 300
		}
		cost = {
			trigger = {
				planet = {
					is_homeworld = yes
				}
			}
			minerals = 1000
		}
	}

	ai_weight = {
		weight = 1

		modifier = {
			factor = 2
			planet = {
				is_homeworld = yes
			}
		}
	}
}
