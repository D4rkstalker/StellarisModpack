d_underground_habitat_1 = {
	icon = d_ancient_vault
	time = 365

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	planet_modifier = {
		planet_max_districts_add = -1
	}

	on_cleared = {
		if = {
			limit = {
				any_owned_pop = {
					is_gestalt = no
					is_robot_pop = no
				}
			}
			random_owned_pop = {
				limit = {
					is_gestalt = no
					is_robot_pop = no
				}
				root = {
					create_pop = {
						species = prev
					}
				}
			}
		}
		else = {
			random_owned_pop = {
				root = {
					create_pop = {
						species = prev
					}
				}
			}
		}
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 1000
		}
	}
}
