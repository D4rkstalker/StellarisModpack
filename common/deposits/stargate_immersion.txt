d_stargate_excavation = {
	is_for_colonizeable = yes
	category = deposit_cat_rare

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
		    job_researcher_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
            planet = {
                has_planet_flag = stargate_excavation_startup_funding
            }
		}
		modifier = {
		    job_researcher_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
            planet = {
                has_planet_flag = stargate_excavation_more_funding
            }
		}
		modifier = {
		    job_researcher_add = 1
		}
	}

	resources = {
		category = planet_deposits

		upkeep = {
			trigger = {
				planet = {
				    has_planet_flag = stargate_excavation_startup_funding
				}
			}
			energy = 10
			minerals = 10
		}
		upkeep = {
			trigger = {
				planet = {
				    has_planet_flag = stargate_excavation_more_funding
				}
			}
			energy = 15
			minerals = 15
		}
	}

	drop_weight = {
		weight = 0
	}
}
