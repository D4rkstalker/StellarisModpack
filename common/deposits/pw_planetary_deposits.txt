		planet_jobs_food_produces_mult = 0.05
		district_farming_max = 1

		potential = {
			exists = owner

					owner = { is_anglers_empire = yes }

				}
			}
		}
		modifier = {
			job_farmer_add = 3
		}
	}

		potential = {
			exists = owner

			exists = owner
