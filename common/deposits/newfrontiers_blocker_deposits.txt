		d_ancient_ruins
		d_relic_landfill
		d_serene_coastline
	}

	drop_weight = {
		weight = @blocker_high
		modifier = {
			factor = 3
			is_planet_class = pc_desert
		}
		modifier = {
			factor = 2
			has_modifier = "dune_world"
		}

			factor = 0.5
			has_modifier = "oasis_world"

				has_modifier = "salt_world"
				has_modifier = "prairie_world"
				has_modifier = "steppe_world"
			}
		}
		modifier = {
			factor = 0
			num_free_districts = {
				type = district_city
				value < 2
			}
		}
	}
	
	prerequisites = { "tech_dune_sea" }

d_industrial_boneyard = {
	time = 270
	
	is_for_colonizeable = yes
	category = deposit_cat_blockers
	icon = d_metal_boneyard
	
	resources = {
		category = deposit_blockers
		cost = {
			energy = 700
		}
	}
	
	potential = {
		is_planet_class = pc_nuked
	}
	
	planet_modifier = {
		planet_max_districts_add = -1
		planet_jobs_engineering_research_produces_mult = 0.10
	}
	
	on_cleared = {
		if = {
			limit = { exists = owner }
			owner = { 
				add_resource = { alloys = 100 } 
				if = {
					limit = { has_origin = origin_scavengers }			
					give_random_eng_technology = yes
				}
			}
		}
	}

	drop_weight = {
		weight = @blocker_high
		modifier = {
			factor = 5
			has_modifier = "depleted_world"
		}
		modifier = {
			factor = 0
			num_free_districts = {
				type = district_city
				value < 2
			}
		}
	}
}
