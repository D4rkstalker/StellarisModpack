	time = 1080
	is_for_colonizeable = yes
	category = deposit_cat_blockers

	resources = {
		category = deposit_blockers
		cost = {
			rare_crystals = @shadowgate_blocker_cost_sm
		}
	}

	potential = { has_planet_flag = shadowgate_planet }

	on_cleared = {
		custom_tooltip = shadowgate.scrap.tooltip
		owner = {
			add_resource = { sr_dark_matter = @shadowgate_district_cost_dm_half }
			add_resource = { sr_living_metal = @shadowgate_district_cost_lm_half }
			if = { limit = { hidden_trigger = { NOR = { has_tech_option = tech_mine_living_metal has_technology = tech_mine_living_metal } } } add_research_option = tech_mine_living_metal }
		}
	}

	planet_modifier = {
		planet_housing_add = -2
		district_harmonic_subspace_resonator_max = -1
		pop_environment_tolerance = -0.05
	}

	drop_weight = { weight = 0 }

	ai_weight = {
		weight = 100000
		modifier = {
			factor = 0
			from = { num_districts = { type = district_harmonic_subspace_resonator value = 0 } }
		}
	}
}
