automate_growth_management = {

	available = {
		AND = {
			free_housing >= 0
			free_building_slots > 0
			owner = { has_policy_flag = ipa_growth_buildings_yes }
			OR = {
				AND = {
					NOT = { has_building = building_clinic }
				}
				AND = {
					NOT = { has_building = building_machine_assembly_plant }
				}
				AND = {
					NOT = { has_building = building_spawning_pool }
				}
			}
		}
	}

	crisis = yes

	buildings = {

		# regular empire
		clinic = {
			building = building_clinic
			available = {
				AND = {
					NOT = { has_building = building_clinic }
					owner = { clinic_upkeep_affordable = yes }
				}
			}
		}
		
		# machine
		machine_assembly_plant = {
			building = building_machine_assembly_plant
			available = {
				AND = {
					NOT = { has_building = building_machine_assembly_plant }
					owner = { machine_assembly_plant_upkeep_affordable = yes }
				}
			}
		}

		# hive
	    spawning_pool = {
			building = building_spawning_pool
			available = {
				AND = {
					NOT = { has_building = building_spawning_pool }
					owner = { spawning_pool_upkeep_affordable = yes }
				}
			}
		}
	}
}
