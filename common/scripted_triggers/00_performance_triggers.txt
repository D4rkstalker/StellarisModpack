z_pop_job_trigger = {
	OR = {
		has_global_flag = z_manual_job_check	#_手动检测
		z_performance_trigger1 = yes			#_全职业
		z_performance_trigger2 = yes			#_失业人口与奴隶
		hfe_trigger_uncapped_job_mod = yes		#_human fallen empire 友情支持
	}
}

#_全职业检测
z_performance_trigger1 = {
	has_global_flag = z_allow_job_check
	NOT = { has_global_flag = z_performance1 }
}

z_performance_trigger2 = {
	OR = {
		has_job = no
		has_job = servant
		has_job = slave_toiler
	}
	OR = {
		has_global_flag = z_allow_job_check2
		has_global_flag = z_allow_nojob_check
	}
	NOT = { has_global_flag = z_performance1 }
}
