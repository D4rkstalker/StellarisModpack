eac_is_planet_resettlement_target = {
    free_housing >= 1
    OR = {
        free_jobs >= 1
        OWNER = { has_policy_flag = eac_apm_freejobs_ignore }
    }
    is_under_colonization = no 
}

    OR = {
        has_modifier = eac_apm_planet_preference_modifier 
        has_modifier = eac_apm_focus_food_modifier 
        has_modifier = eac_apm_focus_energy_modifier 
        has_modifier = eac_apm_focus_minerals_modifier 
        has_modifier = eac_apm_focus_research_modifier 
        has_modifier = eac_apm_focus_toilers_modifier
        has_modifier = eac_apm_focus_no_migration_modifier
        has_modifier = eac_apm_focus_all_overcrowding_modifier
        has_modifier = eac_apm_focus_assimilating_modifier
    }
}
