buildings_simple_allow = {
	hidden_trigger = {
		planet = {
			OR = {
				AND = {
					free_jobs < 10
					num_free_districts = { type = district_mining value = 0 }
					num_free_districts = { type = district_generator value = 0 }
					num_free_districts = { type = district_farming value = 0 }
					num_free_districts = { type = district_industrial value = 0 }
				}
				AND = {
					exists = owner
					owner = { is_ai = no }
				}
			}
		}
	}
}

building_basic_income_check = {
	hidden_trigger = {
		planet = {
			exists = owner
			OR = {
				owner = { is_ai = no }
				AND = {
					free_jobs < 7
					owner = {
						OR = {
							resource_income_compare = {
								resource = energy
								value > -5
							}
							has_resource = { type = energy amount > 1500 }
						}
	
						OR = {
							resource_income_compare = {
								resource = minerals
								value > 0
							}
							has_resource = { type = minerals amount > 1500 }
						}
					}
				}
			}
		}
	}
}

building_relaxed_basic_income_check = {
	hidden_trigger = {
		planet = {
			exists = owner
			OR = {
				owner = { is_ai = no }
				AND = {
					owner = {
						OR = {
							resource_income_compare = {
								resource = energy
								value > -5
							}
							has_resource = { type = energy amount > 1000 }
						}
	
						OR = {
							resource_income_compare = {
								resource = minerals
								value > 0
							}
							has_resource = { type = minerals amount > 1000 }
						}
					}
				}
			}
		}
	}
}
