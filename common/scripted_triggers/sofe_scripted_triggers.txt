	OR = {
		is_country_type = swarm
		is_country_type = extradimensional
		is_country_type = extradimensional_2
		is_country_type = extradimensional_3
		is_country_type = ai_empire
		is_country_type = gray_goo
		is_country_type = awakened_marauders
		is_country_type = original_empire
	}
}

can_destroy_planet_with_ACOT_PLANET_KILLER_NANITE = {
	custom_tooltip = {
		fail_text = is_a_golden_world
		NOR = {
			is_planet_class = pc_golden_world
			is_planet_class = pc_golden_world_habitable		
		}
	}
	custom_tooltip = {
		fail_text = is_not_hostile
		OR = {
			AND = {
				#exists = space_owner
				space_owner = {
					OR = {
						is_same_value = from.owner
						is_at_war_with = from.owner
						is_country_type = swarm
						is_country_type = extradimensional
						is_country_type = extradimensional_2
						is_country_type = extradimensional_3
						is_country_type = ai_empire
					}
				}
			}
			AND = { # Ownelerless
				NOT = {
					exists = space_owner
				}
				NOT = {
					is_at_war_with = from.owner
				}
			}
			AND = { # primitive
				exists = space_owner
				exists = owner
				owner = {
					is_primitive = yes
				}
				space_owner = {
					OR = {
						is_at_war_with = from.owner
						is_country_type = swarm
						is_country_type = extradimensional
						is_country_type = extradimensional_2
						is_country_type = extradimensional_3
						is_country_type = ai_empire
					}
				}
			}
		}
		NOT = {
			is_owned_by = event_target:origin_crisis_country		
		}
	}
}
