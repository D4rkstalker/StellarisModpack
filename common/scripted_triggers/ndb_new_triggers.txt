is_valid_ndb_hostile_target_country = {
	has_communications = root
					OR = {
						is_at_war_with = root
						is_hostile = root
					}
					OR = {
						AND = {
							any_system_within_border = { ndb_can_target_system = yes }
							NOT = { is_country_type = primitive }
						}
						AND = {
							is_country_type = primitive
							any_owned_planet = {
								exists = solar_system
								solar_system = {
									ndb_can_target_system = yes
									OR = {
										NOT = { exists = owner }
										AND = {
											exists = owner
											owner = { is_hostile = root }
										}
									}
								}
							}
						}
					}
}

is_valid_ndb_neutral_target_country = {
	has_communications = root
					NOR = {
						is_at_war_with = root
						is_hostile = root
						is_same_value = root
						has_defensive_pact = root
						is_in_federation_with = root
						is_overlord_to = root
						AND = {
							is_subject = yes
							exists = overlord
							overlord = { is_same_value = root }
						}
					}
					OR = {
						AND = {
							any_system_within_border = { ndb_can_target_system = yes }
							NOT = { is_country_type = primitive }
						}
						AND = {
							is_country_type = primitive
							any_owned_planet = {
								exists = solar_system
								solar_system = {
									ndb_can_target_system = yes
									OR = {
										NOT = { exists = owner }
										AND = {
											exists = owner
											owner = { is_hostile = root }
										}
									}
								}
							}
						}
					}
}
is_valid_ndb_friendly_target_country = {
	OR = {
		AND = {
			is_same_value = root
			any_system_within_border = { ndb_can_target_system = yes }
			NOT = { is_country_type = primitive }
		}
		AND = {
			has_communications = root
			is_country_type = primitive
			any_owned_planet = {
				exists = solar_system
				solar_system = {
					ndb_can_target_system = yes
					exists = owner
					owner = { is_same_value = root }
				}
			}
		}
	}
