frontier_planet_slave_tick_check = {
	is_capital = no
	is_colony = yes
	is_occupied_flag = no
	planet_stability < 25
	has_ground_combat = no
	has_orbital_bombardment = no
	has_planet_flag = frontier_planet_@root
	NOT = { planet_devastation >= 30 }
	NOT = { has_planet_flag = plan_sep_active }
	NOT = { has_modifier = planet_culture_shock }
	NOT = { has_planet_flag = recent_sep_rebellion }
	NOT = { has_planet_flag = rebellion_suppressed }
	NOT = { has_planet_flag = plan_slave_sep_active }

		limit = { exists = sector }
		sector = {
			NOT = { has_sector_flag = sector_sep_active }
			NOT = { has_sector_flag = sec_slave_sep_active }
		}
	}
	count_owned_pop = {
		limit = {
			can_think = yes
			pop_has_happiness = yes
		}
		count >= 10
	}
	pop_percentage = {
		percentage > 0.3
		limit = {
			is_enslaved_or_purged = yes
			happiness < 60
		}
		exclude = { can_think = no }
	}
}

	NOT = { has_modifier = planet_culture_shock }
	NOT = { has_planet_flag = recent_sep_rebellion }
	NOT = { has_planet_flag = rebellion_suppressed }

		NOT = { exists = sector }
		sector = {
			NOR = {
				has_sector_flag = sector_sep_active

			}
		}

	count_owned_pop = {
		limit = {
			can_think = yes
			pop_has_happiness = yes
		}
		count >= 3
	}
	if = { # So a slave rebellion is preferred over a geneneral separatist rebellion
		limit = { any_owned_pop = { is_enslaved_or_purged = yes } }
		pop_percentage = {
			percentage <= 0.3
			limit = {
				is_enslaved_or_purged = yes
				happiness < 60
			}
			exclude = { can_think = no }
		}
	}
	OR = {
		planet_stability < 30
		AND = {
			planet_crime  >= 60
			OR = {
				NOT = { exists = sector.leader }
				AND = {
					exists = sector.leader
					sector.leader = { NOT = { has_trait = leader_trait_corrupt } }
				}
			}
		}
	}
}
