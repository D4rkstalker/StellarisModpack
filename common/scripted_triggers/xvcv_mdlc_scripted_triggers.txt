xvcv_mdlc_synth_empire_trigger = {
	host_has_dlc = "Utopia"
	NOT = { has_ethic = ethic_gestalt_consciousness }
	OR = {
		owner.species = { has_trait = trait_mechanical }
		has_country_flag = synthetic_empire
		has_country_flag = xvcv_mdlc_synth_empire
	}
}

	}
}
