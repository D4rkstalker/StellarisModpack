slave_may_be_student = {
	or = {
		complex_worker_job_check_trigger = yes 
		and = {
			is_enslaved = yes
			has_slavery_type = { type = slavery_indentured }
			planet = { has_modifier = education_programs }
			exists = owner
			owner = {
				not = {
					has_policy_flag = state_education_none				
				}
			}
		}
	}
}
