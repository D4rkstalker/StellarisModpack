	NOR = {
		has_deposit_for = shipclass_research_station
		solar_system = {
			NOT = {
				any_system_planet = {
					OR = {
						has_planet_flag = has_living_metal_deposit
						has_planet_flag = has_zro_deposit
					}
				}
			}
		}
		event_target:current_system_creator = { NOT = { has_technology = tech_mine_zro } }
