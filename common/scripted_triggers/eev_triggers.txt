eev_is_valid_system = {
	solar_system = {
		exists = star
		NOR = {
			any_system_planet = {
				OR = {
					exists = archaeological_site
					has_planet_flag = hatchling_egg
					has_planet_flag = suppress_archaeological_sites
				}
			}
			has_special_star_flag_trigger = yes
			has_star_flag = empire_cluster
			is_fe_cluster = yes
			has_star_flag = drone_home_system
		}
	}
}

# Listing habitats
eev_is_habitat = {
	OR = {
		is_planet_class = pc_habitat
		# Starborn: Improved Voidtouched
		uses_district_set = starborn_start
		is_planet_class = pc_habitat
		uses_district_set = habitat
		# Acquisition of Technology
		is_aot_habitat = yes
		# Blue Archive
		uses_district_set = BA_auto_habitat
		# DarkSpace
		is_ds_skycatcher = yes
		# Gigastructural Engineering & More
		is_giga_habitat = yes
		# Habitation
		has_planet_flag = hab_init
		# Living Space
		has_planet_flag = stpg_planetary_habitat
		# More Events Mod
		is_pc_mem_vazuran_habitat = yes
		# Nyto Species Pack
		uses_district_set = GF_nyto_devil_furnace_world
		# Planetary Diversity
		is_pd_habitat = yes
		# NSC2 Modjam Freebooters Origin
		uses_district_set = nsc_freebooter_habitat
		# Twinks Atlantean
		uses_district_set = skypalace
		# ZOFE
		uses_district_set = utopian_habitat
	}
}
