non_landlocked_planet_criteria = {
	NOR = {
		exists = owner
		has_anomaly = yes
		solar_system = { has_star_flag = guardian }
		solar_system = { has_star_flag = hostile_system }
		is_planet_class = pc_habitat
		is_planet_class = pc_ringworld_habitable
	}
	OR = {
		AND = {
			is_surveyed = { who = event_target:landlocked_saviors status = yes }
			NOT = { exists = space_owner }
		}
		AND = {
			exists = space_owner
			space_owner = { is_country = event_target:landlocked_saviors }
		}
	}
}

is_innermost_moon = {
	orbit = {
		NOT = {
			any_moon = {
				orbit = {
					compare_distance = {
						closer_object = prev
						further_object = root
					}
				}
			}
		}
	}
}

is_mzilli_shadow = {
	is_planet_class = pc_mzilli_shadow
}

is_valid_star_for_infection = {
	is_star = yes
	NOR = {
		is_planet_class = pc_t_star
		is_planet_class = pc_black_hole
		is_planet_class = pc_neutron_star
		is_planet_class = pc_pulsar
		is_planet_class = pc_d_star
		is_planet_class = pc_yd_star
		is_planet_class = pc_l_star
		is_planet_class = pc_y_star
		is_planet_class = pc_toxoid_star
		is_planet_class = pc_mzilli_rogue
	}
}
