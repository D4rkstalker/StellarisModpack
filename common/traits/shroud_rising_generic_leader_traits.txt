
# All leaders will have a common trait-point value set in defines, which they use to grab traits

# leader_trait = yes - this tags the trait as a leader-trait and not a species-trait
# initial = no - if set to no, this trait can not be granted to leaders when they are generated
# leader_class = { admiral scientist } -- OR - if not set, it can be given to all leader-types except Rulers
# requires_traits = { trait_psionic } -- AND - leader-trait can only be randomly given on generation if the leaders species has this trait
# requires_governments = { despotic_hegemony ai_overlordship } -- OR - leader-trait can only be randomly given on generation if the Empire has this type of government form
# prerequisites = { tech_psionic_theory } - AND - leader-trait can only be given if the technology specified has been researched by the Empire

leader_trait_cultist_eater = {
	councilor_trait = yes
	inline_script = {
		script = trait/icon_character_trait
		CLASS = leader
		ICON = "GFX_leader_trait_cultist"
		RARITY = common
		COUNCIL = yes
		TIER = none
	}
	self_modifier = {		
	}
	modifier = {		
		pop_ethic_spiritualist_attraction_mult = 0.20
		planet_stability_add = -5
		planet_crime_add = 5		
	}
	leader_potential_add = {		
		NOT = { from = { has_ethic = "ethic_gestalt_consciousness" } }
	}
	leader_class = { admiral scientist general governor }	
	initial = no
	randomized = no
	#opposites = { "leader_trait_shroud_inquisitor" "leader_trait_shroud_loyalist" }	
}
