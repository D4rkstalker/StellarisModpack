can_modify_species = {
	root.owner = {
		OR = {
			is_ai = no	
			AND = {
				is_ai = yes
				OR = {
					AND = {
						count_exact_species = {
							count > 10
							limit = {
								OR = {
									is_same_species = root
									is_subspecies = root
								}
							}
						}
					}
				}
			}
		}
		#vanilla
		OR = {
			NOT = { has_trait = trait_self_modified }
			is_same_species = root
		}
	}
}
