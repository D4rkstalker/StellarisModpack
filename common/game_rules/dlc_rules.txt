# This = planet
can_crossbreed_species_on_planet = {

	OR = {

	AND = {
	exists = owner
	owner = { 
	has_ascension_perk = ap_xeno_compatibility 
	}	
	count_species = {
		limit = { 
			is_crossbreeding_possible = yes
		}
		count >= 2
	}
	}
	AND = {
	exists = owner
	owner = { 
	has_ascension_perk = ap_life_masters
	}	
	count_species = {
		limit = { 
			has_trait = trait_hive_mind
		}
		count >= 2
	}
	}
	}
}

# Checks if a pop can be assembled
# This = Species
# Root = Planet
can_pops_assemble_on_planet = {
	NOR = {
		has_trait = trait_do_not_assemble
		has_modifier = planet_robot_assembly_control
		has_modifier = planet_population_control_gestalt
		AND = { # conquered by crisis
			exists = controller
			controller = {
				OR = {
					is_country_type = ai_empire
					is_country_type = swarm
				}
			}
		}
	}
}
