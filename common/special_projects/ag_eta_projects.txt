special_project = {
	key = ag_capture_eta_titan
	days_to_research = 1080
	tech_department = physics_technology
	picture = GFX_evt_ship_in_orbit
	icon = "gfx/interface/icons/situation_log/situation_log_ag_quest.dds"
	
	event_scope = ship_event
	
	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		skill = 7
	}
	
	on_success = {
		if = {
			limit = { root.owner = { NOR = {
				any_controlled_ship = { OR = {
					is_ship_size = ag_eta_titan_0
					is_ship_size = ag_eta_titan_1
				} }
				any_owned_megastructure = { OR = {
					is_megastructure_type = ag_eta_titan_megastructure_0
					is_megastructure_type = ag_eta_titan_megastructure_1
					is_megastructure_type = ag_eta_titan_megastructure_2
					is_megastructure_type = ag_eta_titan_megastructure_3
					is_megastructure_type = ag_eta_titan_megastructure_4
				} }
				# has_special_project = ag_capture_eta_titan
				has_special_project = ag_structural_protection_system_602
			} } }
			ship_event = { id = ag_eta.132 }
			owner = { save_event_target_as = ag_target_country }
			every_country = {
				limit = { NOT = { is_same_value = root.owner } }
				country_event = { id = ag_eta.136 }
			}
		}
		else = { ship_event = { id = ag_eta.133 } }
		from = {
			if = {
				limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_1 }
				solar_system = { every_system_ambient_object = {
					limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_1 }
					destroy_ambient_object = this
				} }
			}
			else_if = {
				limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_2 }
				solar_system = { every_system_ambient_object = {
					limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_2 }
					destroy_ambient_object = this
				} }
			}
			else_if = {
				limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_3 }
				solar_system = { every_system_ambient_object = {
					limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_3 }
					destroy_ambient_object = this
				} }
			}
			else_if = {
				limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_4 }
				solar_system = { every_system_ambient_object = {
					limit = { has_ambient_object_flag = ag_eta_titan_hostile_debris_4 }
					destroy_ambient_object = this
				} }
			}
		}
	}
	
	on_fail = { }
}
