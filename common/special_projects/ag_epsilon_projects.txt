special_project = {
	key = ag_epsilon_ringworld_project
	days_to_research = 360
	tech_department = physics_technology
	picture = GFX_evt_fallen_empire
	icon = "gfx/interface/icons/situation_log/situation_log_ag_ancients.dds"
	
	event_scope = ship_event
	
	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		skill = 3
	}
	
	on_success = {
		from = {
			ag_change_pc = { ag_planet_class = pc_ag_ringworld_alpine }
			ag_set_ancient_ringworld = {
				ag_picture = pc_ag_ringworld_alpine
				ag_color = red
				ag_type = habitable
				ag_surface_type = alpine_02
				ag_generate_ruins = yes
			}
			random_list = {
				10 = { add_deposit = ag_epsilon_ringworld_blocker_1 }
				10 = { add_deposit = ag_epsilon_ringworld_blocker_2 }
				10 = { add_deposit = ag_epsilon_ringworld_blocker_3 }
			}
			reroll_random = yes
			while = {
				count = 2
				random_list = {
					10 = { if = { limit = { NOT = { has_deposit = ag_epsilon_ringworld_blocker_1 } } add_deposit = ag_epsilon_ringworld_blocker_1 } }
					10 = { if = { limit = { NOT = { has_deposit = ag_epsilon_ringworld_blocker_2 } } add_deposit = ag_epsilon_ringworld_blocker_2 } }
					10 = { if = { limit = { NOT = { has_deposit = ag_epsilon_ringworld_blocker_3 } } add_deposit = ag_epsilon_ringworld_blocker_3 } }
					30 = { add_deposit = ag_epsilon_ringworld_blocker_4 }
				}
				reroll_random = yes
			}
			add_deposit = ag_epsilon_ringworld_blocker_4
			add_deposit = ag_epsilon_ringworld_blocker_5
			random_list = {
				75 = { add_modifier = { modifier = "bleak_planet" days = -1 } }
				25 = { }
			}
		}
		if = {
			limit = { owner = { NOT = { has_country_flag = ag_epsilon_4_fired } } }
			owner = { set_country_flag = ag_epsilon_4_fired }
			ship_event = { id = ag_epsilon.4 }
		}
		else = { ship_event = { id = ag_epsilon.5 } }
	}
	on_fail = { }
}
