	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		skill = 3
	}
	
	on_success = {
		if = {

		}
		if = {
			limit = {
				NOT = {
					has_global_flag = apsr_terminal_site01_clear

				country_event = { id = apsr_terminal.201 }
