# Dragon Incubation for AI which doesn't use decisions, stupid iron
special_project = {
	key = "HTBD_DRAGON_AI_INCUBATION"
	energy = 5000
	days_to_research = 180
	tech_department = society_technology
	timelimit = 1800
	picture = GFX_evt_space_dragon
	icon = "gfx/interface/icons/situation_log/situation_log_society.dds"

	event_scope = planet_event

	requirements = {
	}

	AI_wait_days = {
		weight = 20
		modifier = {
			weight = 3
			OR = {
				has_ethic = ethic_fanatic_xenophile
				is_at_war = yes
				resource_stockpile_compare = { resource = energy value > 10000 }
			}
		}
	}

	on_start = {
		if = {
			limit = { htbd_debug_egg = yes }
			log = "[Prev.GetName] started to incube dragon on [This.GetName]!"
		}
		set_planet_flag = htbd_egg_decision
		prev = {
			random_list = {
				20 = {
					modifier = {
						factor = 0
						NOT = { has_technology = tech_htbd_psion }
					}
					prev = { set_planet_flag = htbd_psion_egg }
				}
				20 = {
					modifier = {
						factor = 0
						NOT = { has_technology = tech_htbd_dolly }
					}
					prev = { set_planet_flag = htbd_dolly_egg }
				}
				20 = {
					modifier = {
						factor = 0
						NOT = { has_technology = tech_htbd_cyborg }
					}
					prev = { set_planet_flag = htbd_cyborg_egg }
				}
				20 = {
					modifier = {
						factor = 0
						NOT = { has_technology = tech_htbd_nano }
					}
					prev = { set_planet_flag = htbd_nano_egg }
				}
				5 = {}
			}
		}
	}

	on_success = {
		prev = { save_event_target_as = decision_owner }
		htbd_decision_effect = yes
	}
	on_fail = {
		from = { htbd_egg_decision_done = yes }
	}
	on_cancel = {
		from = { htbd_egg_decision_done = yes }
	}
}
