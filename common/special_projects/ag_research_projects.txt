special_project = {

	cost = {
		base = @ag_locate_ancient_system_project_base_cost

	}
	tech_department = physics_technology
	picture = GFX_evt_ag_ancient_star_chart
	icon = "gfx/interface/icons/situation_log/situation_log_ag_quest.dds"
	
	event_scope = ship_event
	event_chain = ag_locate_ancient_system_project_chain
	
	requirements = {
		shipclass_science_ship = 1
		leader = scientist
	}
	
	on_success = {
		owner = {
			set_country_flag = ag_rp_round_area_physics

		fire_on_action = { on_action = ag_on_research_project_round_completed scopes = { from = root.from } }
	}
	on_fail = { }
}
