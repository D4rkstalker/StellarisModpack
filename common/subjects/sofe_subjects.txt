# Origin Protectorate
origin_protectorate = {
	can_expand = yes
	can_be_integrated = no
	joins_overlord_wars = yes
	calls_overlord_to_war = no
	can_do_diplomacy = no
	
	diplo_threat = 0.75
	
	subject_tax_rate = {
	}
	
	potential = {
		is_country_type = default
		from = { is_country_type = original_empire }
	}
	
	effect = {
		if = {
			limit = {
				from = { 
					NOT = { is_country_type = original_empire }
				}
			}
			set_subject_of = {
				who = from
				subject_type = vassal
			}
		}
	}
}


# Original Empire Banished Ascendancy
origin_banished_ascendancy = {
	can_expand = no
	can_be_integrated = no
	joins_overlord_wars = yes
	calls_overlord_to_war = no
	can_do_diplomacy = no
	
	diplo_threat = 0.75
	
	potential = {
		OR = {
			is_country_type = ascended_empire
			is_country_type = lost_empire			
		}
		from = { is_country_type = original_empire }
	}
	
	subject_tax_rate = {
	}
	
	effect = {
		if = {
			limit = {
				from = { 
					NOT = {
						is_country_type = original_empire
					}
				}
			}
			set_subject_of = {
				who = from
				subject_type = vassal
			}
		}
	}
}
