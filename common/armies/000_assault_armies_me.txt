psi_initiate_army = { #slightly weaker psi army
	damage = 1.05
	morale_damage = 1.35
	health = 1.50
	morale = 2.0
	collateral_damage = 1.00
	war_exhaustion = 2.00
	time = 120
	icon_frame = 9

	resources = {
		category = armies
		cost = {
			minerals = 150
		}

		upkeep = {
			energy = 1.5
		}
	}

	prerequisites = { "tech_telepathy" }
	potential = {
		from = {
			OR = {
				NOT = { host_has_dlc = "Utopia" }
				OR = {
					has_trait = trait_latent_psionic
					has_trait = trait_psionic
				}
			}
			NOR = {
				has_trait = "trait_mechanical"
				has_trait = "trait_machine_unit"
				is_sapient = no
			}
		}
		owner = {
			NOT = { has_authority = auth_machine_intelligence }
		}
		custom_tooltip = {
			text = psi_limit_tooltip
		}
	}

	allow = {
		custom_tooltip = {
			text = psi_limit_tooltip
			owner = {
				NOT = { 
					 check_variable = {
						which = "psi_initiate_count" 
						value < 1
					}
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "psi_initiate_count" 
				value = -1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "psi_initiate_count" 
				value = 1
			}
		}
	}
}
