# Imperial Legions
imperial_legion = {
	damage = 3.0
	health = 3.25
	morale = 5.0
	morale_damage = 1.5
	collateral_damage = 0.5
	war_exhaustion = 3.0
	time = 180
	has_species = no
	icon = GFX_army_type_imperial

	resources = {
		category = armies
		cost = {
			minerals = 400
		}

		upkeep = {
			energy = 4
			mult = galactic_emperor.value:imperial_legion_oversize
		}
	}
	potential = {
		owner = {
			is_galactic_emperor = yes
			is_active_resolution = "resolution_emperor_imperial_legions"
		}
		any_country = {
			is_galactic_community_member = yes
			NOT = { has_ethic = ethic_gestalt_consciousness }
		}
	}
	allow = {
		custom_tooltip = {
			text = imperial_legion_limit_tooltip
			owner = {
				is_galactic_emperor = yes
				check_variable = {
					which = imperial_legion_soldier_count
					value < var_imperial_legion_cap
				}
			}
		}
	}
	on_queued = {
		owner = {
			change_variable = {
				which = imperial_legion_soldier_count
				value = 1
			}
			set_variable = { which = var_imperial_legion_cap value = value:imperial_legion_cap } #显示
		}
	}
	on_unqueued = {
		owner = {
			change_variable = {
				which = imperial_legion_soldier_count
				value = -1
			}
			set_variable = { which = var_imperial_legion_cap value = value:imperial_legion_cap } #显示
		}
	}

	ai_weight = {
		base = 1000
		modifier = {
			set = 0
			NAND = {
				is_active_resolution = "resolution_emperor_imperial_legions"
				is_galactic_emperor = yes
				check_variable = {
					which = imperial_legion_soldier_count
					value < var_imperial_legion_cap
				}
			}
		}
	}
}

	damage = 2.5
	health = 3.25
	morale = 4.0
	morale_damage = 1.5
	collateral_damage = 0.5
	war_exhaustion = 3.0
	time = 180
	has_species = no
	icon = GFX_army_type_imperial

	resources = {
		category = armies
		cost = {
			minerals = 400
		}

		upkeep = {
			energy = 4
		}
	}
	potential = {
		is_active_resolution = "resolution_emperor_imperial_legions_extensive_recruitment"
		planet = {
			has_holding = {
				holding = holding_imperial_training_center
				owner = galactic_emperor
			}
		}
	}
	allow = {
		owner = { has_loyalty > 0 }
		custom_tooltip = {
			text = "imperial_vassal_legion_limit_tooltip"
			owner = {
				check_variable = {
					which = imperial_vassal_legion_count
					value < modifier:imperial_vassal_legion_capacity_add
				}
			}
		}
	}
	on_queued = {
		owner = {
			change_variable = {
				which = imperial_vassal_legion_count
				value = 1
			}
		}
	}
	on_unqueued = {
		owner = {
			change_variable = {
				which = imperial_vassal_legion_count
				value = -1
			}
		}
	}

	ai_weight = {
		base = 1000
		modifier = {
			set = 0
			NAND = {
				is_active_resolution = "resolution_emperor_imperial_legions_extensive_recruitment"
				check_variable = {
					which = imperial_vassal_legion_count
					value < modifier:imperial_vassal_legion_capacity_add
				}
				any_owned_planet = {
					has_holding = {
						holding = holding_imperial_training_center
						owner = galactic_emperor
					}
				}
			}
		}
	}
}
