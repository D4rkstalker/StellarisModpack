}
effect_check_core_trait = {
    potential = {
        always = yes
    }
    allow = {
        always = yes
    }
    effect = {
        hidden_effect = {
            if = {
                limit = { 
                    NOT = { has_technology = tech_central_core_modifying_point_1 }
                    check_variable = {
                        which = core_trait_points
                        value = 0
                    }
                }
                custom_tooltip = trait_check_desc_a
            }
            if = {
                limit = { 
                    NOT = { has_technology = tech_central_core_modifying_point_1 }
                    check_variable = {
                        which = core_trait_points
                        value <= 5
                    }
                    check_variable = {
                        which = core_trait_points
                        value > 0
                    }
                }
                custom_tooltip = trait_check_desc_b
            }
            if = {
                limit = {
                    has_technology = tech_central_core_modifying_point_1
                    check_variable = {
                        which = core_trait_points
                        value = 0
                    }
                }
                custom_tooltip = trait_check_desc_c
            }
            if = {
                limit = { 
                    has_technology = tech_central_core_modifying_point_1
                    check_variable = {
                        which = core_trait_points
                        value <= 6
                    }
                    check_variable = {
                        which = core_trait_points
                        value > 0
                    }
                }
                custom_tooltip = trait_check_desc_d
            }
        }
    }
}
effect_add_core_trait_1 = {
    potential = {
        always = yes
    }
    allow = {
        custom_tooltip_fail = {
            text = effect_add_core_trait.fail.a
            hidden_trigger = {
                has_country_resource = {
                    type = energy
                    amount >= 500
                }
                has_country_resource = {
                    type = minerals
                    amount >= 500
                }
            }
        }
        custom_tooltip_fail = {
            text = effect_add_core_trait.fail.b
            hidden_trigger = {
                any_owned_leader = {
                    leader_class = ruler
                    NOT = { has_trait = leader_trait_flexible_programming_machinedlc }
                }
            }
        }
        custom_tooltip_fail = {
            text = effect_add_core_trait.fail.c
            hidden_trigger = {
                check_variable = {
                    which = core_trait_points
                    value > 0
                }
            }
        }
        custom_tooltip_fail = {
            text = effect_add_core_trait.fail.d
            hidden_trigger = {
                check_variable = {
                    which = core_trait_number
                    value > 0
                }
            }
        }
    }
    effect = {
        custom_tooltip = CORE_TOOLTIP_TRAIT_1 #trait desc
		hidden_effect = {
			every_owned_leader = {
				limit = { leader_class = ruler }
				add_ruler_trait = leader_trait_flexible_programming_machinedlc
			}
			subtract_variable = { #minus
				which = core_trait_points
				value = 1
            }
			subtract_variable = { #minus
				which = core_trait_number
				value = 1
            }
		}
    }
}

effect_reset_core_trait = {
    potential = {
        always = yes
    }
    allow = {
        always = yes
    }
    effect = {
        custom_tooltip = effect_reset_core_trait_tooltip
        hidden_effect = {
            every_owned_leader = {
                limit = { leader_class = ruler }
                clear_central_core_trait = yes
            }
            set_variable = {
                which = core_trait_number
                value = 5 #max traits number
            }
            if = {
                limit = { NOT = { has_country_flag = researched_tech_central_core_modifying_point_1 } }
                set_variable = {
                    which = core_trait_points
                    value = 5 #max modification points
                }
            }
            else_if = {
                limit = { has_country_flag = researched_tech_central_core_modifying_point_1 }
                set_variable = {
                    which = core_trait_points
                    value = 6 #max modification points
                }
            }
        }
    }
}
