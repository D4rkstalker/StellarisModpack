		limit = { is_scope_type = planet check_variable = { which = ag_num_dimensional_obelisk_districts_temp value > 0 } }

ag_dimensional_obelisk_max_district = {
	potential = { hidden_trigger = { from = { NOT = { has_country_flag = ag_dimensional_obelisk_max_district } } } }
	allow = { hidden_trigger = { is_scope_type = planet has_active_building = ag_dimensional_obelisk has_planet_flag = ag_dimensional_obelisk_settings_flag } }
	effect = { hidden_effect = {
		while = {
			limit = { 
				check_variable = { which = ag_num_dimensional_obelisk_districts_temp value <= 24 }
				check_variable = { which = ag_num_dimensional_obelisk_districts_temp value < trigger:planet_size }
			}
			change_variable = { which = ag_num_dimensional_obelisk_districts_temp value = 1 }
		}
		from = { set_country_flag = ag_dimensional_obelisk_max_district }
	} }
}

ag_dimensional_obelisk_use_as_default = {
	potential = { hidden_trigger = { always = yes } }
	allow = { hidden_trigger = { is_scope_type = planet has_active_building = ag_dimensional_obelisk has_planet_flag = ag_dimensional_obelisk_settings_flag } }
	effect = { hidden_effect = { from = {
		if = { limit = { NOT = { has_country_flag = ag_dimensional_obelisk_default_value } } set_country_flag = ag_dimensional_obelisk_default_value }
		if = {
			limit = { has_country_flag = ag_dimensional_obelisk_max_district }
			set_country_flag = ag_dimensional_obelisk_default_max
			clear_variable = ag_dimensional_obelisk_default
		}
		else = {
			set_variable = { which = ag_dimensional_obelisk_default value = ag_num_dimensional_obelisk_districts_temp }
			remove_country_flag = ag_dimensional_obelisk_default_max
		}
	} } }
}
