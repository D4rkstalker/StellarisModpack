#subclass_general_marshall
xvcv_mdlc_core_modifying_traits_general_subclass_general_marshall_add_button_effect = {
    potential = { ruler = { NOT = { has_trait = subclass_general_marshall } } }
    allow = {
        custom_tooltip = xvcv_mdlc_core_modifying_tooltip_add_general_subclass_general_marshall
        xvcv_mdlc_core_modifying_requires_not_ruler_subclass_or_focus_trigger = { CLASS = governor ID = subclass_governor_visionary }
        xvcv_mdlc_core_modifying_requires_not_ruler_subclass_or_focus_trigger = { CLASS = scientist ID = subclass_scientist_researcher }
        xvcv_mdlc_core_modifying_requires_not_ruler_subclass_or_focus_trigger = { CLASS = admiral ID = subclass_admiral_strategist }
        xvcv_mdlc_core_modifying_trait_cost_trigger = yes
        xvcv_mdlc_core_modifying_trait_points_trigger = yes
        xvcv_mdlc_core_modifying_trait_max_number_trigger = yes
    }
    effect = {
        xvcv_mdlc_core_modifying_set_subclass_effect = { CLASS = general ID = subclass_general_marshall }
		hidden_effect = { xvcv_mdlc_core_modifying_trait_add_effect = yes }
    }
}
xvcv_mdlc_core_modifying_traits_general_subclass_general_marshall_remove_button_effect = {
    potential = { ruler = { has_trait = subclass_general_marshall } }
    allow = { always = yes }
    effect = {
        custom_tooltip = xvcv_mdlc_core_modifying_tooltip_remove_general_subclass_general_marshall
		hidden_effect = {
            ruler = { remove_trait = subclass_general_marshall }
            xvcv_mdlc_core_modifying_trait_remove_effect = yes
		}
    }
}
