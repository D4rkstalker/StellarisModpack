        xvcv_mdlc_core_modifying_requires_no_ruler_subclass_or_focus_trigger = { CLASS = admiral ID = subclass_admiral_strategist }
        xvcv_mdlc_core_modifying_requires_no_ruler_subclass_or_focus_trigger = { CLASS = general ID = subclass_general_marshall }
        xvcv_mdlc_core_modifying_trait_cost_trigger = yes
        xvcv_mdlc_core_modifying_trait_points_trigger = yes
        xvcv_mdlc_core_modifying_trait_max_number_trigger = yes
        has_paragon_dlc = yes
    }
    effect = {
        xvcv_mdlc_core_modifying_set_subclass_effect = { CLASS = governor ID = subclass_governor_visionary }
		hidden_effect = { xvcv_mdlc_core_modifying_trait_add_effect = yes }
    }
}
