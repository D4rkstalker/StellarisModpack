
ag_current_ancient_knowledge = {
	potential = { hidden_trigger = { always = yes } }
	allow = { hidden_trigger = { always = yes } }
	effect = {
		from = {
			if = {
				limit = {
					OR = {
						NOT = { is_variable_set = ag_ancient_knowledge_var }
						check_variable = { which = ag_ancient_knowledge_var value <= 0 }
					}
				}
				custom_tooltip = "ag_current_no_ancient_knowledge_desc"
			}
			else = { custom_tooltip = "ag_current_ancient_knowledge_desc" }
		}
	}
}
