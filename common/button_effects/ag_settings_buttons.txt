ag_scaled_difficulty_add = {
	potential = { hidden_trigger = { always = yes } }
	allow = { hidden_trigger = { exists = event_target:global_event_country event_target:global_event_country = { check_variable = { which = ag_scaled_difficulty value < 5.0 } } } }
	effect = { hidden_effect = { event_target:global_event_country = {
		set_global_flag = ag_scaled_difficulty_changed
		if = {
			limit = { check_variable = { which = ag_scaled_difficulty value < 2.0 } }
			change_variable = { which = ag_scaled_difficulty value = 0.25 }
		}
		else = { change_variable = { which = ag_scaled_difficulty value = 0.5 } }
		set_variable = { which = ag_scaled_difficulty_tooltip value = ag_scaled_difficulty }
		change_variable = { which = ag_scaled_difficulty_tooltip value = 1 }
		set_variable = { which = ag_scaled_difficulty_2x value = ag_scaled_difficulty }
		multiply_variable = { which = ag_scaled_difficulty_2x value = 2 }
		prev.from = { if = {
			limit = { NOT = { check_variable = { which = ag_debug_mode_counter value > 4 } } }
			set_variable = { which = ag_debug_mode_counter value = 0 }
		} }
	} } }
}
