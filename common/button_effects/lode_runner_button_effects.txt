lode_runner_single = {
	potential = {
		always = yes
	}
	allow = {
		custom_tooltip = {
			fail_text = "lode_runner_single_tooltip"
			OR = {
				has_global_flag = all_midgame_allowed
				has_global_flag = all_midgame_consecutive
			}
		}
	}

	effect = {
		custom_tooltip = "lode_runner_single_tooltip"
		if = {
			limit = {
				has_global_flag = all_midgame_allowed
			}
			remove_global_flag = all_midgame_allowed
		}
		if = {
			limit = {
				has_global_flag = all_midgame_consecutive
			}
			remove_global_flag = all_midgame_consecutive
		}
	}
}

lode_runner_consecutive = {
	potential = {
		always = yes
	}
	allow = {
		custom_tooltip = {
			fail_text = "lode_runner_consecutive_tooltip"
			NOT = {
				has_global_flag = all_midgame_consecutive
			}
		}
	}

	effect = {
		custom_tooltip = "lode_runner_consecutive_tooltip"
		if = {
			limit = {
				has_global_flag = all_midgame_allowed
			}
			remove_global_flag = all_midgame_allowed
		}
		set_global_flag = all_midgame_consecutive
	}
}
