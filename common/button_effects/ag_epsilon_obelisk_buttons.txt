
ag_epsilon_obelisk_ctrl_0_left = {
	potential = { always = yes }
	allow = { hidden_trigger = { from = { OR = {
		AND = {
			has_country_flag = ag_epsilon_obelisk_1_flag
			event_target:global_event_country = { check_variable = { which = ag_epsilon_obelisk_1_0_selected value > 0 } }
		}
		AND = {
			has_country_flag = ag_epsilon_obelisk_2_flag
			event_target:global_event_country = { check_variable = { which = ag_epsilon_obelisk_2_0_selected value > 0 } }
		}
		AND = {
			has_country_flag = ag_epsilon_obelisk_3_flag
			event_target:global_event_country = { check_variable = { which = ag_epsilon_obelisk_3_0_selected value > 0 } }
		}
	} } } }
	effect = { hidden_effect = { from = {
		if = {
			limit = { has_country_flag = ag_epsilon_obelisk_1_flag }
			event_target:global_event_country = { subtract_variable = { which = ag_epsilon_obelisk_1_0_selected value = 1 } }
		}
		else_if = {
			limit = { has_country_flag = ag_epsilon_obelisk_2_flag }
			event_target:global_event_country = { subtract_variable = { which = ag_epsilon_obelisk_2_0_selected value = 1 } }
		}
		else_if = {
			limit = { always = yes }
			event_target:global_event_country = { subtract_variable = { which = ag_epsilon_obelisk_3_0_selected value = 1 } }
		}
	} } }
}
