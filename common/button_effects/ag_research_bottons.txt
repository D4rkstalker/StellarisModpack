
# Visibility
# Variable ag_research_[research_number] = 1

# Status
# Variable ag_research_[research_number]_status
# variable = 0 Unavailable
# variable = 1 Available
# variable = 2 In Progress
# variable = 3 Completed

# Research Difficulty: 
# Variable: ag_research_[research_number]_difficulty
# 1: Very Easy
# 2: Easy
# 3: Normal
# 4: Hard
# 5: Very Hard

# Research Requirements
# Variable: ag_research_[research_number]_requirement
# variable = 1 Science Ship: 1
# variable = 2 Construction Ship: 1

# Research Area
# Variable: ag_research_[research_number]_area
# variable = 1 Physics
# variable = 2 Society
# variable = 3 Engineering
# variable = 4 Physics and Society
# variable = 5 Physics and Engineering
# variable = 6 Society and Engineering
# variable = 7 Physics, Society and Engineering

# Research Progress
# Variable: ag_research_[research_number]_progress
# variable = ag_research_[research_number]_total_progress / ag_research_[research_number]_required_progress

ag_research_1_select_botton = {
	potential = { hidden_trigger = { from = { is_variable_set = ag_research_1 check_variable = { which = ag_research_1 value = 1 } } } }
	allow = {
		hidden_trigger = {
			from = {
				is_variable_set = ag_research_1_status
				is_variable_set = ag_research_1_id
				OR = {
					check_variable = { which = ag_research_1_status value = 1 }
					check_variable = { which = ag_research_1_status value = 2 }
					check_variable = { which = ag_research_1_status value = 3 }
				}
				OR = {
					check_variable = { which = ag_current_research value = ag_research_1_id }
					ag_research_project_trigger = { ag_research_number = 1 }
				}
			}
		}
	}
	effect = {
		from = {
			if = {
				limit = { check_variable = { which = ag_research_1_status value = 3 } }
				custom_tooltip = "ag_research_completed_desc"
			}
			else = {
				if = {
					limit = { check_variable = { which = ag_current_research value = ag_research_1_id } }
					custom_tooltip = "ag_cancel_research_project_effect"
					hidden_effect = {
						remove_country_flag = ag_research_project_ongoing
						set_variable = { which = ag_research_1_status value = 1 }
						ag_save_current_progress = { clear_current_research = yes }
					}
				}
				else = {
					custom_tooltip = "ag_select_research_project_effect"
					hidden_effect ={
						set_country_flag = ag_research_project_ongoing
						if = {
							limit = { check_variable = { which = ag_current_research value = ag_research_2_id } }
							set_variable = { which = ag_research_2_status value = 1 }
						}
						if = {
							limit = { check_variable = { which = ag_current_research value = ag_research_3_id } }
							set_variable = { which = ag_research_3_status value = 1 }
						}
						if = {
							limit = { check_variable = { which = ag_current_research value = ag_research_4_id } }
							set_variable = { which = ag_research_4_status value = 1 }
						}
						ag_save_current_progress = { clear_current_research = yes }
						ag_select_research_project = { ag_research_number = 1 }
						ag_set_research_lvl = { ag_research_frame = 1 }
					}
				}
			}
		}
	}
}

ag_research_to_first_page_botton = {
	potential = { hidden_trigger = { always = yes } }
	allow = {
		custom_tooltip = {
			fail_text = "ag_is_first_page"
			from = { NOT = { check_variable = { which = ag_research_current_page value = 1 } } }
		}
	}
	effect = {
		custom_tooltip = "ag_research_to_first_page_effect"
		custom_tooltip = "ag_research_current_page_desc"
		hidden_effect = {
			from = {
				set_variable = { which = ag_research_current_page value = 1 }
				ag_load_page = yes
				ag_check_locate_ancient_system_project_state = yes
			}
		}
	}
}
