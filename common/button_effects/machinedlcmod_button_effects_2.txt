effect_add_leader_trait_governor_1 = {
    potential = {
        always = yes
    }
    allow = {
        custom_tooltip_fail = {
            text = machine_leader_trait_cost.fail
            hidden_trigger = {
                has_country_resource = {
                    type = energy
                    amount >= 500
                }
                has_country_resource = {
                    type = minerals
                    amount >= 500
                }
            }
        }
        custom_tooltip_fail = {
            text = machine_leader_already_picked_trait.fail
            hidden_trigger = {
                NOT = { has_country_flag = leader_trait_governor_1 }
            }
        }
        custom_tooltip_fail = {
            text = machine_leader_trait_points.fail
            hidden_trigger = {
                check_variable = {
                    which = machine_leader_trait_points_set
                    value > 0
                }
            }
        }
        custom_tooltip_fail = {
            text = machine_leader_already_picked_class.fail
            hidden_trigger = {
                NOR = {

                    has_country_flag = leader_class_set_to_scientist
                    has_country_flag = leader_class_set_to_admiral
                    has_country_flag = leader_class_set_to_general
                }
            }
        }
    }       
    effect = {
        custom_tooltip = tooltip_leader_trait_governor_1
        hidden_effect = {
            set_country_flag = leader_trait_governor_1
            if = {
                limit = { NOT = { has_country_flag = leader_class_set_to_governor } }
                set_country_flag = leader_class_set_to_governor
            }
            subtract_variable = { #minus
				which = machine_leader_trait_points_set
				value = 1
            }
        }        
    }
}
