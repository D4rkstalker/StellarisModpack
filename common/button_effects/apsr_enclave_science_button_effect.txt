apsr_enclave_science_button_01_buy = {
	potential = {
		has_country_flag = apsr_enclave_science_button_01_pressed
	}
	allow = {
		custom_tooltip = {
			fail_text = "apsr_enclave_science_button_01_failtext"
			resource_stockpile_compare = { resource = apsr_knowledge value >= 900 }
		}
	}
	effect = {
		custom_tooltip = apsr_enclave_science_buy_desc
		hidden_effect = {	
			add_resource = { apsr_knowledge = -900 }
			set_country_flag = apsr_enclave_science_button_01_purchase
			give_technology = { tech = tech_apsr_enclave_science_button_01_bonus message = no }	
			remove_country_flag = apsr_enclave_science_button_01_pressed
			event_target:global_event_country = {
				change_variable = {
					which = apsr_ancients_count
					value = 1
				}
			}
			change_variable = {
				which = apsr_achimod_improvements_purchased_count
				value = 1
			}
			change_variable = {
				which = apsr_enclave_science_improvements_purchased_count
				value = 1
			}
			if = {
				limit = {
					check_variable = {
						which = apsr_enclave_science_improvements_purchased_count
						value = 5
					}
				}
				random_list = {
					10 = {
						set_country_flag = apsr_enclave_science_set_1_bonus_is_01
						give_technology = { tech = tech_apsr_enclave_science_bonus_01 message = no }	
					}
					10 = {
						set_country_flag = apsr_enclave_science_set_1_bonus_is_02
						give_technology = { tech = tech_apsr_enclave_science_bonus_02 message = no }
					}
					10 = {
						set_country_flag = apsr_enclave_science_set_1_bonus_is_03
						give_technology = { tech = tech_apsr_enclave_science_bonus_03 message = no }

					10 = {
						set_country_flag = apsr_enclave_science_set_1_bonus_is_04

					}
					10 = {
						set_country_flag = apsr_enclave_science_set_1_bonus_is_05
						give_technology = { tech = tech_apsr_enclave_science_bonus_05 message = no }
					}
				}
			}
			if = {
				limit = {
					check_variable = {
						which = apsr_enclave_science_improvements_purchased_count
						value = 10
					}
				}
				random_list = {
					10 = {
						modifier = {
							factor = 0
							has_country_flag = apsr_enclave_science_set_1_bonus_is_01
						}
						set_country_flag = apsr_enclave_science_set_2_bonus_is_01
						give_technology = { tech = tech_apsr_enclave_science_bonus_01 message = no }	
					}
					10 = {
						modifier = {
							factor = 0
							has_country_flag = apsr_enclave_science_set_1_bonus_is_02
						}
						set_country_flag = apsr_enclave_science_set_2_bonus_is_02
						give_technology = { tech = tech_apsr_enclave_science_bonus_02 message = no }
					}
					10 = {
						modifier = {
							factor = 0
							has_country_flag = apsr_enclave_science_set_1_bonus_is_03
						}
						set_country_flag = apsr_enclave_science_set_2_bonus_is_03
						give_technology = { tech = tech_apsr_enclave_science_bonus_03 message = no }
					}

						modifier = {
							factor = 0
							has_country_flag = apsr_enclave_science_set_1_bonus_is_04
						}
						set_country_flag = apsr_enclave_science_set_2_bonus_is_04
