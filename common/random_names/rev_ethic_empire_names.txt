## Extra Megacorperation
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 5
			OR = {
				has_government = "gov_megacorporation"
				has_government = "gov_dystopian_capitalism"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.extracorp.1"
	noun = "[This.Capital.System.GetName]"
	prefix_format = "[This.Capital.System.GetName] <extra_megacorp>"
	# Empire of Sol
}
