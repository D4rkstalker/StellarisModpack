# For now, at least, it's just a copy of the Democratic Generic name formats

# Democratic Generic 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 1
			has_government = "ver_gov_single_party_republic"
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.dem_gen.1"
	noun = "[This.Capital.System.GetName]"
	prefix_format = "[This.Capital.System.GetName] <democratic_gen>"
	# Empire of Sol
}
