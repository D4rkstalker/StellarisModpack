# #### BASIC GOVERNMENTS

# ### Imperial Generic

# Imperial Generic 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 15
			OR = {
				has_government = "gov_despotic_empire"
				has_government = "gov_feudal_empire"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "{<lrsk_impdic_filler>{AofB{<lrsk_imperial_gen> [This.Capital.System.GetName]}}}"
	noun = "{[This.Capital.System.GetName]}"
	prefix_format = "{AofBpfx{[This.Capital.System.GetName] <lrsk_imperial_gen>}}"
	# Galactic Dynasty of Sol
}
