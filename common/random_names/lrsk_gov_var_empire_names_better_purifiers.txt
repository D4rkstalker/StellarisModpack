# Fanatical Purifiers
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 20
			OR = {
				has_government = "lrsk_gov_var_gov_doom_horde"
				has_government = "lrsk_gov_var_gov_doom_horde_imp"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "{[This.GetSpeciesAdj] {<lrsk_hrd_pur_desc> {<lrsk_hrd_purifiers>}}}"
	adjective = "{[This.GetSpeciesAdj]}"
	prefix_format = "{[This.GetSpeciesAdj] {<lrsk_hrd_purifiers>}}"
}
