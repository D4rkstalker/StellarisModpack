# Capitalism is great. Money is god. Exploit everyone and everything to get richer.
# Opposition Preservationist
# Pacifist: Materialist Xenophobe Authoritarian

industrialist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_pacifist
	
	unique = yes
	support_multiplier = 1
	
	is_potential = {
		years_passed > 25
		has_encountered_other_species = yes
		or = {
			has_ethic = ethic_pacifist
			has_ethic = ethic_fanatic_pacifist
		}
		NOT = { has_modifier = banned_industrialist }
		NOT = { has_country_flag = cf_fac_industrialist_died }
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	resources = {
		category = pop_factions
		produces = {
			influence = 3
		}		
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no 
	}
	can_join_faction = {
		not = { is_pop_category = worker }
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_job = merchant
			has_job = noble
			has_job = executive
			has_ethic = ethic_materialist
			has_ethic = ethic_xenophobe
			has_ethic = ethic_authoritarian
		}
	}
	
	attraction = {
		base = 105
		
		modifier = {
			factor = 0
			exists = from
			from = {
				has_modifier = banned_faction
			}
		}
		
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}
		
		modifier = {
			factor = 0.9
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		
		modifier = {
			factor = 1.5
			has_ethic = ethic_materialist
		}
		
		modifier = {
			factor = 0.9
			NOT = { is_same_species = owner_species }
		}
	}
	
	leader = {
		base = 100
		
		modifier = {
			factor = 0.0
			OR = {
				leader_class = admiral
				leader_class = general
			}			
		}
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}		
	}

	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_pacifist_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_pacifist_outraged_gc
			}
		}
	}
	
	demand = {
		title = "TOTALITARIAN_STRATIFICATION"
		unfulfilled_title = "TOTALITARIAN_NO_STRATIFICATION"
		desc = "TOTALITARIAN_STRATIFICATION_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					NOT = {
						any_owned_pop = {
							NOR = {
								has_living_standard = { type = living_standard_stratified }
								has_living_standard = { type = living_standard_academic_privilege }
							}
						}
					}
					any_owned_species = {
						OR = {
							has_citizenship_type = { type = citizenship_slavery country = prev }
							has_citizenship_type = { type = citizenship_purge country = prev }
							has_citizenship_type = { type = citizenship_robot_servitude country = prev }
						}
					}
				}
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_ELITISM"
		unfulfilled_title = "TOTALITARIAN_NO_ELITISM"
		desc = "TOTALITARIAN_ELITISM_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_technology = tech_selected_lineages
					has_technology = tech_capacity_boosters
				}
			}
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = leader_enhancement_selected_lineages
					has_policy_flag = leader_enhancement_capacity_boosters
				}
			}
		}
	}

	demand = {
		title = "ISOLATIONIST_DIPLO_STANCE"
		unfulfilled_title = "ISOLATIONIST_DIPLO_STANCE"
		desc = "ISOLATIONIST_DIPLO_STANCE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				is_homicidal = no
				is_country_type = default
				is_megacorp = no
			}
		}

		trigger = {
			owner = {
				has_policy_flag = diplo_stance_isolationist
			}
		}
	}

	demand = {
		title = "PROSPERITY_PEACE"
		unfulfilled_title = "PROSPERITY_NO_PEACE"
		desc = "PROSPERITY_PEACE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				is_at_war = no
			}
		}
	}

	demand = {
		title = "AGGRESSIVE_DIPLO_STANCE"
		unfulfilled_title = "AGGRESSIVE_DIPLO_STANCE"
		desc = "AGGRESSIVE_DIPLO_STANCE_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				is_homicidal = no
				is_country_type = default
				OR = {
					has_policy_flag = diplo_stance_belligerent
					has_policy_flag = diplo_stance_supremacist
				}
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_policy_flag = diplo_stance_belligerent
					has_policy_flag = diplo_stance_supremacist
				}
			}
		}
	}

	on_create = {
		pop_faction_event = { id = factions.1 }
	}
	on_destroy = {
		set_timed_country_flag = {
			flag = cf_fac_industrialist_died
			days = 3600
		}
		if = {
			limit = {
				has_modifier = promoted_industrialist
			}
			remove_modifier = promoted_industrialist
		}
		if = {
			limit = {
				has_modifier = suppressed_industrialist
			}
			remove_modifier = suppressed_industrialist
		}
	}

	actions = {
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_industrialist
						has_modifier = promoted_industrialist
					}
				}
			}

			effect = {
				add_modifier = { modifier = promoted_faction days = 1800 }
				parameter:empire = {
					add_modifier = { modifier = promoted_industrialist days = -1 }
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
								not = { has_modifier = promoted_opponents }
							}
							add_modifier = { modifier = promoted_opponents days = 3600 } 
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_industrialist
				}
			}

			effect = {
				remove_modifier = promoted_faction
				parameter:empire = {
					remove_modifier = promoted_industrialist
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
							}
							remove_modifier = promoted_opponents 
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_industrialist
						has_modifier = promoted_industrialist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_industrialist days = -1 }
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
								not = { has_modifier = suppressed_opponents }
							}
							add_modifier = { modifier = suppressed_opponents days = 1800 } 
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_industrialist
				}
				not = { has_modifier = banned_faction }
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_industrialist
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
							}
							remove_modifier = suppressed_opponents 
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		ban_faction = {
			title = "BAN_FACTION"
			description = "BAN_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					has_modifier = suppressed_industrialist
					# not = { has_ethic = ethic_pacifist }
					# not = { has_ethic = ethic_fanatic_pacifist}
				}
				not = { has_modifier = banned_faction }
			}

			effect = {
				hidden_effect = {
					save_event_target_as = MyFaction
					parameter:empire = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
								not = { has_modifier = banned_opposition }
							}
							add_modifier = { modifier = banned_opposition days = 720 } 
						}
						every_owned_pop = {
							limit = {
								member_of_faction = event_target:MyFaction
							}
							random_list = {
								10 = {
									add_modifier = { modifier = pop_faction_banned days = 360 }							
								}
								85 = {
									add_modifier = { modifier = pop_faction_banned days = 720 }
								}
								5 = {
									add_modifier = { modifier = pop_faction_banned days = 1800 }
								}
							}
						}
					}
				}
				add_modifier = { modifier = banned_faction days = 1800 }
				parameter:empire = {
					if = {
						limit = {
							not = { has_modifier = banned_industrialist }
						}
						add_modifier = { modifier = banned_industrialist days = -1 }
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_ban_faction = {
			title = "CANCEL_BAN_FACTION"
			description = "CANCEL_BAN_FACTION_DESC"
			
			potential = {
				exists = owner
				has_modifier = banned_faction
			}

			effect = {
				remove_modifier = banned_faction
				parameter:empire = {
					remove_modifier = banned_industrialist
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "preservationist"
							}
							remove_modifier = banned_opposition
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
}
