nat_separatist_18 = {
	election_header = "GFX_faction_header_blue"
	guiding_ethic = ethic_militarist # Irrelevant for Separatists factions
	use_guiding_ethic_as_pop_filter = no #this means that the faction will let pops that aren't militarist in too

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = -0.5
		}
	}

	is_potential = {
		is_faction_allowed = yes
		exists = event_target:nat_separatist_18
		count_owned_pop = { limit = { has_pop_flag = sep_pop_nat_@event_target:nat_separatist_18 } count >= 5 }
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		has_pop_flag = sep_pop_nat_@event_target:nat_separatist_18
		has_pop_flag = sep_pop_nat
		happiness < 60
		can_think = yes
		pop_was_saved = no
		is_shackled_robot = no
		owner = {
			NOT = { is_same_value = event_target:nat_separatist_18 }
			OR = {
				NOT = { exists = overlord }
				AND = {
					exists = overlord
					overlord = { NOT = { is_same_value = event_target:nat_separatist_18 } }
				}
			}
		}
	}

	attraction = {
		base = 80

		modifier = {
			factor = 0
			happiness >= 0.7
		}

		modifier = {
			factor = 0.25
			has_citizenship_type = { type = citizenship_full }
		}

		modifier = {
			factor = 1.05
			happiness < 0.6
			happiness > 0.5
		}

		modifier = {
			factor = 1.1
			happiness <= 0.5
			happiness > 0.4
		}

		modifier = {
			factor = 1.2
			happiness <= 0.4
			happiness > 0.3
		}

		modifier = {
			factor = 1.3
			happiness <= 0.3
			happiness > 0.2
		}

		modifier = {
			factor = 1.4
			happiness <= 0.2
			happiness > 0.1
		}

		modifier = {
			factor = 1.55
			happiness <= 0.1
		}

		modifier = { #Outcasts and radicals
			factor = 1.5
			OR = {
				NOT = { has_citizenship_type = { type = citizenship_full } }
				has_trait = trait_unruly
			}
		}

		modifier = {
			factor = 2
			owner = { is_at_war_with = event_target:nat_separatist_18 }
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0
			NOT = { has_leader_flag = sep_leader }
		}
	}

	# Reunification
	demand = {
		title = "NATIONALISM_DEMAND_REUNION_x"
		unfulfilled_title = "NATIONALISM_DEMAND_REUNION_x"
		desc = "NATIONALISM_DEMAND_REUNION_18_desc"

		unfulfilled_effect = -30
		fulfilled_effect = 0.001

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = no
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						has_pop_flag = sep_pop_nat_@event_target:nat_separatist_18
					}
				}
			}
		}
	}

	# Reconstitution of dead country
	demand = {
		title = "NATIONALISM_DEMAND_RECONSTITUTION_x"
		unfulfilled_title = "NATIONALISM_DEMAND_RECONSTITUTION_x"
		desc = "NATIONALISM_DEMAND_RECONSTITUTION_18_desc"

		unfulfilled_effect = -30
		fulfilled_effect = 0.001

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = yes
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						has_pop_flag = sep_pop_nat_@event_target:nat_separatist_18
					}
				}
			}
		}
	}

	# Hosting Government-in-Exile
	demand = {
		title = "NATIONALISM_DEMAND_HOSTING_x"
		unfulfilled_title = "NATIONALISM_DEMAND_HOSTING_x"
		desc = "NATIONALISM_DEMAND_HOSTING_18_desc"

		unfulfilled_effect = -0.001
		fulfilled_effect = 15

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = yes
				NOT = { has_country_flag = dead_nation }
			}
		}

		trigger = {
			event_target:nat_separatist_18 = {
				any_owned_fleet = {
					is_ship_size = enclave_mercenary_station
					exists = solar_system.space_owner
					solar_system.space_owner = { is_same_empire = root.owner }
				}
			}
		}
	}

	# Federalism - Government Policy - Ethics and Civics
	demand = {
		title = "NATIONALISM_DEMAND_FEDERAL_x"
		unfulfilled_title = "NATIONALISM_DEMAND_FEDERAL_x"
		desc = "NATIONALISM_DEMAND_FEDERAL_18_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 10

		potential = {
			has_global_flag = cultural_overhaul_active
		}

		trigger = {
			owner = {
				has_policy_flag = federal_state
			}
		}
	}

	# Migration Treaty with Former Homeland
	demand = {
		title = "NATIONALISM_DEMAND_MIGRATION_x"
		unfulfilled_title = "NATIONALISM_DEMAND_MIGRATION_x"
		desc = "NATIONALISM_DEMAND_MIGRATION_18_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 15

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = no
			}
		}

		trigger = {
			owner = {
				has_migration_access = event_target:nat_separatist_18
			}
		}
	}

	# Commercial Pact with Former Homeland
	demand = {
		title = "NATIONALISM_DEMAND_COMMERCIAL_x"
		unfulfilled_title = "NATIONALISM_DEMAND_COMMERCIAL_x"
		desc = "NATIONALISM_DEMAND_COMMERCIAL_18_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 5

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = no
			}
		}

		trigger = {
			owner = {
				has_commercial_pact = event_target:nat_separatist_18
			}
		}
	}

	# Defensive Pact with Former Homeland
	demand = {
		title = "NATIONALISM_DEMAND_DEFENSE_x"
		unfulfilled_title = "NATIONALISM_DEMAND_DEFENSE_x"
		desc = "NATIONALISM_DEMAND_DEFENSE_18_desc"

		unfulfilled_effect = -0.001
		fulfilled_effect = 15

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = no
			}
		}

		trigger = {
			owner = {
				has_defensive_pact = event_target:nat_separatist_18
			}
		}
	}

	# No War with Former Homeland
	demand = {
		title = "NATIONALISM_DEMAND_NO_WAR_x"
		unfulfilled_title = "NATIONALISM_DEMAND_NO_WAR_x"
		desc = "NATIONALISM_DEMAND_NO_WAR_18_desc"

		unfulfilled_effect = -25
		fulfilled_effect = 10

		potential = {
			event_target:nat_separatist_18 = {
				is_exile_country = no
			}
		}

		trigger = {
			owner = {
				NOT = {
					is_at_war_with = event_target:nat_separatist_18
				}
			}
		}
	}

	# Citizenship for former Citizens
	demand = {
		title = "NATIONALISM_DEMAND_CITIZENSHIP_x"
		unfulfilled_title = "NATIONALISM_DEMAND_CITIZENSHIP_x"
		desc = "NATIONALISM_DEMAND_CITIZENSHIP_18_desc"

		unfulfilled_effect = -20
		fulfilled_effect = 5

		potential = {
			owner = {
				any_owned_pop = {
					has_pop_flag = citizen_of_@event_target:nat_separatist_18
				}
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						has_pop_flag = citizen_of_@event_target:nat_separatist_18
						has_citizenship_rights = no
					}
				}
			}
		}
	}

	# No Slavery for former Citizens
	demand = {
		title = "NATIONALISM_DEMAND_NO_SLAVERY_x"
		unfulfilled_title = "NATIONALISM_DEMAND_NO_SLAVERY_x"
		desc = "NATIONALISM_DEMAND_NO_SLAVERY_18_desc"

		unfulfilled_effect = -40
		fulfilled_effect = 0.001

		potential = {
			owner = {
				any_owned_pop = {
					OR = {
						has_pop_flag = citizen_of_@event_target:nat_separatist_18
						has_pop_flag = resident_of_@event_target:nat_separatist_18
					}
					# is_enslaved = yes
				}
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						has_pop_flag = sep_pop_nat_@event_target:nat_separatist_18
						is_enslaved = yes
					}
				}
			}
		}
	}

	# No Purging for former Citizens
	demand = {
		title = "NATIONALISM_DEMAND_NO_PURGING_x"
		unfulfilled_title = "NATIONALISM_DEMAND_NO_PURGING_x"
		desc = "NATIONALISM_DEMAND_NO_PURGING_18_desc"

		unfulfilled_effect = -60
		fulfilled_effect = 0.001

		potential = {
			owner = {
				any_owned_pop = {
					OR = {
						has_pop_flag = citizen_of_@event_target:nat_separatist_18
						has_pop_flag = resident_of_@event_target:nat_separatist_18
					}
					# is_being_purged = yes
				}
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
						OR = {
							has_pop_flag = citizen_of_@event_target:nat_separatist_18
							has_pop_flag = resident_of_@event_target:nat_separatist_18
						}
						is_being_purged = yes
					}
				}
			}
		}
	}

	on_create = {
		event_target:nat_separatist_18 = { save_event_target_as = nat_country_loc }
		nat_faction_first_effect = yes
		set_name = {
			key = nat_separatist_real_name
			variable_string = "[nat_country_loc.GetAdj]"
			variable_string = "[last_created_leader.GetName]"
		}
		nat_faction_second_effect = yes
	}

	on_destroy = { faction_kill_leader_effect = yes }

	actions = {}
}
