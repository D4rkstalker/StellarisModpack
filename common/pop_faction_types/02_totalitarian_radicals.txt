totalitarian_radicals = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_authoritarian

	unique = yes
	support_multiplier = 0

	resources = {
		category = pop_radical_factions
		produces = {
			# totalitarian_militancy = -12
			# totalitarian_support = 12
			# ideological_purity_desired_value_change = -50
		}
	}

	is_potential = {
		#years_passed > 11
		#has_encountered_other_species = yes
		owner = { has_country_flag = first_faction } 
		OR = { 
			has_ethic = ethic_fanatic_egalitarian
			has_ethic = ethic_egalitarian
		}
		has_modifier = politics
		count_owned_pop = {
			count >= 5
			limit = {
				has_ethic = ethic_authoritarian
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	on_create = {
		pop_faction_event = { id = faction_spawn.2 }
		pop_faction_event = { id = faction_spawn.3 }
	}
	on_destroy = {
		# Removing faction exist static modifier
		remove_modifier = totalitarian_radicals_exist
	}

	can_join_faction = {
		is_enslaved = no
		is_shackled_robot = no
		is_being_purged = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		# Should really check for citizenship instead
		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
		modifier = {
			factor = 0.0
			leader_class = ruler
		}
	}

	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_authoritarian_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_authoritarian_outraged_gc
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_AUTOCRACY"
		unfulfilled_title = "TOTALITARIAN_NO_AUTOCRACY"
		desc = "TOTALITARIAN_AUTOCRACY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				NOT = { has_authority = auth_democratic }
			}
		}

		trigger = {
			owner = {
				OR = {
					has_authority = auth_imperial
					has_authority = auth_dictatorial
				}
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_DEMOCRACY"
		unfulfilled_title = "TOTALITARIAN_NO_DEMOCRACY"
		desc = "TOTALITARIAN_DEMOCRACY_DESC"

		unfulfilled_effect = -25
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				NOR = {
					has_authority = auth_imperial
					has_authority = auth_dictatorial
				}
			}
		}

		trigger = {
			owner = {
				NOT = { has_authority = auth_democratic }
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_STRATIFICATION"
		unfulfilled_title = "TOTALITARIAN_NO_STRATIFICATION"
		desc = "TOTALITARIAN_STRATIFICATION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -20

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					NOT = {
						any_owned_pop = {
							NOR = {
								has_living_standard = { type = living_standard_stratified }
								has_living_standard = { type = living_standard_academic_privilege }
							}
						}
					}
					any_owned_species = {
						OR = {
							has_citizenship_type = { type = citizenship_slavery country = prev }
							has_citizenship_type = { type = citizenship_purge country = prev }
							has_citizenship_type = { type = citizenship_robot_servitude country = prev }
						}
					}
				}
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_ELITISM"
		unfulfilled_title = "TOTALITARIAN_NO_ELITISM"
		desc = "TOTALITARIAN_ELITISM_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				OR = {
					has_technology = tech_selected_lineages
					has_technology = tech_capacity_boosters
				}
			}
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = leader_enhancement_selected_lineages
					has_policy_flag = leader_enhancement_capacity_boosters
				}
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_EXTRANATIONAL_AUTHORITY"
		unfulfilled_title = "TOTALITARIAN_NO_EXTRANATIONAL_AUTHORITY"
		desc = "TOTALITARIAN_EXTRANATIONAL_AUTHORITY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
			}
		}

		trigger = {
			owner = {
				is_overlord = yes
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_UNDERLING"
		unfulfilled_title = "TOTALITARIAN_NOT_UNDERLING"
		desc = "TOTALITARIAN_UNDERLING_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				is_subject = yes
			}
		}

		trigger = {
			owner = {
				is_subject = yes
			}
		}
	}
	actions = {
		conduct_operation = {
			title = "CONDUCT_OPERATION"
			description = "CONDUCT_OPERATION_DESC"

			potential = {
				exists = owner
			}

			effect = {
				pop_faction_event = { id = operations_totalitarian.2 }
			}

			ai_weight = {
				base = 0
			}
		}
	}
