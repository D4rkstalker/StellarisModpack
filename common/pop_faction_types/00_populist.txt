##############################
# Pluralist Populist Faction
##############################

populist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_pluralist
	visual_ethic = "populist_visual_ethics"

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = 0.5
		}
	}

	is_potential = {
		is_faction_allowed = yes
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		can_think = yes
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { is_ruler = yes }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	demand = {
		title = "POPULIST_GOVERNMENTAL"
		unfulfilled_title = "FACTIONS_FUCK_ME"
		desc = "POPULIST_GOVERNMENTAL_DESC"

		fulfilled_effect = 20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = { is_bugged_governmental_pluralist = yes }
		}

		trigger = {
			owner = { is_bugged_governmental_pluralist = yes }
		}
	}

	demand = {
		title = "POPULIST_NO_SLAVERY"
		unfulfilled_title = "POPULIST_SLAVERY"
		desc = "POPULIST_NO_SLAVERY_DESC"

		fulfilled_effect = 15
		unfulfilled_effect = -25

		potential = {
			exists = owner
			owner = {
					has_policy_flag = slavery_allowed
				NOR = {
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
				}
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_policy_flag = slavery_allowed
				}
			}
		}
	}

	demand = {
		title = "POPULIST_SHARED_BURDEN"
		unfulfilled_title = "FACTIONS_FUCK_ME"
		desc = "POPULIST_SHARED_BURDEN_DESC"

		fulfilled_effect = 10

		potential = {
			exists = owner
			owner = {
				is_bugged_government_intervention = yes
			}
		}

		trigger = {
			owner = {
				is_bugged_government_intervention = yes
			}
		}
	}

	demand = {
		title = "FACTIONS_FUCK_ME"
		unfulfilled_title = "POPULIST_NO_STRATIFIED"
		desc = "POPULIST_NO_STRATIFIED_DESC"

		
		fulfilled_effect = 0.01
		unfulfilled_effect = -25

		potential = {
			exists = owner
			owner = {
				any_owned_pop = {
				    OR = {
						has_living_standard = { type = living_standard_stratified }
						has_living_standard = { type = living_standard_academic_privilege }
						has_living_standard = { type = living_standard_dystopian_society }
					}
				}
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {
					    OR = {
							has_living_standard = { type = living_standard_stratified }
							has_living_standard = { type = living_standard_academic_privilege }
							has_living_standard = { type = living_standard_dystopian_society }
						}
					}
				}
			}
		}
	}

	demand = {
		title = "POPULIST_WELFARE_STANDARD"
		unfulfilled_title = "POPULIST_WELFARE_STANDARD"
		desc = "POPULIST_WELFARE_STANDARD_DESC"

		fulfilled_effect = 15
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				NOT = {
					any_owned_pop = {
					    OR = {
							has_living_standard = { type = living_standard_stratified }
							has_living_standard = { type = living_standard_academic_privilege }
							has_living_standard = { type = living_standard_dystopian_society }
						}
					}
				}
			}
		}

		trigger = {
			owner = {
				any_owned_pop = {
					OR = {
						has_living_standard = { type = living_standard_shared_burden }
						has_living_standard = { type = living_standard_utopian }
					}
				}
			}
		}
	}

	demand = {
		title = "POPULIST_NO_LEADER_ENHANCEMENT"
		unfulfilled_title = "POPULIST_LEADER_ENHANCEMENT"
		desc = "POPULIST_NO_LEADER_ENHANCEMENT_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_technology = tech_selected_lineages
					has_technology = tech_capacity_boosters
				}
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_policy_flag = leader_enhancement_selected_lineages
					has_policy_flag = leader_enhancement_capacity_boosters
				}
			}
		}
	}

	demand = {
		title = "POPULIST_BENEVOLENT_SUBJUGATION_POLICY"
		unfulfilled_title = "POPULIST_BENEVOLENT_SUBJUGATION_POLICY"
		desc = "POPULIST_BENEVOLENT_SUBJUGATION_POLICY_DESC"

		unfulfilled_effect = -0.001
		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				is_subject = no
				has_policy_flag = benevolent_terms
			}
		}

		trigger = {
			owner = {
				has_policy_flag = benevolent_terms
			}
		}
	}

	demand = {
		title = "POPULIST_OPPRESSIVE_SUBJUGATION_POLICY"
		unfulfilled_title = "POPULIST_OPPRESSIVE_SUBJUGATION_POLICY"
		desc = "POPULIST_OPPRESSIVE_SUBJUGATION_POLICY_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				is_subject = no
				NOT = { has_policy_flag = benevolent_terms }
			}
		}

		trigger = {
			owner = {
				has_policy_flag = oppressive_terms
			}
		}
	}

	demand = {
		title = "FACTIONS_FUCK_ME"
		unfulfilled_title = "POPULIST_NO_HEGEMONY"
		desc = "POPULIST_NO_HEGEMONY_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				has_federation = yes
				federation = {
					has_federation_type = hegemony_federation
				}
				is_same_empire = root.owner.federation.federation_leader
			}
		}

		trigger = {
			NOT = {
				owner = {
					has_federation = yes
					federation = {
						has_federation_type = hegemony_federation
					}
					is_same_empire = root.owner.federation.federation_leader
				}
			}
		}
	}

	demand = {
		title = "POPULIST_GALACTIC_UNION"
		unfulfilled_title = "POPULIST_NO_GALACTIC_UNION"
		desc = "POPULIST_GALACTIC_UNION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 3
			}
		}

		trigger = {
			owner = {
				has_federation = yes
				federation = {
					has_federation_type = default_federation
				}
			}
		}
	}

	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_pluralist
			}
			remove_modifier = promoted_pluralist
		}
		if = {
			limit = {
				has_modifier = suppressed_pluralist
			}
			remove_modifier = suppressed_pluralist
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"

			cost = {
				influence = 500
			}

			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { any_agreement = { agreement_preset = preset_dominion } }
					}
				}
			}

			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_pluralist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_pluralist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = { id = factions.2000 }
						}
					}
				}
			}

			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_pluralist
					}
				}
				modifier = {
					factor = 0
					NOT = {
						owner = {
							exists = leader
							leader = {
								is_ruler = yes
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_pluralist
						has_modifier = promoted_pluralist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_pluralist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_pluralist
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_pluralist
				}
			}

			ai_weight = {
				base = 0
			}
		}
		
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_pluralist
						has_modifier = promoted_pluralist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_pluralist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_pluralist
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_pluralist
				}
			}

			ai_weight = {
				base = 0
			}
		}
	}
}
