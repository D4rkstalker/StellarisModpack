	guiding_ethic = ethic_socialism

		has_encountered_other_species = yes
		NOT = { has_ethic = ethic_gestalt_consciousness }
	}

			valid_objects = {

			}
		}
	}

		category = pop_factions
		produces = {
			influence = 2
		}		
	}		

	can_join_faction = {

		has_ethic = ethic_socialism

		modifier = {
			factor = 1.25
			exists = from
			from = {

				leader = { leader_class = ruler }
			}
		}
		
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		
		
	}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}

	demand = {

		unfulfilled_title = "NO_HEALTHCARE_CAMPAIGN"
		desc = "HEALTHCARE_CAMPAIGN_DESC"
		
		fulfilled_effect = 10
		
		potential = {
			exists = owner

		trigger = {

				has_edict = healthcare_campaign
			}

	demand = {
		title = "EDUCATION_CAMPAIGN"
		unfulfilled_title = "NO_EDUCATION_CAMPAIGN"
		desc = "EDUCATION_CAMPAIGN_DESC"

		potential = {
			exists = owner

		}

		desc = "SOCIALISM_DEVELOPED_DESC"

		fulfilled_effect = 7

		potential = {

		}

		trigger = {

                           has_valid_civic = civic_shared_burden					   
						   
						   } 
				   }

		title = "SOCIAL_SECURITY"
		unfulfilled_title = "NO_SOCIAL_SECURITY"
		desc = "SOCIAL_SECURITY_DESC"

		fulfilled_effect = 10

		potential = { 
			exists = owner
		}

		trigger = {
			owner = { 
				any_owned_pop = {				
				    OR = {
					      has_living_standard = { type = living_standard_shared_burden }
					      has_living_standard = { type = living_standard_good }
						  has_living_standard = { type = living_standard_utopian }
						  has_living_standard = { type = living_standard_chemical_bliss }

				}

		trigger = {
			owner = {
				OR = {

		pop_faction_event = { id = factions.1 }
	}
	
	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_socialism
			}
			remove_modifier = promoted_socialism
		}
		if = {
			limit = {
				has_modifier = suppressed_socialism
			}
			remove_modifier = suppressed_socialism
		}
	}	


	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			
			cost = {
				influence = 500
			}
			
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}
			
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_socialism" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_socialism

						add_modifier = { modifier = embraced_faction_timer days = 3600 }

				hidden_effect = {
					save_event_target_as = TargetFaction

						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default			
								has_communications = prev
							}
							country_event = { id = factions.2000 }
						}
					}				
				}				
			}
			
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50

						has_ethic = ethic_socialism

					factor = 0
					NOT = {

							exists = ruler

					}

				}
			}

				parameter:empire = {
					add_modifier = { modifier = promoted_socialism days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}

			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"

					has_modifier = promoted_socialism

					remove_modifier = promoted_socialism
				}
			}
			
			ai_weight = {
				base = 0
			}
		}		
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					NOR = { 
						has_modifier = suppressed_socialism

					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }

					add_modifier = { modifier = suppressed_socialism days = -1 }
				}
			}
			
			ai_weight = {
