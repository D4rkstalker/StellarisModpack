# Separatist _4 Faction
##############################

plan_separatist_4 = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_militarist # Irrelevant for Separatists factions
	use_guiding_ethic_as_pop_filter = no #this means that the faction will let pops that aren't militarist in too

	unique = no
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = -0.5
		}
	}

	is_potential = {
		years_passed >= 5
		exists = event_target:plan_separatist_4
		any_owned_planet = { is_same_value = event_target:plan_separatist_4 }	
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		planet = {
			is_same_value = event_target:plan_separatist_4
			has_planet_flag = plan_sep_active
		}
		has_pop_flag = sep_pop
		is_shackled_robot = no
		can_think = yes
		pop_was_saved = no
	}

	attraction = {
		base = 150

		modifier = {
			factor = 0
			happiness >= 60
		}

		modifier = {
			factor = 1.2
			happiness <= 0.4
			happiness > 0.3
		}

		modifier = {
			factor = 1.3
			happiness <= 0.3
			happiness > 0.2
		}

		modifier = {
			factor = 1.4
			happiness <= 0.2
			happiness > 0.1
		}

		modifier = {
			factor = 1.5
			happiness <= 0.1
		}

		modifier = { #Outcasts and radicals
			factor = 1.5
			OR = {
				NOT = { has_citizenship_type = { type = citizenship_full } }
				has_trait = trait_unruly
			}
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			OR = {
				is_event_leader = yes
				has_leader_flag = appointed_heir
				AND = {
					leader_class = governor
					exists = sector
					sector = { any_owned_planet = { NOT = { is_same_value = event_target:plan_separatist_4 } } }
				}
				AND = {
					leader_class = ruler
					NOT = { has_leader_flag = sep_won_election }
				}
			}
		}
	}

	# Independence
	demand = {
		title = "SEPARATISM_DEMAND_IND_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_IND_p_x"
		desc = "SEPARATISM_DEMAND_IND_p_4_desc"

		unfulfilled_effect = -30
		fulfilled_effect = 0.001

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = { is_owned_by = root.owner }
		}

		trigger = {
			owner = {
				NOR = {
					any_owned_planet = {
						is_same_value = event_target:plan_separatist_4
					}
				}
			}
		}
	}

	# No Military Personnel Jobs
	demand = {
		title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		desc = "SEPARATISM_DEMAND_MIL_JOBS_p_4_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 5

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = {
				NOR = {					
					has_available_jobs = soldier
					has_available_jobs = warrior_drone
					num_assigned_jobs = {
						job = soldier
						value >= 1
					}
					num_assigned_jobs = {
						job = warrior_drone
						value >= 1
					}
				}
			}
		}

		trigger = {
			event_target:plan_separatist_4 = {
				NOR = {					
					has_available_jobs = soldier
					has_available_jobs = warrior_drone
					num_assigned_jobs = {
						job = soldier
						value >= 1
					}
					num_assigned_jobs = {
						job = warrior_drone
						value >= 1
					}
				}
			}
		}
	}

	# # Military Personnel Exist - 1
	demand = {
		title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		desc = "SEPARATISM_DEMAND_MIL_JOBS_p_4_desc"

		unfulfilled_effect = -2.5
		fulfilled_effect = 0.001

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = {
				owner = {
					is_same_empire = root.owner
				}
				count_owned_pop = {
					limit = {
						OR = {
							has_job = soldier
							has_job = warrior_drone
						}
					}
					count = 1
				}
			}
		}

		trigger = {
			event_target:plan_separatist_4 = {
				NOT = {
					count_owned_pop = {
						limit = {
							OR = {
								has_job = soldier
								has_job = warrior_drone
							}
						}
						count = 1
					}
				}
			}
		}
	}

	# # Military Personnel Exist - 2
	demand = {
		title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		desc = "SEPARATISM_DEMAND_MIL_JOBS_p_4_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 0.001

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = {
				owner = {
					is_same_empire = root.owner
				}
				count_owned_pop = {
					limit = {
						OR = {
							has_job = soldier
							has_job = warrior_drone
						}
					}
					count = 2
				}
			}
		}

		trigger = {
			event_target:plan_separatist_4 = {
				NOT = {
					count_owned_pop = {
						limit = {
							OR = {
								has_job = soldier
								has_job = warrior_drone
							}
						}
						count = 2
					}
				}
			}
		}
	}

	# # Military Personnel Exist - 3
	demand = {
		title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_MIL_JOBS_p_x"
		desc = "SEPARATISM_DEMAND_MIL_JOBS_p_4_desc"

		unfulfilled_effect = -7.5
		fulfilled_effect = 0.001

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = {
				owner = {
					is_same_empire = root.owner
				}
				count_owned_pop = {
					limit = {
						OR = {
							has_job = soldier
							has_job = warrior_drone
						}
					}
					count >= 3
				}
			}
		}

		trigger = {
			event_target:plan_separatist_4 = {
				NOT = {
					count_owned_pop = {
						limit = {
							OR = {
								has_job = soldier
								has_job = warrior_drone
							}
						}
						count >= 3
					}
				}
			}
		}
	}

	# # No Military Buildings
	demand = {
		title = "SEPARATISM_DEMAND_MIL_BUILD_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_MIL_BUILD_p_x"
		desc = "SEPARATISM_DEMAND_MIL_BUILD_p_4_desc"

		unfulfilled_effect = -20
		fulfilled_effect = 5

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = { is_owned_by = root.owner }
		}

		trigger = {
			event_target:plan_separatist_4 = {
				NOR = {
					has_building = building_military_academy
					has_building = building_dread_encampment
					has_building = building_stronghold
					has_building = building_fortress
				}
			}
		}
	}

	# Desire Planetary Shield Generator
	# Reasoning: In case they rebel, protection from enemies
	demand = {
		title = "SEPARATISM_DEMAND_SHIELD_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_SHIELD_p_x"
		desc = "SEPARATISM_DEMAND_SHIELD_p_4_desc"

		unfulfilled_effect = -5
		fulfilled_effect = 10

		potential = {
			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = {
				owner = {
					is_same_empire = root.owner
					has_technology = tech_planetary_shield_generator
				}
			}
		}

		trigger = {
			event_target:plan_separatist_4 = {
				has_building = building_planetary_shield_generator
			}
		}
	}

	# Federal Government Policy - Ethics and Civics
	demand = {
		title = "SEPARATISM_DEMAND_FEDERAL_p_x"
		unfulfilled_title = "SEPARATISM_DEMAND_FEDERAL_p_x"
		desc = "SEPARATISM_DEMAND_FEDERAL_p_4_desc"

		unfulfilled_effect = -10
		fulfilled_effect = 5

		potential = {
			has_global_flag = cultural_overhaul_active

			exists = event_target:plan_separatist_4
			event_target:plan_separatist_4 = { is_owned_by = root.owner }
		}

		trigger = {
			owner = {
				has_policy_flag = federal_state
			}
		}
	}

	on_create = {
		event_target:plan_separatist_4 = { save_event_target_as = plan_sep_loc }
		plan_faction_first_effect = yes
		set_name = {
			key = plan_separatist_real_name
			variable_string = "[plan_sep_loc.GetAdj]"
			variable_string = "[last_created_leader.GetName]"
		}
		plan_faction_second_effect = yes
	}

	on_destroy = {
		# Allows planets to resume getting separatists, which in turn will allow this to re-form if necessary
		# faction_kill_leader_effect = yes
		if = {
			limit = { exists = event_target:plan_separatist_4 }
			event_target:plan_separatist_4 = { remove_planet_flag = plan_sep_active }
			clear_global_event_target = plan_separatist_4
		}
	}

	actions = {}
}
