			has_ethic = ethic_egalitarian

		NOT = { has_modifier = banned_elitist }	
		NOT = { has_country_flag = cf_fac_elitist_died }

		is_being_purged = no

		not = { is_pop_category = worker }
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			and = {
				has_ethic = ethic_egalitarian
				is_pop_category = ruler

			has_ethic = ethic_materialist
			has_ethic = ethic_militarist
			has_ethic = ethic_xenophobe
		}
	}
	
	attraction = {
		base = 110

		modifier = {
			factor = 0
			exists = from
			from = {
				has_modifier = banned_faction
			}
		}
		
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}
		
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		
		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}

		base = 100
		
		modifier = {
			factor = 0
			leader_class = ruler
			owner = { has_authority = auth_imperial }
		}
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}

	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_egalitarian_outraged_gc
			}
		}

		trigger = {

				has_country_flag = faction_egalitarian_outraged_gc

		trigger = {

				NOT = { has_policy_flag = diplo_stance_cooperative }

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				OR = {
					has_technology = tech_selected_lineages
					has_technology = tech_capacity_boosters
				}
			}
		}

		trigger = {
			owner = {
				OR = {
					has_policy_flag = leader_enhancement_selected_lineages
					has_policy_flag = leader_enhancement_capacity_boosters
				}
			}
		}
	}
	
	demand = {
		title = "TECHNOLOGIST_RESEARCH_TREATIES"
		unfulfilled_title = "TECHNOLOGIST_FEW_RESEARCH_TREATIES"
		desc = "TECHNOLOGIST_RESEARCH_TREATIES_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
			}
		}

		trigger = {
			owner = {
				OR = {
					AND = {
						has_federation = yes
						federation = {
							has_federation_perk = research_federation_passive

					}

			}
		}

				has_modifier = promoted_elitist

		if = {
			limit = {
				has_modifier = suppressed_elitist
			}
			remove_modifier = suppressed_elitist
		}
	}

			description = "PROMOTE_FACTION_DESC"

				exists = owner

						has_modifier = promoted_elitist

				}
			}

			effect = {
				add_modifier = { modifier = promoted_faction days = 1800 }
				parameter:empire = {
					add_modifier = { modifier = promoted_elitist days = -1 }

						every_pop_faction = {
							limit = {

								not = { has_modifier = promoted_opponents }

				base = 0
			}

			description = "CANCEL_PROMOTE_FACTION_DESC"

					has_modifier = promoted_elitist

			}

				remove_modifier = promoted_faction

					remove_modifier = promoted_elitist
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "labourite"
							}

					}

			}

				base = 0
			}
		}		
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					NOR = { 
						has_modifier = suppressed_elitist
						has_modifier = promoted_elitist 
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_elitist days = -1 }
					hidden_effect = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "labourite"
								not = { has_modifier = suppressed_opponents }
							}
							add_modifier = { modifier = suppressed_opponents days = 1800 } 
						}
					}
				}

			description = "CANCEL_SUPPRESS_FACTION_DESC"

				exists = owner
				parameter:empire = { 
					has_modifier = suppressed_elitist
				}
				not = { has_modifier = banned_faction }																																					  
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_elitist

						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "labourite"
							}
							remove_modifier = suppressed_opponents 
						}
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}

			title = "BAN_FACTION"
			description = "BAN_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					has_modifier = suppressed_elitist
					# not = { has_ethic = ethic_egalitarian }
					# not = { has_ethic = ethic_fanatic_egalitarian }
				}
				not = { has_modifier = banned_faction }
			}

			effect = {
				hidden_effect = {
					save_event_target_as = MyFaction
					parameter:empire = {
						every_pop_faction = {
							limit = { 
								is_pop_faction_type = "labourite"
								not = { has_modifier = banned_opposition }
							}
							add_modifier = { modifier = banned_opposition days = 720 } 
						}
						every_owned_pop = {
							limit = {
								member_of_faction = event_target:MyFaction
							}
							random_list = {

								}
								85 = {
									add_modifier = { modifier = pop_faction_banned days = 720 }
								}

									add_modifier = { modifier = pop_faction_banned days = 1800 }
								}
							}
						}
					}
				}
				add_modifier = { modifier = banned_faction days = 1800 }
				
				parameter:empire = {
					if = {
						limit = {
							not = { has_modifier = banned_elitist }
						}
						add_modifier = { modifier = banned_elitist days = -1 }
					}
				}
			}

				base = 0
			}

			title = "CANCEL_BAN_FACTION"

							limit = {

							}
