# Protect what we have. Live off the land. Back to basics. Survival of the fittest.

survivalist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_militarist
	
	unique = yes
	support_multiplier = 1
	
	resources = {
		category = pop_factions
		produces = {
			influence = 1
		}
	}

	is_potential = {
		years_passed > 25
		has_encountered_other_species = yes
		or = {
			has_ethic = ethic_militarist
			has_ethic = ethic_fanatic_militarist
			has_ethic = ethic_egalitarian
			has_ethic = ethic_fanatic_egalitarian
		}
		nor = {
			has_ethic = ethic_pacifist
			has_ethic = ethic_fanatic_pacifist
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
		NOT = { has_modifier = banned_survivalist }
		NOT = { has_country_flag = cf_fac_survivalist_died }
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	
	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no 
	}
	
	can_join_faction = {
		not = { is_pop_category = ruler }
		is_shackled_robot = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			and = {
				has_ethic = ethic_militarist
				is_pop_category = worker			
			}
			has_ethic = ethic_egalitarian
			has_ethic = ethic_xenophobe
		}
		nor = {
			has_job = soldier
			has_job = enforcer
		}
	}
	
	attraction = {
		base = 105
		
		modifier = {
			factor = 0
			exists = from
			from = {
				has_modifier = banned_faction
			}
		}
		
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}
		
		modifier = {
			factor = 1.1
			exists = from
			or = {
				has_job = farmer
				has_job = miner
			}
		}
		
		modifier = {
			factor = 0.9
			exists = from
			has_ethic = ethic_militarist
		}
		
		#modifier = {
		#	factor = 2.0
		#	exists = from
		#	has_job = ranger
		#}
		
		modifier = {
			factor = 1.1
			has_citizenship_rights = yes
		}
		
		modifier = {
			factor = 1.1
			NOT = { is_same_species = owner_species }
		}
	}
	
	leader = {
		base = 100
		
		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"

		fulfilled_effect = -20
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_militarist_outraged_gc
			}
		}

		trigger = {
			owner = {
				has_country_flag = faction_militarist_outraged_gc
			}
		}
	}
	
	demand = {
		title = "SUPREMACIST_ISOLATIONIST_NO_FEDERATION"
		unfulfilled_title = "SUPREMACIST_ISOLATIONIST_FEDERATION"
		desc = "SUPREMACIST_ISOLATIONIST_NO_FEDERATION_DESC"
		
		unfulfilled_effect = -15
		fulfilled_effect = 0.001
		
		potential = {
			exists = owner
			owner = {
				num_communications > 0
			}
		}
		
		trigger = {
			owner = {
				has_federation = no
			}
		}
	}
	
	demand = {
		title = "PROGRESSIVE_REPRODUCTIVE_FREEDOM"
		unfulfilled_title = "PROGRESSIVE_NO_REPRODUCTIVE_FREEDOM"
		desc = "PROGRESSIVE_REPRODUCTIVE_FREEDOM_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -20

		potential = {
			exists = owner
		}

		trigger = {
			NOT = {
				owner = { has_policy_flag = population_controls_allowed }
			}
		}
	}
	
	demand = {
		title = "PROGRESSIVE_NO_STRATIFIED"
		unfulfilled_title = "PROGRESSIVE_STRATIFIED"
		desc = "PROGRESSIVE_NO_STRATIFIED_DESC"

		unfulfilled_effect = -15
		fulfilled_effect = 0.001

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				NOT = {
					any_owned_pop = {						has_living_standard = { type = living_standard_stratified }		  
					}
				}
			}
		}
	}
	
	demand = { # "TRADITIONALIST_NO_ROBOTS"
		title = "TRADITIONALIST_NO_ROBOTS"
		unfulfilled_title = "TRADITIONALIST_ROBOTS"
		desc = "TRADITIONALIST_NO_ROBOTS_DESC"
		
		unfulfilled_effect = -10
		fulfilled_effect = 5
		
		potential = {
			exists = owner
		}
		
		trigger = {
			owner = {
				has_policy_flag = robots_outlawed
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		set_timed_country_flag = {
			flag = cf_fac_survivalist_died
			days = 3600
		}
		if = {
			limit = {
				has_modifier = promoted_survivalist
			}
			remove_modifier = promoted_survivalist
		}
		if = {
			limit = {
				has_modifier = suppressed_survivalist
			}
			remove_modifier = suppressed_survivalist
		}
	}
	
	actions = {
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_survivalist
						has_modifier = promoted_survivalist
					}
				}
			}

			effect = {
				add_modifier = { modifier = promoted_faction days = 1800 }
				parameter:empire = {
					add_modifier = { modifier = promoted_survivalist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { has_modifier = promoted_survivalist }
			}

			effect = {
				remove_modifier = promoted_faction
				parameter:empire = {
					remove_modifier = promoted_survivalist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_survivalist
						has_modifier = promoted_survivalist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_survivalist days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					has_modifier = suppressed_survivalist
				}
				not = { has_modifier = banned_faction }
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_survivalist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		ban_faction = {
			title = "BAN_FACTION"
			description = "BAN_FACTION_DESC"
			
			potential = {
				exists = owner
				parameter:empire = { 
					has_modifier = suppressed_survivalist 
					# not = { has_ethic = ethic_militarist }
					# not = { has_ethic = ethic_fanatic_militarist }
				}
				not = { has_modifier = banned_faction }
			}

			effect = {
				hidden_effect = {
					save_event_target_as = MyFaction
					parameter:empire = {
						every_owned_pop = {
							limit = {
								member_of_faction = event_target:MyFaction
							}
							random_list = {
								10 = {
									add_modifier = { modifier = pop_faction_banned days = 360 }							
								}
								85 = {
									add_modifier = { modifier = pop_faction_banned days = 720 }
								}
								5 = {
									add_modifier = { modifier = pop_faction_banned days = 1800 }
								}
							}
						}
					}
				}
				add_modifier = { modifier = banned_faction days = 1800 }
				parameter:empire = {
					if = {
						limit = {
							not = { has_modifier = banned_survivalist }
						}
						add_modifier = { modifier = banned_survivalist days = -1 }
					}
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_ban_faction = {
			title = "CANCEL_BAN_FACTION"
			description = "CANCEL_BAN_FACTION_DESC"
			
			potential = {
				exists = owner
				has_modifier = banned_faction
			}

			effect = {
				remove_modifier = banned_faction
				parameter:empire = {
					remove_modifier = banned_survivalist
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
}
