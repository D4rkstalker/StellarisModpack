#High Priest
high_priest = {
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_citadel_of_faith
	clothes_texture_index = 1

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_ruler_job
	possible = {
		NOT = { has_ethic = ethic_materialist }
	}

	resources = {
		category = planet_priests
		produces = {
			unity = 6
		}
		upkeep = {
			consumer_goods = 2
		}
		#Bug Branch
			produces = {
				trigger = {
					owner = {  is_bugged_governmental_spiritualist = yes  }
				}
				unity = 1
			}
		#Bug Branch
	}

	#Bug Branch
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = { 
					is_bugged_machine_cult = yes
				}
			}
			modifier = {
				planet_jobs_engineering_research_produces_mult = 0.05
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = { 
					is_bugged_life_cult = yes
				}
			}
			modifier = {
				planet_jobs_society_research_produces_mult = 0.05
			}
		}

		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = { 
					is_bugged_gravity_cult = yes
				}
			}
			modifier = {
				planet_jobs_physics_research_produces_mult = 0.05
			}
		}
		
		triggered_planet_modifier = {
			potential = {
				always = yes
			}
			bugged_priest_amenities_mult = 0.1
		}
	#Bug Branch

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 5
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	weight = {
		weight = @ruler_job_weight
		mult = value:job_weights_modifier|JOB|high_priest|RESOURCE|unity|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.1
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 5
			has_job = high_priest
		}
		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
		}
	}
}

		}
		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
			OR = { # contingency won't purge itself
				NOT = { exists = event_target:custodian_bot }
				AND = {
					exists = event_target:custodian_bot

			}
