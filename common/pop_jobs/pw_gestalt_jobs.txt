@Job_Priority_Patched = 1
pw_mining_coordinator = {
	category = complex_drone
	condition_string = DRONE_JOB_TRIGGER
	building_icon = building_mineral_purification_plant
	icon = miner

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
	}

	possible = {
		drone_job_check_trigger = yes
	}

	resources = {
		category = planet_miners
		produces = {
			minerals = @pw_job_director_resource_base
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_edict = mining_subsidies
				}
			}
			energy = 1
		}
	}

	planet_modifier = {
		planet_jobs_minerals_produces_mult = @pw_job_director_resource_mult
	}

	weight = {
		modifier = { 
			factor = 50 
			has_trait = trait_priority_minerals
			}
		modifier = { 
			factor = 0.1 
			has_trait = trait_negative_priority_minerals
			}

		weight = @synapse_drone_job_weight
		modifier = {
			factor = 0.9
			exists = owner
			years_passed < 1
			owner = { has_valid_civic = civic_machine_assimilator }
			has_trait = trait_cybernetic
		}
	}
}
