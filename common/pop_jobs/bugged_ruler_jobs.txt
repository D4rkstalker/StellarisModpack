#Mogul
bugged_mogul = {
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_galactic_stock_exchange
	clothes_texture_index = 1

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_ruler_job
	possible = {
		owner = { NOT = { has_valid_civic = civic_bugged_distributism } }
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 3
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		trade_value_add = 6
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_trade_mult|
	}

	#planet_jobs_energy_produces_mult

	#civic_merchant_guilds
	
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		trade_value_add = 1
		mult = value:bugged_stock_exchange_tv
	}
	
	triggered_planet_modifier = {
		potential = {
			OR = {
				owner = { has_valid_civic = civic_bugged_state_monopoly }
			}
		}
		modifier = {
		    planet_technician_produces_mult = 0.05
			planet_miners_produces_mult = 0.05
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = economic_stance_bugged_planned_economy
			}
		}
		modifier = {
			planet_jobs_worker_produces_mult = 0.05
		}
	}

	weight = {
		weight = @ruler_job_weight
		mult = value:job_weights_modifier|JOB|executive|RESOURCE|unity|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 5
			has_job = bugged_mogul
		}		
	}
}
