miner = {
	category = worker
	condition_string = WORKER_JOB_TRIGGER
	building_icon = building_mineral_purification_plant
	clothes_texture_index = 2

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_worker_job
	possible = {
		hidden_trigger = { exists = owner }
		owner = {
			is_gestalt = no
		}
	}

	#Lithoid Supremacy Stuff
	resources = {
		category = planet_miners
		produces = {
			minerals = 4
		}
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_material_requisitorium
				}
			}
			minerals = @requisitorium_minerals_reduction
		}
		produces = {
			trigger = {
				exists = owner
				owner = { 
					OR = {
						has_valid_civic = civic_mining_cities
						has_valid_civic = civic_mining_cities_corp
					}
				}
			}
			engineering_research = 1
			#Engineering + 1
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { 
					OR = {
						has_valid_civic = civic_mining_cities
						has_valid_civic = civic_mining_cities_corp
					}
				}
			}
			consumer_goods = 0.4
			#CG = -0.4
			#This will cause an initial net zero of CG (1.7+)
		}
	}

	#add amenities
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				OR = {
					has_valid_civic = civic_mining_cities
					has_valid_civic = civic_mining_cities_corp
				}
			}
		}
		modifier = {
			planet_amenities_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				OR = {
					has_valid_civic = civic_mining_cities
					has_valid_civic = civic_mining_cities_corp
				}
			}
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			planet_amenities_add = 0.4
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				OR = {
					has_valid_civic = civic_mining_cities
					has_valid_civic = civic_mining_cities_corp
				}
			}
			has_trait = trait_repugnant
		}
		modifier = {
			planet_amenities_add = -0.4
		}
	}
	###

	overlord_resources = {
		category = planet_requisitioned_minerals
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_material_requisitorium
				}
			}
			minerals = @requisitorium_minerals_overlord
		}
	}

	weight = {
		weight = @worker_job_weight
		mult = value:job_weights_modifier|JOB|miner|RESOURCE|minerals|
		modifier = {
			factor = 10
			is_enslaved = yes
			can_take_servant_job = no
			NOT = { has_slavery_type = { type = slavery_indentured } }
		}
		modifier = {
			factor = 3
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { has_technology = tech_droid_workers }
		}
		modifier = {
			factor = 200
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { NOT = { has_technology = tech_droid_workers } }
		}
		modifier = {
			factor = 0.25
			can_take_servant_job = yes
		}

		# low income modifier
		modifier = {
			factor = 1.1
			exists = planet
			planet = {
				has_available_jobs = "miner"
			}
			exists = owner
			owner = {
				is_ai = yes
				has_monthly_income = {
					resource = minerals
					value < 0
				}
			}
		}
	}
}
