head_researcher = {
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_institute
	clothes_texture_index = 1
	icon = researcher

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
	}

	possible = {
		not = { is_pop_category = worker }
		ruler_job_check_trigger = yes
	}
	
	resources = {
		category = planet_researchers
		produces = {
			physics_research = 5
			engineering_research = 5
			society_research = 5
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_shadow_council }
			}
			consumer_goods = 1
		}		
	}
	
	planet_modifier = {
		planet_amenities_add = 5
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_shadow_council }
		}
		modifier = {
			pop_government_ethic_attraction = 0.1
			pop_ethics_shift_speed_mult = 0.1
		}
	}

	triggered_planet_modifier = {
		potential = {
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			planet_amenities_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_repugnant
		}
		modifier = {
			planet_amenities_add = -1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_genetic_inheritance }
		}
		modifier = {
			pop_growth_speed = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_science_is_sorcery }
		}
		modifier = {
			planet_crime_add = 20
		}
	}
	
	weight = {
		weight = @ruler_job_weight
		modifier = {
			factor = 0.1
			is_non_citizen_non_mech = yes	
		}
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_academic_privilege }
		}		
		modifier = {
			factor = 5
			has_job = head_researcher
		}
		modifier = {
			factor = 0.01
			is_being_purged_by_crisis = yes
		}
		modifier = {
			factor = 1.2
			good_amenities_producer = yes
		}
		modifier = {
			factor = 4
			good_science_producer = yes
		}
		modifier = {
			factor = 0.1
			poor_science_producer = yes
		}
		modifier = {
			factor = 0.5
			poor_amenities_producer = yes
		}
		modifier = {
			factor = 0.1
			exists = owner
			owner = { has_valid_civic = civic_science_is_sorcery }
		}
	}
}
