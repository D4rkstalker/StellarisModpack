	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_mineral_purification_plant
	clothes_texture_index = 2
	icon = miner

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
	}

	possible = {
		ruler_job_check_trigger = yes
	}

	resources = {
		category = planet_miners
		produces = {
			minerals = @pw_job_director_resource_base
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_edict = mining_subsidies
				}
			}
			energy = 1
		}
	}

	planet_modifier = {
		planet_jobs_minerals_produces_mult = @pw_job_director_resource_mult
	}

	weight = {
		modifier = { 
			factor = 50 
			has_trait = trait_priority_minerals
			}
		modifier = { 
			factor = 0.1 
			has_trait = trait_negative_priority_minerals
			}

		weight = @ruler_job_weight
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 3
			has_trait = trait_erudite
		}
		modifier = {
			factor = 2
			OR = {
				has_trait = trait_industrious
			}
		}
		modifier = {
			factor = 5
			has_job = pw_mining_director
		}
		modifier = {
			factor = 1.25
			has_trait = trait_presapient_proles
		}
		modifier = {
			factor = 1.5
			has_trait = trait_robust
		}
		modifier = {
			factor = 1.5
			has_trait = trait_necrophage
		}
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
		}
	}
}
