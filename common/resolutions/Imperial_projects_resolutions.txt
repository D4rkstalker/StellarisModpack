resolution_Empyreal_Travel = {
	icon = "GFX_resolution_imperial_institutions"
	resources = {
		category = resolutions
		cost = {
			influence = 500
		}
	}
	target = no
	level = 1
	effect = {
		hidden_effect = {
			set_global_flag = galactic_community_resolution_passed
			country_event = {
				id = Imperial_projects.0001
			}
		}
	}
	fail_effects = {
		hidden_effect = {
			add_modifier = {
				modifier = resolution_failed_diplomatic_weight
				days = 2200
			}
		}
	}
	potential = {
		has_nemesis = yes
		OR = {
			is_galactic_custodian = yes
			is_galactic_emperor = yes
		}
		OR = {
			has_global_flag = grand_custodianship_declared
			has_galactic_emperor = yes
		}
	}
	allow = {
		OR = {
			has_global_flag = grand_custodianship_declared
			imperial_authority >= 100
		}
		OR = {
			is_galactic_emperor = yes
			is_galactic_custodian = yes
		}
		custom_tooltip = {
			fail_text = "resolution_already_active"
			NOR = {
				is_active_resolution = "resolution_Empyreal_Travel"
			}
		}
	}
	ai_weight = {
		weight = 1
		modifier = {
			factor = 2
			OR = {
				is_megacorp = yes
				has_valid_civic = civic_corporate_dominion
			}
		}
		modifier = {
			factor = 1.2
			OR = {
				has_valid_civic = civic_free_traders
				has_valid_civic = civic_trading_posts
			}
		}
		modifier = {
			factor = 0.1
			is_gestalt = yes
		}
	}
}
