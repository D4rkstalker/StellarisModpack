resolution_mutualdefense_repeal_1 = {
	icon = "GFX_repeal_red"

	resources = {
		category = resolutions

			trigger = { NOT = { is_active_resolution = "resolution_emperor_imperial_ministry_of_defence" } }

				days = 2200
			}
		}
	}

	potential = {
		is_active_resolution = "resolution_mutualdefense_the_readied_shield"
	}
	
	allow = {
		custom_tooltip = {
			fail_text = "resolution_already_in_lower_house_session"
			NOT = { has_global_flag = lh_recent_resolution_mutualdefense_repeal_1 }
