resolution_mischief_repeal_1 = {
	icon = "GFX_repeal_blue"

	resources = {
		category = resolutions
		cost = {
			influence = @resolution_cost_t1
		}
	}
	target = no

	fire_and_forget = yes

	effect = {
		custom_tooltip = resolution_mischief_repeal_1_effect_success
		hidden_effect = {
			cancel_resolution = "resolution_mischief_1"
			set_timed_country_flag = { flag = galactic_community_resolution_passed_mischief_repeal days = @resolution_flag_timer }
		}
	}

	fail_effects = {
		hidden_effect = {
			set_timed_country_flag = { flag = galactic_community_resolution_failed_mischief_repeal days = @resolution_flag_timer }
		}
	}

	potential = {
		is_active_resolution = "resolution_mischief_1"
	}

	allow = {}

	ai_weight = {
		base = @resolution_weight_unpopular

		modifier = {
			factor = 0
			AND = {
				has_valid_civic = civic_criminal_heritage
				is_megacorp = yes
			}
			desc = cr_her_eu
		}

		modifier = {
			factor = 1.25
			OR = {
				has_valid_civic = civic_free_traders
				has_valid_civic = civic_trading_posts
			}
			desc = gal_com_opposing_civics
		}
	}
}
