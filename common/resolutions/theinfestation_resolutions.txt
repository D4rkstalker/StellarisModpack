resolution_exomycosis_infestation = {
	icon = "GFX_resolution_exomycosis_infestation"

	resources = {
		category = resolutions
		cost = {
			influence = @resolution_cost_t1
		}
	}
	target = no

	effect = {
		custom_tooltip = resolution_exomycosis_infestation_effect_success

		hidden_effect = {
			every_playable_country = {
				limit = {
					is_galactic_community_member = yes
					NOR = {
						has_origin = origin_exomycosis
						has_country_flag = exomycosis_research_complete
					}
				}
				country_event = { id = theinfestation_exomycosis_event.44 }
			}
		}
	}

	fail_effects = {
	}

	potential = {
		has_country_flag = exomycosis_research_complete
	}

	active = {
	}

	ai_weight = {
		base = @resolution_weight_normal
		
		modifier = {
			factor = 10

			any_owned_planet = {
				OR = {
					is_planet_class = pc_mycelium
					has_modifier = mold_contaminated
					has_modifier = exomycosis_infestation
				}
			}
			desc = gal_com_has_exomycosis_planet
		}

		modifier = {
			factor = 5

			any_neighbor_country = {
				has_origin = origin_exomycosis
			}
			desc = gal_com_is_neighboring_exomycosis
		}

		modifier = {
			factor = 0.75

			NOT = {
				any_neighbor_country = {
					has_origin = origin_exomycosis
				}
			}
			desc = gal_com_is_not_neighboring_exomycosis
		}

		modifier = {
			factor = 0.5

			any_owned_planet = {
				NOR = {
					is_planet_class = pc_mycelium
					has_modifier = mold_contaminated
					has_modifier = exomycosis_infestation
				}
			}
			desc = gal_com_not_has_exomycosis_planet
		}
	}
}
