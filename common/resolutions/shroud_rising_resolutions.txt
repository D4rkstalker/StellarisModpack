resolution_eye_of_terror_rifts_1 = {
	icon = "GFX_resolution_eye_of_terror"

	resources = {
		category = resolutions
		cost = {
			influence = 100
		}
	}
	target = no

	level = 1

	effect = {
		custom_tooltip = resolution_eye_of_terror_rifts_1_tooltip

		hidden_effect = {
			set_global_flag = resolution_eye_of_terror_rifts_1
			set_global_flag = galactic_community_resolution_passed
			set_timed_country_flag = { flag = galactic_community_resolution_passed_general days = 360  }
			add_modifier = {
				modifier = resolution_passed_diplomatic_weight
				days = 2200
			}
			
		}
	}

	modifier = {
		resolution_eye_of_terror_rifts_1 = 1
	
	}
	fail_effects = {
		hidden_effect = {
			set_timed_country_flag = { flag = galactic_community_resolution_failed_general days = 360  }
			add_modifier = {
				modifier = resolution_failed_diplomatic_weight
				days = 2200
			}
		}
	}

	potential = {
		has_global_flag = eye_of_terror_open		
	}
		
	allow = {
		custom_tooltip = {
			fail_text = "resolution_already_active"
			NOT = {
				is_active_resolution = "resolution_eye_of_terror_rifts_1"
			}
		}
	}	

	active = { # Resolution only in effect while this trigger is met
		
	}

	ai_weight = {
		base = 6

		modifier = {
			factor = 2
			desc = gal_com_had_world_devoured			 
			has_country_flag = had_world_devoured			
		}
		modifier = {
			factor = 2
			desc = gal_com_had_world_invaded_by_cult			
			has_country_flag = had_world_invaded_by_cult				
		}
		modifier = {
			factor = 5
			desc = gal_com_has_ap_defender_of_the_galaxy
			has_ascension_perk = ap_defender_of_the_galaxy
		}

		modifier = {
			factor = 1.5

		}

		modifier = {

		modifier = {

		}

		modifier = {
			factor = 0.5

			desc = ethic_xenophobe
		}
		modifier = {
			factor = 0
			has_civic = civic_shroud_worship

		}
