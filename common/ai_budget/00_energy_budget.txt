energy_expenditure_deposit_blockers = {
	resource = energy 
	type = expenditure
	category = deposit_blockers
	
	potential = {
		years_passed > 3
	}
	
	fraction = {
		weight = 0.7
	}

	static_min = {
		base = 0
		modifier = {
			add = 400
			years_passed > 3			
			any_owned_planet = {
				num_uncleared_blockers > 0
				num_free_districts = { type = any value < 1 }
			}
		}
		modifier = {
			add = 2000	
			any_owned_planet = {
				num_uncleared_blockers > 0
				num_free_districts = { type = any value < 1 }
			}
			years_passed > 20
		}
	}
}
