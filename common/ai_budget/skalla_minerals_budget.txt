minerals_expenditure_stations_early = {
	resource = minerals
	type = expenditure
	category = stations

	potential = {
		years_passed < 5
		any_planet_within_border = {
			OR = {
    				has_deposit_for = shipclass_mining_station
				has_deposit_for = shipclass_research_station
			}
    			has_orbital_station = no
		}
	}

	fraction = {
		weight = 1
	}
}
