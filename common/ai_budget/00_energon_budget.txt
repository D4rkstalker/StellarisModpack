#ENERGON CRYSTALS

energon_crystals_expenditure_buffer = { # Buffer, will not be spent
	resource = energon_crystals
	type = expenditure
	category = buffer

	potential = {
		always = yes
	}

	weight = {
		weight = 0.3
	}
}

energon_crystals_expenditure_colonies_expand = {
	resource = energon_crystals
	type = expenditure
	category = colonies

	potential = {
		OR = {
			is_cybertronian_empire = yes 
			has_trait = trait_energon_dependence 
			has_trait = trait_dark_energon_dependence
		}
		NOT = { has_origin = origin_tree_of_life }
		ai_colonize_plans > 0
		has_resource = { type = energon_crystals amount > 400 }
		has_monthly_income = {
			resource = energon_crystals
			value > 0
		}
	}

	weight = {
		weight = 2.0
	}

	desired_max = {
		base = 550
	}
}
