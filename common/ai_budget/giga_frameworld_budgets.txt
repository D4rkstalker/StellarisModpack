alloys_expenditude_frameworld_expand = {
	resource = alloys
	type = expenditure
	category = megastructures_frameworld

	potential = {
		has_origin = origin_frameworld
	}

	weight = {
		weight = 0.1

		# defer a bit if we're fighting because we probably need ships
		modifier = {
			factor = 0.1
			is_at_war = yes
		}

		modifier = {
			factor = 2
			exists = capital_scope
			capital_scope = {
				free_district_slots < 5
			}
		}

		modifier = {
			factor = 2
			exists = capital_scope
			capital_scope = {
				free_district_slots < 3
			}
		}

		modifier = {
			factor = 1.5
			exists = capital_scope
			capital_scope = {
				free_district_slots < 2
			}
		}

		modifier = {
			factor = 1.5
			exists = capital_scope
			capital_scope = {
				OR = {
					free_district_slots < 1
					free_housing < -10
				}
			}
		}
	}
}
