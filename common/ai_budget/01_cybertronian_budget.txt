# Because Cybertronian leader budget
energy_expenditure_cybertronian_leaders = {
	resource = energy 
	type = expenditure
	category = leaders
	
	potential = {
		species = {
			has_trait = trait_cybertronian
		}
		resource_stockpile_compare = {
			resource = energy
			value < 3000
		}
	}
	
	weight = {
		weight = 1.0

		modifier = {
			factor = 5.0
			any_owned_ship = {
				is_ship_class = shipclass_science_ship
				NOT = { exists = leader }
			}
		}
	}
}
