	}

		modifier = {
				add = 100
				resource_stockpile_compare = {

				}
				resource_stockpile_compare = {
					resource = minerals
					value > 4000
				}
				resource_income_compare = {
					resource = minerals
					value > 200
				}
		}
		modifier = {
				add = 400

					resource = energy
					value < 1000
				}

					resource = minerals
					value > 10000
				}
				resource_income_compare = {
					resource = minerals
					value > 200
				}
		}
	}
}
