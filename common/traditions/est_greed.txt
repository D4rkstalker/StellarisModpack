tr_est_greed_1 = {
	on_enabled = {
		if = {
			limit = {
				has_country_resource = { 
					type = energy 
					amount < 500 
				}
			}
		}
		else = {
			if = {
				limit = {
					has_country_resource = { 
						type = energy 
						amount < 6000 
					}
				}
				add_modifier = {
					modifier = mod_est_greed_1_1
					days = -1
				}
			}
			else = {
				if = {
					limit = {
						has_country_resource = { 
							type = energy 
							amount < 25000 
						}
					}
					add_modifier = {
						modifier = mod_est_greed_1_2
						days = -1
					}
				}	
				else = {
					if = {
						limit = {
							has_country_resource = { 
								type = energy 
								amount < 80000
							}
						}
						add_modifier = {
							modifier = mod_est_greed_1_3
							days = -1
						}
					}
					else = {
						add_modifier = {
							modifier = mod_est_greed_1_4
							days = -1
						}
					}
				}				
			}
		}
		if = {
			limit = {
				Not = {
					has_global_flag = est_greed_1_picked_flag
				}
			}
			set_global_flag = est_greed_1_picked_flag
		}
	}
	
	tradition_swap = {
		name = tr_est_greed_1_AI
		inherit_effects = no
		inherit_icon = yes
		inherit_name = yes
		trigger = {
			is_ai = yes
		}	
		modifier = {
			country_base_influence_produces_add = 1
		}
		weight = {factor =  1}
	}
	ai_weight = {
		factor = 4000
	}
}
