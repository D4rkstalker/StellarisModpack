tr_est_mysticism_adopt = {
	modifier = {
		planet_sensor_range_add = 1
		ship_sensor_range_add = 1
	}

	on_enabled = {
		every_neighbor_country = {
				limit = {
					Not = {
						is_in_federation_with = prev
						is_overlord_to = prev
						prev = {
							is_overlord_to = prev
						}
					}
				}
				set_timed_country_flag = {
					flag = est_mysticism_adopt_flag
					days = 1
				}
				if = {
					limit = {
						has_modifier = mod_est_mysticism_adopt
					}
					add_modifier = {
						modifier = "mod_est_mysticism_adopt"
						days = -1
					}
				}
			}
		if = {
			limit = {
				Not = {
					has_global_flag = est_mysticism_adopt_picked_flag
				}
			}
			set_global_flag = est_mysticism_adopt_picked_flag
		}
	}
	
	tradition_swap = {
		name = tr_est_mysticism_adopt_passive_AI
		inherit_effects = no
		inherit_icon = yes
		inherit_name = yes
		trigger = {
			Not = {
				has_ethic = ethic_gestalt_consciousness
			}
			est_AI_no_passive_aggressive_traditons = yes
		}	
		modifier = {
			planet_sensor_range_add = 2
			ship_sensor_range_add = 2
		}
		weight = {factor =  1}
	}
	tradition_swap = {
		name = tr_est_philosophy_adopt
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			has_ethic = ethic_gestalt_consciousness
		}	
		modifier = {
			planet_sensor_range_add = 1
			ship_sensor_range_add = 1
		}
		weight = {
			factor = 1
		}
	}
	
	ai_weight = {
		factor = 10
		modifier = {
			factor = 2
			has_ai_personality = decadent_hierarchy
		}	
		modifier = {
			factor = 2
			has_civic = civic_shadow_council
		}	
		modifier = {
			factor = 2
			has_civic = civic_machine_introspective
		}	
		modifier = {
			factor = 0
			est_no_fed_federation_builders = yes			
		}
	}
}
