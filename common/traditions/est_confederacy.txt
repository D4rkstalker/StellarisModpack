tr_est_confederacy_2 = {
	custom_tooltip = tr_est_confederacy_2_desc

	possible = { has_tradition = tr_est_confederacy_1 }

	on_enabled = {
		every_pool_leader = {
			limit = {
				leader_class = governor
				Not = { has_trait = leader_trait_governor_est_confederacy_2 }
			}
			add_trait = leader_trait_governor_est_confederacy_2
		}
		every_owned_leader = {
			limit = {
				leader_class = governor
				Not = { has_trait = leader_trait_governor_est_confederacy_2 }
			}
			add_trait = leader_trait_governor_est_confederacy_2
		}
	}

	ai_weight = { factor = 5000 }
}
