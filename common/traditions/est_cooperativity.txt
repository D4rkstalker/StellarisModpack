tr_est_cooperativity_adopt = {
	custom_tooltip = tr_est_cooperativity_adopt_desc

	on_enabled = {
		every_owned_planet = {
			if = {
				limit = {
					num_pops > 23
				}
				add_modifier = {
					modifier = "mod_est_cooperativity_adopt_3"
					days = -1
				}
			}
			else = {
				if = {
					limit = {
						num_pops > 19
					}
					add_modifier = {
						modifier = "mod_est_cooperativity_adopt_2"
						days = -1
					}
				}
				else = {
					if = {
						limit = {
							num_pops > 15
						}
						add_modifier = {
							modifier = "mod_est_cooperativity_adopt_1"
							days = -1
						}
					}
				}
			
			}
		}
	}
}
