tr_est_justice_finish = {	
	on_enabled = {
		if = {
			limit = {
				has_resource = {
					type =  influence 
					amount > 749
				}
			}
			if = {
				limit = {
					has_resource = {
						type =  influence 
						amount > 999
					}
				}
				add_modifier = {
					modifier = "mod_est_justice_finish_2"
					days = -1
				}
			}
			else = {
				add_modifier = {
					modifier = "mod_est_justice_finish_1"
					days = -1
				}
			}
		}
		if = {
			limit = {
				Not = {
					has_global_flag = est_justice_finish_picked_flag
				}
			}
			set_global_flag = est_justice_finish_picked_flag
		}
	}
	
	tradition_swap = {
		name = tr_est_destruction_finish
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			has_ethic = ethic_gestalt_consciousness
			est_AI_no_passive_aggressive_traditons = no
		}	
		on_enabled = {
			every_country = {
				limit = {
					is_at_war_with = root
					Not = {
						has_modifier = mod_est_destruction_finish
					}
				}
				add_modifier = {
					modifier = "mod_est_destruction_finish"
					days = -1
				}				
			}
			if = {
				limit = {
					Not = {
						has_global_flag = est_destruction_finish_picked_flag
					}
				}
				set_global_flag = est_destruction_finish_picked_flag
			}	
		}
		modifier = {
			ascension_perks_add = 1
		}
		weight = {factor = 1 }
	}
	
	tradition_swap = {
		name = tr_est_destruction_finish_passive_ai
		inherit_effects = no
		inherit_icon = yes
		inherit_name = no
		trigger = {
			has_ethic = ethic_gestalt_consciousness
			est_AI_no_passive_aggressive_traditons = yes
		}	
		modifier = {
			country_war_exhaustion_mult = -0.15
			ascension_perks_add = 1

		weight = {factor = 1 }
	}
