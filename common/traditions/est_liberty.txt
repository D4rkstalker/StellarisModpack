tr_est_liberty_2 = {
	custom_tooltip = tr_est_liberty_2_desc

	possible = {
		has_tradition = tr_est_liberty_1
	}
	
	on_enabled = {
		if = {
			limit = {
				count_pop_faction = {
					limit = {
						has_modifier = suppressed_faction
					} 
					count < 1 
				}
			}
			every_pop_faction = {
				limit = {
					Not = {
						has_modifier = mod_est_liberty_2
					}
				}
				add_modifier = {
					modifier = "mod_est_liberty_2"
					days = -1
				}
			}
		}
	}

	tradition_swap = {
		name = tr_est_liberty_2_AI
		inherit_icon = yes
		inherit_name = yes
		trigger = {
			is_ai = yes
		}	
		on_enabled = {
			every_pop_faction = {
				add_modifier = {
					modifier = "mod_est_liberty_2_AI"
					days = -1
				}
			}
		}
		weight = {factor = 1 }
	}

	ai_weight = { factor = 5000 }
}
