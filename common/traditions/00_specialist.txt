				}
				AND = {
					has_tradition = tr_prosperity_adopt
					NOT = { has_tradition = tr_prosperity_finish }
				}
				AND = {
					has_tradition = tr_supremacy_adopt
					NOT = { has_tradition = tr_supremacy_finish }
