###	铁血联盟	---禁用|3.4|
AAR_policy_FU = {

	potential = { always = no
		has_ascension_perk = AAR_AP_ForcedUnion
	}
	allow = { always = no }

	option = {#允许宣称
		name = "AAR_policy_FU_allow_claim"
		policy_flags = { AAR_flg_policy_FU_allow_claim }

		potential = { always = yes } 
		valid = { always = yes }

		on_enabled = {
			every_subject = { limit = { has_country_flag = AAR_flg_cntr_no_claim }
				remove_country_flag = AAR_flg_cntr_no_claim
			}
		}
		
		AI_weight = { factor = 1 }
	}
	option = {#禁止宣称
		name = "AAR_policy_FU_no_claim"
		policy_flags = { AAR_flg_policy_FU_no_claim }
		
		potential = { always = yes } 
		valid = { always = yes }

		on_enabled = {
			every_subject = { limit = { is_subject_type = AAR_SBJ_tributary }	# 过期
				set_country_flag = AAR_flg_cntr_no_claim
				# 清除当前宣称
				every_system = { limit = { prev = { has_claim = prev } }
					remove_claims = { who = prev }
				}
			}
		}
		
		AI_weight = { factor = 0 }
	}
}
