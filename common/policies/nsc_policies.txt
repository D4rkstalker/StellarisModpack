		name = "black_sky"

			has_technology = tech_Battlecruiser_1

			ship_home_territory_fire_rate_mult = -0.1
			ship_fire_rate_mult = 0.2
			ship_orbital_bombardment_mult = 0.25
		}

		name = "privateer_operation"
