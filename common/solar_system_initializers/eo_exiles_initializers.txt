@distance = 50
@jumps = 3
@planet_min_size = 10
@planet_max_size = 25
@base_moon_distance = 10
@moon_min_size = 6
@moon_max_size = 10
@homeworld_min_size = 18
@homeworld_max_size = 21

# Basic 1
eo_exiles_init = {
	class = "rl_standard_stars"

	usage = misc_system_init
	usage_odds = {
		base = 0
		modifier = {
			any_country = {
				has_origin = origin_exiles
				NOT = {
					has_country_flag = parent_system_spawned
				}
			}
			add = 99999
		}
	}

	planet = {
		count = 1
		class = star
		orbit_distance = 0
		orbit_angle = 1
		size = { min = 20 max = 30 }
		has_ring = no
	}

	change_orbit = 45

	planet = {
		name = "NAME_Exiles_Homeworld"
		home_planet = yes
		class = pc_continental
		orbit_angle = { min = 90 max = 270 }
		size = { min = @homeworld_min_size max = @homeworld_max_size }

		init_effect = {
			prevent_anomaly = yes
			save_event_target_as = exiles_homeworld

			random_country = {
				limit = {
					has_origin = origin_exiles
					NOT = {
						has_country_flag = parent_system_spawned
					}
				}
				set_country_flag = parent_system_spawned
				save_event_target_as = exiles_empire
			}
			change_pc = event_target:exiles_empire.owner_species
			event_target:exiles_empire = {
				owner_species = {
					set_species_homeworld = event_target:exiles_homeworld
				}
				if = {
					limit = {
						OR = {
							has_valid_civic = civic_machine_assimilator
							has_valid_civic = civic_machine_servitor
						}
					}
					random_owned_species = {
						limit = {
							NOT = { is_species_class = MACHINE }
						}
						set_species_homeworld = event_target:exiles_homeworld
					}
				}
			}
			remove_planet = yes
		}
	}

	planet = {
		count = { min = 2 max = 10 }
		orbit_distance = 20
		orbit_angle = { min = 90 max = 270 }

		change_orbit = @base_moon_distance

		moon = {
			count = { min = 0 max = 1 }
			orbit_angle = { min = 90 max = 270 }
			orbit_distance = 5
		}
	}
}
