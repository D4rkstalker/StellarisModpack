				create_starting_leaders = yes
				set_country_flag = infiltrators_parent
				set_country_flag = special_intro_for@event_target:infiltrators_child
				set_country_flag = infiltrators_parent_of@event_target:infiltrators_child
				country_event = { id = game_start.9 }

				set_name = random # Fixes some naming issues
