# This = subject
# From = subject's overlord
on_becoming_subject = {
	events = {
		exalted_subjects.0 # Set Vassal type
		exalted_subjects.1 # Set Tributary type
		#exalted_subjects.3 # Force open borders between Overlord and subjects
	}
}

# this = country

		exalted_subjects.919 # No longer Remote Archive

	}
}
