# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# замена зданий (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.4001 # сообщение - этап 1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.4001 # сообщение - этап 1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.4002 # сообщение - этап 2 построен
		apsr.4003 # сообщение - этап 3 построен
	}
}

on_survey = {
	events = {
		apsr.4011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.4013 # сообщение при захвате планеты
		apsr.4801 # замена зданий при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.4013 # сообщение при получении планеты
		apsr.4801 # замена зданий при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.4013 # сообщение при колонизации планеты
		apsr.4801 # замена зданий при колонизации планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.4014 # отображение найденного ресурса, не показаного раньше
		apsr.4901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
