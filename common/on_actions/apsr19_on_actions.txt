# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# фейк-замена (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.19001 # сообщение - этап био1 построен
		apsr.19004 # сообщение - этап мех1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.19001 # сообщение - этап био1 построен
		apsr.19004 # сообщение - этап мех1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.19002 # сообщение - этап био2 построен
		apsr.19003 # сообщение - этап био3 построен
		apsr.19005 # сообщение - этап мех2 построен
		apsr.19006 # сообщение - этап мех3 построен
		apsr.19007 # конверсия в мех-версию
	}
}

on_survey = {
	events = {
		apsr.19011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.19013 # сообщение при захвате планеты
		apsr.19801 # замена зданий на фейки при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.19013 # сообщение при получении планеты
		apsr.19801 # замена зданий на фейки при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.19013 # сообщение при колонизации планеты
		apsr.19801 # замена зданий на фейки при колонизации планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.19014 # отображение найденного ресурса, не показаного раньше
		apsr.19901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}


# ивенты не запускаемые через экшн:
# apsr.19601 - при изучении аномалии с ресурсом
