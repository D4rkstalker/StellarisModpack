# A ship has surveyed a planet.
# Scope = Ship
# From = Planet

	events = {
		newfrontiers_action.100
		newfrontiers_decision_event.3
	}
}

		newfrontiers_colony_event.1     # Ancient Machine

	}

	random_events = {
		300 = 0
	}
}
