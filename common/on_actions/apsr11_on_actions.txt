# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# фейк-замена (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.11001 # сообщение - этап био-1 построен
		apsr.11004 # сообщение - этап мех-1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.11001 # сообщение - этап био-1 построен
		apsr.11004 # сообщение - этап мех-1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.11002 # сообщение - этап био-2 построен
		apsr.11003 # сообщение - этап био-3 построен
		apsr.11005 # сообщение - этап мех-2 построен
		apsr.11006 # сообщение - этап мех-3 построен
		apsr.11007 # конверсия в мех-версию
	}
}

on_survey = {
	events = {
		apsr.11011 # сообщение при изучении научным кораблем (страница 1)
	}
}

on_planet_conquer = {
	events = {
		apsr.11013 # сообщение при захвате планеты
		apsr.11801 # замена зданий при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.11013 # сообщение при получении планеты
		apsr.11801 # замена зданий при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.11013 # сообщение при колонизации планеты
		apsr.11801 # замена зданий при колонизации планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.11014 # отображение найденного ресурса, не показаного раньше
		apsr.11901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
