# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)
# сюжет (2хх)



on_building_complete = {
	events = {
		apsr.7000 # сообщение - кордон построен
	}
}

on_building_replaced = {
	events = {
		apsr.7000 # сообщение - кордон построен
	}
}

on_building_upgraded = {
	events = {
		apsr.7001 # сообщение - этап 1 построен
		apsr.7002 # сообщение - этап 2 построен
		apsr.7003 # сообщение - этап 3 построен
	}
}

on_survey = {
	events = {
		apsr.7011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.7013 # сообщение при захвате планеты
	}
}

on_planet_transfer = {
	events = {
		apsr.7013 # сообщение при получении планеты
	}
}

on_colonization_started = {
	events = {
		apsr.7013 # сообщение при колонизации планеты
	}
}

on_yearly_pulse = {
	events = {
		apsr.7014 # отображение найденного ресурса, не показаного раньше
		apsr.7901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}

on_colony_1_year_old = {
	events = {
		apsr.7200 # запуск ситуации после 1 года
	}
}

on_tech_increased = {
	events = { 
		apsr.7204 # Технология кордона изучена, когда есть захваченная планета с уже построенными зданиями
	}
}



on_planet_defenders_lose = {
	events = {
		apsr.7212 # поражение на этапе 1
		apsr.7215 # поражение на этапе 2
		apsr.7218 # поражение на этапе 3
		apsr.7221 # поражение на этапе 4
	}
}

on_planet_defenders_win = {
	events = {
		apsr.7213 # победа на этапе 1
		apsr.7216 # победа на этапе 2
		apsr.7219 # победа на этапе 3
		apsr.7222 # победа на этапе 4
	}
}
