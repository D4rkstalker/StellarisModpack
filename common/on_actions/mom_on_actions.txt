# Triggers when the game starts
on_game_start = {
	events = {
		mom_house_divided.1
		mom_severance.1
		mom_breakaway_civ.1
		mom_exiles.1
		mom_god_emperor.1
		mom_crowded.1
		action.90
	}
}

#This = country
on_game_start_country = {
	events = {
		mom_climate.1
		mom_climate.2
		mom_junction.1
	}
}


# A colony has been destroyed. Called just before owner and controller is cleared
# Scope = Planet
on_colony_destroyed = {
	events = {
		mom_breakaway_civ.2
	}
}

#From = Country scope
#This = Planet scope
on_planet_ownerless = {
	events = {
		mom_breakaway_civ.2
	}
}

#Fired whenever a new owner is set for a planet,
#be it after a war or through a trade
#From = Country scope (new owner)
#This = Planet scope
on_planet_transfer = {
	events = {
		mom_breakaway_civ.2
	}
}

# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey = {
    events = {
        mom_exiles.4
    }
}

# Triggers event when ship enters a system
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system = {
    events = {
        mom_exiles.3
    }
}

# A planet has been colonized.
# Scope = Planet
on_colonized = {
	events = {
		mom_exiles.7
	}
}

# Triggers country_event for the defender upon victory
# This = country, planet owner
# From = country, attack leader
# FromFrom = planet
on_planet_defenders_win = {
    events = {
        mom_exiles.12
		mom_exiles.17
    }
}

# Executed as a leader has died
# This = Country
# From = Leader
on_leader_death = {
	events = {
		mom_god_emperor.4
	}
}

# this = country
on_yearly_pulse_country = {
	events = {
		mom_god_emperor.6
		mom_crowded.2
	}
}
