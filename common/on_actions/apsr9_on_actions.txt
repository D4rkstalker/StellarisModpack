# обозначения:
# постройка и улучшение здания, эксперименты (00х)
# нахождение ресурса (01х)
# замена зданий (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.9001 # сообщение - этап 1 построен. запуск экспериментов
	}
}

on_building_replaced = {
	events = {
		apsr.9001 # сообщение - этап 1 построен. запуск экспериментов
	}
}

on_building_upgraded = {
	events = {
		apsr.9009 # сообщение - этап 2 построен
	}
}

on_survey = {
	events = {
		apsr.9011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.9013 # сообщение при захвате планеты
		apsr.9801 # замена зданий на руины или этап 1 при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.9013 # сообщение при получении планеты
		apsr.9801 # замена зданий на руины или этап 1 при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.9013 # сообщение при колонизации планеты
		apsr.9801 # замена зданий на руины или этап 1 при колонизации планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.9014 # отображение найденного ресурса, не показаного раньше
		apsr.9901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}


# ивенты не запускаемые через экшн:
# apsr.9601 - при изучении аномалии с ресурсом
#
# apsr.9002 - успешное завершение 1 эксперимента
# apsr.9003 - успешное завершение 2 эксперимента
# apsr.9004 - успешное завершение 3 эксперимента
# apsr.9005 - успешное завершение 4 эксперимента. доступ к улучшению здания
#
# apsr.9301 - ПРОВАЛЬНОЕ завершение эксперимента (сообщение)
# apsr.9302 - ПРОВАЛЬНОЕ завершение эксперимента (взрыв планеты)
