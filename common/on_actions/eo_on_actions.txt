# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey = {
	events = {
		eo_exofungus.17 # Exofungus origin discovers one of their guaranteed primitives
		eo_exofungus.18 # Exofungus origin discovers previously unknown primitive
		eo_exofungus.19 # Exofungus origin discovers primitive with exofungal colonizers
		eo_exofungus.20 # Other empire discovers primitive with exofungal colonizers
		eo_exofungus.21 # Other empire discovers primitive of previously encountered exofungus
		eo_exofungus.22 # Other empire discovers primitive of of unknown exofungus
	}
}
