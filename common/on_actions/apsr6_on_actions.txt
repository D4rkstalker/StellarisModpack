# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# фейк-замена (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.6001 # сообщение - этап 1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.6001 # сообщение - этап 1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.6002 # сообщение - этап 2 построен
		apsr.6003 # сообщение - этап 3 построен
		apsr.6004 # сообщение - этап 4 построен
	}
}

on_survey = {
	events = {
		apsr.6011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.6013 # сообщение при захвате планеты
		apsr.6801 # замена зданий при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.6013 # сообщение при получении планеты
		apsr.6801 # замена зданий при захвате планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.6013 # сообщение при колонизации планеты
		apsr.6801 # замена зданий при захвате планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.6014 # отображение найденного ресурса, не показаного раньше
		apsr.6901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
