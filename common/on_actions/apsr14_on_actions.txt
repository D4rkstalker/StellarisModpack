# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# фейк-замена (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.14001 # сообщение - этап био-1 построен
		apsr.14004 # сообщение - этап мех-1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.14001 # сообщение - этап био-1 построен
		apsr.14004 # сообщение - этап мех-1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.14002 # сообщение - этап био-2 построен
		apsr.14003 # сообщение - этап био-3 построен
		apsr.14005 # сообщение - этап мех-2 построен
		apsr.14006 # сообщение - этап мех-3 построен
		apsr.14007 # сообщение - этап мех-4 построен
		apsr.14008 # конверсия в мех-версию
	}
}

on_survey = {
	events = {
		apsr.14011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.14013 # сообщение при захвате планеты
		apsr.14801 # замена зданий при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.14013 # сообщение при получении планеты
		apsr.14801 # замена зданий при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.14013 # сообщение при колонизации планеты
		apsr.14801 # замена зданий при колонизации планеты (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.14014 # отображение найденного ресурса, не показаного раньше
		apsr.14901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
