
### Получение планеты с Кубом - добавление реликвии или просто сообщение

on_planet_conquer = {
	events = {
		apsr_arcrel.1 # сообщение при захвате планеты
	}
}

on_planet_transfer = {
	events = {
		apsr_arcrel.1 # сообщение при получении планеты
	}
}

on_colonization_started = {
	events = {
		apsr_arcrel.1 # сообщение при колонизации планеты
	}
}
