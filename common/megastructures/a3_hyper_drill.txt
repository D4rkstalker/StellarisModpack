hyper_drill_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -5 y = -5 }
	build_time = 720
	resources = {
		category = megastructures
		cost = {
			alloys = 1000
			influence = 150
		}
		
		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_technology = tech_big_drill
		no_scope = {
			NOT = {
				has_global_flag = drill_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "cant_build_on_habitable"
				colonizeable_planet = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {			# prevent leaking habitability information
					who = prev.from
					status = yes
				}
				colonizeable_planet = no	# prevent the spaceport colliding with the habitat
			}
			custom_tooltip = {
				fail_text = "already_stripmined"
				NOR = { 
					has_planet_flag = stripminedplanet
					is_planet_class = pc_core_mined
				}
			}
			custom_tooltip = {
				fail_text = "cant_build_around_gas"
				NOR = {
					is_planet_class = pc_shrouded
					is_planet_class = pc_gas_giant
					is_planet_class = pc_gigabrown_dwarf
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		} # use these for all non-star megastructures
	}
	
	ai_weight = {
		factor = 5
		
		modifier = {
			factor = 3
			fromfrom.planet = { planet_size > 20 }
		}
		
		modifier = {
			factor = 0.1
			from = {
				OR = {
					has_megastructure = hyper_drill_0
					has_megastructure = hyper_drill_1
					has_megastructure = hyper_drill_2
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
	}
}

hyper_drill_2 = {
	entity = "hyper_drill_1_part_big_entity"
	construction_entity = ""
	portrait = "GFX_megastructure_barren"
	construction_blocks_others = no
	upgrade_from = {
		hyper_drill_1
	}
	build_time = 720
	
	resources = {
		category = megastructures
		cost = {
			alloys = 1500
			energy = 5000
		}
		
		upkeep = {
			energy = 75
		}
		
		produces = {
			minerals = 200
		}
	}
	
	potential = {
		has_technology = tech_big_drill
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "cant_activate_two_stripmines"
			from = {
				NOT = { has_country_flag = building_hyperdrill }
			}
		}
	}
	
	on_build_start = {
		fromfrom.planet = {
			space_owner = {
				set_country_flag = building_hyperdrill
			}
		}
	}
	on_build_complete = {
		from = {
			set_country_flag = has_built_or_repaired_megastructure
		}
		fromfrom.planet = {
			space_owner = {
				remove_country_flag = building_hyperdrill
			}
		}
	}
}
