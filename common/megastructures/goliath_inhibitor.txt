# Inhibitor Frame
goliath_inhibitor_0 = {
	entity = "goliath_inhibitor_phase_01_entity"
	construction_entity = "goliath_inhibitor_phase_01_entity"
	portrait = "GFX_megastructure_construction_background"	
	build_time = 150
	resources = {
		category = megastructures
		cost = {
			alloys = 10000
			unity = 5000
		}

		upkeep = {
			energy = 15
		}
	}

	prerequisites = { "tech_goliath_rift_inhibitor" }
	
	construction_blocks_and_blocked_by = self_type
	
	possible = {		
		custom_tooltip = {			
			fail_text = "requires_inside_border_or_subject_ally"				
			OR = {
				AND = {
					has_global_flag = resolution_eye_of_terror_rifts_1
					exists = owner
					owner = {
						is_galactic_community_member = yes
					}
				}
				is_inside_border = from
				AND = {
					exists = owner
					owner = {
						has_defensive_pact = from
					}
				}
				AND = {
					exists = owner
					owner = {
						is_in_federation_with = from
					}
				}
				
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_value = from }
					}
				}			
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_empire = from }
					}
				}
				AND = {
					exists = owner
					owner = {	
						is_subject = yes
						overlord = { is_overlord_to = from	}
					}
				}
				AND = {
					exists = owner
					owner = {	
						is_overlord_to = from
					}
				}														
						
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"			
			from = {
				NOT = {
					any_owned_megastructure = {
						OR = {
							is_megastructure_type = goliath_inhibitor_0
							is_megastructure_type = goliath_inhibitor_1
							is_megastructure_type = goliath_inhibitor_2
							is_megastructure_type = goliath_inhibitor_3
							is_megastructure_type = goliath_inhibitor_4							
							is_megastructure_type = goliath_inhibitor_permanently_ruined
						}
					}
				}
			}
		}
		
		custom_tooltip = {
			# fail_text = "requires_no_active_rifts"
			# has_global_flag = rift_2_sealed
			# has_global_flag = rift_3_sealed
			# has_global_flag = rift_4_sealed
			
		}
		
	}	
	

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_eye_of_terror"
				is_planet_class = pc_eye_of_terror
			}
			
		}
		
	}
	
	ai_weight = {
			factor = 100			
			modifier = {
				factor = 2
				has_global_flag = eye_of_terror_open
			}
			modifier = {
				factor = 0
				from = {					
						has_civic = civic_shroud_worship 
				
				}
			}
	}
		
	
	
	on_build_cancel = {}
	on_build_complete = {
		
		fromfrom.planet = {
			if = {
				limit = { has_orbital_station = yes }
				orbital_station = {
					dismantle = yes
				}
			}
		}
		
		from = {
			country_event = { id = shroud_rising_eye_of_terror.14 }
		}
	}
}
