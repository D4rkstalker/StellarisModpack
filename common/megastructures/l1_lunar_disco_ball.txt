				fail_text = "requires_no_anomaly"

				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				AND = {
					is_surveyed = {	
						who = prev.from
						status = yes
					}
					colonizeable_planet = no #yall cant be trusted smh
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_asteroid"
				NOT = {
					is_asteroid = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		} # use these for all non-star megastructures

	ai_weight = {
		factor = 0
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
	}
}

lunar_disco_ball_2 = {
	entity = ""
	construction_entity = ""
	portrait = "GFX_megastructure_disco_moon"
	construction_blocks_others = no
	upgrade_from = {
		lunar_disco_ball_1
	}
	build_time = 2500
	
	resources = {
		category = megastructures
		cost = {
			alloys = 2000
			rare_crystals = 1500
			energy = 10000
		}
		
		upkeep = {
			energy = 100
		}
		
		produces = {
			unity = 50
		}
	}
	
	potential = {
		has_technology = tech_disco_moon
	}
	
	on_build_complete = {
		from = {
			set_country_flag = has_built_or_repaired_megastructure
		}
		fromfrom.planet = {
		    change_pc = pc_disco_moon
			set_planet_flag = discomoontarget
			solar_system = {
				every_planet = {
					limit = {
						any_moon = { has_planet_flag = discomoontarget }
					}
					add_modifier = { modifier = disco_moon_modifier days = -1 }
					every_moon = {
						limit = {
							NOT = {
								has_planet_flag = discomoontarget
							}
						}
						add_modifier = { modifier = disco_moon_modifier days = -1 }
					}
				}
			}
			remove_planet_flag = discomoontarget
		}
		remove_megastructure = fromfrom
	}
}
