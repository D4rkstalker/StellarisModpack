gasgiant_ignitor_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -5 y = -5 }
	build_time = 2700
	resources = {
		category = megastructures
		cost = {
			alloys = 5000
			influence = 75
		}
		
		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_technology = tech_ignite_browndwarf
		no_scope = {
			NOT = {
				has_global_flag = compressor_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_dwarf"
				OR = {
					is_planet_class = pc_gigabrown_dwarf
					is_planet_class = pc_t_star
					is_planet_class = pc_t_star_big
					AND = {
						has_planet_flag = helioformingcandidate
						is_planet_class = pc_gas_giant
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {	
					who = prev.from
					status = yes
				}
				colonizeable_planet = no
			}

 			# balance for habitats
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
		} # use these for all non-star megastructures
	}
	
	ai_weight = {
		factor = 15
		
		modifier = {
			factor = 0.1
			from = {
				OR = {
					has_megastructure = gasgiant_ignitor_0
					has_megastructure = gasgiant_ignitor_1
					has_megastructure = gasgiant_ignitor_2
					has_megastructure = gasgiant_ignitor_4
					has_megastructure = gasgiant_ignitor_5
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
	}
}
