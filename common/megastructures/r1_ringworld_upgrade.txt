ringworld_upgrade_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -5 y = -5 }
	build_time = 1800

	resources = {
		category = megastructures
		cost = {
			alloys = 3500
		}

		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_technology = tech_ringworld_mastery
		no_scope = {
			NOT = {
				has_global_flag = upgrade_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_ringworld_tech"
				is_planet_class = pc_ringworld_tech
			}
			custom_tooltip = {
				fail_text = "cant_upgrade_this"
				NOT = { has_planet_flag = cant_upgrade }
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_upgrade"
				NOT = { has_planet_flag = has_ringworld_upgrade }
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		}
	}
	
	ai_weight = {
		factor = 1
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
			set_planet_flag = has_ringworld_upgrade
		}
	}
}

ringworld_upgrade_1 = {
	entity = "invisible_turret_entity"
	construction_entity = "ringworld_habitable_entity_01_entity"
	portrait = "GFX_megastructure_continental_planet"
	construction_blocks_others = no
	upgrade_from = {
		ringworld_upgrade_0
	}
	build_time = 2700
	
	resources = {
		category = megastructures
		cost = {
			alloys = 4250
		}
	}
	
	potential = {
		has_technology = tech_ringworld_mastery
	}
	
	on_build_complete = {
		fromfrom.planet = {
			remove_planet_flag = has_ringworld_upgrade
			change_pc = pc_ringworld_habitable
			set_planet_entity = {
				entity = "ringworld_habitable_entity_01_entity"
				graphical_culture = fromfrom
			}
			while = { 
				limit = { 
					planet_size > 5
				} 
				change_planet_size = -1			
			}
			set_name = "Habitable Section"
		}
		remove_megastructure = fromfrom
	}
}
