ring_shipyard_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	build_time = 1800

	resources = {
		category = megastructures
		cost = {
			alloys = 2500
			influence = 200
		}
	}
	construction_blocks_others = no

	potential = {
		has_technology = tech_molten_shipyard
		no_scope = {
			NOT = {
				has_global_flag = shipyard_disabled
			}
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "smaller_than_25"
				planet_size < 26
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "must_build_around_molten"
				is_surveyed = {
					who = prev.from
					status = yes
				}
				is_planet_class = pc_molten
			}
			custom_tooltip = {
				fail_text = "only_one_equatorial_shipyard"
				NOT = {
					solar_system = {
						any_system_planet = {
							has_planet_flag = system_equatorial_shipyard
						}
					}
				}
			}

			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
			
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		}
	}
	
	ai_weight = {
		factor = 1
		
		modifier = {
			factor = 0.1
			owner = {
				any_planet_within_border = {
					has_planet_flag = system_equatorial_shipyard
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
			set_planet_flag = system_equatorial_shipyard
		}
	}
}

ring_shipyard_2 = {
	entity = "big_planetary_ring_game_entity"
	construction_entity = "planetary_ring_game_construction_entity"
	portrait = "GFX_megastructure_molten_world"
	construction_blocks_others = no
	upgrade_from = {
		ring_shipyard_1
	}
	build_time = 1800
	
	resources = {
		category = megastructures
		cost = {
			alloys = 5000
		}
		upkeep = {
			energy = 50
		}
	}
	
	potential = {
		has_technology = tech_molten_shipyard
	}
	
	on_build_complete = {
		from = {
			set_country_flag = has_built_or_repaired_megastructure
		}
		fromfrom.planet = {
			set_planet_flag = has_equatorial_shipyard
		}
		spawn_planet = {
			class = "pc_ringworld_alloys"
			location = fromfrom.planet
			orbit_location = yes
			orbit_angle_offset = 0
			orbit_distance_offset = 0
			size = 12
			has_ring = no
			
			init_effect = {
				set_name = "Processing Ring"
				set_planet_entity = {
					entity = "planetary_ring_game_entity"
				}
				surveyed = {
					set_surveyed = yes
					surveyor = FROM
				}
				clear_blockers = yes
				set_planet_flag = megastructure
				set_planet_flag = habitat
				set_planet_flag = colony_event
				add_modifier = {
					modifier = ringyard_alloys
					days = -1
				}
			}
		}
	}
}
