barren_terraform_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -5 y = -5 }
	build_time = 1800
	
	resources = {
		category = megastructures
		cost = {
			alloys = 3500
			influence = 75
		}
		
		upkeep = {
			energy = 5
		}
	}

	potential = {
		has_technology = tech_fix_barren
		no_scope = {
			NOT = {
				has_global_flag = barren_terraform_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_barren"
				OR = {
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_frozen
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					has_planet_flag = dynamicignitor
					has_planet_flag = atmospherepurifier
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {	
					who = prev.from
					status = yes
				}
				colonizeable_planet = no
			}

 			# balance for habitats
			custom_tooltip = {
				fail_text = "requires_not_asteroid"
				NOT = {
					is_asteroid = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		} # use these for all non-star megastructures
	}
	
	ai_weight = {
		factor = 15
		
		modifier = {
			factor = 5
			fromfrom.planet = { planet_size > 20 }
		}
		
		modifier = {
			factor = 0.1
			from = {
				OR = {
					has_megastructure = barren_terraform_1
					has_megastructure = barren_terraform_2
					has_megastructure = barren_terraform_3
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
			set_planet_flag = dynamicignitor
		}
	}
}

barren_terraform_3 = {
	entity = "invisible_turret_entity"
	construction_entity = ""
	portrait = "GFX_megastructure_barren"
	construction_blocks_others = no
	upgrade_from = {
		barren_terraform_2
	}
	build_time = 3600
	
	resources = {
		category = megastructures
		cost = {
			energy = 5000
		}
	}
	
	potential = {
		has_technology = tech_fix_barren
	}
	
	on_build_complete = {
		from = {
			set_country_flag = has_built_or_repaired_megastructure
		}
		fromfrom.planet = {
			remove_planet_flag = has_megastructure
		    change_pc = pc_ll_temperate
			add_modifier = { modifier = lifeless_temperate }

			planet_event = { id = gigaeng.3 }
		}
		remove_megastructure = fromfrom
	}
}
