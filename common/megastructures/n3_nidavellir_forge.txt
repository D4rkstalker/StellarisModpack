nidavellir_forge_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = yes
	place_entity_on_planet_plane = no
	entity_offset = { x = -7 y = -7 }
	build_time = 1800
	
	resources = {
		category = megastructures
		cost = {
			alloys = 3500
			influence = 150
		}

		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_ascension_perk = ap_hyper_engineering
		has_technology = tech_nidavellir
		no_scope = {
			NOT = {
				has_global_flag = nidavellir_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "too_small_star"
				NOT = {
					planet_size < 11
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					solar_system = {
						has_star_flag = has_giga
					}
				}
			}
			custom_tooltip = {
				fail_text = "must_build_around_neutron_star"
				OR = {
					is_planet_class = pc_neutron_star
					is_planet_class = pc_pulsar
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = has_giga
		set_star_flag = has_megastructure
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
		from = { set_country_flag = built_nidavellir }
	}
}
