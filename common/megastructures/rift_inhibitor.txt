rift_inhibitor_0 = {
	entity = "dyson_sphere_phase_01_entity"
	construction_entity = "dyson_sphere_frame_entity"
	portrait = "GFX_megastructure_construction_background"	
	build_time = 150
	resources = {
		category = megastructures
		cost = {
			alloys = 1000
			influence = 100
		}

		upkeep = {
			energy = 15
		}
	}
	
	construction_blocks_and_blocked_by = self_type
	prerequisites = { "tech_rift_inhibitor" }
	
	possible = {		
		custom_tooltip = {			
		fail_text = "can_not_build_in_cult_territory"	
			NOT = {					
				AND = {
					exists = owner						
					owner = {
						is_ai = yes
						has_civic = civic_shroud_worship
					}
				}
			}
		}
		custom_tooltip = {			
			fail_text = "requires_inside_border_or_subject_ally"				
			OR = {	
				AND = {
					has_global_flag = resolution_eye_of_terror_rifts_1
					exists = owner
					owner = {
						is_galactic_community_member = yes
					}
				}
				is_inside_border = from
				AND = {
					exists = owner
					owner = {
						has_defensive_pact = from
					}
				}
				AND = {
					exists = owner
					owner = {
						is_in_federation_with = from
					}
				}
				
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_value = from }
					}
				}			
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_empire = from }
					}
				}
				AND = {
					exists = owner
					owner = {	
						is_subject = yes
						overlord = { is_overlord_to = from	}
					}
				}
				AND = {
					exists = owner
					owner = {	
						is_overlord_to = from
					}
				}														
						
			}
		}
		
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_shroud_rift"
				has_planet_flag = shroud_rift_planet				
			}
			custom_tooltip = {
				fail_text = "cannot_build_around_eye_of_terror"
				NOT = { is_planet_class = pc_eye_of_terror }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_inhibitor"
				#can_build_megastructure_on_planet = yes
				solar_system = {
					NOR = {					
						has_megastructure = rift_inhibitor_0
						has_megastructure = rift_inhibitor_1
						has_megastructure = rift_inhibitor_2
						has_megastructure = rift_inhibitor_permanently_ruined									
					}
				}
			}
		}
		
		
		
	}

	# root = system
	# from = country	
	
	ai_weight = {
			factor = 20		
			modifier = {
				factor = 10
				has_global_flag = eye_of_terror_open
			}
			modifier = {
				factor = 0
				from = {					
					has_civic = civic_shroud_worship 					
				}
			}
	}
		
	
	on_build_cancel = {}
	on_build_complete = {
		
		fromfrom.planet = {
			if = {
				limit = { has_orbital_station = yes }
				orbital_station = {
					dismantle = yes
				}
			}
		}
	}
}
