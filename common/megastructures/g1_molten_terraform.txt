molten_terraform_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -5 y = -5 }
	build_time = 1800
	resources = {
		category = megastructures
		cost = {
			alloys = 3500
			influence = 75
		}
		
		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_technology = tech_fix_molten
		no_scope = {
			NOT = {
				has_global_flag = geothermal_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_molten"
				is_planet_class = pc_molten
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					has_planet_flag = dynamicignitor
					has_planet_flag = atmospherepurifier
					has_planet_flag = geothermalstabilizer
					has_planet_flag = crystal_drill
					has_planet_flag = system_equatorial_shipyard
					solar_system = {
						has_star_flag = ring_world_built
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {	
					who = prev.from
					status = yes
				}
				colonizeable_planet = no
			}

 			# balance for habitats
			custom_tooltip = {
				fail_text = "requires_not_asteroid"
				NOT = {
					is_asteroid = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
		} # use these for all non-star megastructures
	}
	
	ai_weight = {
		factor = 5
		
		modifier = {
			factor = 5
			fromfrom.planet = { planet_size > 20 }
		}

		modifier = {
			factor = 0.1
			from = {
				OR = {
					has_megastructure = molten_terraform_1
					has_megastructure = molten_terraform_2
					has_megastructure = molten_terraform_3
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
			set_planet_flag = geothermalstabilizer
		}
	}
}

	construction_blocks_others = no
	upgrade_from = {
		molten_terraform_2
	}
	build_time = 3600
	
	resources = {
		category = megastructures
		cost = {
			energy = 12500
		}
	}
	
	potential = {
		has_technology = tech_fix_molten
	}
	
	on_build_complete = {
		from = {
			set_country_flag = has_built_or_repaired_megastructure
		}
		fromfrom.planet = {
			remove_planet_flag = has_megastructure
			remove_planet_flag = geothermalstabilizer
		    change_pc = pc_ll_torrid
			add_modifier = { modifier = lifeless_torrid }
			add_modifier = { modifier = lifeless_torrid_tectonics days = 1800 }

			planet_event = { id = gigaeng.4 }
		}
		remove_megastructure = fromfrom
	}
