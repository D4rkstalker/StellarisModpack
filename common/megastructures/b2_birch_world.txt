birch_world_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = yes
	place_entity_on_planet_plane = no
	entity_offset = { x = -150 y = -7 }
	build_time = 1800
	
	resources = {
		category = megastructures
		cost = {
			alloys = 100000
			influence = 150
		}

		upkeep = {
			energy = 5
		}
	}
	
	potential = {
		has_ascension_perk = ap_birch_world
		has_technology = tech_birch_world
		no_scope = {
			NOT = {
				has_global_flag = birch_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_no_existing_birch"
			from = { NOT = { has_country_flag = built_birch } }
		}
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					solar_system = {
						has_star_flag = has_giga
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "must_build_around_core_black_hole"
				is_planet_class = pc_core_black_hole
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = has_giga
		set_star_flag = has_megastructure
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
		from = { set_country_flag = built_birch }
	}
}

birch_world_3 = {
	entity = "birch_world_entity"
	construction_entity = "birch_world_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	upgrade_from = {
		birch_world_2
	}
	build_time = 10800
	
	resources = {
		category = megastructures
		cost = {
			alloys = 400000
			volatile_motes = 7500
			exotic_gases = 7500
			rare_crystals = 7500
		}
	}
	
	potential = {
		has_technology = tech_birch_world
	}
	
	on_build_complete = {
		spawn_planet = {
			class = "pc_birch"
			location = fromfrom
			size = 1
			has_ring = no
			
			init_effect = {				
				set_name = "Birch World"
				surveyed = {
					set_surveyed = yes
					surveyor = FROM
				}
				set_all_comms_surveyed = yes
				clear_blockers = yes
				set_planet_flag = megastructure
				set_planet_flag = habitat
				add_district = operations_structure
				set_planet_flag = colony_event
			}
		}
		fromfrom.planet = {
			set_planet_flag = has_megastructure
			change_planet_size = -10
		}
		remove_megastructure = fromfrom
	}
}
