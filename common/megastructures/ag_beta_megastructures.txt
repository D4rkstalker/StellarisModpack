	entity = "ag_void_entity"

	show_galactic_map_icon = no

	build_time = @ag_beta_station_type1_megastructure_buildtime
	
	upgrade_desc = hide
	
	entity_offset = { x = @entity_x y = @entity_y }
	
	resources = {
		category = megastructures
		cost = {
			sr_living_metal = @ag_beta_station_type1_megastructure_cost_sr_living_metal
			nanites = @ag_beta_station_type1_megastructure_cost_nanites
			sr_dark_matter = @ag_beta_station_type1_megastructure_cost_sr_dark_matter
			sr_zro = @ag_beta_station_type1_megastructure_cost_sr_zro
		}
		cost = {
			trigger = { owner = { NOT = { has_country_flag = ag_ship_subsystem_enabled_201_13 } } }
			sr_living_metal = @ag_beta_station_type1_megastructure_extra_cost_sr_living_metal
			nanites = @ag_beta_station_type1_megastructure_extra_cost_nanites
			sr_dark_matter = @ag_beta_station_type1_megastructure_extra_cost_sr_dark_matter
			sr_zro = @ag_beta_station_type1_megastructure_extra_cost_sr_zro
		}
	}
	
	potential = {
		NOT = { has_global_flag = ag_ancient_empire_mod_shipyard_building }
		has_technology = tech_ag_beta_station_type1
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}		
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "ag_requires_tech_ag_beta_station_type1"
			from = { has_technology = tech_ag_beta_station_type1 }
		}
	}
	
	placement_rules = {
		planet_possible = {
			ag_is_construction_section = yes
			custom_tooltip = {
				fail_text = "ag_requires_usable_ancient_construction_section"
				NOT = { has_planet_flag = has_megastructure }
			}
		}
	}
	
	on_build_start = {
	}
	on_build_cancel = {
	}
	on_build_complete = {
		from = { country_event = { id = ag_ancient_knowledge.26 } }
		if = {
			limit = { from = { NOT = { has_country_flag = ag_beta_station_type1_built } } }
			from = {
				set_country_flag = ag_beta_station_type1_built
				country_event = { id = ag_beta.131 }
			}
		}
		else = {
			create_message = {
				type = "MEGASTRUCTURE_BUILT"
				localization = "ag_beta_station_type1_built_message_desc"
				days = 30
				target = from
				variable = {
					type = name
					localization = "ag_megastructure_system"
					scope = fromfrom.solar_system
				}
			}
		}
		from = {
			change_variable = {
				which = ag_num_beta_station_type1_cores
				value = 1
			}
		}
		remove_megastructure = fromfrom
	}
	
	construction_blocks_and_blocked_by = self_type
	
	ai_weight = {
		weight = 0
	}
}

		set_star_flag = ag_psionic_inhibitor_platform_in_system
		if = {
			limit = { NOT = { any_ship_in_system = { is_ship_size = ag_psionic_inhibition_station } } }
			space_owner = { save_event_target_as = ag_station_owner }
			random_system_planet = {
				limit = { is_primary_star = yes }
				create_fleet = {
					name = "NAME_ag_psionic_inhibition_station_fleet"
					settings = { spawn_debris = no }
					effect = {
						set_owner = event_target:ag_station_owner
						create_ship = {
							name = "NAME_ag_psionic_inhibition_station_ship"
							design = "NAME_ag_psionic_inhibition_station"
							prefix = no
							effect = {
								ag_difficulty_bonus = yes
							}
						}
						set_location = {
							target = root.fromfrom
							distance = 0
							angle = random
						}
						fleet_event = { id = ag_beta.52 days = 180 random = 60 }
					}
				}
			}
		}
		remove_megastructure = fromfrom
	}
	
	construction_blocks_and_blocked_by = self_type
		
	ai_weight = {
		weight = 0
		modifier = {
			weight = 100
			OR = {
				has_star_flag = ag_beta_psionic_system_3
				has_star_flag = ag_beta_psionic_system_2
				has_star_flag = ag_beta_psionic_system_1
			}
			any_system_planet = { is_colony = yes }
		}
		modifier = {
			weight = 1000
			OR = {
				has_star_flag = ag_beta_psionic_system_3
				has_star_flag = ag_beta_psionic_system_2
				has_star_flag = ag_beta_psionic_system_1
			}
			any_system_planet = { is_capital = yes }
		}
	}
