planet_ring_g_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	build_time = 900
	resources = {
		category = megastructures
		cost = {
			alloys = 1500
			influence = 75
		}

		upkeep = {
			energy = 5
			alloys = 5
		}
	}
	
	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
	}
	
	potential = {
		has_technology = tech_planet_rings
		no_scope = {
			NOT = {
				has_global_flag = planet_rings_disabled
			}
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_surveyed_planet"
				is_surveyed = {			# prevent leaking habitability information
					who = prev.from
					status = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				#can_build_megastructure_on_planet = yes
				NOR = {
					solar_system = {
						has_star_flag = ring_world_built
					}
					has_planet_flag = has_megastructure
					is_planet_class = pc_ringworld_habitable
					is_planet_class = pc_habitat
					is_planet_class = pc_gigaorbital_ring
					is_planet_class = pc_habitable_gas_giant
					is_planet_class = pc_interstellar_habitat
					is_planet_class = pc_interstellar_ringworld_habitable
					is_planet_class = pc_giga_ringworld_habitable
					is_planet_class = pc_ringworld_habitable_damaged
					is_planet_class = pc_ringworld_tech
					is_planet_class = pc_ringworld_tech_damaged
					is_planet_class = pc_ringworld_seam
					is_planet_class = pc_ringworld_seam_damaged
					is_planet_class = pc_birch
				}
			}

 			# balance for habitats
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
				}
			}
			custom_tooltip = {
				fail_text = "must_build_around_habitable_planet"
				is_colony = yes
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
			custom_tooltip = {
				fail_text = "must_not_have_ring_already"
				NOT = {
					has_planet_flag = has_giga_planet_ring
				}
			}
		} # use these for all non-star megastructures
	}
	
	construction_blocks_others = no
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_giga_planet_ring
		}
	}
}
