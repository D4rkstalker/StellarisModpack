htbd_hole_stabilizing = {
	entity = "htbd_wormhole_omni_entity"
	construction_entity = ""
	portrait = "GFX_htbd_hole_background"
	place_entity_on_planet_plane = no
	build_outside_gravity_well = no
	show_galactic_map_icon = no

	entity_offset = { x = 0 y = 0 }

	potential = {
		has_technology = tech_htbd_hole_sustenance
	}

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "htbd_requires_ether_hole"
			solar_system = {
				any_system_megastructure = {
					OR = {
						is_megastructure_type = htbd_hole
						is_megastructure_type = htbd_hole_end
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "htbd_requires_hole_sustenance"
			from = { has_technology = tech_htbd_hole_sustenance }
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "htbd_requires_ether_hole_star"
				has_planet_flag = htbd_ether_hole_star
			}
			custom_tooltip = {
				fail_text = "htbd_requires_good_star"
				htbd_etherhole_source = yes
			}
		}
	}

	build_time = 600
	resources = {
		category = megastructures
		cost = {
			alloys = 2000
			energy = 4000
		}
	}
	construction_blocks_others = no

	bypass_type = htbd_pass

	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		# replace ether hole with our one
		every_system_megastructure = {
			limit = {
				OR = {
					is_megastructure_type = htbd_hole
					is_megastructure_type = htbd_hole_end
				}
			}
			remove_megastructure = this
		}
		if = {
			limit = { has_star_flag = htbd_pass_to }
			htbd_spawn_stable_hole_end = yes
		}
		else = {
			htbd_spawn_stable_hole = yes
		}
		remove_star_flag = htbd_pass_stay
		set_star_flag = htbd_pass_stay
		remove_megastructure = fromfrom
	}
}
