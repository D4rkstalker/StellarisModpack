ultra_shipyard_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	construction_blocks_others = no
	place_entity_on_planet_plane = no
	entity_offset = { x = -7 y = -7 }
	build_time = 500
	resources = {
		category = megastructures
		cost = {
			alloys = 4000
			influence = 300
		}
		
		upkeep = {
			energy = 10
		}
	}
	
	potential = {
		has_ascension_perk = ap_hyper_engineering
		has_technology = tech_ultra_shipyard
		no_scope = {
			NOT = {
				has_global_flag = ultrashipyard_disabled
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_ultra_shipyard"
			from = { NOT = { has_country_flag = built_ultrashipyard } }
		}
		custom_tooltip = {
			fail_text = "requires_no_colonized_planets"
			NOT = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_big_star"
			OR = {
				is_star_class = sc_b
				is_star_class = sc_a
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			AND = {
				has_no_non_gate_megastructure = yes
				NOT = {
					any_system_planet = {
						OR = {
							has_planet_flag = has_giga
							has_planet_flag = has_megastructure
						}
					}
				}
			}
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = {
					solar_system = {
						has_star_flag = has_giga
					}
				}
			}
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
		}
	}
	
	ai_weight = {
		factor = 15

		modifier = {
			factor = 0.1
			from = {
				OR = {
					has_megastructure = ultra_shipyard_0
					has_megastructure = ultra_shipyard_1
					has_megastructure = ultra_shipyard_2
					has_megastructure = ultra_shipyard_3
					has_megastructure = ultra_shipyard_4
				}
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = has_giga
		from = { set_country_flag = built_ultrashipyard }
	}
}
