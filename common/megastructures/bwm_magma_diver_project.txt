@phase_0_entity_x = 0
@phase_0_entity_y = 0
@entity_x = 0
@entity_y = 7

bwm_magma_diver_project_00 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	build_outside_gravity_well = no
	entity_offset = { x = @phase_0_entity_x y = @phase_0_entity_y }
	build_time = 360

	resources = {
		category = megastructures
		cost = {
			alloys = 6000
			energy = 1000
			influence = 50
		}

		upkeep = {
			energy = 10
		}
	}
	
	prerequisites = { "tech_bwm_magma_diver" }

	potential = {
		OR = {
			is_ai = no
			has_country_flag = bwm_ai_tech_toggle_3
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_magma_diver"
			from = {
				NOT = {
					any_owned_ship = {
						OR = {
							is_ship_size = bwm_conduit_avatar
							is_ship_size = bwm_conduit_avatar_bound
						}
					}
					any_owned_megastructure = {
						OR = {
							is_megastructure_type = bwm_magma_diver_project_00
							is_megastructure_type = bwm_magma_diver_project_01
						}
					}
				}
			}
		}
	}
	
	placement_rules = {	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure
		}
		from = {
			if = {
				limit = {
					NOT = {
						has_country_flag = bwm_magma_diver_builder
					}
				}
				set_country_flag = bwm_magma_diver_builder
			}
		}
	}
}

bwm_magma_diver_project_01 = {
	entity = "bwm_conduit_avatar_construction_entity"
	construction_entity = "bwm_conduit_avatar_construction_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	entity_offset = { x = @entity_x y = @entity_y }
	upgrade_from = {
		bwm_magma_diver_project_00
	}
	build_time = 1800
	resources = {
		category = megastructures
		cost = {
			alloys = 24000
			energy = 5000
			influence = 50
		}

		upkeep = {
			energy = 20
		}
	}
	
	potential = {
		OR = {
			is_ai = no
			has_country_flag = bwm_ai_tech_toggle_3
		}
	}

	possible = {
		from = {
			OR = {
				has_country_flag = bwm_magma_diver_builder
				has_galactic_wonders = yes
			}
		}
	}
	
	on_build_complete = {
		create_fleet = {
			name = "Avatar"
			settings = {
				spawn_debris = no
				is_boss = yes
			}
			effect = {
				set_owner = from
				create_ship = {
					name = "Conduit"
					design = "Matet"
					graphical_culture = fromfrom
					prefix = no
				}
				set_location = {
					target = fromfrom
					distance = 0
					angle = 0
				}
			}
		}
		fromfrom = {
			create_ambient_object = {
				type = bwm_effect_explosion_4_object
				location = THIS
				duration = 10
					
				use_3d_location = yes
					
				entity_offset = { min = 0 max = 0 }
				entity_offset_angle = { min = 0 max = 0 }
				entity_offset_height = { min = 0 max = 0 }
					
				entity_scale_to_size = yes
				scale = 100
			}
		}
		from = {
			if = {
				limit = {
					has_country_flag = bwm_magma_diver_builder
				}
				remove_country_flag = bwm_magma_diver_builder
			}
		}
		remove_megastructure = fromfrom
		fromfrom.planet = {
			remove_planet_flag = has_megastructure
		}
	}
}
