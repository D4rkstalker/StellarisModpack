acemod_sfex_jumpdrive_order_calculate_effective_range_effect = {
	optimize_memory
	set_variable = {
		which = acemod_sfex_fleet_order_var_jumpdrive_effective_range
		value = acemod_sfex_fleet_order_var_jumpdrive_custom_max_range_set
	}
	every_owned_ship = {
		export_modifier_to_variable = {
			modifier = ship_ftl_jumpdrive_range_mult
			variable = acemod_sfex_jumpdrive_order_var_ship_ftl_jumpdrive_range_mult
		}
	}
	ordered_owned_ship = {
		position = 0
		order_by = acemod_sfex_jumpdrive_order_var_ship_ftl_jumpdrive_range_mult
		inverse = yes
		fleet = {
			set_variable = {
				which = acemod_sfex_fleet_order_var_fleet_ftl_jumpdrive_range_mult
				value = acemod_sfex_fleet_order_var_jumpdrive_effective_range
			}
			multiply_variable = {
				which = acemod_sfex_fleet_order_var_fleet_ftl_jumpdrive_range_mult
				value = prev.acemod_sfex_jumpdrive_order_var_ship_ftl_jumpdrive_range_mult
			}
			change_variable = {
				which = acemod_sfex_fleet_order_var_jumpdrive_effective_range
				value = acemod_sfex_fleet_order_var_fleet_ftl_jumpdrive_range_mult
			}
			clear_variable = acemod_sfex_fleet_order_var_fleet_ftl_jumpdrive_range_mult
		}
	}
	every_owned_ship = {
		clear_variable = acemod_sfex_jumpdrive_order_var_ship_ftl_jumpdrive_range_mult
	}
}

acemod_sfex_jumpdrive_order_apply_cooldown_effect = {
	optimize_memory
	if = {
		limit = {
			NOR = {
				has_global_flag = acemod_sfex_jumpdrive_order_global_flag_jump_drive_cooldown_disabled
				owner = {
					has_country_flag = acemod_sfex_jumpdrive_order_country_flag_jump_drive_cooldown_disabled
				}
				has_fleet_flag = acemod_sfex_jumpdrive_order_fleet_flag_jump_drive_cooldown_disabled
			}
		}
		acemod_sfex_core_create_global_country_if_needed = yes
		event_target:acemod_sfex_core_global_country = {
			if = {
				limit = {
					NOT = {
						is_variable_set = acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
					}
				}
				set_variable = {
					which = acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
					value = 200
				}
			}
		}
		owner = {
			if = {
				limit = {
					NOT = {
						is_variable_set = acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
					}
				}
				set_variable = {
					which = acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
					value = event_target:acemod_sfex_core_global_country.acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
				}
			}
		}
		every_owned_ship = {
			add_modifier = {
				modifier = acemod_sfex_jump_drive_order_cooldown
				days = 1
				time_multiplier = owner.acemod_sfex_jumpdrive_order_var_acemod_sfex_jump_drive_cooldown_time
			}
		}
	}
}
