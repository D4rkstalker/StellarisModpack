
ag_lost_ancient_ship_diplo_portrait_check = {
	optimize_memory
	if = {
		limit = { NOT = { exists = event_target:ag_lost_ancient_ship_diplo_portrait } }
		event_target:global_event_country = {
			ag_lost_ancient_ship_leader_sci_spawn = { ag_save_global_event_target = no }
			last_created_leader = { save_global_event_target_as = ag_lost_ancient_ship_diplo_portrait }
		}
	}
	event_target:ag_lost_ancient_ship_diplo_species = {
		if = { limit = { exists = event_target:ag_lost_ancient_ship_owner event_target:ag_lost_ancient_ship_owner = { has_country_flag = SRA_ag_StarLightII_changed } } change_species_portrait = ag_StarlightII_anim_portrait }
		else = { change_species_portrait = ag_ancient_ai_red_alt }
	}
	event_target:ag_lost_ancient_ship_diplo_portrait = {
		if = { limit = { exists = event_target:ag_lost_ancient_ship_owner event_target:ag_lost_ancient_ship_owner = { has_country_flag = SRA_ag_StarLightII_changed } } change_leader_portrait = ag_StarlightII_anim_portrait }
		else = { change_leader_portrait = ag_ancient_ai_red_alt }
	}
	if = {
		limit = { exists = event_target:ag_starlight_2_ship event_target:ag_starlight_2_ship = { is_scope_type = leader } }
		event_target:ag_starlight_2_ship = {
			if = { limit = { exists = event_target:ag_lost_ancient_ship_owner event_target:ag_lost_ancient_ship_owner = { has_country_flag = SRA_ag_StarLightII_changed } } change_leader_portrait = ag_StarlightII_anim_portrait }
			else = { change_leader_portrait = ag_ancient_ai_red_alt }
		}
	}
}
# 1 ~ 4
ag_set_anim_StarlightII_clothes_diff = {
	optimize_memory
	ag_lost_ancient_ship_diplo_portrait_check = yes
	event_target:ag_lost_ancient_ship_diplo_portrait = { set_variable = { which = ag_anim_StarlightII_clothes_diff value = $ag_index$ } }
}

	optimize_memory
	if = {
		limit = { NOT = { exists = event_target:ag_epsilon_titan_diplo_portrait } }
		event_target:global_event_country = {
			ag_epsilon_titan_leader_adm_spawn = { ag_save_global_event_target = no }
			last_created_leader = { save_global_event_target_as = ag_epsilon_titan_diplo_portrait }
		}
	}
	event_target:ag_epsilon_titan_diplo_species = {
		if = { limit = { always = no } change_species_portrait = ag_StarInAurora_anim_portrait }
		else = { change_species_portrait = ag_ancient_ai_red_alt }
	}
	event_target:ag_epsilon_titan_diplo_portrait = {
		if = { limit = { always = no } change_leader_portrait = ag_StarInAurora_anim_portrait }
		else = { change_leader_portrait = ag_ancient_ai_red_alt }
	}

		limit = { exists = event_target:ag_epsilon_titan_leader event_target:ag_epsilon_titan_leader = { is_scope_type = leader } }
		event_target:ag_epsilon_titan_leader = {
			if = { limit = { always = no } change_leader_portrait = ag_StarlightII_anim_portrait }
			else = { change_leader_portrait = ag_ancient_ai_red_alt }

	}
}
