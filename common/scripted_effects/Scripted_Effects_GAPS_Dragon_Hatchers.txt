Country_Type_GAPS_Pacified_Dragon
# Checks if Mining Drone country (global event target) exists
# Creates the country and saves it if it does not
Create_Pacified_Dragon_Country = {
	if = {
		limit = { NOT = { exists = event_target:Global_Target_GAPS_Pacified_Dragon_Country } }
		create_country = {
			name = "NAME_Mineral_Extraction_Operation"
			type = Country_Type_GAPS_Pacified_Dragon
			flag = {
				icon = {
					category = "zoological"
					file = "flag_zoological_5.dds"
				}
				background= {
					category = "backgrounds"
					file = "00_solid.dds"
				}
				colors={
					"red"
					"red"
					"null"
					"null"
				}
			}
			effect = {
				save_global_event_target_as = Global_Target_GAPS_Pacified_Dragon_Country
			}
		}
	}
}
