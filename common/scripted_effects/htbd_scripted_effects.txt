		type = htbd_horde

		flag = {
			icon = {
				category = "zoological"

				category = "backgrounds"
				file = "00_solid.dds"
			}

				"black"
				"orange"

		}
	}
	last_created_country = {
		set_country_flag = htbd_country
		add_modifier = { modifier = strategic_coordination_center_modifier days = -1 }
		save_global_event_target_as = htbd_country
	}
}

htbd_set_pass_timer = {
	if = {
		limit = {
			root = { is_ship_size = space_dragon_grand }
		}
		set_timed_star_flag = { flag = htbd_pass_stay days = @htbd_pass_time_long }
	}
	else_if = {
		limit = {
			root = { is_ship_size = space_dragon_teen }
		}
		set_timed_star_flag = { flag = htbd_pass_stay days = @htbd_pass_time_short }
	}
	else = {
		set_timed_star_flag = { flag = htbd_pass_stay days = @htbd_pass_time }
	}
}

# this = country
htbd_set_neutral_countries = {
	# Enclave, Sphere and Xanophile Fallen calm dragons down
	every_country = {
		if = {
			limit = {
				htbd_is_neutral_faction = yes
			}
			set_faction_hostility = {
				target = event_target:htbd_country
				set_hostile = no
				set_neutral = yes
				set_friendly = no
			}
			set_country_flag = htbd_dragon_scent
		}
	}
}

htbd_stats_init = {
	if = {
		limit = {
			NOT = {
				any_country = {
					is_country_type = default
					has_event_chain = htbd_way_chain
					prev = {
						htbd_is_waypoint = yes
					}
				}
			}
		}
		# log = "Drop [This.GetName] system stats"
		set_variable = { which = htbd_system_like value = 0 }
	}
}

# Prev = System
htbd_add_waypoint = {
	prev = {
		#htbd_stats_init = yes
		if = {
			limit = { NOT = { has_star_flag = htbd_poi.@prev } }
			set_star_flag = htbd_poi.@prev
		}
	}
}

# This = System
htbd_del_waypoint = {
	if = {
		limit = { has_star_flag = htbd_poi.@prev }
		remove_star_flag = htbd_poi.@prev
	}

	# check system stats
	# log = "Check [This.GetName] system stats for [Prev.GetName]"
	if = {
		limit = {
			check_variable = { which = htbd_system_like value > 0 }
			prev = { NOT = { has_country_flag = htbd_liked_systems_done } }
		}
		enable_special_project = {
			name = HTBD_WAY_LIKE
			owner = prev
			location = event_target:way_point
		}
	}

	if = {
		limit = {
			check_variable = { which = htbd_system_like value < 0 }
			prev = { NOT = { has_country_flag = htbd_avoid_systems_done } }
		}
		enable_special_project = {
			name = HTBD_WAY_AVOID
			owner = prev
			location = event_target:way_point
		}
	}
}

#this = fleet
htbd_report_stay = {
	solar_system = {
		if = {
			limit = {
				root = { check_variable = { which = htbd_stay_level value > 20 } }
			}
			if = {
				limit = { htbd_debug = yes }
				log = "[Root.GetName] likes system [This.GetName]"
			}
			change_variable = { which = htbd_system_like value = 2 }
			random_system_planet = {
				limit = { is_star = yes }
				save_event_target_as = way_point
			}
			every_country = {
				limit = {
					is_country_type = default
					has_event_chain = htbd_way_chain
					intel_level = { level > medium system = prev }
					prev = { htbd_is_waypoint = yes	}
				}
				prev = { htbd_del_waypoint = yes }
			}
		}
	}
}

#this = fleet
htbd_report_leave = {
	solar_system = {
		if = {
			limit = {
				root = { check_variable = { which = htbd_stay_level value < 3 }}
			}
			if = {
				limit = { htbd_debug = yes }
				log = "[Prev.GetName] avoids system [This.GetName]"
			}
			change_variable = { which = htbd_system_like value = -2 }
		}
		else = {
			if = {
				limit = { htbd_debug = yes }
				log = "[Prev.GetName] likes system [This.GetName]"
			}
			change_variable = { which = htbd_system_like value = 3 }
		}
		random_system_planet = {
			limit = { is_star = yes }
			save_event_target_as = way_point
		}
		every_country = {
			limit = {
				is_country_type = default
				has_event_chain = htbd_way_chain
				OR = {
					prev = { is_inside_border = prev }
					intel_level = { level > medium system = prev }
				}
				prev = {
					count_ship_in_system = {
						count = 1
						limit = {
							exists = owner
							owner = { is_country_type = htbd_horde }
						}
					}
					htbd_is_waypoint = yes
				}
			}
			# log = "Delete waypoint for [This.GetName]"
			prev = { htbd_del_waypoint = yes }
		}
	}
}

	remove_planet_flag = htbd_dolly_egg

	remove_planet_flag = htbd_nano_egg

htbd_start_incubation = {
	if = {
		limit = { NOT = { has_planet_flag = egg_incubated } }
		set_timed_planet_flag = { flag = htbd_egg_embryo days = @incubation_embryo }
		set_planet_flag = egg_incubated
		event_target:htbd_country = { change_variable = { which = htbd_incubated_eggs value = 1 } }
	}
}

htbd_stop_incubation = {
	if = {
		limit = { has_planet_flag = egg_incubated }
		remove_planet_flag = htbd_egg_embryo
		remove_planet_flag = egg_incubated
		remove_planet_flag = htbd_hidden_egg
		event_target:htbd_country = { subtract_variable = { which = htbd_incubated_eggs value = 1 } }
	}
}

htbd_planet_depleted = {
	if = {
		limit = { planet_size > 4 }
		change_planet_size = -2
		if = {
			limit = {
				OR = {
					has_planet_flag = htbd_psion_egg
					has_planet_flag = htbd_dolly_egg
					has_planet_flag = htbd_cyborg_egg
					has_planet_flag = htbd_nano_egg
				}
			}
			change_planet_size = -1
		}
	}
	if = {
		limit = { planet_size < 5 }
		random = {
			chance = 70
			change_pc = pc_asteroid
			set_planet_size = 1
		}
	}
}

htbd_decision_effect = {
	event_target:decision_owner = { set_country_flag = htbd_hatching_done }
	random_list = {
		60 = {
			modifier = {
				factor = 0
				NOT = { exists = space_owner }
			}
			modifier = {
				factor = 0.4
				space_owner = { NOT = { is_same_value = event_target:decision_owner } }
			}
			# success, dwarf dragon
			space_owner = {
				if = {
					limit = {
						NOT = { has_country_flag = htbd_dragon_was_hatched }
						is_same_value = event_target:decision_owner
					}
					set_country_flag = htbd_dragon_was_hatched
				}
			}
			planet_event = { id = htbd_ev.999 }
		}
		30 = {
			modifier = {
				factor = 1.5
				planet_size < 8
			}
			modifier = {
				factor = 0.2
				event_target:decision_owner = { has_technology = tech_htbd_hatchery }
			}
			# no luck, no dragon
			event_target:decision_owner = { country_event = { id = htbd_ev.94 } }
		}
		40 = {
			modifier = {
				factor = 0.5
				event_target:decision_owner = { has_technology = tech_htbd_hatchery }
			}
			# bad luck, hostile dragon, need update for new ship size in behavior code
			if = {
				limit = { exists = space_owner }
				space_owner = { country_event = { id = htbd_ev.95 } }
			}
			else = {
				event_target:htbd_country = { country_event = { id = htbd_ev.95 } }
			}
		}
	}
}
