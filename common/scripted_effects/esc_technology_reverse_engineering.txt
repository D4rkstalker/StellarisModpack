	## bio armor 1 >>> biological weapons
	if = {
		limit = {
			has_technology = esc_tech_bio_armor_1
			ESC_TRIGGER_may_reverse_engineer_biological_weapons = yes
			NOT = { has_technology = esc_tech_biological_weapons }
		}
		give_technology = { tech = esc_tech_biological_weapons }
	}
	## armor 7 >>> armor 6
	if = {
		limit = {
			has_technology = esc_tech_armor_7
			NOT = { has_technology = esc_tech_armor_6 }
		}
		give_technology = { message = no tech = esc_tech_armor_6 }
	}
	## dragonscale armor >>> armor 5
	## armor 6 >>> armor 5

		limit = {
			has_technology = esc_tech_psionic_cannon_2
			NOT = { has_technology = esc_tech_psionic_cannon_1 }
			ESC_TRIGGER_may_reverse_engineer_psionic_weapons = yes
		}
		give_technology = { message = no tech = esc_tech_psionic_cannon_1 }
	}
	## psionic torpedo 2 >>> psionic torpedo 1
	if = {
		limit = {
			has_technology = esc_tech_psionic_torpedo_2
			NOT = { has_technology = esc_tech_psionic_torpedo_1 }
			ESC_TRIGGER_may_reverse_engineer_psionic_weapons = yes
		}
		give_technology = { message = no tech = esc_tech_psionic_torpedo_1 }
	}
	## psionic cannon 1 >>> psionic weapons
	## psionic torpedo 1 >>> psionic weapons
	if = {
		limit = {
			OR = {
				has_technology = esc_tech_psionic_torpedo_1
				has_technology = esc_tech_psionic_cannon_1
			}
			NOT = { has_technology = esc_tech_psionic_weapons }
			ESC_TRIGGER_may_reverse_engineer_psionic_weapons = yes
		}
		give_technology = { message = no tech = esc_tech_psionic_weapons }
	}
}
