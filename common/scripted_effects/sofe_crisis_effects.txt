		leader_age_max = 45
		traits = { trait = random_trait }
	}
	create_fleet = {
		name = "NAME_Origin_Auxiliary_Fleet"
		effect = {
			set_owner = event_target:origin_crisis_country
			set_fleet_flag = horde_auxuliaries
			while = {
				count = 4
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_ESCORT"
					prefix = no
				}
			}
			while = {
				count = 1
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_BATTLECRUISER"
					prefix = no
				}
			}
			assign_leader = last_created_leader
			set_location = {
				target = root
				distance = 10
				angle = random
			}
		}
		settings = { 
			can_upgrade = no 
			can_disband = no
			is_boss = yes
			spawn_debris = yes
