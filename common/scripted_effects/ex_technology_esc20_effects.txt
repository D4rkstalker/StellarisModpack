armor_tech_check = {
	## shielding armor 3 >>> shielding armor 2
	if = {
		limit = {
			has_technology = "nhsc_tech_armor_shield_3"
			NOT = { has_technology = "nhsc_tech_armor_shield_2" }
		}
		give_technology = { message = no tech = nhsc_tech_armor_shield_2 }
	}
	## shielding armor 2 >>> shielding armor 1
	if = {
		limit = {
			has_technology = "nhsc_tech_armor_shield_2"
			NOT = { has_technology = "nhsc_tech_armor_shield_1" }
		}
		give_technology = { message = no tech = nhsc_tech_armor_shield_1 }
	}
	## nanite armor 2 >>> nanite armor 1
	if = {
		limit = {
			has_technology = "nhsc_tech_armor_nanite_2"
			NOT = { has_technology = "nhsc_tech_armor_nanite_1" }
		}
		give_technology = { message = no tech = nhsc_tech_armor_nanite_1 }
	}
	## armor 7 >>> armor 6
	if = {
		limit = {
			has_technology = "nhsc_tech_armor_7"
			NOT = { has_technology = "nhsc_tech_armor_6" }
		}
		give_technology = { message = no tech = nhsc_tech_armor_6 }
	}
	## dragonscale armor >>> armor 5
	## armor 6 >>> armor 5
	## shieldsing armor 1 >>> armor 5
	## nanite armor 1 >>> armor 5
	if = {
		limit = {
			OR = {
				has_technology = "tech_dragon_armor"
				has_technology = "nhsc_tech_armor_6"
				has_technology = "nhsc_tech_armor_nanite_1"
				has_technology = "nhsc_tech_armor_shield_1"
			}
			NOT = { has_technology = "tech_ship_armor_5" }
		}
		give_technology = { message = no tech = tech_ship_armor_5 }
	}
	## nanite armor OR shielding armor >>> sythetic lifeform studies [since they require nanites]
	if = {
		limit = {
			OR = {
				has_technology = "nhsc_tech_armor_nanite_1"
				has_technology = "nhsc_tech_armor_shield_1"
			}
			NOT = { has_technology = "nhsc_tech_synthetic_lifeform_studies" }
		}
		give_technology = { message = no tech = nhsc_tech_synthetic_lifeform_studies }
		## synthetic lifeform studies >>> nanite transmutation [...useful?]
#		if = {
#			limit = {
#				NOT = { has_technology = "tech_nanite_transmutation" }
#			}
#			give_technology = { message = no tech = tech_nanite_transmutation }
#		}
	}
	## armor 5 >>> armor 4
	if = {
		limit = {
			has_technology = "tech_ship_armor_5"
			NOT = { has_technology = "tech_ship_armor_4" }
		}
		give_technology = { message = no tech = tech_ship_armor_4 }
	}
	## armor 4 >>> armor 3
	if = {
		limit = {
				has_technology = "tech_ship_armor_4"
				NOT = { has_technology = "tech_ship_armor_3" }
