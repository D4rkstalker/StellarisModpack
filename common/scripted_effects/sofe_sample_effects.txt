create_origin_scout_fleet = {
	owner = {
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			traits = {
				trait = leader_trait_admiral_ancient_knowledge
				trait = leader_trait_admiral_bionic_implant
				trait = leader_trait_admiral_master_mind
			}
			skill = 10
		}
		last_created_leader = { set_age = 0 }
	}
	create_fleet = {
		settings = {
             spawn_debris = no 
			 can_upgrade = no 
			 uses_naval_capacity = no
	}
		effect = {
			set_owner = root
			while = {
				count = 20
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_ESCORT"
					upgradable = no
				}
			}
			while = {
				count = 2
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_TITAN"
					upgradable = no
				}
			}
			while = {
				count = 1
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_AVATAR"
					upgradable = no
				}
			}
			assign_leader = last_created_leader
			while = {
				count = 10
				create_ship = {
					name = random
					design = "NAME_PROGENITOR_BATTLECRUISER"
					upgradable = no
				}
			}
			set_location = {
				target = prev
				distance = 45
				angle = random
			}
		}
	}
}

create_flagship = {
	owner = {
		create_leader = {
			class = admiral
			species = owner_main_species
			name = random
			traits = {
				trait = leader_trait_admiral_ancient_knowledge
				trait = leader_trait_admiral_bionic_implant
				trait = leader_trait_admiral_master_mind
			}
			skill = 10
		}
		last_created_leader = { set_age = 0 }
	}
	create_fleet = {
		settings = {
             spawn_debris = no 
			 is_boss = yes
			 can_upgrade = no 
			 can_disband = no
			 uses_naval_capacity = no
	}
		effect = {
			set_owner = root
			while = {
				count = 1
				create_ship = {
					name = random
					design = "ORIGIN_GAIA_MAIN"
					upgradable = no
					effect = {
						set_ship_flag = arkship_first_life
						set_disable_at_health = 0.01
					}
				}
			}
			assign_leader = last_created_leader
			set_location = {
				target = prev
				distance = 45
				angle = random
			}
		}
	}
}
