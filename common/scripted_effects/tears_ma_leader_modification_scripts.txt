tears_sam_add_leader_traits_after_modification = {
	tears_sam_add_leader_traits_intelligence = yes
	tears_sam_add_leader_traits_military = yes
}
tears_sam_remove_leader_traits_after_modification = {
	tears_sam_remove_leader_traits_intelligence = yes
	tears_sam_remove_leader_traits_military = yes
}

tears_sam_add_leader_traits_intelligence = {
	if = {
		limit = { 
			species = { 
				has_trait = tears_sam_trait_machine_intelligence 
			} 
		}
		if = {
			limit = { 
				leader_class = admiral 
				NOT = { has_trait = tears_sam_trait_admiral_machine_intelligence } 
			}
			add_trait = tears_sam_trait_admiral_machine_intelligence
			add_ruler_trait = tears_sam_trait_ruler_machine_intelligence
		}
		if = {
			limit = { 
				leader_class = general 
				NOT = { has_trait = tears_sam_trait_general_machine_intelligence } 
			}
			add_trait = tears_sam_trait_general_machine_intelligence
			add_ruler_trait = tears_sam_trait_ruler_machine_intelligence
		}
		if = {
			limit = { 
				leader_class = governor 
				NOT = { has_trait = tears_sam_trait_governor_machine_intelligence } 
			}
			add_trait = tears_sam_trait_governor_machine_intelligence
			add_ruler_trait = tears_sam_trait_ruler_machine_intelligence
		}
		if = {
			limit = { 
				leader_class = scientist 
				NOT = { has_trait = tears_sam_trait_scientist_machine_intelligence } 
			}
			add_trait = tears_sam_trait_scientist_machine_intelligence
			add_ruler_trait = tears_sam_trait_ruler_machine_intelligence
		}
		if = {
			limit = { 
				leader_class = ruler 
				NOT = { has_trait = tears_sam_trait_ruler_machine_intelligence } 
			}
			add_trait = tears_sam_trait_ruler_machine_intelligence
			add_ruler_trait = tears_sam_trait_ruler_machine_intelligence

		}
	}
}
