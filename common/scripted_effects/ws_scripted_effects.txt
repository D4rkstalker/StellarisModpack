######### Hyperlane effect######

create_wild_space_hyperlane = {
	every_system = {
	limit = {
		is_star_class = sc_dn
	}
	every_neighbor_system = {
			ignore_hyperlanes = yes
			limit = {
				NOT = {
					has_hyperlane_to = PREV
				}
				is_star_class = sc_dn
			}
			add_hyperlane = { 
				from = PREV
				to = THIS 
			}
		}
	}
}
