origin_nsc_freebooter_start_setup = {
	set_planet_entity = {
		entity = "habitat_phase_03_entity"
		graphical_culture = space_owner
	}
	solar_system = {
		every_system_planet = {
			limit = { has_planet_flag = mining_planet }
			create_mining_station = { owner = space_owner }
		}
	}
	#add_deposit = d_nsc_freebooter_home #disable this code for now
	add_deposit = d_nsc_freebooter_habitat
	while = {
		count = 28 #changed to match that of the vanilla
		create_pop = { species = owner.species }
	}
	add_building = building_capital
	add_building = building_bureaucratic_1
	add_building = building_research_lab_1
	add_building = building_foundry_1
	while = {
		count = 2
		add_district = district_nsc_freebooter_hab_housing
	}
	while = {
		count = 2
		add_district = district_nsc_freebooter_hab_foundry
	}
	if = {
		limit = { owner = { is_lithoid_empire = no } }
		while = { add_district = district_nsc_freebooter_hab_farming count = 2 }
		else = { while = { add_district = district_nsc_freebooter_hab_mining count = 2 } }
	}
	add_district = district_nsc_freebooter_hab_factory
	owner = {
		give_technology = { tech = tech_nsc_freebooter_boarding_pod_1 message = no }
		if = {
			limit = {
				OR = {
					has_valid_civic = nsc_freebooter_civic_freebooter_trait_change
					has_valid_civic = nsc_freebooter_civic_freebooter_trait_change_corp
				}
			}
			every_owned_pop = {
				limit = { is_same_species = owner.species }
				modify_species = {
					species = this
					remove_trait = trait_nsc_freebooter_1
					add_trait = trait_nsc_freebooter_1_alt
				}
			}
		}
	}
}
nsc_freebooter_empire_admiral_market_target_selection = {
	random_country = {
		limit = {
			is_ai = yes
			is_country_type = default
			NOT = { is_at_war_with = root }
			has_resource = { type = energy amount >= 200 }
		}
		if = {
			limit = { is_at_war = yes }
			random_war = {
				limit = {
					any_war_participant = {
						is_war_leader = yes
						OR = {
							relative_power = {
								who = this
								category = fleet
								value = equivalent
							}
							relative_power = {
								who = this
								category = fleet
								value = superior
							}
							relative_power = {
								who = root
								category = fleet
								value = overwhelming
							}
						}
					}
				}
				random_war_participant = {
					limit = {
						is_war_leader = yes
						OR = {
							relative_power = {
								who = this
								category = fleet
								value = equivalent
							}
							relative_power = {
								who = this
								category = fleet
								value = superior
							}
							relative_power = {
								who = root
								category = fleet
								value = overwhelming
							}
						}
					}
					save_event_target_as = nsc_freebooter_customer_empire_biggest_enemy
				}
			}
		}
		weights = {
			# can use add or mult
			base = 5
			modifier = {
				add = 5
				is_at_war = yes
			}
			modifier = {
				add = 20
				is_at_war = yes
				OR = {
					relative_power = {
						who = event_target:nsc_freebooter_customer_empire_biggest_enemy
						category = fleet
						value = inferior
					}
					relative_power = {
						who = event_target:nsc_freebooter_customer_empire_biggest_enemy
						category = fleet
						value = pathetic
					}
				}
			}
			modifier = {
				add = 10
				num_fleets >= 5
			}
			modifier = {
				add = 10
				OR = {
					relative_power = {
						who = root
						category = fleet
						value = inferior
					}
					relative_power = {
						who = root
						category = fleet
						value = pathetic
					}
				}
			}
			modifier = {
				add = 20
				any_owned_fleet = { NOT = { exists = leader } }
			}
			modifier = {
				mult = 0.5
				has_country_flag = already_bought_admiral_from_nsc_freebooter_empire_before
			}
		}
		save_event_target_as = nsc_freebooter_empire_admiral_market_target
	}
}
nsc_freebooter_admiral_market_selected_target_count_effect = {
	#clear the variable and the leader flag for recounting
	set_variable = {
		which = nsc_freebooter_admiral_market_selected_target_count
		value = 0
	}
	every_owned_leader = {
		limit = { has_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag }
		remove_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
	}
	while = {
		count = 6
		change_variable = {
			which = nsc_freebooter_admiral_market_selected_target_count
			value = 1
		}
		random_owned_leader = {
			limit = {
				leader_class = admiral
				NOR = {
					has_trait = nsc_freebooter_admiral_trait_great_khan #no, we can't sell the khan lolol
					has_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
				}
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 1 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_1 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 2 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_2 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 3 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_3 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 4 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_4 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 5 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_5 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
			if = {
				limit = { root = { check_variable = { which = nsc_freebooter_admiral_market_selected_target_count value = 6 } } }
				save_event_target_as = nsc_freebooter_admiral_market_selected_leader_6 set_leader_flag = nsc_freebooter_admiral_market_selected_leader_flag
			}
		}
	}
}

			10 = { root = { add_resource = { energy = 2000 influence = 500 } } event_target:nsc_freebooter_empire_admiral_market_target = { add_resource = { energy = -2000 } } }
		}
		#if the player uses some kinds of mod to increase max leader level...
		if = {
			limit = { has_skill > 10 }
			root = { add_resource = { energy = 2200 influence = 550 } } event_target:nsc_freebooter_empire_admiral_market_target = { add_resource = { energy = -2200 } }

	if = {

		event_target:nsc_freebooter_empire_admiral_market_target = {
			set_country_flag = already_bought_admiral_from_nsc_freebooter_empire_before
			clone_leader = { target = event_target:nsc_freebooter_admiral_market_selected_leader_1 class = admiral }
			event_target:nsc_freebooter_admiral_market_selected_leader_1 = { kill_leader = { show_notification = no } }
			if = {
				limit = { has_country_flag = nsc_freebooter_empire_admiral_market_target_pay_double }
				add_resource = { energy = -1000 }
				remove_country_flag = nsc_freebooter_empire_admiral_market_target_pay_double
			}
		}
	}
}

nsc_freebooter_planet_raiding_effect_1 = {
	if = {
		limit = {
			root = {
				OR = {
					colony_type = col_generator
					colony_type = col_habitat_energy
					colony_type = col_generator
				}
			}
		}
		add_resource = { energy = 50 }
		if = {
			limit = { NOT = { is_at_war_with = root.owner } }
			create_message = {
				type = NSC_FREEBOOTERS_PIRATE_RAID_MESSAGE_TYPE #the notification message for a freebooter empire
				localization = NSC_FREEBOOTER_PIRATE_RAIDING_DONE_ENERGIES
				days = 15
				target = root
				variable = {
					type = name
					localization = NSC_FREEBOOTER_PIRATE_RAIDING_TARGET_PLANET
					scope = root
				}
			}
			else = {
				create_message = {
					type = NSC_FREEBOOTERS_PIRATE_RAID_MESSAGE_TYPE  #the notification message for a freebooter empire #alternative ver
					localization = NSC_FREEBOOTER_PIRATE_RAIDING_DONE_ENERGIES_ALT
					days = 15
					target = root
					variable = {
						type = name
						localization = NSC_FREEBOOTER_PIRATE_RAIDING_TARGET_PLANET
						scope = root
					}
				}
			}
		}
	}
	else_if = {
		limit = {
			root = {
				OR = {
					colony_type = col_mining
					colony_type = col_habitat_mining
				}
			}
		}
		add_resource = { minerals = 50 }

				type = NSC_FREEBOOTERS_PIRATE_RAID_MESSAGE_TYPE
				localization = NSC_FREEBOOTER_PIRATE_RAIDING_DONE_MINERALS
				days = 15
				target = root

					type = name
					localization = NSC_FREEBOOTER_PIRATE_RAIDING_TARGET_PLANET
					scope = root
				}
			}
			else = {
