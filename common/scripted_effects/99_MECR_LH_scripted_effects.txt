## 开始会议前阶段
# 开始会议前初始化每一项议案的得分
lower_house_next_resolution_init = {
	if = {
		limit = { any_country = { is_proposing_resolution = $resolution$ } }
		random_country = {
			limit = { is_proposing_resolution = $resolution$ }
			save_global_event_target_as = proposer_$resolution$
		}
		# scope: event_target:global_event_country
		set_variable = { which = lh_s_$resolution$ value = 0 }
	}
	else = {
		clear_variable = lh_s_$resolution$
	}
}
# MECRLH100初始化，根据评分预设支持的议案列表
lower_house_init_MECRLH100_resolution_s_or_o = {
	if = {
		limit = { any_country = { is_proposing_resolution = $resolution$ } }
		if = {
			limit = { check_variable = { which = lh_support value <= value:lhw_$resolution$ } }
			set_country_flag = lh_starting_$resolution$_support
		}
		else = {
			remove_country_flag = lh_starting_$resolution$_support
		}
	}
	else = {
		remove_country_flag = lh_starting_$resolution$_support
	}
}
# 选项效果改变支持或不支持
lower_house_change_resolution_s_or_o = {
	if = {
		limit = { has_country_flag = lh_starting_$resolution$_support }
		remove_country_flag = lh_starting_$resolution$_support
	}
	else = {
		set_country_flag = lh_starting_$resolution$_support
	}
}

# 将分数送入支持的议案总分
lower_house_add_score = {
	if = {
		limit = { has_country_flag = lh_starting_$resolution$_support }
		event_target:global_event_country = {
			change_variable = { which = lh_s_$resolution$ value = prev.value:LH_votes }
		}
		remove_country_flag = lh_starting_$resolution$_support
	}
}
# 开始会议，检查每一项议案的得票找出最高者
lower_house_next_resolution_check = {
	if = {
		limit = { any_country = { is_proposing_resolution = $resolution$ } }
		# scope: event_target:global_event_country
		if = {
			limit = { check_variable = { which = lh_next_resolution_score value < lh_s_$resolution$ } }
			set_variable = { which = lh_next_resolution_score value = lh_s_$resolution$ }
			remove_all_lh_next_resolution_flags = yes
			set_global_flag = lh_next_$resolution$
		}
		clear_variable = lh_s_$resolution$

## 会议进行中阶段
# 会议中检查进度
lower_house_voting_check = {
	# 计算分数
	event_target:global_event_country = {
		set_variable = { which = lh_recent_support value = 0 }
		set_variable = { which = lh_recent_oppose value = 0 }
	}
	every_galcom_member = {
		limit = {
			any_situation = {
				is_situation_type = situation_galactic_lower_house
				current_situation_approach = lower_house_approach_support
			}
		}
		hidden_effect = {
			event_target:global_event_country = {
				change_variable = { which = lh_recent_support value = prev.value:LH_votes }
			}
		}
		custom_tooltip = supporter_tooltip
	}
	every_galcom_member = {
		limit = {
			any_situation = {
				is_situation_type = situation_galactic_lower_house
				current_situation_approach = lower_house_approach_oppose
			}
		}
		hidden_effect = {
			event_target:global_event_country = {
				change_variable = { which = lh_recent_oppose value = prev.value:LH_votes }
			}
		}
		custom_tooltip = opposer_tooltip
	}
	every_galcom_member = {
		limit = {
			any_situation = {
				is_situation_type = situation_galactic_lower_house
				NOR = {
					current_situation_approach = lower_house_approach_support
					current_situation_approach = lower_house_approach_oppose
				}
			}
		}
		custom_tooltip = abstainer_tooltip
	}
	event_target:global_event_country = {
		set_variable = { which = lh_recent_all value = lh_recent_support }
		change_variable = { which = lh_recent_all value = lh_recent_oppose }
		if = {
			limit = { check_variable = { which = lh_recent_all value = 0 } }
			set_variable = { which = lh_recent_all value = 1 }
		}
		set_variable = { which = lh_recent_support_percentage value = lh_recent_support }
		divide_variable = { which = lh_recent_support_percentage value = lh_recent_all }
		multiply_variable = { which = lh_recent_support_percentage value = 100 }
		if = {
			limit = { check_variable = { which = lh_recent_support_percentage value = 0 } }
			set_variable = { which = lh_recent_support_percentage value = 0.1 }
		}
		else_if = {
			limit = { check_variable = { which = lh_recent_support_percentage value = 100 } }
			set_variable = { which = lh_recent_support_percentage value = 99.9 }
		}
	}
	# 调整局势
	hidden_effect = {
		every_galcom_member = {
			if = {
				limit = {
					NOT = { any_situation = { is_situation_type = situation_galactic_lower_house } }
				}
				start_situation = {
					type = situation_galactic_lower_house
					target = This
				}
			}
			every_situation = {
				limit = { is_situation_type = situation_galactic_lower_house }
				export_trigger_value_to_variable = {
					trigger = situation_progress
					variable = situation_progress_var
				}
				multiply_variable = { which = situation_progress_var value = -1 }
				add_situation_progress = situation_progress_var
				add_situation_progress = event_target:global_event_country.lh_recent_support_percentage
			}
		}
		every_country = {
			limit = {
				is_galactic_community_member = no
				any_situation = { is_situation_type = situation_galactic_lower_house }
			}
			every_situation = {
				limit = { is_situation_type = situation_galactic_lower_house }
				destroy_situation = This
			}
		}
	}
	# 生成投票权重
	monthly_set_lh_voting_weight = yes
}
# 每月重设投票权重，避免一直计算
monthly_set_lh_voting_weight = {
	every_galcom_member = {
		set_variable = { which = lhw_resolution_recent value = value:lhw_resolution_recent }
		change_variable = { which = lhw_resolution_recent value = lower_house_random }
		set_variable = { which = lhw_resolution_recent_display value = 100 }
		multiply_variable = { which = lhw_resolution_recent_display value = lhw_resolution_recent }
		round_variable = lhw_resolution_recent_display
		multiply_variable = { which = lhw_resolution_recent_display value = 0.01 }
	}
}
# 生成随机变动
generate_weight_random_change = {
	every_galcom_member = {
		if = {
			limit = { is_ai = yes }
			random_list = {
				1 = { set_variable = { which = lower_house_random value = 0.1 } }
				1 = { set_variable = { which = lower_house_random value = 0.2 } }
				1 = { set_variable = { which = lower_house_random value = 0.3 } }
				1 = { set_variable = { which = lower_house_random value = 0.4 } }
				1 = { set_variable = { which = lower_house_random value = 0.5 } }
				1 = { set_variable = { which = lower_house_random value = 0.6 } }
				1 = { set_variable = { which = lower_house_random value = 0.7 } }
				1 = { set_variable = { which = lower_house_random value = 0.8 } }
				1 = { set_variable = { which = lower_house_random value = 0.9 } }
				1 = { set_variable = { which = lower_house_random value = 1.0 } }
			}
			random_list = {
				1 = { set_variable = { which = lower_house_random_2 value = 0.1 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.2 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.3 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.4 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.5 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.6 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.7 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.8 } }
				1 = { set_variable = { which = lower_house_random_2 value = 0.9 } }
				1 = { set_variable = { which = lower_house_random_2 value = 1.0 } }
			}
			multiply_variable = { which = lower_house_random value = lower_house_random_2 }
			random_list = {
				1 = { multiply_variable = { which = lower_house_random value = 1 } }
				1 = { multiply_variable = { which = lower_house_random value = -1 } }
			}
			clear_variable = lower_house_random_2
		}
		else = {
			set_variable = { which = lower_house_random value = 0 }
		}
	}
}


## 会议结束阶段
# 通过或否决议案
lower_house_pass_resolution = {
	if = {
		limit = {
			any_federation = {
				has_federation_perk = galactic_confederation_passive
				has_federation_law = centralization_very_high_confederation
			}
		}
		pass_resolution_no_cooldown = $resolution$
	}
	else = {
		pass_resolution = $resolution$
	}
	every_galcom_member = {
		create_message = {
			type = MESSAGE_LOWER_HOUSE_RESOLUTION_PASSED
			localization = MESSAGE_LOWER_HOUSE_RESOLUTION_PASSED_DESC
			days = 12
		}
	}
	remove_global_flag = lh_recent_$resolution$
}

}
