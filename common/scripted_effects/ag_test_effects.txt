	set_variable = { which = ag_num_beta_station_type1_cores value = 2147438647 }
	# Gamma Area
	set_country_flag = ag_gamma_crisis_relic_actived
	if = {
		limit = { NOT = { has_global_flag = ag_has_gamma_relic_design } }
		set_global_flag = ag_has_gamma_relic_design
		create_ship_design = { design = "NAME_ag_gamma_battlecruiser_0" }
		last_created_design = { save_global_event_target_as = ag_gamma_battlecruiser_0_design }
		create_ship_design = { design = "NAME_ag_gamma_battlecruiser_1" }
		last_created_design = { save_global_event_target_as = ag_gamma_battlecruiser_1_design }
		create_ship_design = { design = "NAME_ag_gamma_escort_0" }
		last_created_design = { save_global_event_target_as = ag_gamma_escort_0_design }
		create_ship_design = { design = "NAME_ag_gamma_escort_1" }
		last_created_design = { save_global_event_target_as = ag_gamma_escort_1_design }
	}
	add_ship_design = event_target:ag_gamma_battlecruiser_0_design
	add_ship_design = event_target:ag_gamma_battlecruiser_1_design
	add_ship_design = event_target:ag_gamma_escort_0_design
	add_ship_design = event_target:ag_gamma_escort_1_design
	create_ship_design = { design = "NAME_ag_gamma_station_extended_shipyard" } add_ship_design = last_created_design
	# Delta Area
	ag_give_technology = { ag_tech = tech_ag_delta_platform }
	# Eta Area
	ag_give_technology = { ag_tech = tech_ag_ancient_eta_gateway }
	# Zeta Crisis
	set_country_flag = ag_zeta_crisis_relic_actived
	if = {
		limit = { NOT = { has_country_flag = ag_has_zeta_relic_design } }
		set_country_flag = ag_has_zeta_relic_design
		create_ship_design = { design = "NAME_ag_zeta_battleship" }
		add_ship_design = last_created_design
		create_ship_design = { design = "NAME_ag_zeta_destroyer" }
		add_ship_design = last_created_design
	}
	else = {
		create_ship_design = { design = "NAME_ag_zeta_battleship" } add_ship_design = last_created_design
