#Lithoid Supremacy Effects

add_leader_traits_after_modification_lithoid_supremacy = {
	#Add Sandstone Hewn (Governor)
	if = {
		limit = {
			species = {
				has_trait = trait_sandstone_hewn
			}
		}
		if = {
			limit = {
				leader_class = governor
				NOT = { has_trait = leader_trait_governor_sandstone }
			}
			add_trait = leader_trait_governor_sandstone
		}
	}
	#Add Chroma Stones (Admiral)
	else_if = {
		limit = { 
			species = {
				has_trait = trait_chroma_stones
			}
		}
		if = {
			limit = {
				leader_class = admiral
				NOT = { has_trait = leader_trait_admiral_chroma_stones }
			}
			add_trait = leader_trait_admiral_chroma_stones
		}
	}
	#Add Internal Ballistics (Admiral)
	else_if = {
		limit = { 
			species = {
				has_trait = trait_internal_ballistics
			}
		}
		if = {
			limit = {
				leader_class = admiral
				NOT = { has_trait = leader_trait_admiral_internal_ballistics }
			}
			add_trait = leader_trait_admiral_internal_ballistics
		}
	}
	#Add Zro Infused
	else_if = {
		limit = { 
			species = {
				has_trait = trait_zro_infused
			}
		}
		if = {
			limit = {
				leader_class = admiral
				NOT = { has_trait = leader_trait_admiral_zro_infused }
				NOT = { has_trait = leader_trait_admiral_zro_overload }
			}
			add_trait = leader_trait_admiral_zro_infused
		}
		if = {
			limit = {
				leader_class = governor
				NOT = { has_trait = leader_trait_governor_zro_infused }
				NOT = { has_trait = leader_trait_governor_zro_overload }
			}
			add_trait = leader_trait_governor_zro_infused
		}
		if = {
			limit = {
				leader_class = scientist
				NOT = { has_trait = leader_trait_scientist_zro_infused }
				NOT = { has_trait = leader_trait_scientist_zro_overload }
			}
			add_trait = leader_trait_scientist_zro_infused
		}
		if = {
			limit = {
				leader_class = general
				NOT = { has_trait = leader_trait_general_zro_infused }
				NOT = { has_trait = leader_trait_general_zro_overload }
			}
			add_trait = leader_trait_general_zro_infused
		}
	}
}
