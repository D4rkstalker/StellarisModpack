# change vanilla scripted effect
nuke_primitives_effect = {
	destroy_colony = yes
	set_planet_flag = primitives_nuked_themselves
	change_pc = pc_nuked
	reroll_planet = yes
	every_tile = {
		limit = { has_building = yes }
		if = {
			limit = {
				has_building = "building_primitive_farm"
			}
			remove_building = yes
			else = {
				set_ruined = yes
			}			
		}
	}
	set_planet_entity = { entity = nuked_planet_tomb_entity }
	set_planet_flag = ig_nuclear_fire_nuked_special
}
