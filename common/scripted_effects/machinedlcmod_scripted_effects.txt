calculate_servitor_morale = {
	if = {
		limit = { pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage >= 0.45 } }
		add_modifier = { modifier = servitor_morale_max days = -1 }
	}
	if = {
		limit = {
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage < 0.45 }
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage >= 0.3 }
		}
		add_modifier = { modifier = servitor_morale_high days = -1 }
	}
	if = {
		limit = {
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage < 0.3 }
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage >= 0.15 }
		}
		add_modifier = { modifier = servitor_morale_normal days = -1 }
	}
	if = {
		limit = {
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage < 0.15 }
			pop_percentage = { limit = { has_living_standard = { type = living_standard_organic_trophy } } percentage >= 0 }
		}
		add_modifier = { modifier = servitor_morale_low days = -1 }
	}
}
clear_mod_servitor_morale = {
	switch = {
		trigger = has_modifier
		servitor_morale_max = { remove_modifier = servitor_morale_max }
		servitor_morale_high = { remove_modifier = servitor_morale_high }
		servitor_morale_normal = { remove_modifier = servitor_morale_normal }
		servitor_morale_low = { remove_modifier = servitor_morale_low }
	}
}
