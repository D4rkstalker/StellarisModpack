enable_storm_cluster = {
	if = {
		limit = {
			NOT = { has_global_flag = scluster_allowed }
		}
		set_global_flag = scluster_allowed
	}
}

disable_storm_cluster = {
	if = {
		limit = {
			has_global_flag = scluster_allowed
		}
		remove_global_flag = scluster_allowed
	}
}

set_storm_cluster_variables = {
	event_target:global_event_country = {
		[[MAX_MEGASTRUCTURES]
			set_variable = {
				which = scluster_megastructures_max
				value = $MAX_MEGASTRUCTURES$
			}
		]
		[[MAX_GUARDIANS]
			set_variable = {
				which = scluster_guardians_max
				value = $MAX_GUARDIANS$
			}
		]
	}
}

start_space_storm = {
	event_target:global_event_country = {
		country_event = { id = galactic_features.401 }
	}
}

scluster_trigger_great_khan = {
	if = {
		limit = {
			NOT = { has_global_flag = marauder_crisis_ongoing }
		}
		random_country = {
			limit = {
				is_country_type = dormant_marauders
				has_country_flag = scluster_marauders
			}
			set_global_flag = horde_triggered
			country_event = {
				id = achievement.1
			}
		}
	}
}
