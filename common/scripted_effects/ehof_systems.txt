ehof_balance_system = {
	if = {
		limit = {
			OR = {
				is_star_class = sc_black_hole
				event_target:ehof_system_creator = { is_ai = yes }
			}
		}
		ehof_remove_habitable_planets = yes
	}
	else = {
		set_variable = { which = ehof_habitable_planets value = 0 }
		every_system_planet = {
			limit = { ehof_is_habitable = yes }
			prev = { change_variable = { which = ehof_habitable_planets value = 1 } }
		}

		random_list = {
			15 = { while = { limit = { check_variable = { which = ehof_habitable_planets value > 0 } } random_system_planet = { limit = { ehof_is_habitable = yes } ehof_change_planet = yes } } }

ehof_generate_gateways = {
	if = {
		limit = {
			ehof_compound_cluster = no
			OR = {
				ehof_trader_system = no
				AND = {
					ehof_trader_system = yes
					ehof_trader_hub = yes
				}
			}
		}
		if = {
			limit = {
				ehof_trader_system = no
				ehof_creator_system = no
			}
			if = {
				limit = {
					exists = event_target:ehof_system_creator
					event_target:ehof_system_creator = { has_technology = tech_ehof_use_gateway }
				}
				spawn_megastructure = {
					type = "ehof_gateway_disabled"
					location = solar_system
					orbit_angle = 225
					orbit_distance = 300
				}
			}
			else = {
				spawn_megastructure = {
					type = "ehof_gateway_unknown"
					location = solar_system
					orbit_angle = 225
					orbit_distance = 300
				}
			}
		}
		else = {
			if = {
				limit = { ehof_trader_hub = yes }
				spawn_megastructure = {
					type = "ehof_gateway"
					location = solar_system
					orbit_angle = 225
					orbit_distance = 300
					owner = event_target:ehof_trader_country
				}
			}
			else = {
				spawn_megastructure = {
					type = "ehof_gateway"
					location = solar_system
					orbit_angle = 225
					orbit_distance = 300
				}
			}

	}
