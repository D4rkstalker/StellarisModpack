### MANUALLY resources spawn scripts from console



### 1 ###

apsr_console_spawn_1 = {
	random_galaxy_planet = {
		limit = {
			AND = {
				NOT = { has_planet_flag = "apsr_planet" }
				NOT = { has_planet_flag = "legw_planet" } # MOD: Legendary Worlds
				NOT = { has_planet_flag = hatchling_egg }
				NOT = { has_planet_flag = gpm_is_precursor_world } # MOD: Guilli Planet Modifiers
				NOT = { has_planet_flag = gpm_is_wondrous_planet } # MOD: Guilli Planet Modifiers
				NOT = { has_planet_flag = gpm_ego_gaia_planet } # MOD: Guilli Planet Modifiers
				NOT = { has_global_flag = apsr1_spawned }
				is_colony = no
				is_capital = no
				planet_size > 9
				OR = {
					is_planet_class = pc_gaia		
					is_planet_class = pc_nuked
					is_planet_class = pc_continental
					is_planet_class = pc_alpine
					is_planet_class = pc_tropical
					is_planet_class = pc_ocean
					is_planet_class = pc_savannah
					is_planet_class = pc_arid
					is_planet_class = pc_desert
					is_planet_class = pc_arctic
					is_planet_class = pc_tundra
				}
				solar_system = {
					NOT = {
						any_system_planet = {
							has_apsr_modifiers = yes
						}	
					}
					NOT = {
						any_system_megastructure = { is_megastructure_type = dyson_sphere_ruined }
					}
				} 
			}
		}
		apsr_change_ancient_not_complex_type = yes
		add_modifier = { modifier = apsr1_planet_modifier days = -1 }
		add_deposit = apsr1_deposit 
		set_planet_entity = { 
			picture = pc_apsr1_planet_picture
			entity = "apsr_aura_planet_ancient_not_complex" 
		}
		set_planet_flag = "apsr_planet"	
		set_global_flag = apsr1_spawned
		save_global_event_target_as = apsr1_planet_target
	}
}

apsr_console_remove_1 = {
	every_galaxy_planet = {
		if = { 
			limit = {
				has_modifier = apsr1_planet_modifier
			}
			remove_modifier = apsr1_planet_modifier
			remove_planet_flag = "apsr_planet"
		}
		if = { 
			limit = {
				has_deposit = apsr1_deposit 
			}
			remove_deposit = apsr1_deposit 
			remove_building = apsr1_building_part1
			remove_building = apsr1_building_part2
			remove_building = apsr1_building_part3
			apsr_console_change_normal_planet_type = yes
		}
	}
	remove_global_flag = apsr1_spawned
	clear_global_event_target = apsr1_planet_target
}
