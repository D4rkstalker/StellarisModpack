nsc_modmenu_game_content_add_modifier_effect = {
	optimize_memory
	if = {
		limit = { has_modifier = nsc_modmenu_$CATEGORY$_$SUBCAT$_$ID$_$MODIFIER$_mod }
		remove_modifier = nsc_modmenu_$CATEGORY$_$SUBCAT$_$ID$_$MODIFIER$_mod
	}
	if = {
		limit = { event_target:nsc_modmenu_data_storage = { check_variable = { which = nsc_modmenu_$CATEGORY$_$SUBCAT$_$ID$_$MODIFIER$_var value != 0 } } }
		add_modifier = { modifier = nsc_modmenu_$CATEGORY$_$SUBCAT$_$ID$_$MODIFIER$_mod multiplier = event_target:nsc_modmenu_data_storage.nsc_modmenu_$CATEGORY$_$SUBCAT$_$ID$_$MODIFIER$_var }
	}
}

	}
}
