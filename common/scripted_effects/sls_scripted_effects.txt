effect_sls_set_base_3_status = { #$SLS_CATEGORY$, $POINT$
	if = {
		limit = {
			check_variable = {
				which = sls_leader_$SLS_CATEGORY$_points
				value = $SLS_POINT$
			}
		}
		add_trait_no_notify = sls_trait_research_$SLS_CATEGORY$_$SLS_POINT$
	}
}

effect_sls_clean_base_3_status = { #$SLS_CATEGORY$, $POINT$
	remove_trait = sls_trait_research_$SLS_CATEGORY$_$SLS_POINT$
}
