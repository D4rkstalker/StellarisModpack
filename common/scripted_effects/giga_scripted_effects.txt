# Flusion
generate_flusion_primitives_on_planet = {
	create_species = {
		name = "Pl'antes"
		class = PLANT
		portrait = pla8
		homeworld = THIS
		traits = { trait = random_traits }
	}
	create_country = {
		name = "Flusionian Nationstates"
		authority = random
		civics = {
			civic = civic_increasing_urbanization
			civic = civic_atmospheric_pollution
		}
		species = last_created
		ethos = random
		flag = {
			icon = {
				category = "special"
				file = "primitive.dds"
			}
			background= {
				category = "backgrounds"
				file = "new_dawn.dds"
			}
			colors= {
				"beige"
				"brown"
				"null"
				"null"
			}
		}
		type = primitive
	}
	last_created_country = {
		set_graphical_culture = industrial_01
		set_country_flag = atomic_age
		set_primitive_age = atomic_age
	}
	setup_primitive_planet = yes
}

setup_paluush_primitive_planet = {
	set_owner = last_created_country
	set_capital = yes
	while = { count = 25 create_pop = { species = last_created } }
	add_building = building_giga_paluush_homes
	add_building = building_giga_paluush_homes
	add_building = building_giga_paluush_homes
	add_building = building_giga_paluush_homes
	add_building = building_giga_paluush_homes
	add_building = building_giga_paluush_capital
	remove_building = building_colony_shelter

}

create_garyx = {
	create_species = {
		name = "Garyx"
		plural = "Garyx"
		class = GARYX
		portrait = garyx
	}
	last_created_species = { save_global_event_target_as = garyx_species }
	create_country = {
		name = "Garyx"
		type = faction
		species = last_created
		effect = { establish_communications_no_message = root.owner }
	}
	last_created_country = {
		create_leader = {
			class = ruler
			species = event_target:garyx_species
			name = "Garyx"
			skill = 5
		}
	}
}

create_eeloo = {
	create_country = {
		name = "Eeloowinies"
		type = faction
	}
	last_created_country = {
		set_country_flag = eeloowinies
		create_fleet = {
			name = "Rogue Weaponized Eeloo"
			settings = {
				spawn_debris = no
				is_boss = yes
			}
			effect = {
				set_owner = PREV
				set_fleet_flag = eeloofleet
				create_ship = {
					name = "Eeloo"
					design = "RogueEeloo"
				}
				set_location = event_target:gigaeeloo
			}
		}
	}
	random_system = {
		limit = { has_star_flag = kerbol_system }
		random_system_planet = {
			limit = { has_planet_flag = giga_eeloo }
			remove_planet = yes
		}
	}
}
