setup_paluush_primitive_planet = {
	set_owner = last_created_country
	set_capital = yes
	
	last_created_country = {
		if = {
			limit = { has_country_flag = machine_age }
			prev = {
				while = {
					count = 20
					create_pop = {
						species = last_created
					}					
				}
				add_building = building_paluush_homes
				add_building = building_paluush_homes
				add_building = building_paluush_homes
				add_building = building_paluush_homes
				add_building = building_primitive_capital
				remove_building = building_colony_shelter
			}
		}
	}
	
	if = {
		limit = {
			last_created_country = {
				OR = {
					has_country_flag = machine_age
					has_country_flag = atomic_age
					has_country_flag = early_space_age				
				}
			}
		}
		create_paluush_primitive_blockers = yes
	}
}

create_paluush_primitive_armies = {
	last_created_country = {
		if = {
			limit = { has_country_flag = machine_age }
			PREV = {
				while = {
					count = 11
					create_army = {
						name = "Colossal Paluush Legions"
						owner = last_created
						species = last_created
						type = "paluush_defender"
					}
				}
				while = {
					count = 1
					create_army = {
						name = "The Grandbunny"
						owner = last_created
						species = last_created
						type = "royal_paluush"
					}
				}
				while = {
					count = 1
					create_army = {
						name = "Poh'tin"
						owner = last_created
						species = last_created
						type = "royal_paluush"
					}
				}
				while = {
					count = 1
					create_army = {
						name = "Sparkbun"
						owner = last_created
						species = last_created
						type = "royal_paluush"
					}
				}
			}
			break = yes
		}
	}
}
