
ag_system_selector_initialization = {
	set_variable = { which = ag_system_selector_countries_page value = 1 }
	set_variable = { which = ag_system_selector_star_classes_page value = 1 }
	if = {
		limit = { NOT = { has_global_flag = "has_real_space_mod" } }
		set_variable = { which = ag_system_selector_star_classes_page_total value = 2 }
	}
	else = { set_variable = { which = ag_system_selector_star_classes_page_total value = 6 } }
	event_target:global_event_country = {
		set_variable = { which = ag_num_system_selector_countries value = 2 }			# Page1.1 means system without space_owner, Page1.2 means unknown targets.
		set_variable = { which = ag_system_selector_countries_page_total value = 1 }
	}
	every_country = {
		limit = { OR = {
			AND = { is_same_value = root count_system_within_border = { count > 0 } }
			AND = { NOT = { is_same_value = root } has_communications = root count_system_within_border = { count > 0 } }
		} }
		set_country_flag = ag_system_selector_country
		event_target:global_event_country = {
			change_variable = { which = ag_num_system_selector_countries value = 1 }
			if = {
				limit = { check_variable = { which = ag_num_system_selector_countries value = 10 } }
				set_variable = { which = ag_num_system_selector_countries value = 1 }
				change_variable = { which = ag_system_selector_countries_page_total value = 1 }
			}
			if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 1 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 1 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 2 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 2 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 3 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 3 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 4 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 4 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 5 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 5 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 6 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 6 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 7 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 7 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 8 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 8 } }
			else_if = { limit = { check_variable = { which = ag_num_system_selector_countries value = 9 } } ag_system_selector_set_target_country_flag = { ag_country_seq = 9 } }
		}
	}
}

ag_system_selector_after_effect = {
	every_country = {
		limit = { has_country_flag = ag_system_selector_country }
		remove_country_flag = ag_system_selector_country
		remove_country_flag = ag_countries_list_selected
		ag_system_selector_clear_target_country_flag = yes
	}
	every_system = {
		limit = { has_star_flag = ag_system_selector_selected_system }
		remove_star_flag = ag_system_selector_selected_system
	}
	ag_system_selector_clear_targets_country = yes
	ag_system_selector_clear_selected_country_flag = yes
	ag_system_selector_clear_selected_star_class_flag = yes
	ag_system_selector_clear_selected_star_country_flag = yes
	remove_country_flag = ag_system_selector_disable_country_triggers
	remove_country_flag = ag_system_selector_disable_star_classes_triggers
	remove_global_flag = ag_system_selector_initialized
	set_variable = { which = ag_num_system_selector_selected_systems value = 0 }
	clear_variable = ag_num_system_selector_selected_systems
}

ag_system_selector_select_unknown_country = {
	if = {
		limit = { NOT = { has_country_flag = ag_countries_list_selected_1_2 } }
		set_country_flag = ag_countries_list_selected_1
		set_country_flag = ag_countries_list_selected_1_2

			limit = { NOR = { is_same_value = root has_communications = root } }
