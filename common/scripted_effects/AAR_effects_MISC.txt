AAR_prmt_eft_all_MISC_ambt = {
	create_ambient_object = {
		type = $TYPE$
		entity_scale_to_size = no
		scale = $SCALE$
		location = $LOC$
		duration = $DAYS$
	}
}




###_[AAR年度功能DB]
#_evt	<aar年检>
#_list	AAR遗珍DB
AAR_eft_cntr_MISC_feature_DB = {
	##	AAR遗珍DB
	#	"魍魉之盒"
	if = { limit = { has_relic = r_defender			 NOT = { has_ascension_perk = eap_defender_of_the_galaxy } }
		remove_relic = r_defender
	}
	else_if = { limit = { has_ascension_perk = eap_defender_of_the_galaxy		 NOT = { has_relic = r_defender } }
		add_relic = r_defender
	}
	#	"维度魔方"
	if = { limit = { has_relic = AAR_R_webway		 NOT = { has_ascension_perk = AAR_AP_webway } }
		remove_relic = AAR_R_webway
	}
	else_if = { limit = { has_ascension_perk = AAR_AP_webway		 NOT = { has_relic = AAR_R_webway } }
		add_relic = AAR_R_webway
	}
	#	"宇宙立方"
	if = { limit = { has_relic = AAR_R_jumper		 NOT = { has_ascension_perk = AAR_AP_jumper } }
		remove_relic = AAR_R_jumper
	}
	else_if = { limit = { has_ascension_perk = AAR_AP_jumper		 NOT = { has_relic = AAR_R_jumper } }
		add_relic = AAR_R_jumper
	}
}
