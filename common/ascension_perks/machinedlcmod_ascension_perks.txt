ap_single_mind = {

	potential = {
        host_has_dlc = "Synthetic Dawn Story Pack"
        
		OR = {has_country_flag = synthetic_empire has_authority = auth_machine_intelligence}
		NOR = {
            has_ascension_perk = ap_single_mind      
            has_country_flag = ap_united_mind
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_tech_machine_leaders"
			has_technology = tech_machine_leaders
		}				
    }

	on_enabled = {
		custom_tooltip = "ap_single_mind_tooltip"
		hidden_effect = {
            set_country_flag = ap_single_mind
            capital_scope = {
                enable_special_project = {
                    name = "MACHINE_NON_GESTALT_PROJECT"
                    location = this
                    owner = root
                }
            }
		}
    }
	
	modifier = {
		ROBOT_species_trait_points_add = 2
	}
	
	ai_weight = {
		factor = 0
	}
}
