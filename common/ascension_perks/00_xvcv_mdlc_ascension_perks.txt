xvcv_mdlc_ap_self_assembling_colony_ship = {
	modifier = {
		colony_start_num_pops_add = 1
		planet_colony_development_speed_mult = 0.5
	}

	on_enabled = {
		custom_tooltip = xvcv_mdlc_ap_self_assembling_colony_ship_tooltip_effects
		if = {
			limit = { has_authority = auth_machine_intelligence }
			custom_tooltip = xvcv_mdlc_ap_self_assembling_colony_ship_tooltip_effects_2
		}
		hidden_effect = {
			create_fleet = {
				effect = {
					set_owner = root
					create_ship = {
						name = random
						random_existing_design = colonizer
						colonizer_species = owner.species
						graphical_culture = root
					}
					set_location = root.capital_scope.starbase
				}
			}
		}
		add_resource = {
			energy = -200
			alloys = -400
		}
	}

	potential = {
		has_authority = auth_machine_intelligence #requires "Synthetic Dawn Story Pack" dlc
		NOT = { has_ascension_perk = xvcv_mdlc_ap_self_assembling_colony_ship }
	}

	possible = {
		custom_tooltip = {
			fail_text = "xvcv_mdlc_ap_self_assembling_colony_ship_tooltip_requirements"
			resource_stockpile_compare = { resource = energy value >= 200 }
			resource_stockpile_compare = { resource = alloys value >= 400 }
		}
	}

	ai_weight = {
		factor = 10
	}
}
