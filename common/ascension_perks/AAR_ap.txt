	on_enabled = {
		custom_tooltip = AAR_tips_hostile_territory

			set_update_modifiers_batch = begin	#_|修正更新|暂停
			
			set_country_flag = AAR_flg_cntr_AP_VIGILANCE	# |AAR防御飞升添加|
			every_system_within_border = { AAR_eft_sstm_VIGILANCE_starbase_mod_DB = yes }	# AAR防御星基修正DB
			give_technology = { tech = AAR_tech_AP_VIGILANCE message = yes }

				add_modifier = { modifier = AAR_mod_cntr_HOSTILE_ai days = -1 }	# AAR敌对海域AI加成
			}
			
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}

	potential = {
		NOT = { has_ascension_perk = AAR_AP_hostile_territory }
	}
	possible = {
		custom_tooltip = { fail_text = "AAR_tips_AP_requires_ap_eternal_vigilance"

		}

		}
