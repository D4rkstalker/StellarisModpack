ap_birch_world = {
	on_enabled = {
		custom_tooltip = "add_research_options_birch_world"
		hidden_effect = {
			if = {
				limit = { 
					NOT = { has_technology = tech_birch_world }
				}
				add_research_option = tech_birch_world
			}		
		}
	}
	
	modifier = {
		country_megastructure_build_cap_add = 1
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_ap_hyper_engineering"
			has_ascension_perk = ap_hyper_engineering
		}
		has_technology = tech_gargantuan_ringworld
		has_ascension_perk = ap_master_builders
		custom_tooltip = {
			fail_text = "requires_found_giga_black_hole"
			has_country_flag = found_giga_black_hole
		}
	}				
	
	potential = {
		host_has_dlc = "Utopia"
		NOT = {
			has_ascension_perk = ap_birch_world
		}
		NOT = {
			no_scope = {
				has_global_flag = birch_disabled
			}
		}
	}
	
	ai_weight = {
		factor = 10
		modifier = {
			factor = 2
			has_ethic = ethic_pacifist
		}
		modifier = {
			factor = 3
			has_ethic = ethic_fanatic_pacifist
		}
	}
}
