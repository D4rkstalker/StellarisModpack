#CYBERTRONIAN
ap_cybertronian_worlds = {
	potential = {
		NOT = { has_ascension_perk = ap_cybertronian_worlds }
		OR = {
			has_trait = trait_cybertronian
			has_trait = trait_bw_cybertronian
			has_trait = trait_transtech_cybertronian
		}
	}
	possible = {
		OR = {
			has_trait = trait_cybertronian
			has_trait = trait_bw_cybertronian
			has_trait = trait_transtech_cybertronian
		}
		has_technology = tech_housing_2
	}
	
	on_enabled = {
		custom_tooltip = "allow_cybertronian_worlds"
		custom_tooltip = "describe_cybertronian_worlds"
	}

	ai_weight = {
		factor = 10
		modifier = {
			factor = 5
			has_trait = trait_cybertronian
		}
		modifier = {
			factor = 5
			has_trait = trait_bw_cybertronian
		}
		modifier = {
			factor = 5
			has_trait = trait_transtech_cybertronian
		}
	}
}

#INSECTICON
ap_insecticon_hive_worlds = {
	potential = {
		host_has_dlc = "Utopia"
		has_trait = trait_robot_insecticon
		NOT = {
			has_ascension_perk = ap_insecticon_hive_worlds
		}
		NOT = { # Lithoid devouring swarms consume worlds and cannot improve them (since they never get terraforming tech)
			AND = {
				is_lithoid_empire = yes
				has_valid_civic = civic_hive_devouring_swarm
			}
		}
	}
	possible = {
		has_trait = trait_robot_insecticon
		has_technology = tech_housing_2
	}
	on_enabled = {
		custom_tooltip = "allow_insecticon_hive_worlds"
		custom_tooltip = "describe_insecticon_hive_worlds"
	}

	ai_weight = {
		factor = 10
		modifier = {
			factor = 5
			has_trait = trait_robot_insecticon
		}
	}
}
