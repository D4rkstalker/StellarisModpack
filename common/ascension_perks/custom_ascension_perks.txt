custom_master_builders = {
	potential = {
		AND = {
			OR = { 
				has_civic = civic_machina
				has_civic = civic_ancient_hive
				has_civic = civic_godly_hive
			}
			NOT = { 
				OR = {
					has_ascension_perk = custom_dominion
					has_ascension_perk = custom_hive_dominion
				}
			}
		}
	}

	possible = {
        custom_tooltip = {
			fail_text = "req_perk_3_godly"
			AND = {
				OR = { 
					has_civic = civic_ancient_hive
					has_civic = civic_machina
					has_civic = civic_godly
					has_civic = civic_divine_inheritance
					has_civic = civic_godly_hive
				}
				num_ascension_perks > 2
			}
		}
	}

	modifier = {
		megastructure_build_speed_mult = 2
		megastructures_cost_mult = -0.20
	}
	
	ai_weight = {
		weight = 0
		factor = 0
	}
}

custom_dominion = {
	potential = {
		AND = {		
			NOT = { 
				OR = {
					has_ascension_perk = custom_dominion
					has_ascension_perk = custom_hive_dominion
				}
			}
			NOT = {
				authority = {
					value = auth_hive_mind
				}
			}
		}	
	}
	
	possible = {

		custom_tooltip = {
			fail_text = "req_perk_2_godly_dom"
			AND = {
				OR = { 
					has_civic = civic_ancient_hive
					has_civic = civic_machina
					has_civic = civic_godly
					has_civic = civic_divine_inheritance
					has_civic = civic_godly_hive
				}
				has_ascension_perk = ap_interstellar_dominion
				num_ascension_perks > 1
			}
		}
	}

	modifier = {
		country_starbase_influence_cost_mult = -0.25
		country_claim_influence_cost_mult = -0.25
		country_admin_cap_add = 500
	}
	
	ai_weight = {
		weight = 0
		factor = 0
	}
}
