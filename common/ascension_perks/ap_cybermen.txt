ap_cybermen = {
	on_enabled = {
		give_technology = { tech = tech_cybermen message = yes }
		custom_tooltip = "ap_cybermen_upgraded_traits"
		hidden_effect = {
			country_event = { id = ascension_traits_upgrade.4 days = 0 random = 0 }
			set_policy = {
				 policy = p_cybermen_integration
				 option = cybermen_integration_off
				 cooldown = no
			}
		}
	}

	potential = {
		host_has_dlc = "Synthetic Dawn Story Pack"
		has_authority = auth_machine_intelligence
		has_civic = civic_machine_assimilator
		NOT = {
			has_ascension_perk = ap_cybermen
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_2"
			num_ascension_perks > 1
		}
		has_technology = tech_gene_tailoring
	}
	
	ai_weight = {
		factor = 10
	}
}
