	on_enabled = {
		#custom_tooltip = ap_esap_artificial_procreation_tooltip
	}
	possible = {
		custom_tooltip = {
			fail_text = "requires_ap_engineered_evolution"
			has_ascension_perk = ap_engineered_evolution
		}
		custom_tooltip = {
			fail_text = "requires_not_ap_esap_bio_cyber_psi"
			Or = {
				has_ascension_perk = ap_esap_bio_cyber_psi_2
				Not = {
					has_ascension_perk = ap_esap_bio_cyber_psi 
				}
			}
		}
	}
	modifier = {
		pop_growth_speed = 0.30
		planet_pops_food_upkeep_mult = -0.25
	}
	potential = {
		host_has_dlc = "Utopia"
		NOT = {
			has_ascension_perk = ap_esap_artificial_procreation
			has_authority = "auth_machine_intelligence"
		}
		Or = {
			And = {
				is_mechanical_empire = no
				is_cyborg_empire = no
			}
			has_ascension_perk = ap_esap_bio_cyber_psi 
		}
		Or = {
			And = {
				NOT = { has_ascension_perk = ap_mind_over_matter }
				NOT = { has_ascension_perk = ap_the_flesh_is_weak }
			}
			has_ascension_perk = ap_esap_bio_cyber_psi
		}
	}
	ai_weight = {
		factor = 5
	}
}
