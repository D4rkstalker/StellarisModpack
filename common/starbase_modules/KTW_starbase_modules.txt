




KTW_hanger_module = {
	section = "KTW_HANGAR_STARBASE_SECTION"
	icon = "GFX_sm_KTW_hanger_module"
	construction_days = 240


	potential = {
		exists = owner
		owner = { has_technology = tech_KTW_weapons }
		is_orbital_ring = no
	}

	resources = {
		category = starbase_modules
		cost = { alloys = 350 }
		upkeep = { alloys = 1 energy = 1 }
	}
	
	station_modifier = {
		ship_hull_mult = 0.15
		ship_armor_mult = 0.15
		ship_weapon_damage = 0.1
		starbase_trade_protection_range_add = 1
		starbase_trade_protection_add = 10
		starbase_defense_platform_capacity_add = 1
	}
	
	ai_build_at_chokepoint = yes
	ai_build_outside_chokepoint = no


	ai_weight = {
		weight = 100
		### AI shouldn't use this component if it can't afford at least some upgrades right away
		modifier = {
			factor = 0.0
			exists = owner
			owner = { NOT = { graphical_culture = fe_01 } }
		}
	}


	show_in_tech = "tech_KTW_weapons"
}
