# Custom federal edicts

@Edict1Cost = 100
@Edict2Cost = 200
@Edict3Cost = 300

@EdictPerpetual = -1
@Edict10Years = 3600

ifo_fed_ethic_authoritarian = {
	length = @Edict10Years
	icon = "GFX_edict_type_policy"
	edict_cap_usage = 1

	resources = {
		category = edicts
		cost = {
			influence = @Edict2Cost
		}
	}

	modifier = {
		#
	}

	potential = {
		has_origin = ifo_origin_fleet
		exists = federation
		or = {
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
	}

	effect = {
		custom_tooltip = edict_ifo_fed_ethic_authoritarian_effect
		hidden_effect = {
			every_federation_ally = {
				limit = { not = { is_same_empire = prev } }
					add_modifier = { modifier = ifo_fed_ethic_authoritarian days = @Edict10Years }
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}
