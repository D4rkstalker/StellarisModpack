@Edict2Cost = 200

@EdictDuration = 3600

exalted_benevolent_intervention = {
	length = @EdictDuration

	potential = {
		has_country_flag = exalted_domination_subject_edicts
	}

	allow = {
		any_subject = {
			relative_power = {
				who = FROM
				value <= equivalent
			}
		}
	}

	resources = {
		category = edicts
		cost = {
			influence = @Edict2Cost
		}
	}

	effect = {
		every_subject = {
			limit = {
				relative_power = {
					who = FROM
					value <= equivalent
				}
			}

			if = {
				limit = {
					NOT = { has_modifier = modifier_exalted_benevolent_intervention }
				}

				add_modifier = {
					modifier = modifier_exalted_benevolent_intervention
				}
			}

			hidden_effect = {
				set_country_flag = exalted_benevolent_intervention_isactive

				add_opinion_modifier = {
					who = FROM
					modifier = opinion_subject_provided_aid
				}
			}
		}
	}

	ai_weight = {
		weight = @EdictMedPrio

		modifier = {
			weight = @EdictHighPrio
			has_ascension_perk = ap_shared_destiny
		}

		modifier = {
			factor = 0.1
			has_country_resource = {
				type = influence
				amount < 500
			}
		}

		modifier = {
			factor = 0.5
			NOR = {
				has_ethic = ethic_egalitarian
				has_ethic = ethic_fanatic_egalitarian
				has_ethic = ethic_pacifist
				has_ethic = ethic_fanatic_pacifist
				has_ethic = ethic_xenophile
				has_ethic = ethic_fanatic_xenophile
			}
		}

		modifier = {
			factor = 5
			has_civic = civic_feudal_realm
		}
	}
}
