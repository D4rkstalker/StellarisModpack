

Edict_GAPS_On_The_Dragon_Trail = {
	length =  0
	resources = {
		cost = {
			energy = 500
		}
	}
	
	allow = { any_system = { OR = { has_star_flag = guardians_dragon_system has_star_flag = rubricator_system } } }
	potential = { 
		has_origin = Origin_GAPS_Dragon_Hatchers
		NOR = { has_country_flag = CountryFlag_Used_Edict_GAPS_On_The_Dragon_Trail }
	}

	effect = {  
		set_country_flag = CountryFlag_Used_Edict_GAPS_On_The_Dragon_Trail
		begin_event_chain = {
			event_chain = "Event_Chain_GAPS_Dragon_Systems"
			target = root
		}
		every_system_within_border = {		
			limit = { any_system_planet = { is_capital = yes } }
			random_system_planet = { limit = { is_primary_star = yes } save_event_target_as = Target_Planet_GAPS_Dragon_Project }
			enable_special_project = { name = Special_Project_GAPS_Find_Dragon_Systems owner = root location = event_target:Target_Planet_GAPS_Dragon_Project }
		}
	}
}
