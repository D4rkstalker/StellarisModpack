
Edict_GAPS_Tiyank_Matriarch_Pacification = {
	length =  0
	resources = {
		cost = {
			exotic_gases = 500
		}
	}
	
	allow = {
		has_country_flag = encountered_elderly_tiyanki
		any_system = { has_star_flag = elderly_tiyanki_system }
	}
	potential = { has_origin = Origin_GAPS_Ranchers }
	
	effect = {  
		begin_event_chain = {
			event_chain = "Event_Chain_GAPS_Matriarch_Pacification"
			target = root
		}
		every_system = {		
			limit = { owner = { is_same_value = root } any_system_planet = { is_capital = yes } }
			random_system_planet = { limit = { is_primary_star = yes } save_event_target_as = Events_GAPS_Ranchers }
			enable_special_project = { name = Special_Project_GAPS_Matriarch_Pacification owner = root location = event_target:Events_GAPS_Ranchers }
		}
	}
}
