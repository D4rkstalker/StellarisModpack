ag_deploy_delta_platform = {
	length = 0
	icon = "GFX_ag_action_edicts"
	resources = {
		category = edicts
		cost = {
			energy = 100
		}
	}
	
	potential = {
		NOT = { has_country_flag = ag_delta_platform_deploying }
		has_technology = tech_ag_delta_platform
		any_owned_fleet = { has_fleet_flag = ag_delta_platform_player }
	}
	
	effect = {
		custom_tooltip = edict_ag_deploy_delta_platform_custom
		hidden_effect = {
			if = {
				limit = { has_country_flag = ag_ship_subsystem_enabled_402_14 }
				set_timed_country_flag = { flag = ag_delta_platform_deploying days = 2 }
				every_owned_fleet = {
					limit = { has_fleet_flag = ag_delta_platform_player }
					ag_deploy_ancient_turret_preparation = { ag_delay = 2 }
				}
			}
			else = {
				set_timed_country_flag = { flag = ag_delta_platform_deploying days = 5 }
				every_owned_fleet = {
					limit = { has_fleet_flag = ag_delta_platform_player }
					ag_deploy_ancient_turret_preparation = { ag_delay = 5 }
				}
			}
		}
	}
	
	ai_weight = {
		weight = 0
	}
}
