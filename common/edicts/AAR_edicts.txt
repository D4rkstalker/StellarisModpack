	length = 360
	icon = "GFX_AAR_edict_DOMINION_expand"

	resources = {
		category = edicts

			unity = 500
			influence = 25
		#	multiplier = value:sacrifice_cost	# 3*unity月产出		不管用
			multiplier = value:edict_size_effect
		}
	}

#	modifier = {
#		custom_tooltip = "edict_sacrifice_togetherness_tooltip" #This string should be updated if the values below are changed
#		show_only_custom_tooltip = no
#	}

	potential = {
		has_ascension_perk = AAR_AP_dominion_EXPAND
	}
	allow = {
		custom_tooltip = {
			fail_text = "AAR_tips_AP_dominion_EXPAND_CD"
			hidden_trigger = {
				NOT = { has_country_flag = AAR_flg_cntr_AP_dominion_EXPAND_CD }	# 玩家=1年/AI=10年
			}
		}
	}

#	prerequisites = { "tech_cultural_heritage" }
#	show_tech_unlock_if = { always = no }

	effect = {
		hidden_effect = {
			set_update_modifiers_batch = begin	#_|修正更新|暂停
			
			#	AI添加CD
			if = { limit = { is_ai = yes }
				set_timed_country_flag = { flag = AAR_flg_cntr_AP_dominion_EXPAND_CD days = 3600 }
			}
			#	强制开边
			every_country = { limit = { has_closed_borders = prev }
				set_closed_borders = {
					who = prev
					status = no
					forced = @AAR_svar_dominion_border
				}
				add_opinion_modifier = { modifier = AAR_opinion_AP_dominion_EXPAND who = prev }
			}
			
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}

	ai_weight = {
		weight = 100	# @EdictHighPrio
	}
}
