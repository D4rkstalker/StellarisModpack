# Edicts that cost influence, but don't count towards the Edict cap.

#Peace Crisis edicts:
pw_festival_plaza_this_too_shall_pass = {
	length = @pw_EdictPerpetual
	icon = "GFX_edict_type_policy"

	prerequisites = {
		pw_tech_brace_against_the_storm
	}

	resources = {
		category = edicts
		cost = { #Activation Cost
			trigger = {
				NOT = { has_edict = pw_festival_plaza_this_too_shall_pass }
			}
			influence = 250
		}
		cost = { #Deactivation Cost
			trigger = {
				has_edict = pw_festival_plaza_this_too_shall_pass
			}
			influence = 50
		}
	}

	potential = {
		any_owned_planet = {
			has_active_building = pw_building_festival_plaza
		}
	}

	allow = {
		is_pacifist = yes
		any_owned_planet = {
			has_active_building = pw_building_festival_plaza
		}
		NOT = {
			has_edict = pw_festival_plaza_inexorable_peace
		}
	}

	modifier = {
		pw_mod_damage_against_end_game_crisis = 0.25
	}

	ai_weight = {
		weight = @EdictHighPrio
		modifier = {
			factor = 2
			OR = {
				has_global_flag = galactic_crisis_happened
				has_global_flag = gray_goo_crisis_active
			}
		}
	}
}
