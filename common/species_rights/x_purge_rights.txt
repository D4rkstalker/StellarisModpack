
purge_normal = {
	token = purge_normal
	pop_escape_chance = 0.05
	pop_decline_rate = 200	
	pop_modifier = {
		pop_decline_speed = 6
		pop_happiness = -10
	}	
	
	potential = {
		exists = from
		from = { 
		NOT = { has_valid_civic = civic_hive_devouring_swarm } 
		NOT = { has_valid_civic = civic_mutualistic_behavior }
		}
	}
	
	allow = {
		hidden_trigger = { exists = from }
		if = { 
			limit = { is_lonely_hive_mind_pop = yes } 
			always = yes
		}
		else = {
			custom_tooltip = {
				fail_text = PURGES_ALLOWED
				OR = {
					is_robotic_species = yes
					from = {
						OR = {
							has_ethic = ethic_gestalt_consciousness
							has_valid_civic = civic_fanatic_purifiers
							has_policy_flag = purge_allowed
						}			
					}
				}
			}	
		}
	}
	ai_will_do = {
		factor = 3
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOT = {
					has_ai_personality_behaviour = purger
				}
			}
		}		
	}
}

purge_necrophage = {

	pop_escape_chance = 0.1
	pop_decline_rate = 50
	pop_modifier = {
		pop_decline_speed = 3
		pop_happiness = -0.5
	}	
	
	potential = {
		exists = from
		from = {
			OR = { 
				has_valid_civic = civic_hive_parasitic 
				AND = {
					has_origin = origin_necrophage
					has_trait = trait_necrophage
				}
			}
		}
	}
	
	allow = {
		hidden_trigger = { exists = from }

		custom_tooltip = {
			fail_text = CANNOT_NECROPHAGE_MACHINES

		}
		if = { 
			limit = { is_lonely_hive_mind_pop = no } 
			custom_tooltip = {
				fail_text = PURGES_ALLOWED
				from = {
					OR = {
						has_valid_civic = civic_hive_parasitic 
						has_valid_civic = civic_fanatic_purifiers
						has_policy_flag = purge_allowed
					}			
				}
			}
		}
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOT = {
					has_ai_personality_behaviour = purger
				}
			}
		}		
	}
}
