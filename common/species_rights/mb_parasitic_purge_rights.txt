
purge_normal = {
	token = purge_normal
	pop_escape_chance = 0.05
	
	pop_modifier = {
		pop_decline_speed = 6
		pop_happiness = -10
	}	
	
	potential = {
		exists = from
		from = { 
		NOT = { has_valid_civic = civic_hive_devouring_swarm } 
		NOT = { has_valid_civic = civic_mutualistic_behavior }
		}
	}
	
	allow = {
		hidden_trigger = { exists = from }
		if = { 
			limit = { is_lonely_hive_mind_pop = yes } 
			always = yes
		}
		else = {
			custom_tooltip = {
				fail_text = PURGES_ALLOWED
				OR = {
					is_robotic_species = yes
					from = {
						OR = {
							has_ethic = ethic_gestalt_consciousness
							has_valid_civic = civic_fanatic_purifiers
							has_policy_flag = purge_allowed
						}			
					}
				}
			}	
		}
	}
	ai_will_do = {
		factor = 3
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOT = {
					has_ai_personality_behaviour = purger
				}
			}
		}		
	}
}

purge_processing = {
	token = purge_processing
	pop_escape_chance = 0.20
	
	pop_modifier = {
		pop_decline_speed = 3
		pop_happiness = -10
	}

	potential = {
		host_has_dlc = "Utopia"
		exists = from
		from = { 
		NOT = { has_authority = auth_machine_intelligence }
		NOT = { has_valid_civic = civic_mutualistic_behavior }
		}
	}
	
	
	
	allow = {
		hidden_trigger = { exists = from }
			
		custom_tooltip = {
			fail_text = CAN_NOT_EAT_MACHINES
			OR = {
				NOR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
				}
				from = { has_valid_civic = civic_hive_devouring_swarm }
				# devouring swarms can CRONCH. in edge cases.
			}
		}
		custom_tooltip = {
			fail_text = "lonely_hive_mind_pops_kill_themselves_purge_type"
			is_lonely_hive_mind_pop = no
		}
		custom_tooltip = {
			fail_text = PURGES_ALLOWED
			from = {
				OR = {
					has_ethic = ethic_gestalt_consciousness
					has_valid_civic = civic_fanatic_purifiers
					has_policy_flag = purge_allowed
					AND = {
						has_policy_flag = pre_sapients_eat
						prev = {
							is_sapient = no
						}
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 50
		modifier = {
			factor = 0
			NOR = { 
				has_trait = trait_delicious 
				from = { 

OR  = {
has_valid_civic = civic_hive_parasitic 
has_valid_civic = civic_hive_devouring_swarm 
}
}
				AND = {
					is_sapient = no
					from = {
						has_policy_flag = pre_sapients_eat
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOR = {
					has_ai_personality_behaviour = purger
				}
			}
		}
	}
}
