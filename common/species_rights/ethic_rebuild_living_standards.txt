			}

				AND = {

					}	
					NOT = { has_purge_type = { country = from type = purge_neutering } }
				}
				text = is_purged_not_neutered
			}
		}
		custom_tooltip = {

			NOT = {

					type = citizenship_assimilation

	ai_will_do = {
