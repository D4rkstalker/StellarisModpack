centralization_minimal_confederation = {
	unlocks_centralization = 1
	potential = { has_federation_perk = galactic_confederation_passive }
	allow = {
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = centralization_minimal_confederation }
					has_modifier = federation_law_category_centralization_confederation_cooldown
					has_any_federation_law_in_category = law_category_centralization_confederation
				}
			}
		}
		custom_tooltip = {
			OR = {
				has_federation_law = centralization_minimal_confederation
				has_federation_law = centralization_low_confederation
			}
		}
	}
	cohesion_growth = -1.5
	modifier = { description = "centralization_minimal_confederation_effect" }
	on_enact = {
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_federation_law_cooldown = { modifier = federation_law_category_centralization_confederation_cooldown }
		}
		set_variable = { which = centralization_institution_upgrades_cap value = 0 }
		add_cohesion = -10
	}

	ai_weight = {
		base = 1
		################################## 从高到低
		modifier = { # 基础支持度					+1
			desc = federation_acceptance_generally_positive
			add = 25
		}
		modifier = { #团结度同意度修正	-100~+100	+1~-1
			desc = federation_acceptance_neg_coh_modifier
			add = value:fedcoh_neg_modifier
		}
		modifier = { # 联邦团结度>0，反对降低		-1
			desc = federation_acceptance_dont_lower_centralization
			add = -25
			federation = { federation_cohesion > 0 }
		}
		modifier = { # 联邦团结度<-50，支持降低		+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion < -50 }
			federation = { federation_cohesion_growth < 1.5 }
		}
		modifier = { # 联邦团结度=-100，支持降低	+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion <= -99 }
			federation = { federation_cohesion_growth < 0 }
		}
		modifier = { # 主席的意见					-4
			desc = federation_acceptance_president
			add = -100
			is_federation_leader = yes
		}
		modifier = { # 议会的意见					-1.5
			desc = federation_acceptance_council
			add = -37
			is_federation_leader = no
			is_part_of_galactic_council = yes
		}
		complex_trigger_modifier = { # 对主席的好感	-2~+2
			desc = federation_acceptance_opinion_to_president
			trigger = opinion
			parameters = { who = federation.federation_leader }
			mode = add
			mult = -0.05
			potential = {
				is_federation_leader = no
			}
		}
		modifier = { # 成员对监管人好感度，正		-1
			desc = federation_acceptance_like_custodian
			add = -25
			is_federation_leader = no
			opinion_level = { who = galactic_custodian level >= good }
		}
		modifier = { # 成员对监管人好感度，负		+2
			desc = federation_acceptance_dislike_custodian
			add = 50
			is_federation_leader = no
			opinion_level = { who = galactic_custodian level <= poor }
		}
	}
}
