centralization_imperium_low = {
	display_level = 1
	unlocks_centralization = 2
	cohesion_growth = -1
	potential = { has_federation_perk = imperium_federation_passive }
	allow = {
		custom_tooltip = {
			OR = {
				has_federation_law = centralization_imperium_low
				has_federation_law = centralization_imperium_mid
			}
		}
		custom_tooltip = {
			fail_text = "unified_imperium_established"
			NOT = { has_federation_flag = flag_centralization_imperium_unified_imperium }
		}
		custom_tooltip = {
			NOT = {
				AND = {
					has_any_federation_law_in_category = law_category_centralization_imperium
					NOT = { has_federation_law = centralization_imperium_low }
					OR = {
						has_modifier = federation_law_category_centralization_imperium_cooldown
						has_modifier = federation_law_category_imperium_institutions_cooldown
						has_modifier = federation_law_category_imperium_laws_cooldown
					}
				}
			}
		}
	}
	on_enact = {
		hidden_effect = {
			set_variable = { which = imperial_institutions_upgrades_support value = @imperial_institutions_num_1 }
			if = {
				limit = { has_federation_perk = imperium_elite_bureaucrats }
				change_variable = { which = imperial_institutions_upgrades_support value = @imperial_institutions_num_add }
			}
			add_imperium_federation_law_cooldown_centralization = yes
		}
	}
	modifier = {
		federation_total_level_of_core_institutions_capacity_add = @imperial_institutions_num_1
	}
	ai_weight = {
		base = 1
		################################## 从高到低
		modifier = { # 基础支持度					+1
			desc = federation_acceptance_generally_positive
			add = 25
		}
		modifier = { #团结度同意度修正	-100~+100	+1~-1
			desc = federation_acceptance_neg_coh_modifier
			add = value:fedcoh_neg_modifier
		}
		modifier = { # 联邦团结度>0，反对降低		-1
			desc = federation_acceptance_dont_lower_centralization
			add = -25
			federation = { federation_cohesion > 0 }
		}
		modifier = { # 联邦团结度<-50，支持降低		+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion < -50 }
			federation = { federation_cohesion_growth < 1.5 }
		}
		modifier = { # 联邦团结度=-100，支持降低	+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion <= -99 }
			federation = { federation_cohesion_growth < 0 }
		}
		modifier = { # 皇帝的意见					-4
			desc = federation_acceptance_emperor
			add = -100
			is_federation_leader = yes
		}
		complex_trigger_modifier = { # 皇帝的附庸	-1~+1
			desc = federation_acceptance_emperor_vassal
			trigger = has_loyalty
			mode = add
			mult = -0.25
			potential = {
				is_subject = yes
				overlord = { is_galactic_emperor = yes }
			}
		}
		complex_trigger_modifier = { # 皇帝的附庸（月度）
			desc = federation_acceptance_emperor_vassal_monthly_loyalty
			trigger = has_monthly_loyalty
			mode = add
			mult = -1
			potential = {
				is_subject = yes
				overlord = { is_galactic_emperor = yes }
			}
		}
		modifier = { # 议会的意见					-1.5
			desc = federation_acceptance_council
			add = -37
			is_federation_leader = no
			is_part_of_galactic_council = yes
		}
		modifier = { # 成员对皇帝好感度，正			-1
			desc = federation_acceptance_loyal_to_emperor
			add = -25
			is_federation_leader = no
			opinion_level = { who = galactic_emperor level >= good }
		}
		modifier = { # 成员对皇帝好感度，负			+2
			desc = federation_acceptance_disloyal_to_emperor
			add = 50
			is_federation_leader = no
			opinion_level = { who = galactic_emperor level <= poor }
		}
	}
}

			set_variable = { which = imperial_institutions_upgrades_support value = @imperial_institutions_num_2 }
			if = {
				limit = { has_federation_perk = imperium_elite_bureaucrats }
				change_variable = { which = imperial_institutions_upgrades_support value = @imperial_institutions_num_add }
			}
			if = {
				limit = { NOT = { has_federation_flag = flag_centralization_imperium_mid_first_time } }
				every_country = {
					limit = { is_galactic_community_member = no }
					country_event = { id = MECR.1003 days = 7 }
				}
				set_federation_flag = flag_centralization_imperium_mid_first_time
			}
			add_imperium_federation_law_cooldown_centralization = yes
		}
	}
	modifier = {
		federation_total_level_of_core_institutions_capacity_add = @imperial_institutions_num_2
		resolutions_cost_mult = -0.1
	}
	ai_weight = {
		base = 1
		################################## 从低到高
		modifier = { # 基础支持度					+1
			desc = federation_acceptance_generally_positive
			add = 25
			federation = { fed_centralization_imperium_mid_lower = yes }
		}
		modifier = { # 联邦团结度=100，支持提高		+1
			desc = federation_acceptance_high_cohesion
			add = 25
			federation = { federation_cohesion >= 99 }
			federation = { fed_centralization_imperium_mid_lower = yes }
		}
		modifier = { # 联邦团结度<80，反对提高		-2
			desc = federation_acceptance_better_cohesion
			add = -50
			is_federation_leader = no
			federation = { federation_cohesion < 80 }
			federation = { fed_centralization_imperium_mid_lower = yes }
		}
		modifier = { # 皇帝的意见					+3
			desc = federation_acceptance_emperor
			add = 75
			is_federation_leader = yes
			federation = { fed_centralization_imperium_mid_lower = yes }
		}
		complex_trigger_modifier = { # 皇帝的附庸	-1~+1
			desc = federation_acceptance_emperor_vassal
			trigger = has_loyalty
			mode = add
			mult = 0.25
			potential = {
				is_subject = yes
				overlord = { is_galactic_emperor = yes }
				federation = { fed_centralization_imperium_mid_lower = yes }
			}
		}
		complex_trigger_modifier = { # 皇帝的附庸（月度）
			desc = federation_acceptance_emperor_vassal_monthly_loyalty
			trigger = has_monthly_loyalty
