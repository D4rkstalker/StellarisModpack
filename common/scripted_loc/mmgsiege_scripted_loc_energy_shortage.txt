defined_text = {
	name = MmgsiegeEnergyShortageRobotPopHappiness
	random = no
	text = {
		localization_key = "mmgsiege_shortage_robot_pop_happiness"
		trigger = {
			exists = owner
			any_owned_species = {
				OR = {
					is_archetype = ROBOT
					is_archetype = MACHINE
				}
				species_has_happiness_with_owner = prev.owner
			}
		}
	}
	default = "BLANK_STRING"
}
