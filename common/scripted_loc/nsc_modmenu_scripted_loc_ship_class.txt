#StrikeCruiser
defined_text = {
	name = Get_nsc_ship_class_StrikeCruiser_enable_disable_status
	random = no

	text = {
		trigger = { nsc_modmenu_game_content_on_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser } }
		localization_key = nsc_modmenu_setting_page_ship_category_class_StrikeCruiser_enabled_desc
	}
	text = {
		trigger = { NOT = { nsc_modmenu_game_content_on_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser } } }
		localization_key = nsc_modmenu_setting_page_ship_category_class_StrikeCruiser_disabled_desc
	}
}
defined_text = {
	name = Get_nsc_ship_class_StrikeCruiser_build_cost
	random = no

	text = { #'SYMBOL = = VALUE = 0' means 'check_variable = { value = 0 }' and 'SYMBOL = > VALUE = 0' means 'check_variable = { value > 0 }' #do you see the difference?
		trigger = { nsc_modmenu_game_content_check_variable_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser MODIFIER = build_cost SYMBOL = = VALUE = 0 } }
		localization_key = nsc_modmenu_setting_page_ship_category_class_StrikeCruiser_build_cost_desc_neutral
	}
	text = {
		trigger = { nsc_modmenu_game_content_check_variable_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser MODIFIER = build_cost SYMBOL = < VALUE = 0 } }
		localization_key = nsc_modmenu_setting_page_ship_category_class_StrikeCruiser_build_cost_desc_minus
	}
	text = {
		trigger = { nsc_modmenu_game_content_check_variable_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser MODIFIER = build_cost SYMBOL = > VALUE = 0 } }
		localization_key = nsc_modmenu_setting_page_ship_category_class_StrikeCruiser_build_cost_desc_plus
