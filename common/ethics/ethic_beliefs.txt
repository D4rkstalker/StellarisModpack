ethic_no_religion = {
	cost = 1
	category = "god"
	category_value = 0
	category_opposite = yes
	
	tags = {
	}
	
	
	random_weight = {
		value = 100
	}	
	
	pop_attraction_tag = {
		desc = POP_ATTRACTION_NO_RELIGION_POS
		trigger = {
			OR = {
				has_ethic = ethic_no_religion
			}
		}
	}
	
	
	country_attraction = {
		value = 1
		modifier = {
			factor = 0
			NOR = {
				is_country_type = default
				has_ethic = ethic_no_religion
				AND = {
					is_country_type = primitive
					capital_scope = {
						has_observation_outpost = yes
						observation_outpost = {
							has_mission = native_indoctrination		
						}
						observation_outpost_owner = { 
								has_ethic = ethic_no_religion						
						}				
					}
				}				
			}
		}
		# Effects of indoctrination mission
		modifier = {
			factor = 5
			is_country_type = primitive
			capital_scope = {
				has_observation_outpost = yes
				observation_outpost = {
					has_mission = native_indoctrination		
				}
				observation_outpost_owner = { has_ethic = ethic_no_religion }
			}
		}
		

	}

	# from = planet
	pop_attraction = {
		value = 1.5
		
		modifier = {
			factor = 0
			has_trait = trait_hive_mind
		}

		modifier = {
			factor = 1
			owner = { has_ethic = ethic_no_religion }
		}		
		
		modifier = {
		factor = 2
		from = {	
			any_owned_pop = {
				pop_is_heretic = yes
				has_ethic = ethic_no_religion
				}					
			}	
		}	
		modifier = {
		factor = 5
		from = {	
			any_owned_pop = {
				has_pop_flag = heretic_preacher
				has_ethic = ethic_no_religion
				}					
			}	
		}	
		modifier = {
		factor = 8
			from = { 
				AND = {
					has_building = building_monolith
					owner = {has_ethic = ethic_no_religion}
				}	
			}
		}		
		modifier = {
		factor = 10
			from = { 
				AND = {
					has_building = building_spiritual_monolith

			}
		}
