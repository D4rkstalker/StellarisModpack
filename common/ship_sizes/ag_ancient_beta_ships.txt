	potential_construction = {
		exists = owner

			has_technology = tech_ag_beta_station_type1
			is_variable_set = ag_num_beta_station_type1_cores
			check_variable = { which = ag_num_beta_station_type1_cores value > 0 }
		}
	}

	possible_construction = {
		custom_tooltip = {
			fail_text = "ag_requires_ancient_cores"
			owner = {
				check_variable = {
					which = ag_num_beta_station_type1_cores
					value > 0
				}
			}
		}
		custom_tooltip = {
			fail_text = "ag_requires_less_than_six_stations_in_system"
			solar_system = {
				NOT = {
					count_ship_in_system = {
						limit = {
							OR = {
								is_ship_size = ag_beta_station_type1_0
								is_ship_size = ag_beta_station_type1_1
							}
						}
						count > 5
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "ag_requires_not_outpost"
			OR = {
				is_scope_type = megastructure
				AND = {
					is_scope_type = starbase
					has_starbase_size > starbase_outpost
				}
			}
		}
	}

	resources = {
		category = station_gatherers
