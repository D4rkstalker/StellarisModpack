# Supernova
supernova_category = {
	desc = "supernova_category_desc"
	picture = GFX_evt_ship_in_orbit
	max_once_global = yes
	level = 3
	null_spawn_chance = 0.5
	spawn_chance = {
		modifier = {
			factor = 0
			OR = {
				has_deposit_for = shipclass_mining_station
				has_deposit_for = shipclass_research_station
			}
		}
		modifier = {
			add = 1
			mid_game_years_passed < 0
			AND = {
				has_global_flag = eev_kaboom
				OR = {
					is_planet_class = pc_a_star
					is_planet_class = pc_m_giant_star
					# Real Space
					is_planet_class = pc_k_giant_star
					is_planet_class = pc_g_giant_star
				}
				is_binary_star = no
				is_trinary_star = no
				eev_is_valid_system = yes
			}
		}
	}
	on_success = {
		0 = {
			modifier = {
				add = 1
				has_nemesis = yes
			}
			anomaly_event = supernova.1
		}
		0 = {
			modifier = {
				add = 1
				has_nemesis = no
			}
			anomaly_event = supernova.4
		}
		3 = {
			anomaly_event = supernova.5
		}
	}
	on_spawn = {
		clear_deposits = yes
	}
}
