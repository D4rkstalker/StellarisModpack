##
# Dragon Egg category to mark possible Egg planets.
#
HTBD_EGG_CATEGORY = {
	desc = "htbd_egg_category_desc"
	picture = GFX_evt_space_dragon
	level = 7

	should_ai_use = yes

	spawn_chance = {
		modifier = {
			add = 10

			has_planet_flag = egg_incubated
			exists = FROM
			FROM = {
				exists = owner
				owner = {
					has_country_flag = htbd_dragonborn_done
				}
			}
		}
	}

	on_spawn = {
		log = "[This.Owner.GetName] found Egg anomaly at [From.GetName]"
	}

	on_success = {
		1 = {
			max_once = no
			anomaly_event = htbd_ev.9
		}
	}
}
