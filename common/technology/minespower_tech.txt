		factor = 2

				}
				research_leader = {
					area = engineering
					has_trait = "leader_trait_expertise_biology"
				}
			}
		}
		modifier = {
			factor = 0.5
			OR = {
				any_owned_planet = {
					OR = {
						has_building = building_hydroponics_farm
						has_building = building_hydroponics_farm_2
						has_building = building_hydroponics_farm_3
					}
				}				
				any_owned_starbase = {
					OR = {
						has_starbase_module = starm_farming_hub
					}
				}	
			}
		}
	}
	
	ai_weight = {
		modifier = {
			factor = 1.25
			research_leader = {
				area = society
				has_trait = "leader_trait_expertise_biology"
			}
		}
	}
}
