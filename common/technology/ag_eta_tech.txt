tech_ag_ancient_eta_gateway_activation = {
	area = engineering
	start_tech = no
	cost = @tier5cost3
	is_dangerous = yes
	tier = 5
	category = { particles }
	prerequisites = { "tech_gateway_activation" }
	is_reverse_engineerable = no
	gateway = gateway_ag_eta_gateway
	potential = { has_country_flag = ag_eta_3_fired }
	prereqfor_desc = {
		custom = {
			title = "tech_ag_ancient_eta_gateway_activation_custom"
			desc = "tech_ag_ancient_eta_gateway_activation_custom_desc"
		}
	}
	weight_modifier = {
		modifier = {
			factor = 1.5
			has_ascension_perk = ap_technological_ascendancy
		}
		modifier = {
			factor = 0
			NOR = {
				AND = {
					NOT = { has_global_flag = ag_eta_titan_event_ended }
					exists = event_target:ag_ancient_eta_star
					event_target:ag_ancient_eta_star.solar_system = {
						exists = space_owner
						space_owner = { is_same_value = prevprev }
						any_system_megastructure = { is_megastructure_type = ag_ancient_eta_gateway_megastructure_damaged }
					}
				}
				AND = {
					has_global_flag = ag_eta_titan_event_ended
					exists = event_target:ag_ancient_eta_shielded_world
					event_target:ag_ancient_eta_shielded_world = {
						exists = owner
						owner = { is_same_value = prevprev }
						has_building = ag_ancient_eta_archive
						has_ground_combat = no
						ag_has_researcher = yes
					}
				}
			}
		}
	}
	weight = 1
	ai_weight = { factor = 1000 }
}
