tech_shroud_knowledge_1 = {
	area = society
	cost = @tier5cost3
	tier = 5
	is_rare = yes
	prerequisites = { tech_shroud_monolith }
	category = { psionics  }
	ai_update_type = all
	weight = @tier5weight3

	potential = {
			has_country_flag = shroud_monolith_discovered
	}	
	
	
	modifier = {		
		damage_vs_country_type_shroud_entities_mult = 0.15
		damage_vs_country_type_shroud_horrors_mult = 0.15
	}
	
	weight_modifier = {		
		modifier = {
			factor = 0.5
			is_spiritualist = no
		}
		modifier = {
			factor = 2
			has_ethic = "ethic_fanatic_spiritualist"
		}		
		modifier = {
			factor = @ap_technological_ascendancy_rare_tech
			has_ascension_perk = ap_technological_ascendancy
		}
		modifier = {
			factor = @federation_perk_factor
			has_federation = yes
			federation = {
				has_federation_perk = rare_tech_boost
				any_member = { has_technology = tech_psionic_theory }
			}
		}
		modifier = {
			factor = 2
			any_owned_pop = {
				pop_has_trait = trait_psionic
			}
		}
		modifier = {
			factor = 1.25
			any_owned_pop = {
				pop_has_trait = trait_latent_psionic
			}
			NOT = {
				any_owned_pop = {
					pop_has_trait = trait_psionic
				}
			}
		}
		modifier = {
			factor = 0
			AND = {
				is_materialist = yes
				NOT = {					
					any_owned_pop = {
						OR = {
							pop_has_trait = trait_psionic
							pop_has_trait = trait_latent_psionic
						}
					}
				}
			}
		}
	}

	ai_weight = {
		factor = 5 #very rare tech
		modifier = {
			factor = 1.5
			OR = {
				has_ethic = ethic_spiritualist
				has_ethic = ethic_fanatic_spiritualist
			}
		}		
		modifier = {
			factor = 2
			is_ai = no #auto-research
		}
	}
}
