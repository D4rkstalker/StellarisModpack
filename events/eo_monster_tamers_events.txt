namespace = eo_monster_tamers

#####Allow more titanic life armies for Monster Tamers
country_event = {
	id = eo_monster_tamers.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_origin = origin_monster_tamers
		check_variable = {
			which = "titanic_life_soldier_count"
			value > 0
		}
	}
	
	immediate = {
		change_variable = {
			which = "titanic_life_soldier_count"
			value = -1
		}
	}
}
