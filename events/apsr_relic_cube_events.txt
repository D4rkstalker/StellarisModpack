# проверка условий для показа одного из двух типов сообщений
planet_event = {
	id = apsr_arcrel.300
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT ={
			OR = {
				has_deposit = apsr_relic_cube_deposit
				has_planet_flag = apsr_archsite_already_spawn
			}
		}
	}
	
	immediate = {
		owner = {
			save_event_target_as = apsr3_planet_owner_target
		}
		# если: есть материалист И игрок не спирит И игрок не воюет с материалистом И отношение к материалисту нормальное
		if = {
			limit = {
				AND = {
					any_country = {
						OR = {
							is_country_type = fallen_empire
							is_country_type = awakened_fallen_empire 
						}
						has_ethic = ethic_fanatic_materialist
						has_communications = root.owner
						NOT = { is_at_war_with = root.owner }
						opinion = { who = root.owner value > 39 }
					}
					owner = {
						NOT ={
							OR = {
								has_ethic = ethic_spiritualist
								has_ethic = ethic_fanatic_spiritualist
							}
						}
					}
				}
			}
			save_event_target_as = apsr3_planet_target
			random_country = {
				limit = { 
					AND = {
						OR = {
							is_country_type = fallen_empire
							is_country_type = awakened_fallen_empire 
						}
						has_ethic = ethic_fanatic_materialist
					}
				}
				save_event_target_as = apsr3_fallen_empire_target
			}
			owner = {
				country_event = { id = apsr_arcrel.311 }
			}
		}
		# если нет
		if = {
			limit = {
				OR = {
					owner = {
						OR = {
							has_ethic = ethic_spiritualist
							has_ethic = ethic_fanatic_spiritualist
						}
					}
					NOT = {
						any_country = {
							OR = {
								is_country_type = fallen_empire
								is_country_type = awakened_fallen_empire 
							}
							has_ethic = ethic_fanatic_materialist
							has_communications = root.owner
							NOT = { is_at_war_with = root.owner }
							opinion = { who = root.owner value > 39 }
						}
					}
				}
			}
			planet_event = { id = apsr_arcrel.310 }
		}
		set_planet_flag = apsr_archsite_already_spawn
	}
}
