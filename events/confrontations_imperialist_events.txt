namespace = confrontations_imperialist

# Imperialist Confrontations

# Calculating which confrontation to use
country_event = {
    id = confrontations_imperialist.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        NOT = { has_modifier = recent_political_event }
        has_modifier = politics

        check_variable = {
            which = current_imperialist_support
            value > current_ideological_purity
        }
    }

    immediate = {
        country_event = { id = confrontations_imperialist.3 }
    }
}

# Dock imperialist_support
country_event = {
    id = confrontations_imperialist.2
    hide_window = yes

    is_triggered_only = yes

    immediate = {
        add_modifier = {
            modifier = recent_political_event
            months = 3
        }

        add_monthly_resource_mult = {
            resource = imperialist_support
            value = -36
            min = 0
            max = -100
        }
    }
}

# Continue to trigger events, split out so you can call it from console
country_event = {
    id = confrontations_imperialist.3
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        export_resource_stockpile_to_variable = {
            resource = imperialist_militancy_current_value
            variable = current_imperialist_militancy
        }
        while = {
            count = 100
            limit = {
                NOT = { has_modifier = recent_political_event }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_imperialist
                        is_pop_faction_type = imperialist_radicals
                        is_pop_faction_type = imperialist
                    }
                }
                pop_faction_event = { id = confrontations_imperialist.4 }
            }
        }
    }
}

# Pick events based on militancy
pop_faction_event = {
    id = confrontations_imperialist.4
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = {
            limit = {
                owner = {
                    check_variable = {
                        which = current_imperialist_militancy
                        value > 75
                    }
                }
            }
            pop_faction_event = { id = confrontations_imperialist_militancy_desperate.1 }
        }
        else_if = {
            limit = {
                owner = {
                    check_variable = {
                        which = current_imperialist_militancy
                        value > 50
                    }
                }
            }
            pop_faction_event = { id = confrontations_imperialist_militancy_furious.1 }            
        }
        else_if = {
            limit = {
                owner = {
                    check_variable = {
                        which = current_imperialist_militancy
                        value > 25
                    }
                }
            }
            pop_faction_event = { id = confrontations_imperialist_militancy_upset.1 }
        }
        else = {
            pop_faction_event = { id = confrontations_imperialist_militancy_peaceful.1 }
        }
    }
}
