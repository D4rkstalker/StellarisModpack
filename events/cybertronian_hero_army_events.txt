namespace = cybertronian_army

# De-increment variable for counting Autobot/Decepticon armies
country_event = {
	id = cybertronian_army.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = { army_type = autobot_optimus_prime }
	}

	immediate = {
		subtract_variable = {
			which = "autobot_optimus_prime_soldier_count"
			value = 1
		}
	}
}
