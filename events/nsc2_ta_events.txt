# Check for tech_voidcraft_breakthrough
country_event = {
	id = nsc2_ta_event.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	 
	trigger = {
		last_increased_tech = tech_voidcraft_breakthrough
	}
	 
	immediate = {
		if = { limit = { has_technology = tech_pd_destroyer_1 }
			give_technology = { tech = tech_nsc2_ta_pd_check message = no }
		}
		if = { limit = { has_technology = tech_explorationship_2 }
			give_technology = { tech = tech_nsc2_ta_exp2_check message = no }
		}
		if = { limit = { has_technology = tech_explorationship_3 }
			give_technology = { tech = tech_nsc2_ta_exp3_check message = no }
		}
		if = { limit = { has_technology = tech_heavycarrier_1 }
			give_technology = { tech = tech_nsc2_ta_hc_check message = no }
		}
		if = { limit = { has_technology = tech_supercarrier_1 }
			give_technology = { tech = tech_nsc2_ta_sc_check message = no }
		}
		if = { limit = { has_technology = tech_spitfire_1 }
			give_technology = { tech = tech_nsc2_ta_sf_check message = no }
		}
		if = { limit = { has_technology = tech_nsc_dlc_check }
			give_technology = { tech = tech_nsc2_ta_dlc_check message = no }
		}
	}
}
