namespace = eev_start_menu

country_event = {
	id = eev_start_menu.1
	title = eev_start_menu.1.name
	desc = eev_start_menu.1.desc
	picture = GFX_evt_hangar_bay
	is_triggered_only = yes
	trigger = {
		has_country_flag = eev_is_host
	}

	# Activate the kaboom
	option = {
		trigger = {
			NOT = { has_global_flag = eev_kaboom }
		}
		name = eev_start_menu.1.kaboom
		custom_tooltip = eev_start_menu.1.kaboomTT
		set_global_flag = eev_kaboom		
		remove_global_flag = eev_no_kaboom
		hidden_effect = {
			country_event = { id = eev_start_menu.1 }
		}
	}

	# Deactivate the kaboom
	option = {
		trigger = {
			NOT = { has_global_flag = eev_no_kaboom }
		}
		name = eev_start_menu.1.no_kaboom
		custom_tooltip = eev_start_menu.1.no_kaboomTT
		set_global_flag = eev_no_kaboom
		remove_global_flag = eev_kaboom
		hidden_effect = {
			country_event = { id = eev_start_menu.1 }
		}
	}

	option = {
		name = eev_start_menu.1.START
		trigger = {
			OR = {
				has_global_flag = eev_kaboom
				has_global_flag = eev_no_kaboom
			}
		}
		default_hide_option = yes
	}
}
