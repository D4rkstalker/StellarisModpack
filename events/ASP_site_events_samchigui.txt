namespace = ASPmod

### Archaeology: When the world sings 

# Spawn Site
planet_event = {
	id = ASPmod.1
	title = "ASPmod.1.name"
	desc = "ASPmod.1.desc"
	picture = GFX_evt_ASP_samchigui_intro
	show_sound = event_dig_site

	pre_triggers = {
		has_owner = yes
		is_homeworld = no
		original_owner = yes
		has_ground_combat = no
		is_capital = no
		is_occupied_flag = no
	}

	trigger = {
		num_pops > 0
		OR = {
			is_planet_class = pc_desert
			is_planet_class = pc_tropical
			is_planet_class = pc_arid
			is_planet_class = pc_continental
			is_planet_class = pc_ocean
			is_planet_class = pc_tundra
			is_planet_class = pc_arctic
			is_planet_class = pc_alpine
			is_planet_class = pc_savannah
		}
		NOT = { exists = archaeological_site }
		NOR = { 
			has_global_flag = arc_ASP_when_the_world_sings
			has_planet_flag = colony_event
		}
	}

	is_triggered_only = yes

	immediate = {
		set_global_flag = arc_ASP_when_the_world_sings
		set_planet_flag = colony_event
	}
	
	option = {
		name = CURIOUS
		create_archaeological_site = site_ASP_when_the_world_sings
	}
}
