namespace = eo_game_start

####Give techs
country_event = {
	id = eo_game_start.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_origin = origin_next_generation
			}
			add_research_option = tech_epigenetic_triggers
			give_technology = { message = no tech = tech_society_1 }
			give_technology = { message = no tech = tech_alien_life_studies }
		}
		if = {
			limit = {
				has_origin = origin_biomedical_renaissance
			}
			give_technology = { message = no tech = tech_gene_tailoring }
		}
		if = {
			limit = {
				OR = {
					has_origin = origin_long_sleep
					has_origin = origin_reactivated
				}
			}
			give_technology = { message = no tech = tech_cryostasis_1 }
		}
		if = {
			limit = {
				has_origin = origin_stellar_cornucopia
			}
			add_research_option = tech_mine_volatile_motes
			add_research_option = tech_mine_exotic_gases
			add_research_option = tech_mine_rare_crystals
		}
	}
}

###Planet setup
# this: capital planet
# from: founder species
planet_event = {
	id = eo_game_start.2
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		eo_pops = yes
		if = {
			limit = {
				NOT = { has_planet_flag = special_start_planet }
####PD workaround
				OR = {
					is_planet_class = pc_desert
					is_planet_class = pc_tropical
					is_planet_class = pc_arid
					is_planet_class = pc_ocean
					is_planet_class = pc_tundra
					is_planet_class = pc_arctic
					is_planet_class = pc_continental
					is_planet_class = pc_alpine
					is_planet_class = pc_savannah
					is_planet_class = pc_gaia
					is_planet_class = pc_nuked
					is_planet_class = pc_relic
				}
			}
			if = {
				limit = {
					NOT = { has_planet_flag = planet_earth }
					NOT = { 
						owner = { has_origin = origin_shattered_ring } 	#setup in initializer 
					}
				}
				eo_deposits_and_blockers = yes
			}
			eo_buildings_and_districts = yes
			eo_modifiers_and_planets = yes
		}
	}
}

country_event = {
	id = eo_game_start.3
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_country = {
			limit = {
				has_origin = origin_infiltrators
			}
			random_owned_species = {
				limit = {
					NOT = {
						has_citizenship_type = { country = prev type = citizenship_full }
					}
				}
				set_citizenship_type = {
					type = citizenship_limited
					country = prev
				}
			}
		}
	}
}
