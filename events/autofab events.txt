#AI add factory
planet_event = {
	id = autofab.26
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		if = {
			limit = {
				NOT = {
					has_planet_flag = has_autofab
				}
				owner = {
					is_ai = yes
					resource_stockpile_compare = {
						resource = minerals
						value >= 200
					}
					resource_stockpile_compare = {
						resource = alloys
						value >= 100
					}
					resource_stockpile_compare = {
						resource = energy
						value >= 200
					}
				}
			}
			owner = {
				add_resource = {
					minerals = 200
				}
				add_resource = {
					alloys = 100
				}
				add_resource = {
					energy = 200
				}
			}
			planet_event = { id = autofab_automation.28 }
		}
		else = {
			planet_event = { id = autofab_automation.26 days = 90 }
		}
	}
}
