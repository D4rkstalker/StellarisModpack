country_event = {
	id = asptest.10
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		while = {
			every_system = {
				limit = { 
					any_system_planet = { is_star = yes }
					has_owner = no
				}
				random_system_planet = {
					limit = {
						NOR = {
							exists = archaeological_site
							has_anomaly = yes
						}
					}
					create_archaeological_site = random
				}
			}
			count = 100
		}
	}
}
