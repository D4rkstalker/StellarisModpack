namespace = esapnsc2
namespace = esap

#Escort Carrier
ship_event = {
	id = esapnsc2.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_ascension_perk = ap_esap_battleship
		}	
		is_ship_size = escortcarrier
	}
	
	immediate = {
		add_modifier = {
			modifier = "mod_esapnsc2_escortcarrier"
			days = -1
		}
	}
}
