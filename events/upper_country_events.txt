namespace = countryupper

#Teleporter Number Variable
country_event = {
	id = countryupper.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = {
			any_planet = {
				has_building = building_teleporter
			}
		}
	}

	immediate = {
		set_variable = {
			which = planet_teleporters
			value = 0
		}
		every_owned_planet = {
			limit = { has_building = building_teleporter }
			prev = {
				change_variable = {
					which = planet_teleporters
					value = 1
				}
			}
		}
	}
}
