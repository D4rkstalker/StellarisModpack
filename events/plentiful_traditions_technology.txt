country_event = {
	id = plentiful_traditions_technology.2
	hide_window = yes
	is_triggered_only = yes
 
	immediate = {
	set_country_flag = tr_plentiful_order_finish_flag
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = num_of_traditions
						value > 0
					}
				}
			}
			set_variable = {
				which = num_of_traditions
				value = 1
			}
			else = {
				change_variable = {
					which = num_of_traditions
					value = 1
				}
			}
		}
	}
}
