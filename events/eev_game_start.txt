event = {
	id = eev_start.1
	hide_window = yes
    is_triggered_only = yes
	
	trigger = {
		NOT = {
			has_global_flag = has_extra_events
		}
	}
	
	immediate = {
		set_global_flag = has_extra_events
	}
}

country_event = {
	id = eev_start.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		NOT = { has_global_flag = eev_host_set }
	}

	immediate = {
		set_global_flag = eev_host_set
		set_country_flag = eev_is_host
	}
}
