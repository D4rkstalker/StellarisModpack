namespace = AutoArmy
country_event = {
	id = AutoArmy.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_country = {
			is_ai = yes
			is_country_type = default
			num_armies < 20
		}
	}

	immediate = {
		every_country = {
			country_event = { id = AutoArmy.2 }
		}

	}
}

countty_event = {
	id = AutoArmy.2
	hide_window = yes
	is_triggered_only = yes
	limit = {
		is_ai = yes
		is_country_type = default
		num_armies < 20
	}
	save_event_target_as = target_country
	random_owned_planet = {
		create_fleet = {
			name = "NAME_Prethoryn_Armies"
			effect = {
				set_owner = event_target:target_country
				while = {
					count = 20
					create_army_transport = {
						ship_name = "NAME_Prethoryn_Transport"
						graphical_culture = target_country
						army_name = "NAME_Prethoryn_Invaders"
						army_type = "swarm_army"
						species = event_target:prethoryn_species
					}
				}
				set_location = {
					target = PREV
					distance = 10
					angle = random
				}
			}
		}
	}

}
