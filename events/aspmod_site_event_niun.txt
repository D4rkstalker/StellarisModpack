#throne
fleet_event = {
	id = asp.57
	title = "asp.57.name"
	desc = "asp.57.desc"
	picture = GFX_evt_asp.57
	show_sound = event_bad_omen
	
	archaeology = yes
	is_triggered_only = yes
	
	immediate = {
		from = { set_site_progress_locked = yes }
	}

	after = {
		from = { set_site_progress_locked = no }
	}

	option = {
		name = asp.57.a
		from.planet = {
			change_pc = pc_gaia
			hidden_effect = {
				reroll_deposits = yes
			}
		}
	}
	option = {
		name = asp.57.b
		owner = {
			if = {
				limit = { has_technology = tech_terrestrial_sculpting }
				add_modifier = { modifier = aspmod_ancient_terraforming_technology days = -1 }
				else = {
					give_technology = { tech = tech_terrestrial_sculpting }
				}
			}
		}
	}
	option = {
		name = asp.57.c
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
				}
			}
		}
		owner = {
			add_resource = {
				unity = 10000
				influence = 300
			}
		}
		from.planet = {
			add_modifier = { modifier = aspmod_tomb_of_the_king days = -1 }
		}
	}
}
