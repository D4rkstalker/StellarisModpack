# Successful integration
country_event = {
    id = vbp_habitat.2
    title = vbp_habitat.2.name
    desc = vbp_habitat.2.desc

    is_triggered_only = yes

    location = FROM
    picture = GFX_evt_satellite_in_orbit
    show_sound = event_celebration

    immediate = {
        FROM = {
            set_owner = root
            remove_all_buildings = yes
            add_modifier = { modifier = infiltration_happiness days = 7300 }
            IF = {
                limit = {
                    ROOT = { is_regular_empire = yes }
                }
                add_building = building_capital
            }
            ELSE_IF = {
                limit = {
                    root = { is_hive_empire = yes }
                }
                add_building = building_hive_capital
            }
            ELSE_IF = {
                limit = {
                    root = { is_machine_empire = yes }
                }
                add_building = building_machine_capital
            }
            add_district = district_hab_housing
            add_district = district_hab_science
            IF = {
                limit = {
                    root = { is_gestalt = no }
                }
                add_district = district_hab_cultural
                add_district = district_hab_commercial
            }
            ELSE = {
                add_district = district_hab_energy
            }
        }
    }

    option = {
        name = EXCELLENT
    }

    option = {
        name = vbp_habitat.2.assimilator

        exclusive_trigger = {
            has_valid_civic = civic_machine_assimilator
        }
    }

    option = {
        name = vbp_habitat.2.servitor

        exclusive_trigger = {
            has_valid_civic = civic_machine_servitor
        }
    }
}

# Habitat extinction
country_event = {
    id = vbp_habitat.3
    hide_window = yes

    mean_time_to_happen = {
        years = 150
    }

    trigger = {
        is_country_type = primitive
        has_country_flag = vbp_voidbound_civilisation
    }

    immediate = {
        capital_scope = {
            save_event_target_as = habitat
            remove_all_buildings = yes
            every_owned_pop = {
                kill_pop = yes
            }
            while = {
                count = 3
                add_deposit = d_vbp_ruined_section
            }
            if = {
                limit = { has_observation_outpost = yes }
                observation_outpost_owner = {
                    country_event = { id = vbp_habitat.4 }
                }
                observation_outpost = {
                    dismantle = yes
                }
            }
        }
    }
}

# Uplift concerns
country_event = {
    id = vbp_habitat.7
    hide_window = yes
    
    mean_time_to_happen = {
        months = 36
    }

    trigger = {
        is_country_type = primitive
        has_country_flag = vbp_voidbound_civilisation
        NOT = { has_country_flag = vbp_habitat_integration_tension_roll }
        capital_scope = {
            has_observation_outpost = yes
            observation_outpost = {
                has_mission = vbp_negotiate_habitat_integration
            }
        }
    }

    immediate = {
        set_country_flag = vbp_habitat_integration_tension_roll
        random_list = {
            80 = {}
            20 = {
                set_country_flag = vbp_habitat_integration_tension
                capital_scope = {
                    save_event_target_as = habitat
                    observation_outpost = {
                        add_mission_progress = -0.20
                    }
                    observation_outpost_owner = {
                        country_event = {
                            id = vbp_habitat.8
                        }
                    }
                }
                random_list = {
                    50 = {}
                    50 = { set_country_flag = vbp_scheduled_detonation }
                }
                modifier = {
                    factor = 2
                    is_xenophobe = yes
                }
                modifier = {
                    factor = 2
                    is_militarist = yes
                }
                modifier = {
                    factor = 5
                    capital_scope = {
                        observation_outpost_owner = {
                            is_gestalt = yes
                        }
                    }
                }
                modifier = {
                    factor = 1.5
                    capital_scope = {
                        observation_outpost_owner = {
                            OR = {
                                is_xenophobe = yes
                                is_militarist = yes
                            }
                        }
                    }
                }
                modifier = {
                    factor = 0.5
                    is_pacifist = yes
                }
                modifier = {
                    factor = 0.5
                    is_xenophile = yes
                }
            }
        }
    }
}
