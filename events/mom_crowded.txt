namespace = mom_crowded

event = {
	id = mom_crowded.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		while = {
			limit = {
				any_country = {
					has_origin = origin_mom_crowded
					NOT = {
						has_country_flag = mom_crowded_setup
					}
				}
			}
			random_country = {
				limit = {
					has_origin = origin_mom_crowded
					NOT = {
						has_country_flag = mom_crowded_setup
					}
				}
				set_country_flag = mom_crowded_setup
				give_technology = {
					tech = tech_housing_1
					message = no
				}
				give_technology = {
					tech = tech_housing_2
					message = no
				}
				capital_scope = {
					set_planet_flag = mom_crowded_planet
					solar_system = {
						random_system_planet = {
							limit = {
								has_mining_station = no
								is_asteroid = no
								is_star = no
								habitable_planet = no
								NOT = {
									is_planet_class = pc_gas_giant
								}								
							}
							clear_deposits = yes
							add_deposit = d_minerals_10
							add_deposit = d_minerals_10
							add_modifier = {
								modifier = mom_crowded_stripmining
								days = -1
							}
							create_mining_station = {
								owner = prevprevprev
							}
						}						
					}
					add_modifier = {
						modifier = mom_crowded_world 
						days = -1
					}
					clear_blockers = yes
					add_district = district_city
					add_district = district_city
					add_district = district_city
					add_district = district_farming
					add_district = district_farming
					add_district = district_farming
					add_district = district_generator
					add_building = building_precinct_house
					add_building = building_bureaucratic_1
					add_building = building_commercial_zone
					add_building = building_commercial_zone
					add_building = building_holo_theatres
					add_building = building_factory_1
					add_building = building_factory_1
					while = {
						limit = {
							count_pops = {
								count < 65
							}
						}
						create_pop = {
							species = owner
						}
					}
				}
			}
		}
	}
}

event = {
	id = mom_crowded.2
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_planet = {
			has_planet_flag = mom_crowded_planet
			OR = {
				has_owner = no
				is_planet_class = pc_city
				AND = {
					has_owner = yes
					owner = {
						NOT = {
							is_country_type = default
						}
					}
				}
			}
		}
	}

	immediate = {
		while = {
			limit = {
				any_planet = {
					has_planet_flag = mom_crowded_planet
					OR = {
						has_owner = no
						is_planet_class = pc_city
						AND = {
							has_owner = yes
							owner = {
								NOT = {
									is_country_type = default
								}
							}
						}
					}
				}
			}
			random_planet = {
				limit = {
					has_planet_flag = mom_crowded_planet
					OR = {
						has_owner = no
						is_planet_class = pc_city
						AND = {
							has_owner = yes
							owner = {
								NOT = {
									is_country_type = default
								}
							}
						}
					}
				}
				remove_planet_flag = mom_crowded_planet
				remove_modifier = mom_crowded_world
			}
		}
	}
}
