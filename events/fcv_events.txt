namespace = fcv

ship_event = {
	id = fcv.1
	title = "fcv.1.name"
	desc = "fcv.1.desc"
	picture = GFX_evt_ship_in_orbit
	show_sound = event_default

	is_triggered_only = yes

	trigger = {
		is_ship_size = fcv
		owner = {
			NOT = { has_country_flag = fcv_constructed }
		}
	}

	immediate = {
		owner = {
			remove_country_flag = fcv_destroyed
			set_country_flag = fcv_constructed
			remove_modifier = fcv_destroyed_modifier
			add_modifier = {
				modifier = fcv_build_modifier 
				days = -1
			}	
		}
		every_playable_country = {
			limit = {
				has_communications = root.owner
				NOT = { is_same_value = root.owner }
				NOT = { has_country_flag = know_dronecv }
			}
			country_event = { id = "fcv.2" }
		}
	}

	after = {
	}

	option = {
		name = fcv.1.a
	}
}

country_event = {
	id = fcv.3
	title = "fcv.3.name"
	desc = "fcv.3.desc"
	picture = GFX_evt_exploding_ship
	show_sound = event_default
	is_triggered_only = yes
	
	trigger = {
		is_ai = no
		fromfrom = { 
			is_ship_size = fcv	
		}	
	}
	
	immediate = {
		owner = {
			remove_country_flag = fcv_constructed
			remove_modifier = fcv_build_modifier
			set_country_flag = fcv_destroyed
			add_modifier = {
				modifier = "fcv_destroyed_modifier"
				days = 3600
			}
		}
	}

	option = {
		name = fcv.3.a
	}
}
