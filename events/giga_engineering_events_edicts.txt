namespace = gigaedict

#Search for moons
country_event = {
	id = gigaedict.1
	title = gigaedict.1.name
	desc = gigaedict.1.desc
	hide_window = no
	picture = GFX_evt_cold_barren
	show_sound = event_sensor_ping
	is_triggered_only = yes
	
	immediate = {
		every_planet_within_border = {
			limit = {
				OR = {
					is_planet_class = pc_molten
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_frozen
				}
				OR = {
					planet_size = 4
					planet_size = 5
					planet_size = 6
				}
				NOT = {
					has_planet_flag = attack_moon_surveyed
				}
			}
			set_planet_flag = attack_moon_surveyed
			add_modifier = {
				modifier = "attack_moon_candidate"
				days = -1
			}
		}
	}
	
	option = {
		name = gigaedict.1.a
	}
}
