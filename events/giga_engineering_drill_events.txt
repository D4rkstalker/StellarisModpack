	id = gigadrill.1001
	hide_window = yes
	is_triggered_only = yes

		planet_event = { id = gigadrill.1002 days = 720 }
		change_planet_size = -1
		change_pc = pc_strip_mined
	}
}

planet_event = {
	id = gigadrill.1002
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { planet_size = 5 }
			set_planet_flag = reached_inner_core
			solar_system.owner = { country_event = { id = gigadrill.1004 days = 1 } }
		}

		if = {
			limit = { planet_size = 2 }
			change_pc = pc_core_mined
			set_planet_flag = depletedstripmine
			remove_planet_flag = has_megastructure
			solar_system.owner = { country_event = { id = gigadrill.1006 days = 1 } }
		}
		else_if = {
			limit = { planet_size > 2 }
			change_planet_size = -1
			planet_event = { id = gigadrill.1002 days = 720 }
		}
	}
}
