namespace = veru_h

# Overlord changes type of new hive-mind empire vassal
country_event = {
	id = veru_h.1
	title = veru.31.name
	desc = veru.31.desc
	picture = GFX_evt_arguing_senate
	show_sound = event_conversation
	location = event_target:subject_1

	is_triggered_only = yes

	# Transform into Pet
	option = {
		add_resource = { influence = -100 }
		name = SUBJECT_exalted_subject_pet
		custom_tooltip = choice_exalted_subject_pet_tooltip
		trigger = {
			OR = {
				has_ascension_perk = ap_shared_destiny
				has_authority = auth_machine_intelligence
				has_ethic = ethic_authoritarian
				has_ethic = ethic_fanatic_authoritarian
				has_ethic = ethic_militarist
				has_ethic = ethic_fanatic_militarist
				has_ethic = ethic_spiritualist
				has_ethic = ethic_fanatic_spiritualist
				has_ethic = ethic_xenophobe
				has_ethic = ethic_fanatic_xenophobe
			}
		}
		hidden_effect = {
			log = "veru_h.1: Setting [Root.GetName] to Pet"
			event_target:subject_1 = {
				set_subject_of = {
					who = ROOT
					subject_type = exalted_subject_pet
				}

				country_event = { id = exalted_subjects.17 }
			}
		}
	}

	# Transform into Symbiont
	option = {
		add_resource = { influence = -100 }
		name = SUBJECT_exalted_subject_symbiont
		custom_tooltip = choice_exalted_subject_symbiont_tooltip
		trigger = {
			OR = {
				has_ascension_perk = ap_shared_destiny
				has_authority = auth_hive_mind
				has_ethic = ethic_egalitarian
				has_ethic = ethic_fanatic_egalitarian
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
				has_ethic = ethic_pacifist
				has_ethic = ethic_fanatic_pacifist
				has_ethic = ethic_xenophile
				has_ethic = ethic_fanatic_xenophile
			}
		}
		hidden_effect = {
			log = "veru_h.1: Setting [Root.GetName] to Symbiont"
			event_target:subject_1 = {
				set_subject_of = {
					who = ROOT
					subject_type = exalted_subject_symbiont
				}

				country_event = { id = exalted_subjects.16 }
			}
		}
	}

	# Leave as vanilla Vassal
	option = {
		name = choice_veru_vassal
		custom_tooltip = choice_exalted_subject_vassal_tooltip

		hidden_effect = {
			log = "veru._h.1: Leaving [Root.GetName] as Vassal"
			event_target:subject_1 = {
				set_subject_of = {
					who = root
					subject_type = vervassal
				}
			}
		}
	}
	
	option = {
		name = veru.3.back
	}
}
