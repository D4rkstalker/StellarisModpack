namespace = lode_runner_events

# Researched L-Gate tech
country_event = {
	id = lode_runner_events.1
	hide_window = yes
	fire_only_once = yes

	is_triggered_only = yes

	trigger = {
		last_increased_tech = tech_lgate_activation
		NOT = { has_global_flag = l_cluster_opened }
	}

	immediate = { 
		if = {
			limit = {
				has_global_flag = gray_goo_crisis_set
			}
			remove_global_flag = gray_goo_crisis_set
		}
		if = {
			limit = {
				has_global_flag = dragon_season
			}
			remove_global_flag = dragon_season
		}
		if = {
			limit = {
				has_global_flag = gray_goo_empire_set
			}
			remove_global_flag = gray_goo_empire_set
		}
		reroll_random = yes
		random_list = {
			10 = {				# gray tempest
				modifier = {
					factor = 0
					NOT = {
						has_global_flag = gray_goo_crisis_allowed
					}
				}
				log = "set l-cluster as gray tempest"
				set_global_flag = gray_goo_crisis_set
			}
			10 = {				# dragon season
				modifier = {
					factor = 0
					NOT = {
						has_global_flag = dragon_season_allowed
					}
				}
				log = "set l-cluster as dragon season"
				set_global_flag = dragon_season
			}
			10 = {				# gray goo empire
				modifier = {
					factor = 0
					NOT = {
						has_global_flag = gray_goo_empire_set_allowed
					}
				}
				log = "set l-cluster as gray empire"
				set_global_flag = gray_goo_empire_set
			}
			0 = {				# no outcome 
				modifier = {
					add = 10
					NAND = {
						has_global_flag = gray_goo_empire_set_allowed
						has_global_flag = dragon_season_allowed
						has_global_flag = gray_goo_crisis_allowed
					}
				}
				log = "set l-cluster as empty"
			}	
		}
	}
}
