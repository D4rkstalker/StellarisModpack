namespace = deathstarbattlestation

ship_event = {
	id = deathstarbattlestation.1
	title = "deathstarbattlestation.1.name"
	desc = "deathstarbattlestation.1.desc"
	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_default

	is_triggered_only = yes

	trigger = {
		is_ship_size = deathstarbattlestation
		owner = {
			NOT = { has_country_flag = deathstarbattlestation_constructed }
		}
	}

	immediate = {
		owner = {
			remove_country_flag = deathstarbattlestation_destroyed
			set_country_flag = deathstarbattlestation_constructed
			remove_modifier = deathstarbattlestation_destroyed_modifier
			add_modifier = {
				modifier = deathstarbattlestation_build_modifier 
				days = -1
			}	
		}
		every_playable_country = {
			limit = {
				has_communications = root.owner
				NOT = { is_same_value = root.owner }
				NOT = { has_country_flag = know_deathstarbattlestation }
			}
			country_event = { id = "deathstarbattlestation.2" }
		}
	}

	after = {
	}

	option = {
		name = deathstarbattlestation.1.a
	}
}

country_event = {
	id = deathstarbattlestation.2
	title = "deathstarbattlestation.2.name"
	desc = "deathstarbattlestation.2.desc"
	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_default
	
	is_triggered_only = yes

	immediate = {
	}

	option = {
		name = deathstarbattlestation.2.a
		hidden_effect = {
			add_opinion_modifier = { 
				who = root.owner
				modifier = deathstarbattlestation_build_opinion 
			}
		}
	}
}

country_event = {
	id = deathstarbattlestation.3
	title = "deathstarbattlestation.3.name"
	desc = "deathstarbattlestation.3.desc"
	picture = GFX_evt_ship_in_orbit_3
	show_sound = event_default
	is_triggered_only = yes
	
	trigger = {
		is_ai = no
		fromfrom = { 
			is_ship_size = deathstarbattlestation	
		}	
		 count_owned_ship = {
			limit = { is_ship_size = deathstarbattlestation }
			count < 1
		}
	}
	
	immediate = {
		owner = {
			remove_country_flag = deathstarbattlestation_constructed
			remove_modifier = deathstarbattlestation_build_modifier
			set_country_flag = deathstarbattlestation_destroyed
			add_modifier = {
				modifier = "deathstarbattlestation_destroyed_modifier"
				days = 3600
			}
		}
	}

	option = {
		name = deathstarbattlestation.3.a
	}
}
