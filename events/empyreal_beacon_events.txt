namespace = Empyreal_beacon
country_event = {
	id = Empyreal_beacon.0001
	hide_window = no
	is_triggered_only = yes
	title = "Beacon_Constructed"
	picture = GFX_evt_galactic_senate
	desc = "Beacon_Constructed_desc"
	option = {
		name = "Empyreal_beacon_done"
		custom_tooltip = "Empyreal_beacon_done_desc"
		hidden_effect = {
			every_country = {
				limit = {
					is_galactic_community_member = yes
				}
				add_modifier = {
					modifier = "empyreal_beacon_modifier"
					days = -1
				}
			}
		}
	}
}

#Mod is added to new people# 
country_event = {
	id = Empyreal_beacon.0003
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_global_flag = empyreal_beacon_exists
			}
			add_modifier = {
				modifier = "empyreal_beacon_modifier"
			}
		}
	}
}

	id = Empyreal_beacon.0002
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {

		}
	}
}
