namespace = civleader

leader_event = { # Kill Dummy Leaders
	id = civleader.1000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = this
	}

	immediate = {
		if = {
			limit = { NOT = { has_leader_flag = dummy_leader } }
			set_leader_flag = dummy_leader
		}
		kill_leader = { show_notification = no }
	}
}

country_event = { # Set Corrupt Governor Flag
	id = civleader.2000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type_with_subjects = yes
		any_owned_leader = {
			has_trait = leader_trait_corrupt
			NOT = { has_leader_flag = corrupt_governor }
		}
	}

	immediate = {
		every_owned_leader = {
			limit = {
				has_trait = leader_trait_corrupt
				NOT = { has_leader_flag = corrupt_governor }
			}
			set_leader_flag = corrupt_governor
		}
	}
}

# on_ruler_set
# Executed as new ruler has been set
# This = Country
country_event = { # Set Corrupt Ruler Trait
	id = civleader.2005
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		ruler = {
			has_leader_flag = corrupt_governor
			NOT = { has_trait = leader_trait_corrupt }
		}
	}

	immediate = {
		ruler = { add_trait = leader_trait_corrupt }
	}
}

# on_ruler_set
# Executed as new ruler has been set
# This = Country
country_event = { # Set Sep Won Election Flag
	id = civleader.2100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		ruler = {
			exists = pop_faction
			pop_faction = { is_separatist_faction = yes }
		}
	}

	immediate = {
		ruler = { set_timed_leader_flag = { flag = sep_won_election years = 7 } }
	}
}
