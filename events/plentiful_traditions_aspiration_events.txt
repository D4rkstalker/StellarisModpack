ship_event = {
	id = plentiful_traditions_aspiration.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			OR = {
				has_swapped_tradition = tr_plentiful_aspiration_gestalt_adopt
				has_non_swapped_tradition = tr_plentiful_aspiration_adopt
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				owner = {
					has_tradition = tr_plentiful_aspiration_adopt
				}
			}
			from = {
				planet_event = {
					id = plentiful_traditions_aspiration.4
				}
			}
				
		}
	}	
}
