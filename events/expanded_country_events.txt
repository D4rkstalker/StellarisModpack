# Science Academy asks to investigate tollok_rong
country_event = {
	id = countryexpanded.1
	title = "anomaly.5025.name"
	desc = "countryexpanded.1.desc"
	
	diplomatic = yes
	
	picture_event_data = {
		room = no_video_feed_room
	}
	
	trigger = {
		has_country_flag = new_baldarak_settled
		OR = {
			NOT = { has_country_flag = new_baldarak_contact_lost }
			has_country_flag = new_baldarak_occupation_4
			has_country_flag = baldarak_revolution_4
		}
		has_country_flag = baldarak_science_academy
		NOT = { has_country_flag = baldarak_planetary_shield }
		is_at_war = no
	}
	
	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.5
			OR = {
				has_country_flag = new_baldarak_occupation_4
				has_country_flag = baldarak_revolution_4
			}
		}
		modifier = {
			factor = 0.5
			mid_game_years_passed >= 50
		}
		modifier = {
			factor = 0.5
			mid_game_years_passed >= 100
		}
	}
	
	immediate = {
		set_country_flag = baldarak_planetary_shield
		random_system = {
			limit = {
				any_system_planet = {
					has_planet_flag = tollok_rong
				}
			}
			random_system_planet = {
				limit = { has_planet_flag = tollok_rong }
				save_event_target_as = is_tollok_rong
			}
		}
	}
	
	option = {
		name = countryexpanded.1.a
		response_text = countryexpanded.1.a.response
		event_target:is_tollok_rong = {
			enable_special_project = {
				name  = INVESTIGATING_TOLLOK_RONG_PROJECT
				location = this
				owner = root
			}
		}
	}
	
	option = {
		name = anomaly.5011.b
		response_text = countryexpanded.1.b.response
	}
}
