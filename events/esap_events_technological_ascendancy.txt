namespace = esap_technological_ascendancy

# Technological Ascendacy bonus 2, adds research bonus when researching repeatable techs (finds appropriate empires).
event = {
	id = esap_technological_ascendancy.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_ascension_perk = ap_esap_technological_ascendancy_bonus_2 
		}	
	}
	
	immediate = {
		every_country = {
			limit =  {
				has_ascension_perk = ap_esap_technological_ascendancy_bonus_2 	
			}
			country_event = {
				id = esap_technological_ascendancy.2
			}
		}
	}
}
