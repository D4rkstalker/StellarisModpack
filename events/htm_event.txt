	is_triggered_only = yes
	hide_window = yes

		if = { # Only generate the list if it doesn't already exist because we went back and forward again
			limit = { 
				NOT = {

			every_system = { 
				limit = { 
					has_hyperlane_to = event_target:htm_system.solar_system
				}
			htm_truncator_list_save = yes
			}
		}
		
		country_event = { id = htm_event.302 }
	}
}
