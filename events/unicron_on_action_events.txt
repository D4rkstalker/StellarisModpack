namespace = unicron_action

planet_event = {
	id = unicron_action.80
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		planet_devastation >= 50
		NOT = {
			has_deposit = d_dark_energon_crystals
		}
	}
	
	immediate = {
		add_deposit = d_dark_energon_crystals
	}
}


# Planet is turned into a dark energon world by special bombing stance
planet_event = {
	id = unicron_action.81
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		OR = {
			has_orbital_bombardment_stance = unicron
		}
		from = {
			OR = {
				is_country_type = default
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
		habitable_planet = yes
	}

	immediate = {
		owner = {
			add_static_war_exhaustion = {
				attacker = from
				location = root
				value_for_planet_destruction = 0.5
			}
			country_event = { id = unicron_action.82 }
		}
		from = { country_event = { id = unicron_action.83 } }
		change_pc = pc_dark_energon
		clear_deposits = yes
		random_list = {
			1 = {
				add_deposit = d_dark_energon_crystals_5
				add_deposit = d_dark_energon_crystals_2
			}
			1 = {
				add_deposit = d_dark_energon_crystals_5
				add_deposit = d_dark_energon_crystals_2
			}
			1 = {
				add_deposit = d_dark_energon_crystals_5
			}
			1 = {
				add_deposit = d_dark_energon_crystals_1
			}
		}
		set_planet_entity = {
			entity = "dark_energon_planet_01_entity"
		}
		set_planet_flag = nuked_planet_anomalies_disabled
		remove_all_buildings = yes
		reroll_deposits = yes
		remove_modifier = "natural_beauty"
		remove_modifier = "atmospheric_aphrodisiac"
		remove_modifier = "atmospheric_hallucinogen"
		remove_modifier = "lush_planet"
		remove_modifier = "dangerous_wildlife"
		set_planet_flag = armageddon_nuked
		add_deposit = d_dark_energon_planet_deposit
	}
}

	picture = GFX_evt_nuclear_explosion
