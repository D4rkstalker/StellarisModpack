country_event = {
	id = alminv.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_country_type = default
	}
	
	immediate = {
		random_owned_planet = {
			limit = {
				is_colony = yes
				nor = { 
					has_building = building_lightmechbay
					has_building = building_mediummechbay
					has_building = building_heavymechbay
					has_building = building_assaultmechbay
				}
			}
			every_planet_army = {
				limit = { army_type = mech_defense_army }
				remove_army = yes
			}
			every_owned_pop = {
				limit = {
					has_pop_flag = popf_spawned_mech@planet
				}
				remove_pop_flag = popf_spawned_mech@planet	
			}			
		}
		random_owned_planet = {
			limit = {
				is_colony = yes
				or = { 
					has_building = building_lightmechbay
					has_building = building_mediummechbay
					has_building = building_heavymechbay
					has_building = building_assaultmechbay
				}
			}
			if = {
				limit = {
					not = {
						any_owned_pop = {
							has_job = mechwarrior
						}
					}
				}
				every_planet_army = {
					limit = { army_type = mech_defense_army }
					remove_army = yes
				}			
			}	
			else = {
				every_owned_pop = {
					limit = {
						has_pop_flag = popf_spawned_mech@planet
						not = { has_job = mechwarrior }
					}
					remove_pop_flag = popf_spawned_mech@planet
					planet = {
						random_planet_army = {
							limit = { army_type = mech_defense_army }
							remove_army = yes
						}						
					}
				}
				while = {
					count = 6
					random_owned_pop = {
						limit = {
							has_job = mechwarrior
							not = { has_pop_flag = popf_spawned_mech@planet }
						}
						set_pop_flag = popf_spawned_mech@planet
						planet = {
							create_army = {
								owner = root
								type = "mech_defense_army"
							}					
						}
					}
				}
			}
		}
		random_owned_planet = {
			limit = {
				is_colony = yes
				not = { has_building = building_aerofighter_base }
			}
			every_planet_army = {
				limit = { army_type = air_defense_army }
				remove_army = yes
			}
			every_planet_army = {
				limit = { army_type = robotic_air_defense_army }
				remove_army = yes
			}
			every_planet_army = {
				limit = { army_type = machine_air_defense }
				remove_army = yes
			}
			every_owned_pop = {
				limit = {
					has_pop_flag = popf_spawned_army@planet
				}
				remove_pop_flag = popf_spawned_army@planet	
			}			
		}
		random_owned_planet = {
			limit = {
				is_colony = yes
				has_building = building_aerofighter_base
			}
			if = {
				limit = {
					not = {
						any_owned_pop = {
							or = {
								has_job = pilot
								has_job = pilot_drone
							}
						}
					}
				}
				every_planet_army = {
					limit = { army_type = air_defense_army }
					remove_army = yes
				}
				every_planet_army = {
					limit = { army_type = robotic_air_defense_army }
					remove_army = yes
				}
				every_planet_army = {
					limit = { army_type = machine_air_defense }
					remove_army = yes
				}				
			}
			else = {
				every_owned_pop = {
					limit = {
						has_pop_flag = popf_spawned_army@planet
						nor = {
							has_job = pilot
							has_job = pilot_drone					
						}
					}
					remove_pop_flag = popf_spawned_army@planet
					if = { 
						limit = {
							has_trait = trait_mechanical
						}
						planet = {
							random_planet_army = {
								limit = { army_type = robotic_air_defense_army }
								remove_army = yes
							}						
						}
					}
					else_if = {
						limit = {
							has_trait = trait_machine_unit
						}
						planet = {
							random_planet_army = {
								limit = { army_type = machine_air_defense }
								remove_army = yes
							}						
						}					
					}
					else = {
						planet = {
							random_planet_army = {
								limit = { army_type = air_defense_army }
								remove_army = yes
							}						
						}					
					}
				}
				while = {
					count = 4
					random_owned_pop = {
						limit = {
							or = {
								has_job = pilot
								has_job = pilot_drone
							}
							not = { has_pop_flag = popf_spawned_army@planet }
						}
						set_pop_flag = popf_spawned_army@planet
						if = {
							limit = {
								has_trait = "trait_machine_unit"
							}
							planet = {
								create_army = {
									owner = root
									type = "machine_air_defense"
								}					
							}
						}
						else_if = {
							limit = {
								has_trait = "trait_mechanical"
							}
							planet = {
								create_army = {
									owner = root
									type = "robotic_air_defense_army"
								}					
							}
						}
						else = {
							planet = {
								create_army = {
									owner = root
									type = "air_defense_army"
								}					
							}					
						}
					}				
				}			
			}
		}
	}
}
