namespace = o_game_pop_added

# A pop has been added to the planet
# Root = pop
# From = planet
pop_event = {
	id = o_game_pop_added.0
	hide_window = yes
	is_triggered_only = yes

	immediate = {					
		if = { limit = { exists = owner owner = { has_origin = origin_hive_overmind } }
		
		if = { limit = { is_same_species = owner_species }
		
		kill_pop = yes
		from = {
			  planet_event = { id = o_planet_pop_added.0 }			
			}
			}			
		}
	}
	}
