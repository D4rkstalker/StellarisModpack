# Stage 3 - randomize stage 3
fleet_event = {
	id = ancrel.7003
	hide_window = yes
	
	archaeology = yes
	is_triggered_only = yes
	
	immediate = {
		fromfrom = { set_site_progress_locked = yes }
		random_list = {
			50 = {
				fleet_event = { id = ancrel.7004 }
			}
			50 = {
				fleet_event = { id = ancrel.7005 }
			}
		}
	}
    
	after = { 
		fromfrom = { 
			set_site_progress_locked = no 
			expire_site_event = ancrel.7003
		} 
	}
}
