namespace=dragons_slaves
# secondary species: assign trait_machine_unit, turn into robotics, then create pops
# this: capital planet
# from: secondary species
planet_event = {
	id = dragons_slaves.1
	hide_window = yes
	is_triggered_only = yes
	trigger = { owner = { has_valid_civic = civic_dragon_slaves } }

	immediate = {
		from = { save_event_target_as = secondary_species }

		if = { limit = { owner = { has_valid_civic = civic_dragon_slaves } }
			event_target:secondary_species = {
				set_citizenship_type = {
					country = root.owner
					type = citizenship_slavery
				}
			#	set_colonization_control = {
			#		country = root.owner
			#		type = colonization_control_yes 
			#	}
			}
			while = {
				count = 3
				random_tile = {
					limit = {
						OR = {
							has_building = building_power_plant_1
							has_building = building_mining_network_1
							has_building = building_hydroponics_farm_1
							has_building = building_basic_science_lab_1 
						}
						has_pop = no
					}
					create_pop = {
						species = event_target:secondary_species
						ethos = random
					}
				}
			}
		}
	}
}
