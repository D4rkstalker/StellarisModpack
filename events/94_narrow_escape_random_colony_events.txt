# Random event chance
country_event = {
	id = narrow_escape_random_colony.11

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_origin = origin_narrow_escape
		mid_game_years_passed >= 0
	}

	immediate = {
		every_owned_planet = {
			limit = {
				NOR = {
					has_planet_flag = narrow_escape_new_colony
					has_planet_flag = narrow_escape_ongoing_event
					has_planet_flag = narrow_escape_stop_colony_events
					has_global_flag = narrow_escape_stop_colony_events_global
				}
			}
			random_list = {
				5 = {
					planet_event = { id = narrow_escape_random_colony.12 }
				}
				95 = {
					modifier = {
						factor = 0
						OR = {
							has_global_flag = narrow_escape_force_colony_events_global
							has_planet_flag = narrow_escape_force_colony_events
						}
					}
				}
				0 = {
					if = {
						limit = { always = no }
						set_global_flag = narrow_escape_force_colony_events_global
						set_planet_flag = narrow_escape_force_colony_events
						set_global_flag = narrow_escape_stop_colony_events_global
						set_planet_flag = narrow_escape_stop_colony_events
						owner = { set_country_flag = narrow_escape_force_archaeology_research }
					}
				}
			}
		}
	}
}
