	fire_only_once = yes

	trigger = {
		is_artificial = no
		NOT = { is_planet_class = pc_nuked }
		years_passed = 15

			is_country_type = primitive

			is_ai = no
			is_homicidal = no
		}
	}

	immediate = {
		random_list = {
			40 = { observation_outpost_owner = { country_event = { id = eev_popup.22 days = 200 random = 100 } } }
			60 = { }
		}
	}
}
