namespace = Incubator
country_event = {
	id = Incubator.0001
	hide_window = no
	is_triggered_only = yes
	title = "Incubator_Constructed"
	picture = GFX_evt_galactic_senate
	desc = "Incubator_Constructed_desc"
	option = {
		name = "Incubator_done"
		custom_tooltip = "Incubator_done_desc"
		hidden_effect = {
			every_country = {
				limit = {
					is_galactic_community_member = yes
				}
				add_modifier = {
					modifier = "Incubator_modifier"
					days = -1
				}
			}
		}
	}
}

#Mod is added to new people# 
country_event = {
	id = Incubator.0003
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_global_flag = Incubator_exists
			}
			add_modifier = {
				modifier = "Incubator_modifier"
			}
		}
	}
}

#mod is removed from leavers
country_event = {
	id = Incubator.0002
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_modifier = "Incubator_modifier"
			}
			remove_modifier = "Incubator_modifier"
		}
	}
}
