namespace = fallen_choices

	id = fallen_choices.1
	title = fallen_choices.1.name
	desc = fallen_choices.1.desc
	picture = GFX_evt_fallen_empire_awakes

	trigger = {
		AND = {
			has_technology = "tech_fallen"
			NOT = { has_technology = tech_fallen_turret }
			NOT = { has_technology = tech_fallen_missile }
		}
	}

	#Turret Only
	option = {
		name = fallen.turret
		give_technology = { tech = tech_fallen_turret message = no }
	}
	
	#Missile Only
	option = {
		name = fallen.missile
		give_technology = { tech = tech_fallen_missile message = no }
	}
	
	#Both
	option = {
		name = fallen.both
		give_technology = { tech = tech_fallen_turret message = no }
		give_technology = { tech = tech_fallen_missile message = no }
	}
}
