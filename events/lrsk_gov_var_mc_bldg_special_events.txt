# Handle special military leader bonus from Military Excellence Academy (Gate for delay)
country_event = {
	id = lrsk_gvp_mc_bldg_special.10
	hide_window = yes
	is_triggered_only = yes

	# Only fire if leader's empire has a bonus and the leader is appropriate
	trigger = {
		OR = {
			AND = {
				check_variable = { which = lrsk_general_training_bonus value > 0 }
				from = { leader_class = general }
			}
			AND = {
				check_variable = { which = lrsk_admiral_training_bonus value > 0 }
				from = { leader_class = admiral }
			}
		}
	}

	immediate = {
		log = "Handling special military leader bonus from Military Excellence Academy  (Gate for delay)"
		log = "Leader = [from.GetTitle] [from.GetName] || General training bonus = [this.lrsk_general_training_bonus] || Admiral training bonus = [this.lrsk_admiral_training_bonus]"

		from = { set_timed_leader_flag = { flag = lrsk_leveling_up_mil_leader days = 3 } }
		# We want to give the default level up event a chance to run. It will temp flag any leader that gets a trait.
		if = { limit = { from = { NOT = { has_leader_flag = has_gained_trait } } } country_event = { id = lrsk_gvp_mc_bldg_special.11 days = 1 } }
	}
}

# Handle pop steal from Reformatory Workhouse
country_event = {
	id = lrsk_gvp_mc_bldg_special.20
	hide_window = yes
	mean_time_to_happen = { months = 12 }

	# Only fire if a criminal slaver corp with built workhouses, and if there are applicable planets/empires
	trigger = {
		has_valid_civic = civic_criminal_heritage
		has_valid_civic = civic_indentured_assets
		check_variable = { which = lrsk_num_built_workhouses value > 0 }
		
		any_country = {
			NOT = { has_country_flag = lrsk_pop_stolen_cooldown }
			check_variable = { which = lrsk_num_owned_workhouses value > 0 }
			any_owned_planet = {
				num_pops > 1 # don't want to steal last pop...
				has_planet_flag = lrsk_could_lose_pops_to@root
			}
		}
	}

	immediate = {
		save_event_target_as = lrsk_slaver_corp_to_steal
		log = "Handling pop steal from Reformatory Workhouse"

		every_country = {
			limit = {
				NOT = { has_country_flag = lrsk_pop_stolen_cooldown }
				check_variable = { which = lrsk_num_owned_workhouses value > 0 }
				any_owned_planet = {
					num_pops > 1 # don't want to steal last pop...
					has_planet_flag = lrsk_could_lose_pops_to@root
				}
			}

			while = {
				limit = {
					NOT = { has_country_flag = lrsk_pop_stolen_cooldown }
					any_owned_planet = {
						num_pops > 1 # don't want to steal last pop...
						has_planet_flag = lrsk_could_lose_pops_to@root
						NOT = { has_planet_flag = lrsk_pop_steal_rolled }
					}
				}

				random_owned_planet = {
					limit = {
						num_pops > 1 # don't want to steal last pop...
						has_planet_flag = lrsk_could_lose_pops_to@root
						NOT = { has_planet_flag = lrsk_pop_steal_rolled }
					}

					set_timed_planet_flag = { flag = lrsk_pop_steal_rolled days = 7 } # marks this planet

					random_list = {
						5 = {
							modifier = {
								add = -4
								num_pops < 5
							}
							modifier = {
								add = 5
								num_pops > 15
							}
							modifier = {
								add = 5
								num_pops > 30
							}
							modifier = {
								add = 5
								num_pops > 45
							}
							modifier = {
								add = 5
								planet_crime > 20
							}
							modifier = {
								add = 5
								planet_crime > 40
							}
							modifier = {
								add = 5
								planet_crime > 60
							}
							modifier = {
								add = 5
								pop_percentage = { percentage > 0.1 limit = { is_unemployed = yes } }
							}
							modifier = {
								add = 5
								pop_percentage = { percentage > 0.3 limit = { is_unemployed = yes } }
							}
							modifier = {
								add = 5
								pop_percentage = { percentage > 0.5 limit = { is_unemployed = yes } }
							}
							modifier = {
								factor = 20
								root = { has_country_flag = lrsk_gvp_testcall }
							}

							# Do the steal

							owner = {
								set_timed_country_flag = { flag = lrsk_pop_stolen_cooldown days = 1800 } # 5 year cooldown
								save_event_target_as = lrsk_slaver_corp_abduct_gov
							}
							save_event_target_as = lrsk_slaver_corp_abduct_location

							# Choice of pop to enslave starts with those who would be "least missed" or are most desperate.
							IF = {
								limit = { any_owned_pop = { OR = { 
									is_unemployed = yes
									is_pop_category = slave
									is_pop_category = purge
								} } }
								random_owned_pop = {
									limit = { OR = { 
										is_unemployed = yes
										is_pop_category = slave
										is_pop_category = purge
									} }
									set_pop_flag = lrsk_pop_to_abduct
								}
							}
							ELSE_IF = { limit = { any_owned_pop = { is_pop_category = worker } }
								random_owned_pop = { limit = { is_pop_category = worker } set_pop_flag = lrsk_pop_to_abduct }
							}
							ELSE_IF = { limit = { any_owned_pop = { is_pop_category = specialist } }
								random_owned_pop = { limit = { is_pop_category = specialist } set_pop_flag = lrsk_pop_to_abduct }
							}
							ELSE = { random_owned_pop = { set_pop_flag = lrsk_pop_to_abduct } }

							
							IF = {
								limit = { any_owned_pop = { has_pop_flag = lrsk_pop_to_abduct } }
								random_owned_pop = { limit = { has_pop_flag = lrsk_pop_to_abduct }
									remove_pop_flag = lrsk_pop_to_abduct
									resettle_pop = {
										pop = this
										planet = root.capital_scope
									}
									# Note: I don't think it makes sense to try to force any slavery category by event.
									# Instead the player is notified that they should check the species rights of the pop.
								}

								root = { country_event = { id = lrsk_gvp_mc_bldg_special.21 } }
								event_target:lrsk_slaver_corp_abduct_gov = { country_event = { id = lrsk_gvp_mc_bldg_special.22 } }
							}
						}
						95 = {
							modifier = {
								add = 4
								num_pops < 5
							}
							modifier = {
								add = -5
								num_pops > 15
							}
							modifier = {
								add = -5
								num_pops > 30
							}
							modifier = {
								add = -5
								num_pops > 45
							}
							modifier = {
								add = -5
								planet_crime > 20
							}
							modifier = {
								add = -5
								planet_crime > 40
							}
							modifier = {
								add = -5
								planet_crime > 60
							}
							modifier = {
								add = -5
								pop_percentage = { percentage > 0.1 limit = { is_unemployed = yes } }
							}
							modifier = {
								add = -5
								pop_percentage = { percentage > 0.3 limit = { is_unemployed = yes } }
							}
							modifier = {
								add = -5
								pop_percentage = { percentage > 0.5 limit = { is_unemployed = yes } }
							}
							modifier = {
								factor = 0
								root = { has_country_flag = lrsk_gvp_testcall }
							}
							
							# Do nothing
						}
					}
				}
			}
		}
	}
}
