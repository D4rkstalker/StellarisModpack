#Basic Menu

country_event = {
	id = gigamenu.1
	title = gigamenu.1.name
	desc = gigamenu.1.desc
	picture = GFX_evt_ringworld
	location = FROMFROM
	
	is_triggered_only = yes
	
	trigger = {
		is_multiplayer = no
	}
	
	#Default, use all megas
	option = {
		name = gigamenu.1.a
		custom_tooltip = gigamenu.1.a.tooltip
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = giga_game_started }
				}
				country_event = { id = gigaflusion.1 days = 1 }
			}
			if = {
				limit = {
					NOT = { has_global_flag = giga_game_started }
					NOT = { has_global_flag = fe_planetcrafts_disabled }
				}
				country_event = { id = fallenplanetcraft.1 days = 1 }
			}
			set_global_flag = giga_game_started #prevents events from running again if menu is resummoned from the edicts screen
		}
	}
	
	#Customize individually
	option = {
		name = gigamenu.1.b
		custom_tooltip = gigamenu.1.b.tooltip
		hidden_effect = {
			country_event = { id = gigamenu.2  }
		}
	}
	
	#Presets
	option = {
		name = gigamenu.1.c
		custom_tooltip = gigamenu.1.c.tooltip
		hidden_effect = {
			country_event = { id = gigamenu.3  }
		}
	}

	#More options (not available after game start)
	option = {
		name = gigamenu.1.d
		custom_tooltip = gigamenu.1.d.tooltip
		trigger = {
			NOT = { has_global_flag = giga_game_started }
		}
		hidden_effect = {
			country_event = { id = gigamenu.4  }
		}
	}
}
