namespace = planet_type

event = {
	id = planet_type.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_planet = {
			is_planet_class = pc_virtual_world
			NOT = { has_planet_flag = modifier_active }	
		}
	}
	
	immediate = {
		every_planet = {
			limit = {
				is_planet_class = pc_virtual_world
				NOT = { has_planet_flag = modifier_active }	
			}
			set_planet_flag = modifier_active
			set_planet_flag = colony_event
			add_modifier = {
				modifier = virtual_world
				days = -1
			}
		}
	}
}
