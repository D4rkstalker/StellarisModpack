# Insult broadcasted
country_event = {
	id = diplomatic.1

	hide_window = yes
	
	trigger = {
		OR = {
			has_ethic = ethic_xenophobe
			has_ethic = ethic_fanatic_xenophobe
		}
		any_owned_fleet = {
			exists = leader
			leader = { leader_class = admiral }
			exists = solar_system
			solar_system = {
				exists = space_owner
				space_owner = {
					has_communications = root
					NOR = {
						is_country = ROOT
						is_at_war_with = ROOT
						is_subject = yes
					}
				}
			}
		}
	}
	
	#is_triggered_only = yes
	mean_time_to_happen = { months = 6000 }
	
	immediate = {
		random_owned_fleet = {
			limit = {
				exists = leader
				solar_system = {
					exists = space_owner
					space_owner = {
						has_communications = root
						NOR = {
							is_country = ROOT
							is_at_war_with = ROOT
						}
					}
				}
			}
			save_event_target_as = broadcasted_insult_fleet
			leader = { save_event_target_as = broadcasted_insult_leader }
			solar_system = {
				save_event_target_as = broadcasted_insult_system
				space_owner = {
					country_event = { id = diplomatic.2 }
				}
			}
		}
	}
}

# Science Ship in territory
country_event = {
	id = diplomatic.9

	hide_window = yes
	
	trigger = {
		any_owned_ship = {
			is_ship_class = shipclass_science_ship
			exists = solar_system
			solar_system = {
				exists = space_owner
				space_owner = {
					has_communications = root
					NOR = {
						is_country = ROOT
						is_at_war_with = ROOT
						is_subject = yes
					}
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
						AND = {
							ROOT = {
								NOT = { has_modifier = alien_genome_research }
								has_technology = tech_genome_mapping
							}
							OR = {
								has_ethic = ethic_xenophobe
								has_ethic = ethic_fanatic_xenophobe
							}
						}
						AND = {
							#exists = leader
							OR = {
								has_ethic = ethic_spiritualist
								has_ethic = ethic_fanatic_spiritualist
							}
							ROOT = {
								NOR = {
									has_ethic = ethic_spiritualist
									has_ethic = ethic_fanatic_spiritualist
								}
							}
						}
					}
				}
			}
		}
	}
	
	#is_triggered_only = yes
	mean_time_to_happen = { months = 3000 }
	
	immediate = {
		random_owned_ship = {
			limit = {
				is_ship_class = shipclass_science_ship
				exists = solar_system
				solar_system = {
					exists = space_owner
					space_owner = {
						has_communications = root
						NOR = {
							is_country = ROOT
							is_at_war_with = ROOT
							is_subject = yes
						}
						OR = {
							has_ethic = ethic_materialist
							has_ethic = ethic_fanatic_materialist
							AND = {
								ROOT = {
									NOT = { has_modifier = alien_genome_research }
									has_technology = tech_genome_mapping
								}
								OR = {
									has_ethic = ethic_xenophobe
									has_ethic = ethic_fanatic_xenophobe
								}
							}
							AND = {
								#exists = leader
								OR = {
									has_ethic = ethic_spiritualist
									has_ethic = ethic_fanatic_spiritualist
								}
								ROOT = {
									NOR = {
										has_ethic = ethic_spiritualist
										has_ethic = ethic_fanatic_spiritualist
									}
								}
							}
						}
					}
				}
			}
			save_event_target_as = encroaching_research_fleet
			if = {
				limit = { exists = leader }
				leader = { save_event_target_as = encroaching_research_leader }
			}
			solar_system = {
				space_owner = {
					switch = {
						trigger = has_ethic
						ethic_materialist = { country_event = { id = diplomatic.10 } }
						ethic_fanatic_materialist = { country_event = { id = diplomatic.10 } }
						ethic_xenophobe = { country_event = { id = diplomatic.12 } }
						ethic_fanatic_xenophobe = { country_event = { id = diplomatic.12 } }
						ethic_spiritualist = { country_event = { id = diplomatic.15 } }
						ethic_fanatic_spiritualist = { country_event = { id = diplomatic.15 } }
					}
				}
			}
		}
	}
}

# Exotic Woodwind
country_event = {
	id = diplomatic.19
	title = "diplomatic.19.name"
	desc = "diplomatic.19.desc"
	picture = GFX_evt_outpost
	show_sound = event_radio_chatter
	location = music_listening_planet
	trackable = yes

	trigger = {
		any_neighbor_country = {
			has_communications = root
			NOT = { has_authority = auth_machine_intelligence }
			NOR = {
				is_at_war_with = ROOT
				is_subject = yes
			}
			exists = capital_scope
			capital_scope = { solar_system = { has_access_fleet = ROOT } }
			any_system_within_border = {
				any_system = {
					distance = {
						source = prev
						use_bypasses = no
						min_jumps = 1
						max_jumps = 3
					}
					any_system_planet = {
						exists = owner
						owner = { is_same_value = ROOT }
						num_pops > 3
						any_owned_pop = { is_same_species = owner }
					}
				}
			}
		}
		is_gestalt = no
	}

	immediate = {
		random_neighbor_country = {
			limit = {
				has_communications = root
				NOT = { has_authority = auth_machine_intelligence }
				NOR = {
					is_at_war_with = ROOT
					is_subject = yes
				}
				capital_scope = { solar_system = { has_access_fleet = ROOT } }
				any_system_within_border = {
					any_system = {
						distance = {
							source = prev
							use_bypasses = no
							min_jumps = 1
							max_jumps = 3
						}
						any_system_planet = {
							exists = owner
							owner = { is_same_value = ROOT }
							num_pops > 3
							any_owned_pop = { is_same_species = owner }
						}
					}
				}
			}
			save_event_target_as = music_origin_country
			ROOT = {
				random_owned_planet = {
					limit = {
						num_pops > 3
						any_owned_pop = { is_same_species = owner }
						solar_system = {
							any_system = {
								distance = {
									source = prev
									use_bypasses = no
									min_jumps = 1
									max_jumps = 4
								}
								exists = space_owner
								space_owner = { is_same_value = event_target:music_origin_country }
							}
						}
					}
					save_event_target_as = music_listening_planet
				}
			}
		}
	}
	
	#is_triggered_only = yes
	mean_time_to_happen = { months = 4800 }
	
	option = {
		name = diplomatic.19.a
		event_target:music_listening_planet = {
			random_owned_pop = {
				limit = {
					is_same_species = owner_species
					NOR = {
						has_ethic = ethic_xenophile
						pop_has_trait = trait_mechanical
					}
					NOT = { has_modifier = alien_music }
				}
				add_modifier = {
					modifier = "alien_music"
					days = 36000
				}
				pop_change_ethic = ethic_xenophile
			}
			random_owned_pop = {
				limit = {
					NOT = { has_modifier = alien_music }
					NOR = {
						has_ethic = ethic_xenophile
						pop_has_trait = trait_mechanical
					}
				}
				add_modifier = {
					modifier = "alien_music"
					days = 36000
				}
				pop_change_ethic = ethic_xenophile
			}
			random_owned_pop = {
				limit = {
					NOT = { has_modifier = alien_music }
					NOR = {
						has_ethic = ethic_xenophile
						pop_has_trait = trait_mechanical
					}
				}
				add_modifier = {
					modifier = "alien_music"
					days = 36000
				}
				pop_change_ethic = ethic_xenophile
			}
		}
	}

	option = {
		name = diplomatic.19.b
		enable_special_project = {
			name = "woodwind_instrument_investigation"
			location = event_target:music_origin_country.capital_scope
			owner = root
		}
	}
