namespace = iisc_military
# reset military orders
country_event = {
	id = iisc_military.10
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		if = {
			limit = {
				exists = event_target:iisc_military_fleet
			}
			if = {
				limit = {
					has_global_flag = iisc_paused
				}
				country_event = {
					id = iisc_military.10
					days = 30
				}
			}
			else_if = {
				limit = {
					has_global_flag = iisc_stopped
				}
				event_target:iisc_military_fleet = {
					delete_fleet = THIS
				}
			}
			else = {
				event_target:iisc_military_fleet = {
					iisc_set_military_fleet_orders = yes
				}
			}
		}
	}
}

# rename
fleet_event = {

	hide_window = yes
	immediate = {
		set_name = random
	}
}
