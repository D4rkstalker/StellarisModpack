namespace = aemre_event

country_event = {
	id = aemre_event.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		AND = {
			root = {
				is_country_type = awakened_fallen_empire
			}
			NOT = {
				AND = {
					exists = event_target:FirstSleeper
					exists = event_target:SecondSleeper
					has_global_flag = war_in_heaven_ongoing
					OR = {
						is_same_value = event_target:FirstSleeper
						is_same_value = event_target:SecondSleeper
					}
				}
			}
		}
	}

	immediate = {
		country_event = { id = aemre_event.2 days = 5 }
	}
}

country_event = {
	id = aemre_event.2
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_subject = {
			every_owned_planet = {
				limit = {
					AND = {
						OR = {
							is_colony = yes
							is_capital = yes
						}
						NOT = {
							any_fleet_in_orbit = {
								has_fleet_flag = aemre_guardian_fleet
							}
						}
					}
				}
				create_fleet = {
					effect = {
						set_owner = root
						set_fleet_flag = aemre_guardian_fleet
						while = {
							count = 4
							create_fallen_empire_platform = yes
						}
						set_location = {
							target = prev
						}
					}
				}
			}
		}
	}
}
