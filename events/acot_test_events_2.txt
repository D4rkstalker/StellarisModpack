country_event = {
	id = acot_test_event.2
	hide_window = yes
	is_triggered_only = yes	
	
	immediate = {
		capital_scope = {	
			owner = {
				create_leader = {
					class = admiral
					species = owner_main_species
					name = random
					skill = 10
				}
				last_created_leader = { set_age = 0 }
			}
			create_fleet = {
				settings = {
					 spawn_debris = no 
					 can_upgrade = no 
					 uses_naval_capacity = no
			}
				effect = {
					set_owner = root
					while = {
						count = 1
						create_ship = {
							name = random
							design = "NAME_Uninstaller"
							graphical_culture = "ai_01"
							upgradable = no
						}
					}
					while = {
						count = 12
						create_ship = {
							name = random
							design = "NAME_Sterilizer"
							graphical_culture = "ai_01"
							upgradable = no
						}
					}
					assign_leader = last_created_leader
					while = {
						count = 6
						create_ship = {
							name = random
							design = "NAME_Euthanizer"
							graphical_culture = "ai_01"
							upgradable = no
						}
					}
					set_location = {
						target = prev
						distance = 45
						angle = random
					}
				}
			}
			set_owner = ROOT
		}
	}
}
