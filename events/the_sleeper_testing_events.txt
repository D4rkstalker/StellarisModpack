#destroy the sleeper, drones countries and TN-3
country_event = {
	id = snoozing.04
	title = snoozing.04.title
	desc = snoozing.04.desc
	
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			limit = {
				OR = {
					is_country_type = hive_researcher_enclave
					is_country_type = scrambled_drones
					is_sleeper = yes
				}
			}
			every_owned_pop = {
				kill_pop = yes
			}
			destroy_country = yes
		}
	}
	
	option = {
		name = snoozing.04.a
	}
}
