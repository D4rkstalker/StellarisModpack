# Asteroid Impacts!
country_event = {
	id = observation.3
	title = "observation.3.name"
	desc = "observation.3.desc"
	picture = GFX_evt_nuclear_explosion
	show_sound = event_super_explosion
	location = event_target:primitive_impact_planet
	
	is_triggered_only = yes
	
	immediate = {
		event_target:primitive_impact_planet = {
			owner = {
				every_owned_pop = {
					kill_pop = yes
				}
			}
			destroy_colony = yes
			add_deposit = d_lithoid_crater
			add_modifier = { modifier = lithoid_crater }
		}
		remove_country_flag = asteroid_sighted_prim_country
	}
	
	option = {
		name = "observation.3.a"
		event_target:primitive_impact_planet = {
			observation_outpost = {
				dismantle = yes
			}
		}
		hidden_effect = {
			if = {
				limit = { has_lithoids = yes }
				random_list = {
					1 = { } # nothing happens
					1 = {
						country_event = { id = primitivesexpanded.68 days = 360 random = 720 }
					}
				}
			}
		}
	}
}
