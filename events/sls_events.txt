####GIVE SLS TRAITS WHEN NEW SCI PPL APPEARS###
country_event = {
	id = sls.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		from = {
			is_ruler = no
			NOT = {
				has_leader_flag = sls_have_traits
			}
			OR = {
				AND = {
					root = {
						is_gestalt = yes
					}
					is_councilor_type = councilor_gestalt_scientist
				}
				AND = {
					root = {
						is_gestalt = no
					}
					leader_class = scientist
				}
			}
		}
	}
	
	immediate = {
		from = {
			save_event_target_as = sls_leader
			sls_set_leader_base_status = yes
		}
	}
}

###WHEN LEADER LEVELS UP###
country_event = {
	id = sls.2
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		from = {
			leader_class = scientist
			has_leader_flag = sls_have_traits
		}
	}
	immediate = {
		from = {
			if = {
				limit = {
					OR = {
						is_researching_area = engineering
						has_trait = leader_trait_expertise_materials
						has_trait = leader_trait_expertise_propulsion
						has_trait = leader_trait_expertise_voidcraft
						has_trait = leader_trait_expertise_industry
					}
				}
				change_variable = { which = sls_leader_engineering_points value = 3 }
			}
			if = {
				limit = {
					OR = {
						is_researching_area = physics
						has_trait = leader_trait_expertise_field_manipulation
						has_trait = leader_trait_expertise_particles
						has_trait = leader_trait_expertise_computing
					}
				}
				change_variable = { which = sls_leader_physics_points value = 3 }
			}
			if = {
				limit = {
					OR = {
						is_researching_area = society
						has_trait = leader_trait_expertise_new_worlds
						has_trait = leader_trait_expertise_statecraft
						has_trait = leader_trait_expertise_archaeostudies
						has_trait = leader_trait_expertise_biology
						has_trait = leader_trait_expertise_military_theory
						has_trait = leader_trait_expertise_psionics
					}
				}
				change_variable = { which = sls_leader_society_points value = 3 }
			}
			change_variable = { which = sls_leader_random_points value = 5 }
			while = {
				count = sls_leader_random_points
				random_list = {
					1 = { 
						if = {
							limit = {
								check_variable = { which = sls_leader_physics_points value < 100 }
							}
							change_variable = { which = sls_leader_physics_points value = 2 }
						}
						else = {
							change_variable = { which = sls_leader_society_points value = 1 }
							change_variable = { which = sls_leader_engineering_points value = 1 }
						}
					}
					1 = { 
						if = {
							limit = {
								check_variable = { which = sls_leader_society_points value < 100 }
							}
							change_variable = { which = sls_leader_society_points value = 2 }
						}
						else = {
							change_variable = { which = sls_leader_engineering_points value = 1 }
							change_variable = { which = sls_leader_physics_points value = 1 }
						}
					}
					1 = { 
						if = {
							limit = {
								check_variable = { which = sls_leader_engineering_points value < 100 }
							}
							change_variable = { which = sls_leader_engineering_points value = 2 }
						}
						else = {
							change_variable = { which = sls_leader_physics_points value = 1 }
							change_variable = { which = sls_leader_society_points value = 1 }
						}
					}
				}
			}
			set_variable = { which = sls_leader_random_points value = 0 }
			effect_sls_move_base_traits_to_right = yes
		}
	}
}

	id = sls.6
	hide_window = no
	is_triggered_only = yes
	trigger = {
	 always = no
	}
	immediate = {
		random_owned_leader = {
			limit = {
				leader_class = scientist
				is_ruler = no
				OR = {
					is_councilor_type = councilor_research
					is_councilor_type = councilor_gestalt_scientist
				}
				has_leader_flag = sls_have_traits
			}
			save_event_target_as = sls_leader
			sls_run_check_research_corps_effect = yes
		}

		root = {
			add_modifier = {
				modifier = sls_physics_add_modifier
				days = 360
				multiplier = event_target:sls_leader.sls_leader_physics_points
			}
		}
		root = {
			add_modifier = {
				modifier = sls_society_add_modifier
				days = 360
				multiplier = event_target:sls_leader.sls_leader_society_points
			}
		}
		root = {
			add_modifier = {
				modifier = sls_engineering_add_modifier
				days = 360
				multiplier = event_target:sls_leader.sls_leader_engineering_points
			}
		}
	}
}
