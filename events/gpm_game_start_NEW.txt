### Initialiser
# if multiplayer -> spawn engine with default settings
# if not multiplayer -> show menu and let user decide
# Multiplayer check
event = {
	id = gpm_game_start.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = {
            has_global_flag = gpm_initialisation_menu
        }
        is_ai = no
	}
	
	immediate = {
		set_global_flag = gpm_initialisation_menu
		set_global_flag = gpm_modifier_amount_normal
		country_event = { id = gpm_start_menu.1 }
	}
}

### Death World civic gamestart
country_event = {
	id = gpm_game_start.1
	hide_window = yes
	is_triggered_only = yes

	trigger = { 
		has_civic = civic_gpm_death_world 
	}

	immediate = {
		set_country_flag = gpm_death_world
		random_owned_planet = {
			limit = { is_capital = yes }
			set_planet_flag = gpm_civic_start_planet
			# remove_modifier = "capital"			- changed this for compatibility reasons
			add_modifier = { modifier = "gpm_death_world" days = -1	}
			gpm_random_dangerous_modifiers = yes
			gpm_random_dangerous_modifiers = yes
			gpm_random_dangerous_modifiers = yes
		}
		every_owned_pop = {
			limit = {
				species = {
					NOT = { has_trait = trait_gpm_mega_adaptive	}
				}
				is_sapient = yes
			}
			modify_species = {
				species = this
				add_trait = trait_gpm_mega_adaptive
				add_traits_at_start_of_list = yes
				base = none
			}
		}
	}
}

### Precursor civic gamestart
country_event = {
	id = gpm_game_start.2
	hide_window = yes
	is_triggered_only = yes

	trigger = { 
		OR = {
			has_civic = civic_gpm_precursor_world 
			has_civic = civic_gpm_precursor_world_machine
		}
	}

	immediate = {
		random_owned_planet ={
			limit = { is_capital = yes }
			set_planet_flag = gpm_civic_start_planet

			every_deposit = {
				remove_deposit = yes
			}
			# set planet size to destrict bonuses x2 for easy balance
			# set_planet_size = 18
			# Cryotube deposit
			add_deposit = gpm_d_precursor_ruins_4_empty

			# Energy Deposits
			random_list = {
				10 = { add_deposit = gpm_d_precursor_ruins_3_empty }		# +1 energy district
				10 = { add_deposit = gpm_d_precursor_ruins_5_empty }		# +1 energy district
			}
			add_deposit = d_geothermal_vent		# +3 energy district

			# Mining Deposits
			random_list = {
				10 = { add_deposit = gpm_d_precursor_ruins_1_empty }		# +1 mining district
				10 = { add_deposit = gpm_d_precursor_ruins_6_empty }		# +1 mining district
			}
			add_deposit = d_veiny_cliffs		# +1 mining district
			# add_deposit = d_rich_mountain		# +3 mining district

			# Farming Deposits
			random_list = {
				10 = { add_deposit = gpm_d_precursor_ruins_2_empty }		# +1 farming district
				10 = { add_deposit = gpm_d_precursor_ruins_7_empty }		# +1 farming district
			}
			add_deposit = d_lichen_fields		# +1 farming district
			add_deposit = d_fungal_caves		# +2 farming district

			add_deposit = d_failing_infrastructure

			random_list = {
				100 = {
					add_modifier = { days = -1 modifier = "gpm_precursor_remnants" }

					# TWeaks #
					# Not with these modifiers #
					modifier = { factor = 0 has_modifier = "gpm_precursor_remnants" }
					set_planet_flag = "gpm_is_precursor_world"
				}
				100 = {
					add_modifier = { days = -1 modifier = "gpm_precursor_city" }
					# On these planets #
					# TWeaks #
					# Not with these modifiers #
					modifier = { factor = 0 has_modifier = "gpm_precursor_city" }
					set_planet_flag = "gpm_is_precursor_world"
				}
				100 = {
					add_modifier = { days = -1 modifier = "gpm_precursor_floating_city" }
					# On these planets #
					# TWeaks #
					# Not with these modifiers #
					modifier = { factor = 0 has_modifier = "gpm_precursor_floating_city" }
					set_planet_flag = "gpm_is_precursor_world"
				}
				

			}
			# Random List with Defensive buffs
			random_list = {
				100 = {
					add_modifier = { days = -1 modifier = "gpm_precursor_planetary_gun" }
					# On these planets #
					# TWeaks #
					# Not with these modifiers #
					modifier = { factor = 0 has_modifier = "gpm_precursor_planetary_gun" }
					set_planet_flag = "gpm_is_precursor_world"
				}

					# Not with these modifiers #
					modifier = { factor = 0 has_modifier = "gpm_precursor_bunker_grid" }
					set_planet_flag = "gpm_is_precursor_world"
				}
