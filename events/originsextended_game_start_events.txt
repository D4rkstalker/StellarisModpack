
### ORIGINS EXTENDED GAME START EVENTS

namespace = originsextended_game_start

# this: capital planet
# from: founder species
planet_event = {
	id = originsextended_game_start.001
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				owner = { has_origin = originsextended_starless }
			}
			clear_deposits = yes # redudant sure, but safe none the less.
			starless_start_deposits_and_blockers = yes # see scripted_effects
		}

		if = {
			limit = {
				owner = { has_origin = originsextended_farming }
			}
			clear_deposits = yes # redudant sure, but safe none the less.
			generate_farming_pops = yes # see scripted_effects
		}

		if = {
			limit = {
				owner = { has_origin = originsextended_core }
			}
			clear_deposits = yes # redudant sure, but safe none the less.
			generate_core_deposits_and_blockers = yes # see scripted_effects
		}
	
		if = {
			limit = {
				owner = { has_origin = originsextended_red }
			}
			clear_deposits = yes # redudant sure, but safe none the less.
			generate_red_deposits_and_blockers = yes # see scripted_effects
		}
	}
}

event = {
	id = originsextended_game_start.003
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		# Outreach Origin
		every_playable_country = {
			limit = { 
				OR = {
					has_origin = originsextended_outreach
				}
			}
			country_event = { id = outreach.00 }
		}
	}
}

event = {
	id = originsextended_game_start.004
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { exists = capital_scope }
			capital_scope = {
				solar_system = {
					every_system_planet = {
						if = {
							limit = { has_planet_flag = originsextended_red_planet_earth }
							clear_deposits = yes
							generate_red_earth_blockers = yes
						}
					}
				}
			}
		}
	}
}
