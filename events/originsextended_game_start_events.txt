
### ORIGINS EXTENDED GAME START EVENTS

namespace = originsextended_game_start

# this: capital planet
# from: founder species
planet_event = {
	id = originsextended_game_start.001
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				owner = { has_origin = originsextended_starless }
			}
			clear_deposits = yes # redudant sure, but safe none the less.
			starless_start_deposits_and_blockers = yes # see scripted_effects
		}
	}
}

# this: capital planet
# from: founder species
planet_event = {
	id = originsextended_game_start.002
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				owner = { has_origin = originsextended_farming }
			}
			generate_farming_pops = yes # see scripted_effects
		}
	}
}
