namespace = starbase_upgrade

event = {
    id = starbase_upgrade.1
    hide_window = yes
    is_triggered_only = yes
    immediate = {
		every_playable_country = {
			limit = {
				is_ai = yes
			}
			if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 150
					}
					any_owned_planet = {
						solar_system = {
							exists = starbase
							starbase = {
								has_starbase_size = starbase_outpost
							}
						 }
						
					}
				}
				country_event = {
					id = starbase_upgrade.99
				}
			}
			if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 800
					}
					has_technology = tech_starbase_3
					any_owned_starbase = {
						has_starbase_size = starbase_starport
					}
				}
				country_event = {
					id = starbase_upgrade.2
				}
			}
			else_if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 2000
					}
					has_technology = tech_starbase_4
					any_owned_starbase = {
						has_starbase_size = starbase_starhold
					}
				}
				country_event = {
					id = starbase_upgrade.3
				}
			}
			else_if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 4000
					}
					has_technology = tech_starbase_5
					any_owned_starbase = {
						has_starbase_size = starbase_starfortress
					}
				}
				country_event = {
					id = starbase_upgrade.4
				}
			}
			else_if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 6000
					}
					used_naval_capacity_percent > 0.95
					has_technology = tech_starbase_6
					any_owned_starbase = {
						has_starbase_size = starbase_citadel
					}
				}
				country_event = {
					id = starbase_upgrade.5
				}
			}
			else_if = {
				limit = {
					resource_stockpile_compare = {
						resource = alloys
						value > 9000
					}
					has_technology = tech_starbase_7
					any_owned_starbase = {
						has_starbase_size = starbase_stronghold
					}
				}
				country_event = {
					id = starbase_upgrade.6
				}
			}
		}
    }
}

country_event = {
	id = starbase_upgrade.99
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		random_owned_planet = {					
			limit = {

				solar_system = {
					exists = starbase
					starbase = {
					has_starbase_size = starbase_outpost
					}
				 }
			}
			exists = starbase
			starbase = {
				set_starbase_size = starbase_starport
			}
		}
	}
}

	id = starbase_upgrade.2
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		random_owned_starbase = {
			limit = { has_starbase_size = starbase_starport }
			set_starbase_size = starbase_starhold
		}
		every_owned_starbase = {
			limit = { 
				has_starbase_size = starbase_starport 
				resource_stockpile_compare = {
					resource = alloys
					value > 1000
				}
			}
			random_list = {
				1 = {
					set_starbase_size = starbase_starhold
				}
				5 = {}
			}
		}
	}
}
