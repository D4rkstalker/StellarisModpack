###	aar权巅主权note
#_dec	<DEC权力之巅>
planet_event = {
	id = AAR_evt_PREROGATIVE.31
	title = AAR_title_CLEAN_claim
	desc = AAR_desc_CLEAN_claim
	picture = GFX_evt_spy_network
	show_sound = event_activating_unknown_technology
	location = root
	is_triggered_only = yes

	## 清除宣称
	option = {
		name = AAR_option_CLEAN_claim
		custom_tooltip = AAR_tips_CLEAN_claim
		allow = {
			hidden_trigger = {
				owner = {
					OR = {
						resource_stockpile_compare = { resource = influence	 value >= 1 mult = prev.TEMP_var_influence }
						resource_stockpile_compare = { resource = influence	 value >= 1000 }
					}
					resource_stockpile_compare = { resource = unity	 value >= 1 mult = prev.TEMP_var_unity }
				}
			}
		}
		hidden_effect = {
			set_update_modifiers_batch = begin	#_|修正更新|暂停
			
			owner = {
				every_system_within_border = {
					limit = {
						any_country = { has_claim = prev }
						distance = {#几何距离=75
							source = root
							type = euclidean
							max_distance < 75
						}
					}
					every_country = { limit = { has_claim = prev }
						prev = { remove_claims = { who = prev num_of_claims = 1 } }
						# 添加态度
						add_opinion_modifier = { modifier = AAR_opinion_malicious_claim who = prevprev }	# 恶意宣称
					}
				}
				# 资源消耗
				add_resource = { influence = 1	mult = prev.TEMP_var_influence_cost }
				add_resource = { unity = 1		mult = prev.TEMP_var_unity_cost }
			}
			# clean
			clear_variable = TEMP_var_SSTM_claimed
			clear_variable = TEMP_var_influence
			clear_variable = TEMP_var_influence_cost
			clear_variable = TEMP_var_unity
			clear_variable = TEMP_var_unity_cost
			# CD
			owner = { set_timed_country_flag = { flag = AAR_flg_cntr_CLEAN_claim days = 1080 } }
			
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}
	## 啥都不做
	option = {
		name = AAR_option_CLEAN_nothing
		hidden_effect = {
			clear_variable = TEMP_var_SSTM_claimed
			clear_variable = TEMP_var_influence
			clear_variable = TEMP_var_unity
		}
	}
}
