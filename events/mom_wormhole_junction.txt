namespace = mom_junction

#Spawns Initial Termini
country_event = {
	id = mom_junction.1
	hide_window = yes
	trigger = {
		has_origin = origin_mom_wormhole_junction
	}
	is_triggered_only = yes
	immediate = {
		capital_scope.solar_system = {
			spawn_megastructure = {
				type = "mom_junction_master"
				location = star orbit_angle = 20
				orbit_distance = 250
				owner = ROOT
				init_effect = {
					set_megastructure_flag = junction_master
				}
			}
		}
		capital_scope.star = {
			set_star_flag = mom_junction_master_system
			save_global_event_target_as = mom_junction_master_system
			set_variable = { which = mom_wormhole_termini_discover_var value = 2 }
		}
		while = {
			count = 2
			random_system = {
				limit = {
					NOR = {
						has_star_flag = mom_junction_master_system
						has_star_flag = mom_junction_termini_system
						any_system_planet = {
							is_colonizable = yes
						}
						any_system_megastructure = {
							is_megastructure_type = mom_junction_termini
							is_megastructure_type = mom_junction_master
						}
						any_ship_in_system = {
							is_ship_class = shipclass_starbase # Empire Check
						}
						has_star_flag = empire_cluster # To not be too close to other's home system
						any_ship_in_system = {
							is_ship_class = shipclass_military # Levaiathan Check
						}
					}
				}
				every_system_planet = {
					surveyed = { set_surveyed = yes surveyor = prevprev }
				}
				set_star_flag = mom_junction_termini_system
				random_list = {
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 36
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 72
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 108
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 144
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 180
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 216
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 252
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 288
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 324
							orbit_distance = 250
						}
					}
					10 = {
						spawn_megastructure = {
							type = "mom_junction_termini"
							location = star
							orbit_angle = 360
							orbit_distance = 250
						}
					}
				}
			}
		}
		country_event = { id = mom_junction.2 }
	}
}

		name = mom_junction.3.a
		
		enable_special_project = {
			name = MOM_DISCOVER_TERMINI_PROJECT

			location = capital_scope
		}
		if = {
			limit = {
				is_ai = no
			}

		}

		hidden_effect = {

		}
		ai_chance = {
			factor = 0
		}
	}
}
