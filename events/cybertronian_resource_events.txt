namespace = energon_sr

# Enable resources for Galactic Market
event = {
	id = energon_sr.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = galactic_market_founded

		OR = {
			NOT = { is_on_market = energon }
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { 
					is_on_market = energon  
				}
				any_playable_country = {
				resource_stockpile_compare = {  resource = energon value >= 1 } 
				}
			}
			enable_on_market = energon
		}
	}
}
