### COLLECTIVISM ###

#Collectivism 3, bonus on planets without crime.
country_event = {
	id = est_coll_indv.2
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_swapped_tradition = tr_est_collectivism_3
	}
	
	immediate = {
		every_owned_planet = {
			if = {
				limit = {
					planet_crime < 1
				}
				if = {
					limit = {
						Not = {
							has_modifier = mod_est_collectivism_3
						}
					}
					add_modifier = {
						modifier = "mod_est_collectivism_3"
						days = -1
					}	
				}
			}		
			else = {
				if = {
					limit = {
						has_modifier = mod_est_collectivism_3
					}
					remove_modifier = "mod_est_collectivism_3"
				}
			}			
		}
	}
}

#Individualism 1 and 2, bonus level to Admirals and Scientist leaders.
country_event = {
	id = est_coll_indv.4
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_swapped_tradition = tr_est_individualism_adopt
		Or = {
			And = {
				has_swapped_tradition = tr_est_individualism_1
				from = { leader_class = scientist }
			}
			And = {
				has_swapped_tradition = tr_est_individualism_2
				from = { leader_class = admiral }
			}
		}
	}
	
	immediate = {
		every_owned_leader = {
			limit = {
				Or = {
					And = {
						owner = { has_swapped_tradition = tr_est_individualism_1 }
						leader_class = scientist
					}
					And = {
						owner = { has_swapped_tradition = tr_est_individualism_2 }
						leader_class = admiral
					}
				}
				Or = {
					has_skill < 2
					has_leader_flag = flag_est_got_academy_bonus_xp
				}
				Not = {  has_leader_flag = flag_est_got_academy_bonus_xp }
			}
			set_timed_leader_flag = { 
				flag = flag_est_got_individualism_bonus_xp
				days = 1
			}
			if = {
				limit = {
					owner = {
						has_non_swapped_tradition = tr_est_academy_2
					}
				}
				if = {
					limit = {
						owner = {
							has_non_swapped_tradition = tr_est_academy_3
						}
					}
					if = {
						limit = {
							owner = {
								has_swapped_tradition = tr_est_individualism_5
							}
						}
						add_experience = 301
					}
					else = {
						add_experience = 358
					}
				}
				else = {
					if = {
						limit = {
							owner = {
								has_swapped_tradition = tr_est_individualism_5
							}
						}
						add_experience = 380
					}
					else = {
						add_experience = 475						
					}
				}
			}
			else = {
				if = {
					limit = {
						owner = {
							has_swapped_tradition = tr_est_individualism_4
						}
					}
					add_experience = 160
				}
				else = {
					add_experience = 200
				}
			}
		}
	}
}

			leader_class = scientist

				has_leader_flag = flag_est_individualism_1_trait_gained

#Perseverance 1, updates when building built.
planet_event = {
	id = est_coll_indv.6
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_swapped_tradition = tr_est_perseverance_1
			has_swapped_tradition = tr_est_perseverance_IP_1
		}
		from = {
			Or = {
				has_building = building_planetary_shield_generator
				last_building_changed = building_planetary_shield_generator
			}
		}
	}
	
	immediate = {
		planet_event = {
			id = est_coll_indv.7
		}
	}
}

#Perseverance 1, updates on planet gained.
planet_event = {
	id = est_coll_indv.7
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_swapped_tradition = tr_est_perseverance_1
			has_swapped_tradition = tr_est_perseverance_IP_1
		}
	}
	
	immediate = {
		if = {
			limit = {
				has_building = building_planetary_shield_generator
			}

				limit = {
					Not = {

					}
				}
				add_modifier = {
					modifier = "mod_est_perseverance_1_shield"
					days = -1
				}
			}
			if = {
				limit = {
					has_modifier = mod_est_perseverance_1_no_shield
				}
				remove_modifier = "mod_est_perseverance_1_no_shield"
			}
		}
		else = {
			if = {
				limit = {
					Not = {
						has_modifier = mod_est_perseverance_1_no_shield
					}
				}
				add_modifier = {
					modifier = "mod_est_perseverance_1_no_shield"
					days = -1
				}
			}
			if = {
				limit = {
					has_modifier = mod_est_perseverance_1_shield
				}
				remove_modifier = "mod_est_perseverance_1_shield"
			}
		}
	}
}

#Perseverance 5, fires when war have finished, fires next event with short delay.
country_event = {
	id = est_coll_indv.8
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = est_perseverance_5_picked_flag
		fromfrom = {
			any_war_participant = {
				Or = {
					has_swapped_tradition = tr_est_perseverance_5
					has_swapped_tradition = tr_est_perseverance_IP_5
				}
			}
		}
	}
	
	immediate = {
		fromfrom = {
			every_war_participant = {
				limit = {
					Or = {
						has_swapped_tradition = tr_est_perseverance_5
						has_swapped_tradition = tr_est_perseverance_IP_5
					}
				}
				country_event = {
					id = est_coll_indv.9
					days = 1
				}
			}
		}
	}
}

#Adds the bonus when loosing war, check that war have been lost.
country_event = {
	id = est_coll_indv.9
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		recently_lost_war = yes
	}
	
	immediate = {
		if = {

				has_modifier = mod_est_perseverance_5
			}
			remove_modifier = "mod_est_perseverance_5"
		}
		add_modifier = {
			modifier = "mod_est_perseverance_5"

		}
	}
