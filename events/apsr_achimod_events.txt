namespace = apsr_achimod



event = {
	id = apsr_achimod.999
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			set_variable = {
				which = apsr_achimod_planets_count
				value = 0
			}
		}
		every_system = {
			limit = {
				any_system_planet = {
					OR = {
						is_planet_class = pc_apsr_ancient
						is_planet_class = pc_apsr_unique
					}
				}
			}
			every_system_planet = {
				limit = { 
					exists = owner
					owner = {
						is_country_type = default
					}
					OR = {
						is_planet_class = pc_apsr_ancient
						is_planet_class = pc_apsr_unique	
					}
				}
				owner = {
					change_variable = {
						which = apsr_achimod_planets_count
						value = 1
					}
				}
			}
		}
	}
}
