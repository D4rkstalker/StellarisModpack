#TN-3's fate, triggered (on the sleeper) when the hive awakens in any way. Rolls until someone accepts him or the station is destroyed.
country_event = {
	id = sleeper_tony.1
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				exists = event_target:hive_enclave_country
				any_country = { has_country_flag = tony_likes }
			}
			random_country = {
				limit = { has_country_flag = tony_likes }
				remove_country_flag = tony_likes
				country_event = { id = sleeper_tony.02 }
			}
		}
		else = {
			every_country = {
				limit = { has_country_flag = tony_likes }
				remove_country_flag = tony_likes
			}
			set_faction_hostility = {
				target = event_target:hive_enclave_country
				set_hostile = yes
				set_neutral = no
				set_friendly = no
			}
		}
	}
}

#Event for taking out the sleeper. TN-3 joins who ever killed it, in case he is still alive
#also sets a global flag to note that the sleeper is dead
#This = destroyed country
#From = optional, destroyer (country)
country_event = {
	id = sleeper_tony.19
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		AND = {
			is_sleeper = yes
			any_country = { is_country_type = hive_researcher_enclave }
		}
	}
	
	immediate = {
		set_global_flag = sleeper_dead
		FROM = {
			country_event = { id = sleeper_tony.20 days = 15 }
		}
	}
}

#drone pops have broken free, TN-3 wants to evacuate them. Only for non genocide/gestalt (except servitors)

	title = sleeper_tony.missions.titles
	desc = sleeper_tony.10.desc
	diplomatic = yes 
	picture_event_data = { 
		portrait = event_target:contact_empire 
		room = "ethic_spaceship_room"
	}
	
	is_triggered_only = yes
	
	immediate = {
		random_country = {
			limit = {
				is_country_type = hive_researcher_enclave
			}
			save_event_target_as = contact_empire
			set_country_flag = sleeper_freed_drone_mission_done
		}
		event_target:the_sleeper = {
			owner_species = { save_event_target_as = sleeper_species }
			random_owned_pop = {
				limit = { has_trait = trait_hive_mind }
				kill_pop = yes
			}
		}
	}
	
#take them in and let him study them. 50% chance that the pops will be killed. if the pop is dead after that year, then nothing will happen
	option = {
		name = sleeper_tony.10.a
		event_target:contact_empire = {
			add_trust = {
				who = root
				amount = 30
			}
		}
		ROOT.capital_scope = {
			create_pop = {
				species = event_target:sleeper_species 
			}
		}
		hidden_effect = {
			last_created_pop = {
				modify_species = {
					species = this
					remove_trait = trait_hive_mind
					add_trait = trait_former_drones
					add_traits_at_start_of_list = yes
				}
				save_event_target_as = drone_pop
			}
			country_event = { id = sleeper_tony.11 }
			random_list = {
				50 = {
					country_event = { id = sleeper_tony.12 days = 360 }
				}
				50 = {
					country_event = { id = sleeper_tony.13 days = 360 }
				}
			}
			event_target:target_planet = {
				change_variable  = { 
					which = sleeper_agitation 
					value = 10
				}
			}
		}
		ai_chance = {
			factor = 60
			modifier = {
				factor = 0
				OR = {
					has_ethic = ethic_egalitarian
					has_ethic = ethic_fanatic_egalitarian
				}
			}
		}
	}
#take them in, but forbid the study
	option = {
		name = sleeper_tony.10.b
		event_target:contact_empire = {
			add_trust = {
				who = root
				amount = 10
			}
		}
		ROOT.capital_scope = {
			create_pop = {
				species = event_target:sleeper_species  
			}
		}
		hidden_effect = {
			last_created_pop = {
				modify_species = {
					species = this
					remove_trait = trait_hive_mind
					add_trait = trait_former_drones
					add_traits_at_start_of_list = yes
				}
			}
			country_event = { id = sleeper_tony.14 }
			country_event = { id = sleeper_tony.15 days = 360 }
			random_country = {
				limit = {
					is_sleeper = yes
				}
				change_variable  = { 
					which = sleeper_agitation 
					value = 2
				}
			}
		}
		ai_chance = {
			factor = 20

				factor = 10
				OR = {
					has_ethic = ethic_egalitarian
					has_ethic = ethic_fanatic_egalitarian
				}
			}
		}
	}
