# collectors gain comms with anyone that has surveyed their planet
country_event = {
	id = collectors.0
	hide_window = yes
	
	trigger = {
		OR = {
		is_country_type = planet_state
		is_country_type = peaceful_planet_state 
		is_country_type = hostile_planet_state 
		is_country_type = consolidated_planet_state 
		}
		any_playable_country = {
			root = {
				NOT = { has_communications = prev }
				capital_scope = { 	
					is_surveyed = { who = prevprev status = yes }
				}				
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
		every_playable_country = {
			limit = { 
				root = {
					NOT = { has_communications = prev }
					capital_scope = { 	
						is_surveyed = { who = prevprev status = yes }
					}				
				}
			}
			establish_communications_no_message = root
			country_event = { id = collectors.13 }
		}	
	}
}


# collectors.1: Spawn collectors (on_game_start)
event = {
	id = collectors.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {

		# Planet-State 4: The Pacifist
		random_system = {
			limit = { is_appropriate_planet_state_system = yes }
			create_collectors = yes
		}
	}
}

# collectors.12: meeting the collectors for the first time (on_entering_system_fleet)
fleet_event = {
	id = collectors.12
	title = collectors.12.name
	desc = collectors.12.desc
	picture = GFX_evt_throne_room
	show_sound = event_radio_chatter
	location = event_target:target_collectors

	is_triggered_only = yes

	trigger = {
		owner = {
			is_country_type = default
			exists = capital_scope
		}
		From = {
			has_star_flag = collectors
			any_system_planet = {
				exists = owner
				owner = {
					is_mad_collector_country = yes
					NOT = { has_communications = root.owner }
				}
			}
		}
	}

	immediate = {
		From = {
			random_system_planet = {
				limit = {
					exists = owner
					owner = { is_mad_collector_country = yes }
				}
				owner = {
					save_event_target_as = target_collectors
				}
			}
		}

		owner = {
			establish_communications_no_message = event_target:target_collectors
			save_event_target_as = target_default_country
			#country_event = { id = story.7 days = 2 }
		}
	}
	
	option = {
		name = SCUM
		trigger = { owner = { has_valid_civic = civic_fanatic_purifiers } }
	}
	option = {
		name = TASTY
		trigger = { owner = { has_valid_civic = civic_hive_devouring_swarm } }
	}
	option = {
		name = EXTERMINATE
		trigger = { owner = { has_valid_civic = civic_machine_terminator } }
	}

	option = {
		name = collectors.12.a
		trigger = { owner = { is_homicidal = no } }
	}

	after = {
		event_target:target_default_country = {
			if = {
				limit = {
					is_xenophile = yes
				}
				add_monthly_resource_mult = {
					resource = influence
					value = @tier1influencecontactxenophile
					min = @tier1influencecontactmin
					max = @tier1influencecontactmax
				}
			}
			else = {
				add_monthly_resource_mult = {
					resource = influence
					value = @tier1influencecontact
					min = @tier1influencecontactmin
					max = @tier1influencecontactmax
				}
			}

			hidden_effect = {
				if = {
					limit = {
						event_target:target_collectors = {
							has_country_flag = collectors
						}
					}
					country_event = { id = collectors.13 }
					break = yes
				}
				
			}
		}
	}
}
