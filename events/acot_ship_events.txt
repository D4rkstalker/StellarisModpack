############DARK MATTER EVENTS


namespace = acot_ships

# Herculean Built
ship_event = {
	id = acot_ships.0
	hide_window = yes
	
	is_triggered_only = yes

	trigger = {
		is_ship_size = precursor_colossus
	}

	immediate = {
		set_ship_flag = herculean_first_life
		set_disable_at_health = 0.02
	}
}

# Herculean First Life
ship_event = {
	id = acot_ships.1
	hide_window = yes
	
	is_triggered_only = yes

	trigger = {
		is_ship_size = precursor_colossus
		has_ship_flag = herculean_first_life
	}

	immediate = {
		set_ship_flag = herculean_second_life
		ship_event = { id = acot_ships.2 days = 1 }
	}
}

# Herculean Second Life Assigned
ship_event = {
	id = acot_ships.2
	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		set_ship_flag = herculean_second_life
		repair_ship = yes
		set_disabled = no	
		remove_ship_flag = herculean_first_life
	}
}

ship_event = {
	id = acot_ships.3
	hide_window = yes
	
	is_triggered_only = yes

	trigger = {
		is_ship_size = precursor_colossus
		has_ship_flag = herculean_second_life
		NOT = {
			has_ship_flag = herculean_first_life
		}
	}

	immediate = {
		destroy_ship = this
	}
}
