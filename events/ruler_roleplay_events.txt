#Yearly tick for ruler
country_event = {
	id = play_as_ruler.90
	hide_window = yes
	is_triggered_only = yes
	trigger = {	
		any_owned_leader = {
			is_ruler = yes
			has_leader_flag = play_as_ruler_ruler_itself
		}
	}
	immediate = {
		country_event = {
			id = play_as_ruler.91 #additional skill point event
		}
		random_owned_leader = {
			limit = {
				is_ruler = yes
				has_leader_flag = play_as_ruler_ruler_itself
			}
			save_event_target_as = rls_ruler
			if = {
				limit = {
					check_variable = {
						which = rls_ruler_idea_exp
						value > 15
					}
				}
				set_variable = {
					which = rls_ruler_idea_exp
					value = 15
				}
			}
			if = {
				limit = {
					check_variable = {
						which = rls_ruler_execution_exp
						value > 15
					}
				}
				set_variable = {
					which = rls_ruler_execution_exp
					value = 15
				}
			}
			#ruled a year and gain one random base level (for AI while they don't use it)
			if = {
				limit = {
					root = {
						is_ai = yes
					}
					
				}
				random_list = {
					1 = {
						change_variable = {
							which = rls_ruler_charm_level
							value = 1
						}
					}
					1 = {
						change_variable = {
							which = rls_ruler_affair_level
							value = 1
						}
					}
					1 = {
						change_variable = {
							which = rls_ruler_idea_level
							value = 1
						}
					}
				}
			}
			if = {
				limit = {
					check_variable = {
						which = rls_ruler_charm_level
						value > 100
					}
				}
				set_variable = {
					which = rls_ruler_charm_level
					value = 100
				}
			}
			if = {
				limit = {
					check_variable = {
						which = rls_ruler_affair_level
						value > 100
					}
				}
				set_variable = {
					which = rls_ruler_affair_level
					value = 100
				}
			}
			if = {
				limit = {
					check_variable = {
						which = rls_ruler_idea_level
						value > 100
					}
				}
				set_variable = {
					which = rls_ruler_idea_level
					value = 100
				}
			}
		}
		every_owned_leader = {
			limit = {
				NOT = {
					is_same_value = event_target:rls_ruler
				}
			}
			change_variable = {
				which = rls_random_foci_counter
				value = 1
			}
			if = {
				limit = {
					check_variable = {
						which = rls_random_foci_counter
						value = 2
					}
				}
				set_variable = {
					which = rls_random_foci_counter
					value = 0
				}
				effect_play_as_ruler_non_ruler_level_their_foci = yes
				effect_play_as_ruler_ruler_get_random_foci = yes
			}
		}
		if = {
			limit = {
				has_country_flag = rls_auto_unlock_enabled
			}
			event_target:rls_ruler = {
				change_variable = {
					which = rls_random_foci_counter
					value = 1
				}
				if = {
					limit = {
						check_variable = {
							which = rls_random_foci_counter
							value = 3
						}
					}
					set_variable = {
						which = rls_random_foci_counter
						value = 0
					}
					effect_play_as_ruler_ruler_auto_unlock_foci_effect = yes
				}
			}
		}
		country_event = {
			id = play_as_ruler.106
		}
		country_event = {
			id = play_as_ruler.105
		}
	}
}

country_event = {
	id = play_as_ruler.120
	hide_window = yes
	is_triggered_only = yes
	trigger = {	
		any_owned_leader = {
			is_ruler = yes
			check_variable = {
				which = rls_ruler_affair_level
				value >= 0
			}
		}
	}

	immediate = {
		effect_play_as_ruler_set_category_num_cap = yes
		set_variable = { #ruler's affair level
			which = rls_affair_monthly_increase
			value = leader.rls_ruler_affair_level
		}
		set_variable = {
			which = rls_ruler_category_num_cap_country
			value = leader.rls_ruler_category_num_cap
		}

		if = {
			limit = {
				NOT = {
					check_variable = {
						which = rls_affair_monthly_increase
						value = 0
					}
				}
			}
			divide_variable = {
				which = rls_affair_monthly_increase
				value = 25
			}
		}
		change_variable = { #monthly affair increase, 4 is base so 25 months to get a set of ruler campaigns
			which = rls_affair_monthly_increase
			value = 4
		}
		if = {
			limit = {
				check_variable = {
					which = rls_country_affair
					value < 0
				}
			}
			set_variable = {
				which = rls_country_affair
				value = 0
			}
		}
		change_variable = { #get monthly increase to counter
			which = rls_country_affair
			value = rls_affair_monthly_increase
		}
		
		###
		if = { #reach 100 to get a set of ruler campaigns
			limit = {
				check_variable = {
					which = rls_country_affair
					value > 100
				}
			}
			
			set_variable = { 
				which = rls_country_affair
				value = 0
			}
			country_event = {
				id = play_as_ruler.1
			}
			random_owned_leader = {
				limit = {
					is_ruler = yes
				}
				change_variable = {
					which = rls_ruler_skill_point_num
					value = 1
				}
			}
		}
		###get approximately month to get a set of ruler campaign
		set_variable = {
			which = rls_country_affair_cap
			value = 100
		}
		subtract_variable = {
			which = rls_country_affair_cap
			value = rls_country_affair
		}
		divide_variable = {
			which = rls_country_affair_cap
			value = rls_affair_monthly_increase
		}
		round_variable = rls_country_affair_cap
	}
}

#Favor Gateway
country_event = {
	id = play_as_ruler.299

	is_triggered_only = yes
	hide_window = yes
	trigger = {
		NOT = {
			has_country_flag = play_as_ruler.300.opened
		}
	}
	immediate = {
		if = { #has spare cap
			limit = {
				any_owned_leader = {
					is_ruler = yes
					check_variable = {
						which = rls_ruler_category_num
						value < rls_ruler_category_num_cap
					}
				}
			}
			country_event = {
				id = play_as_ruler.298
			}
		}
		else_if = { #hit the cap
			limit = {
				any_owned_leader = {
					is_ruler = yes
					check_variable = {
						which = rls_ruler_category_num
						value >= rls_ruler_category_num_cap
					}
				}
			}
			country_event = {
				id = play_as_ruler.297
			}
		}
	}
	
}
