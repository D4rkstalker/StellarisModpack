country_event = {
	id = play_as_ruler.20
	title = play_as_ruler.20.name
	desc = play_as_ruler.20.desc
	picture = GFX_evt_alien_propaganda
	show_sound = event_heir_new
	is_triggered_only = yes
	
	trigger = {
		any_owned_leader = {
			OR = {
				has_trait = trait_witness_ruler_lv1
				has_trait = trait_witness_ruler_lv2
				has_trait = trait_witness_ruler_lv3
				has_trait = trait_witness_ruler_lv4
			}
			NOT = {
				has_leader_flag = play_as_ruler_is_philosopher_king
			}
		}
		has_civic = civic_philosopher_king
	}
	
	immediate = {
        change_variable = { which = play_as_ruler_campaign_amount value = 2 }
		random_owned_leader = {
			limit = { 
				leader_class = ruler
				has_leader_flag = play_as_ruler_ruler_itself
			}
			save_event_target_as = ruler
			set_leader_flag = play_as_ruler_is_philosopher_king
			if = {
				limit = {
					OR = {
						has_trait = trait_witness_ruler_lv1
						has_trait = trait_witness_ruler_lv2
						has_trait = trait_witness_ruler_lv3
						has_trait = trait_witness_ruler_lv4
					}
				}
				remove_trait = trait_witness_ruler_lv1 
				remove_trait = trait_witness_ruler_lv2
				remove_trait = trait_witness_ruler_lv3
				remove_trait = trait_witness_ruler_lv4
				add_trait = trait_witness_ruler_lv5
			}
		}	
	}
	
	option = {
		name = play_as_ruler.20.a
		trigger = {
			leader = {
				leader_class = ruler
				has_trait = trait_witness_ruler_lv4
			}	
		}
		add_monthly_resource_mult = {
			resource = influence
			value = 12
			min = 75
			max = 150
		}
		add_monthly_resource_mult = {
			resource = unity
			value = 6
			min = 125
			max = 2000
		}
		custom_tooltip = play_as_ruler.20.tooltip
	}
}
