namespace = ideological_purity_desired_resource_events

# Update ideological_purity current, desired resting point, and current monthly gain
country_event = {
    id = ideological_purity_desired_resource_events.1
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        country_event = { id = ideological_purity_desired_resource_events.2 }
        country_event = { id = ideological_purity_desired_resource_events.3 }
        country_event = { id = ideological_purity_desired_resource_events.4 }
        country_event = { id = ideological_purity_desired_resource_events.5 }
        country_event = { id = ideological_purity_desired_resource_events.6 }
        country_event = { id = ideological_purity_desired_resource_events.7 }
    }	
}

# Define ideological_purity_new_desired_value based on ideological_purity desired point change production
country_event = {
    id = ideological_purity_desired_resource_events.2
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        # Get the current production
        export_resource_income_to_variable = {
            resource = ideological_purity_desired_value_change
            variable = dre_current_ideological_purity_desired_value_change
        }

        # Define new desired point
        set_variable = {
            which = dre_ideological_purity_new_desired_value
            value = 50
        }
        change_variable = {
            which = dre_ideological_purity_new_desired_value
            value = dre_current_ideological_purity_desired_value_change
        }

        # Clamp desired value to 0 inclusive to 99 inclusive
        if = {
            limit = {
                check_variable = {
                    which = dre_ideological_purity_new_desired_value
                    value > 99
                }
            }
            set_variable = {
                which = dre_ideological_purity_new_desired_value
                value = 99
            }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = dre_ideological_purity_new_desired_value
                    value < 0
                }
            }
            set_variable = {
                which = dre_ideological_purity_new_desired_value
                value = 0
            }
        }
        # ideological_purity_new_desired_value is now the new desired point
    }
}

# Set ideological_purity resting point to ideological_purity_new_desired_value
country_event = {
    id = ideological_purity_desired_resource_events.3
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        # Old desired value
        export_resource_stockpile_to_variable = {
            resource = ideological_purity_desired_value
            variable = dre_current_ideological_purity_desired_value
        }

        # Determine the delta needed to achieve new desired value
        set_variable = {
            which = dre_ideological_purity_desired_value_delta
            value = dre_ideological_purity_new_desired_value 
        }
        subtract_variable = {
            which = dre_ideological_purity_desired_value_delta
            value = dre_current_ideological_purity_desired_value
        }
        # ideological_purity_desired_value_delta is now the delta between new and old desired point

        # Set desired value to new desired value
        add_resource = {
            ideological_purity_desired_value = 1
            mult = dre_ideological_purity_desired_value_delta
        }
    }
}

# Set delta_between_current_desired_ideological_purity
country_event = {
    id = ideological_purity_desired_resource_events.4
    hide_window = yes
    is_triggered_only = yes
    
    immediate = {
        # Get current value of ideological_purity
        export_resource_stockpile_to_variable = {
            resource = ideological_purity_current_value
            variable = dre_current_ideological_purity_current_value
        }
        set_variable = {
            which = dre_delta_between_current_desired_ideological_purity
            value = dre_ideological_purity_new_desired_value
        }
        subtract_variable = {
            which = dre_delta_between_current_desired_ideological_purity
            value = dre_current_ideological_purity_current_value
        }
        # delta_between_current_desired_ideological_purity now represents the difference between the values
    }
}

# Set floored_current_ideological_purity_value and floored_desired_ideological_purity_value
country_event = {
    id = ideological_purity_desired_resource_events.6
    hide_window = yes
    is_triggered_only = yes
    
    immediate = {
        set_variable = {
            which = dre_floored_current_ideological_purity_value
            value = dre_current_ideological_purity_current_value
        }
        floor_variable = dre_floored_current_ideological_purity_value
        
        set_variable = {
            which = dre_floored_desired_ideological_purity_value
            value = dre_ideological_purity_new_desired_value
        }
        floor_variable = dre_floored_desired_ideological_purity_value
    }
}

# Set trending current value
country_event = {
    id = ideological_purity_desired_resource_events.7
    hide_window = yes
    is_triggered_only = yes
    
    immediate = {
        if = {
            # If the value is close enough, do nothing
            limit = {
                check_variable = {
                    which = dre_floored_current_ideological_purity_value
                    value = dre_floored_desired_ideological_purity_value
                }
            }
            remove_modifier = ideological_purity_current_value_trending_up
            remove_modifier = ideological_purity_current_value_trending_down
        }
        else_if = {
            # Adding this months resources would put us over the desired point, so just set it equal to the desired point
            limit = {
                OR = {
                    AND = {
                        check_variable = {
                            which = dre_current_ideological_purity_current_value
                            value < dre_ideological_purity_new_desired_value
                        }
                        check_variable = {
                            which = dre_next_months_current_ideological_purity
                            value >= dre_ideological_purity_new_desired_value
                        }
                    }

                    AND = {
                        check_variable = {
                            which = dre_current_ideological_purity_current_value
                            value > dre_ideological_purity_new_desired_value
                        }
                        check_variable = {
                            which = dre_next_months_current_ideological_purity
                            value <= dre_ideological_purity_new_desired_value
                        }
                    }
                }
                
            }
            remove_modifier = ideological_purity_current_value_trending_up
            remove_modifier = ideological_purity_current_value_trending_down
            add_resource = {
                ideological_purity_current_value = 1
                mult = dre_delta_between_current_desired_ideological_purity
            }
        }
        else_if = {
            # If we are far away from our desired value
            if = {
                limit = {
                    check_variable = {
                        which = dre_ideological_purity_new_desired_value
                        value >= dre_current_ideological_purity_current_value
                    }
                }
                remove_modifier = ideological_purity_current_value_trending_down
                add_modifier = {
                    modifier = ideological_purity_current_value_trending_up
                    days = -1
                }
            }
            else = {
                remove_modifier = ideological_purity_current_value_trending_up
                add_modifier = {
                    modifier = ideological_purity_current_value_trending_down
                    days = -1
                }
            }
        }
    }
}
