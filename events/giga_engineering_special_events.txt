country_event = {
	id = spegigaeng.1001
	title = spegigaeng.1001.title
	desc = spegigaeng.1001.desc
	picture = GFX_evt_exploding_planet
	show_sound = event_super_explosion  
	location = FROMFROM

	is_triggered_only = yes

	option = {
		name = spegigaeng.1001.a
		hidden_effect = {
			add_resource = { influence = 200 }
			remove_megastructure = FROMFROM
			FROMFROM.solar_system = {
				every_system_planet = {
					limit = {
						NOT = { is_star = yes }
						NOT = { is_planet_class = pc_gas_giant }
						NOT = { is_planet_class = pc_asteroid }
						NOT = { is_planet_class = pc_shattered }
						NOT = { is_planet_class = pc_broken }
					}
					change_pc = pc_molten
				}
				every_system_planet = {
					limit = {
						is_star = no
						planet_size < 10
					}
					change_pc = pc_asteroid			
				}
				every_system_planet = {
					limit = {
						planet_size > 10
					}
					change_planet_size = -5			
				}
				every_system_planet = {
					limit = {
						is_planet_class = pc_asteroid
					}
					remove_planet = yes 
				}
				every_system_planet = {
					limit = {
						OR = {
							is_planet_class = pc_gigaorbital_ring
							is_planet_class = pc_gigaorbital_ring_ruined
							is_planet_class = pc_flat_world
						}
					}
					remove_planet = yes 
				}
				random_system_planet = {
					limit = { 
						is_star = yes
					}
					change_pc = pc_neutron_star
				}
				every_system_planet = {
					limit = {
						is_planet_class = pc_gas_giant
					}
					while = { 
						limit = { 
							planet_size > 5
						} 
						change_planet_size = -1			
					}
					change_pc = pc_molten
				}
				every_system_ambient_object = { destroy_ambient_object = this }
				every_fleet_in_system = { destroy_fleet = this }
				random_system_planet = {
					limit = { is_star = yes }
					create_ambient_object = {
						type = "explosion_particle_object"
						location = this
						duration = 50
						
						use_3d_location = yes
						
						entity_offset = { min = 0 max = 0 }
						entity_offset_angle = { min = 0 max = 0 }
						entity_offset_height = { min = 0 max = 0 }
						
						entity_scale_to_size = yes
						scale = 100
					}
				}
				every_system_megastructure = {
					if = {
						limit = {
							exists = this
						}
						remove_megastructure = this
					}
				}
				every_system_planet = {
					limit = {
						is_star = no
					}
					set_deposit = d_iodizium_deposit
				}
			}
  		}
	}
}
