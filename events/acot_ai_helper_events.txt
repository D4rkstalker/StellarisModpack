namespace = acot_ai_helper

##########THE MASTER CHECKLIST#############YEARLY
event = {
	id = acot_ai_helper.1
	hide_window = yes
    is_triggered_only = yes 
	
	trigger = {
		NOT = {
			has_global_flag = acot_ai_helper_disabled
		}
	}

	immediate = {
        every_country = {
            if = {
                limit = {
					is_ai = yes
					is_country_type = "default"
                }
				country_event = { id = acot_ai_helper.5 days = 1 random = 40 }
            }
		}
	}
}
####SPAWNS SOME BETTER TIER ARMIES
country_event = {
	id = acot_ai_helper.5
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		IF = {
			limit = {
				count_armies = {
					limit = { army_type = fe_army }
					count < 8
				}

				limit = {

						is_capital = yes

							owner = ROOT

					}
				}
			}
		}
		IF = {
			limit = {
				count_armies = {
