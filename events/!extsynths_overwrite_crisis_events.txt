# Malfunctioning Synthetic
planet_event = {
	id = crisis.2011
	title = "crisis.2100.name"
	desc = "crisis.2100.desc"
	picture = GFX_evt_robot_assembly_plant
	show_sound = event_robo
	location = root

	is_triggered_only = yes
	
	trigger = {
		any_pop = {
			has_trait = "trait_mechanical"
			NOT = {
				has_trait = "trait_extsynth_synthetic"
			}
		}
	}

	option = {
		name = crisis.2100.a
		random_pop = {
			limit = {
				has_trait = "trait_mechanical"
				NOT = {
					has_trait = "trait_extsynth_synthetic"
				}
			}
 			kill_pop = yes
		}
	}
}

			root.species = {
				NOR = {
					has_trait = trait_extsynth_synthetic
					has_trait = trait_psionic
				}
			}
		}
	}
	desc = {
		text = "crisis.2087.desc_psionic_synth"
		trigger = {
			root.species = {
				has_trait = trait_extsynth_synthetic
				has_trait = trait_psionic
			}
		}
	}
	picture = GFX_evt_robot_assembly_plant
	show_sound = event_robo

	is_triggered_only = yes

	immediate = {
		random_owned_planet = {
			limit = { controller = { is_same_value = root } }
			save_event_target_as = infil_planet
		}
	}

	option = {
		name = crisis.2087.a
	}
}
