country_event = {
	id = throne.0003
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_global_flag = throne_exists
			}
			random_owned_planet = {
				limit = {
					is_capital = yes }

					add_modifier = {
						modifier = "throne_boon"
						days = -1
					}
				}
		}
	}
}

country_event = {
	id = throne.0002
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = { has_modifier = throneworld_contribution }
			remove_modifier = throneworld_contribution
		}
		every_owned_planet = {
			if = {
				limit = {
					has_modifier = "throne_boon"
				}
				remove_modifier = "throne_boon"
			}
		}
	}
}
