		}

		}

#Share with allies
country_event = {
	id = shroud_rising_monolith.9
	hide_window = yes
	is_triggered_only = yes
	
	
	immediate = {	
		
		every_playable_country = {
				if = { 
					limit = {
						NOT = { is_same_value = prev }	
						OR = {							
							has_defensive_pact = prev
							is_in_federation_with = prev
							AND = {
								is_subject = yes
								overlord = { is_same_empire = prevprev }
							}
							AND = {
								is_subject = yes
								overlord = { is_overlord_to = prevprev	}
							}
							is_overlord_to = prev										
						}
					}
					add_opinion_modifier = {
						who = root
						modifier = opinion_shared_monolith_knowledge 
					}
					add_research_option = tech_shroud_monolith
					add_tech_progress = {
						tech = tech_shroud_monolith
						progress = 1.0
						}
				}
				else_if = {
					limit = { 
							NOT = { is_same_value = prev }							
						}
				add_opinion_modifier = {
					who = root				
					modifier = opinion_withheld_monolith_knowledge
				}
				
			}
		
		}
	}
	
}
