namespace = fixit_gamestart

### add AM deposits to ringworld capitals that don't have them
event = {
	id = fixit_gamestart.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_global_flag = gf_alphamod_activated
	}

	immediate = {
		every_planet = {
			limit = {
				is_capital = yes
				is_ringworld_colony = yes
			}

			if = {
				limit = { NOT = { has_deposit = d_fuels_cache } }
				add_deposit = d_fuels_cache
			}
			if = {
				limit = { NOT = { has_deposit = d_biovault } }
				add_deposit = d_biovault
			}
		}
	}
}
