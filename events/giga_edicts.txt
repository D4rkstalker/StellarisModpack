namespace = giga_edicts

#Search for moons
country_event = {
	id = giga_edicts.1
	title = "giga_edicts.1.name"
	desc = "giga_edicts.1.desc"
	picture = GFX_evt_cold_barren
	show_sound = event_sensor_ping
	is_triggered_only = yes
	option = { name = "giga_edicts.1.a" }

	immediate = {
		every_planet_within_border = {
			limit = {
				OR = {
					is_planet_class = pc_molten
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_frozen
				}
				planet_size >= 5
				planet_size <= 7
				NOT = { has_modifier = attack_moon_candidate }
			}
			add_modifier = { modifier = "attack_moon_candidate" days = -1 }
		}
	}
}

#Search for planets
country_event = {
	id = giga_edicts.2
	title = "giga_edicts.2.name"
	desc = "giga_edicts.2.desc"
	picture = GFX_evt_cold_barren
	show_sound = event_sensor_ping
	is_triggered_only = yes
	option = { name = "giga_edicts.2.a" }

	immediate = {
		every_planet_within_border = {
			limit = {
				OR = {
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
				}
				planet_size >= 25
				NOT = { has_modifier = attack_planet_candidate }
			}
			add_modifier = { modifier = "attack_planet_candidate" days = -1 }
		}
	}
}
