#_relic	<r_defender>
country_event = {
	id = AAR_evt_CRISIS.1
	title = eapdefender
	desc = eapdefender_desc
	picture = GFX_AAR_EP_crisis
	show_sound = event_mystic_reveal_light
	location = root
	
	is_triggered_only = yes

	immediate = { 
		set_global_flag = galactic_crisis_happened
	}

	##	虫群
	option = {
		name = defender1
		custom_tooltip = AAR_tips_defender_summon
		trigger = { NOT = { has_global_flag = prethoryn_invasion_happened } }
		hidden_effect = {
			random_country = {
				limit = { is_country_type = global_event } # global_event country is always around
				country_event = { id = crisis.10	 days = 3100 random = 1000 }
			}
			#	添加态度
			set_update_modifiers_batch = begin	#_|修正更新|暂停
			every_country = { limit = { AAR_trgr_cntr_playable = yes }	# |AAR可用国家|
				add_opinion_modifier = { modifier = AAR_opinion_eap_defender who = root }
			}
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}
	##	恶魔
	option = {
		name = defender2
		custom_tooltip = AAR_tips_defender_summon
		trigger = { NOT = { has_global_flag = extradimensional_invasion_happened } }
		hidden_effect = {
			random_country = {
				limit = { is_country_type = global_event } # global_event country is always around
				country_event = { id = crisis.1000	 days = 3100 random = 1000 }
			}
			#	添加态度
			set_update_modifiers_batch = begin	#_|修正更新|暂停
			every_country = { limit = { AAR_trgr_cntr_playable = yes }	# |AAR可用国家|
				add_opinion_modifier = { modifier = AAR_opinion_eap_defender who = root }
			}
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}
	##	肃正
	option = {
		name = defender3
		custom_tooltip = AAR_tips_defender_summon
		trigger = { NOT = { has_global_flag = ai_invasion_happened } }
		hidden_effect = {
			random_country = {
				limit = { is_country_type = global_event } # global_event country is always around
				country_event = { id = crisis.2005	 days = 3100 random = 1000 }
			}
			#	添加态度
			set_update_modifiers_batch = begin	#_|修正更新|暂停
			every_country = { limit = { AAR_trgr_cntr_playable = yes }	# |AAR可用国家|
				add_opinion_modifier = { modifier = AAR_opinion_eap_defender who = root }
			}
			set_update_modifiers_batch = end	#_|修正更新|重启
		}
	}

	##	无
	option = {
		name = AAR_opt_defender_no
		trigger = {
			has_global_flag = prethoryn_invasion_happened
			has_global_flag = extradimensional_invasion_happened
			has_global_flag = ai_invasion_happened
		}
	}
}
