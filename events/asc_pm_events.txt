event = {
	id = asc_pm.3000
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		any_playable_country = {
			any_owned_planet = {
				owner = {
					is_gestalt = yes
				}
				OR = {
					is_training_planet = yes
					is_planet_class = pc_ringworld_eco
					is_planet_class = pc_ringworld_trade
				}
			}
		}
	}

	immediate = {
		random_playable_country = {
			limit = {
				any_owned_planet = {
					owner = {
						is_gestalt = yes
					}
					OR = {
						is_training_planet = yes
						is_planet_class = pc_ringworld_eco
						is_planet_class = pc_ringworld_trade
					}
				}
			}
			random_owned_planet = {
				limit = {
					OR = {
						is_training_planet = yes
						is_planet_class = pc_ringworld_eco
						is_planet_class = pc_ringworld_trade
					}
				}
				remove_modifier = "high_planet_bad"
				remove_modifier = "high_planet_upgraded"
				remove_modifier = "high_planet_p_upgraded"
				remove_modifier = "high_planet_normal"
				remove_modifier = "connected_hm"
				remove_modifier = "recently_changed_ringworld"
				add_modifier = {
					modifier = "cannnot_connect_to_hm"
					days = -1
				}
			}
		}
	}
}
