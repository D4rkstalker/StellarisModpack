namespace = eo_symbiosis

country_event = {
	id = eo_symbiosis.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = yes
		any_owned_pop = {
			is_symbiote = yes
			planet_has_symbiotes = no
			is_being_purged = no
		}
	}

	immediate = {
		every_owned_planet = {
			limit = {
				is_occupied_flag = no
				any_owned_pop = {
					is_symbiote = yes
					is_being_purged = no
					planet_has_symbiotes = no
				}
			}
			random_owned_pop = {
				limit = {
					is_symbiote = yes
					is_being_purged = no
					planet_has_symbiotes = no
				}
				import_symbiote_for_pop = yes
			}
		}
	}
}
