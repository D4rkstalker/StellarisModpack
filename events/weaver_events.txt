#weaver-roll
country_event = {
	id = weaver.5
	hide_window = yes
	trigger = {
		has_country_flag = big_punishment
	}	
	fire_only_once = yes
	mean_time_to_happen = {
		months = 2
	}
	immediate = {
		random_list = {
			10 = {
				if = {
					limit = { 
						NOR = { 
							any_owned_leader = {
								OR = {
									has_trait = leader_trait_admiral_chosen
									has_trait = leader_trait_general_chosen
									has_trait = leader_trait_governor_chosen
									has_trait = leader_trait_ruler_chosen
									has_trait = leader_trait_scientist_chosen
								}
							}
						} 

					}
					country_event = { id = weaver.7 }#dead
				}
				else = { country_event = { id = weaver.6 } }#
			}
			10 = {
				country_event = { id = weaver.6 }#alive
			}
			10 = {
				country_event = { id = weaver.7 } #dead
			}
		}
	}
}
