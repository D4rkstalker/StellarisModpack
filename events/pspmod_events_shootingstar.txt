fleet_event = {
	id = pspmod_sless.3
	title = "pspmod_sless.3.name"
	desc = "pspmod_sless.3.desc"
	picture = GFX_evt_ruined_system
	show_sound = event_wind_ruins
	
	archaeology = yes
	is_triggered_only = yes
	
	immediate = {
		from = { set_site_progress_locked = yes }
		closest_system = {
			limit = {
				has_access_fleet = root.owner
				NOT = { is_same_value = root.solar_system }
				OR = {
					has_owner = no
					owner = { is_same_value = root.owner }
				}
				any_system_planet = {
					has_anomaly = no
					is_colony = no
					OR = {
						is_planet_class = pc_barren
						is_planet_class = pc_barren_cold
						is_planet_class = pc_nuked
					}
					NOR = {
						is_planet_class = pc_gas_giant
						exists = archaeological_site
					}
				}
				any_neighbor_system = {
					any_neighbor_system = {
						any_system_planet = { #2 jumps
							NOT = { has_planet_flag = pspmod_sless_colony }
						}
					}
				}
				NOT = {
					any_system_planet = {
						OR = {
							has_planet_flag = precursor_world
							has_planet_flag = pspmod_sless_colony
						}
					}
				}
			}
			min_steps = 3
			max_steps = 10
			random_system_planet = {
				limit = {
					has_anomaly = no
					is_colony = no
					OR = {
						is_planet_class = pc_barren
						is_planet_class = pc_barren_cold
						is_planet_class = pc_nuked
					}
					NOR = {
						is_planet_class = pc_gas_giant
						exists = archaeological_site
					}
				}
				save_event_target_as = pspmod_sless_colony_2
				change_pc = pc_nuked
				if = {
					limit = { planet_size > 25 }
					set_planet_size = 25
				}
			}
		}
		if = {
			limit = { NOT = { exists = event_target:pspmod_coprean_colony_2 } }
			closest_system = {
				limit = {
					has_access_fleet = root.owner
					NOT = { is_same_value = root.solar_system }
					OR = {
						has_owner = no
						owner = { is_same_value = root.owner }
					}
					any_system_planet = {
						has_anomaly = no
						OR = {
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
							is_planet_class = pc_nuked
						}
						NOT = { exists = archaeological_site }
					}
					NOT = {
						any_system_planet = {
							OR = {
								has_planet_flag = precursor_world
								has_planet_flag = pspmod_sless_colony
							}
						}
					}
				}
				min_steps = 1
				max_steps = 10
				random_system_planet = {
					limit = {
						has_anomaly = no
						OR = {
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
							is_planet_class = pc_nuked
						}
						NOT = { exists = archaeological_site }
					}
					save_event_target_as = pspmod_sless_colony_2
				}
			}
		}
	}

	after = {
		from = { set_site_progress_locked = no }
	}

	option = {
		name = pspmod_sless.3.a
		owner = {
			add_resource = { minor_artifacts = 5 }
		}
		event_target:pspmod_sless_colony_2 = {
			enable_special_project = {
				name = "PSP_SLESS_ALLIANCE_PROJECT_1"
				location = this
				owner = root.owner
			}
		}
	}
}
