fleet_event = {
	id = asp.53
	title = "asp.53.name"
	desc = "asp.53.desc"
	picture = GFX_evt_ancient_tiyanki
	show_sound = elder_tiyanki_idle_vocal
	
	archaeology = yes
	is_triggered_only = yes
	
	immediate = {
		from = { set_site_progress_locked = yes }
	}

	after = {
		from = { set_site_progress_locked = no }
	}

	option = {
		name = asp.53.a
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_xenophile 
					has_ethic = ethic_fanatic_xenophile
				}
			}
		}
		from.planet = {
			add_modifier = { modifier = aspmod_whale_fossil days = -1 }
		}
		owner = {
			add_modifier = { modifier = aspmod_site_whale_lesson days = -1 }
		}
	}
	option = {
		name = asp.53.b
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
				}
			}
		}
		from.planet = {
			add_modifier = { modifier = aspmod_whale_fossil days = -1 }
		}
		owner = {
			add_modifier = { modifier = aspmod_whale_hunt days = -1 }
		}
	}
	option = {
		name = asp.53.c
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_materialist
					has_ethic = ethic_fanatic_materialist
				}
			}
		}
		from.planet = {
			add_modifier = { modifier = aspmod_whale_fossil days = -1 }
			add_deposit = aspmod_d_exotic_gases_layers
		}
	}
	option = {
		name = asp.53.d
		trigger = {
			owner = {
				NOR = {
					has_ethic = ethic_xenophile 
					has_ethic = ethic_fanatic_xenophile
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
					has_ethic = ethic_materialist
					has_ethic = ethic_fanatic_materialist
				}
			}
		}
		from.planet = {
			add_modifier = { modifier = aspmod_whale_fossil days = -1 }
			add_deposit = aspmod_d_exotic_gases_layers
		}
	}
}
