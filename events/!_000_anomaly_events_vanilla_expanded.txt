ship_event = {
	id = anomaly.5
	title = "anomaly.5.name"
	desc = "anomaly.5.desc"
	picture = GFX_evt_treasure_pile
	show_sound = event_ship_bridge
	location = FROM
	
	is_triggered_only = yes

	immediate = {
		owner = {
			country_event = { id = story.5 days = 30 }
			set_country_flag = anomaly_5_happened
		}
	}
	
	option = {
		name = anomaly.5.a
		owner = {
			if = {
				limit = { is_gestalt = no }
				add_monthly_resource_mult = {
					resource = energy
					value = @tier1materialreward
					min = @tier1materialmin
					max = @tier1materialmax
				}
				add_monthly_resource_mult = {
					resource = consumer_goods
					value = @tier1materialreward
					min = @tier1materialmin
					max = @tier1materialmax
				}
			}
			else = {
				add_monthly_resource_mult = {
					resource = energy
					value = @tier2materialreward
					min = @tier2materialmin
					max = @tier2materialmax
				}
			}
			hidden_effect = {
				set_country_flag = lcluster_clue_pirate
				if = {
					limit = {
						ROOT = {
							leader = {
								OR = {
									has_trait = leader_trait_expertise_computing
									has_trait = leader_trait_custom_AI_assistant
									has_trait = leader_trait_sapient_AI_assistant
									has_trait = leader_trait_archaeologist
									has_trait = leader_trait_archaeologist_ancrel
								}
							}
						}
						has_distar = yes
						has_country_flag = encountered_first_lgate
						NOR = {
							has_global_flag = l_cluster_opened
							has_completed_event_chain_counter = {
								event_chain = l_cluster_chain
								counter = clues
							}
						}
					}
					country_event = { id = distar.9000 days = 10 }
				}
				else = {
					pop_lcluster_event = yes
				}
			}
			small_artifact_reward = yes
		}
	}
}

### Junkyard
ship_event = {
	id = anomaly.30
	title = "anomaly.30.name"
	desc = {
		trigger = {
			NOT = { has_ship_flag = anomaly_30_clue }
		}
		text = "anomaly.30.desc"
	}
	desc = {
		trigger = {
			has_ship_flag = anomaly_30_clue
		}
		text = "anomaly.30.desc.clue"
	}
	picture = GFX_evt_junkheap
	show_sound = event_ship_bridge
	location = FROM
	
	is_triggered_only = yes

	immediate = {
		from = { clear_deposits = yes }
		owner = { country_event = { id = story.5 days = 30 } }
		if = {
			limit = {
				has_distar = yes
				NOT = { has_global_flag = l_cluster_opened }
				owner = { has_event_chain = l_cluster_chain }
				leader = {
					OR = {
						has_trait = leader_trait_meticulous
						has_trait = leader_trait_cataloger
						has_trait = leader_trait_expertise_materials
					}
				}
			}
			set_ship_flag = anomaly_30_clue
		}
	}
	
	option = {
		name = anomaly.30.a
		owner = {
			hidden_effect = {
				set_country_flag = lcluster_clue_junkyard
				pop_lcluster_event = yes
			}
		}
	}
	option = {
		name = FASCINATING
		trigger = {
			has_ship_flag = anomaly_30_clue
		}
		allow = {
			leader = {
				OR = {
					has_trait = leader_trait_meticulous
					has_trait = leader_trait_cataloger
					has_trait = leader_trait_expertise_materials
				}
			}
		}
		owner = {	
			generate_lcluster_clue = yes
		}
	}
	after = {
		FROM = { 
			set_deposit = d_minerals_3
		}
		if = {
			limit = { has_ship_flag = anomaly_30_clue }
			FROM = { 
				add_deposit = d_nanites_deposit
			}
		}
		else = {
			FROM = { 
				add_deposit = d_alloys_1
			}
		}
		hidden_effect = {
			FROM = {
				add_modifier = {
					modifier = "junkyard"
					days = -1
				}
			}
			remove_ship_flag = anomaly_30_clue
		}
		owner = { small_artifact_chance = yes }
	}
}

	}
	desc = {
		trigger = {
			NOT = {
				owner = { any_planet_within_border = { is_same_value = event_target:MiningPlanet } }
			}
		}
		text = "anomaly.40.desc.alt"
	}
	picture = GFX_evt_mining_station
	show_sound = event_ship_bridge

	is_triggered_only = yes

	immediate = {
		owner = {
			country_event = { id = story.5 days = 30 }

			save_event_target_as = MyCountry
		}
		from = {
			save_event_target_as = MiningPlanet
		}
	}

	option = {
		name = EXCELLENT

			owner = {
