# Kick starts event chain
country_event = {
	id = core.0
	title = "core.0.name"
	desc = "core.0.desc"
	#hide_window = no
	fire_only_once = yes
	picture = GFX_evt_contingency_planet
	show_sound = event_cityscape
	location = root

	trigger = {
		has_origin = originsextended_core
	}

	option = {
		name = "core.0.option"
		#custom_tooltip = "core.0.tooltip"
	}

	after = {
		begin_event_chain = {
			event_chain = originsextended_core_chain_1
			target = ROOT
		}
		hidden_effect = {
			random_planet_within_border = {
				limit = {
					is_capital = yes
				}
				planet_event = {
					id = core.1
					days = 720
					random = 200
				}
			}
		}
	}
}
