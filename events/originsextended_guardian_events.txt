# Guardian origin, figure out which event chain to throw
# Chain 0 - Setup
country_event = {
	id = guardian.0000
	hide_window = yes
	#is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		has_origin = originsextended_guardian
	}

	immediate = { # randomizes event chain for variety
		every_country = {
			limit = {
				has_origin = originsextended_guardian
			}
			if = {
				limit = {
					owner = {
						has_megastructure = think_tank_ruined
					}
				}
				country_event = {
					id = guardian.1000
				}
			}
			if = {
				limit = {
					owner = {
						has_megastructure = spy_orb_ruined
					}
				}
				country_event = {
					id = guardian.2000
				}
			}
			if = {
				limit = {
					owner = {
						has_megastructure = mega_art_installation_ruined
					}
				}
				country_event = {
					id = guardian.3000
				}
			}
			if = {
				limit = {
					owner = {
						has_megastructure = interstellar_assembly_ruined
					}
				}
				country_event = {
					id = guardian.4000
				}
			}
			if = {
				limit = {
					owner = {
						has_megastructure = strategic_coordination_center_ruined
					}
				}
				country_event = {
					id = guardian.5000
				}
			}
			if = {
				limit = {
					owner = {
						has_megastructure = mega_shipyard_ruined
					}
				}
				country_event = {
					id = guardian.6000
				}
			}
		}
	}
}
