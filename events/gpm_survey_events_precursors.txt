#1 spawn precursor dig site
ship_event = {
	id = gpm_survey_precursors.1
	hide_window = yes
	is_triggered_only = yes
	location = FROM
	
	trigger = {
		FROM = {
			gpm_has_precursor_modifiers = yes
			NOT = { exists = archaeological_site }
			NOR = {
				gpm_is_planet_class_star = yes
				gpm_is_planet_class_asteroid = yes
				gpm_is_planet_class_gas_giant = yes
			}
			OR = {
				solar_system = {
					NOT = { has_star_flag = precursor_system }
				}
				has_planet_flag = gpm_is_precursor_homesystem_planet
			}
		}
		root = { owner = { has_ancrel = yes } }
	}

	immediate = {
		FROM.planet ={
			if = {
				limit = {
					gpm_is_planet_class_normal_uninhabitable = yes
					has_deposit = no
				}
			}
		}
		random_list = {
			50 = { }
			50 = { FROM = { create_archaeological_site = gpm_precursor_generic_digsite } }
		}
	}
}
