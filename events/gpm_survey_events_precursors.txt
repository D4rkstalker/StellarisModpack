#10 Precursor_Remnants discovery
ship_event = {
	id = gpm_survey_precursors.10
	title = "gpm_survey_precursors.10.name"
	desc = {
		trigger = {
			owner = {
				NOR = {
					has_authority = auth_hive_mind
					has_authority = auth_machine_intelligence
				}
			}
		}
		text = "gpm_survey_precursors.10.desc"
	}
	desc = {
		trigger = {	owner = { has_authority = auth_hive_mind } }
		text = "gpm_survey_precursors.10.hive.desc"
	}
	desc = {
		trigger = {	owner = { has_authority = auth_machine_intelligence } }
		text = "gpm_survey_precursors.10.mach.desc"
	}
	picture = GFX_evt_gpm_precursor_remnants
	show_sound = event_scanner
	location = FROM

	is_triggered_only = yes
	
	trigger = {
		FROM = {
			AND = {
				has_modifier = "Precursor_Remnants"
				root = { owner = { NOT = { has_country_flag = gpm_discover_precursor_remnants } } }
			}
		}
	}

	immediate = {	
		root = { owner = { set_country_flag = gpm_discover_precursor_remnants } }
		FROM = {
			if = {
				limit = { NOT = { has_planet_flag = "gpm_is_surveyed" }	}
				gpm_colour_planet_name = yes
				set_planet_flag = "gpm_is_surveyed"
			}
			save_event_target_as = precursor_planet
		}
	}
	
	option = {
		name = gpm_survey.UNBELIEVABLE
		root = {
			owner = {
				if = {
					limit = { has_ancrel = yes }
					medium_artifact_reward = yes
				}
				add_monthly_resource_mult = {
					resource = physics_research
					value = @tier1researchreward
					min = @tier1researchmin
					max = @tier1researchmax
				}
				add_monthly_resource_mult = {
					resource = society_research
					value = @tier1researchreward
					min = @tier1researchmin
					max = @tier1researchmax
				}
				add_monthly_resource_mult = {
					resource = engineering_research
					value = @tier1researchreward
					min = @tier1researchmin
					max = @tier1researchmax
				}
			}
		}
	}
}
