# Spawn next reality code
country_event = {
	id = ehof_reality_code.006
	is_triggered_only = yes
	hide_window = yes
	trigger = { NOT = { has_country_flag = ehof_code_7_complete } }

	immediate = {
		if = {
			limit = {
				OR = {
					AND = {
						OR = {
							has_country_flag = ehof_code_toxic_world
							has_country_flag = ehof_code_broken_world
						}
						check_variable = { which = ehof_phase value >= 1 }
					}
					AND = {
						OR = {
							has_country_flag = ehof_code_frozen_world
							has_country_flag = ehof_code_molten_world
						}
						check_variable = { which = ehof_phase value >= 2 }
					}
					AND = {
						OR = {
							has_country_flag = ehof_code_gas_giant_world
							has_country_flag = ehof_code_star_world
						}
						check_variable = { which = ehof_phase value >= 3 }
					}
				}
			}
			set_global_flag = ehof_spawning_reality_code
			if = { limit = { exists = event_target:current_system_creator } clear_global_event_target = current_system_creator }
			save_global_event_target_as = current_system_creator
			ehof_generate_system = yes

			set_country_flag = ehof_new_system_poi
			change_variable = { which = ehof_new_system_poi value = 1 }
			set_country_flag = ehof_started_reality_code
			if = {
				limit = { is_ai = yes }
				set_country_flag = ehof_ai_system_ready
			}
			remove_global_flag = ehof_spawning_reality_code
		}
		else = { country_event = { id = ehof_reality_code.006 days = 60 } } # Try again in 2 months
	}
}

# Investigate Final digsite
country_event = {
	id = ehof_reality_code.1000
	diplomatic = yes
	custom_gui = "giga_eventui_window"
	picture_event_data = { room = "enigmatic_code_room" }
	title = "ehof_reality_code.1000.name"
	desc = "ehof_reality_code.1000.desc"
	is_triggered_only = yes
	location = event_target:creator_poi

	trigger = { NOT = { has_country_flag = ehof_final_digsite } }

	immediate = {
		hidden_effect = {
			set_country_flag = ehof_final_digsite

			# Save creator
			if = {
				limit = { exists = event_target:current_system_creator }
				clear_global_event_target = current_system_creator
			}
			save_global_event_target_as = current_system_creator

			# Create system
			random_system_within_border = {
				limit = {
					cohesive_system = yes
					has_star_flag = ehof_sector_system_@root
				}
				spawn_system = {
					min_distance >= 5
					max_distance <= 25
					hyperlane = no
					initializer = "ehof_unique_cplanet"
				}
			}

			# Connect hyperlane
			random_system = {
				limit = {
					everchanging_system = yes
					has_star_flag = ehof_system_created_by_@root
				}
				save_event_target_as = creator_poi
				random_system = {
					limit = {
						NOT = { is_same_value = prev }
						has_star_flag = ehof_system_created_by_@root
						has_star_flag = ehof_sector_system_@root
						has_star_flag = empire_has_visited@root
						distance = {
							type = euclidean
							source = prev
							min_distance >= 0
							max_distance <= 30
						}
					}
					add_hyperlane = {
						from = this
						to = prev
					}
				}
			}

			# Create POI
			create_point_of_interest = {
				id = ehof_mega_chain.1
				name = "ehof_mega_chain_1_poi"
				desc = "ehof_mega_chain_1_poi_desc"
				event_chain = ehof_p4_chain
				location = event_target:creator_poi
			}

			country_event = { id = story.5 days = 30 }
		}
	}

	option = {
		custom_gui = "giga_eventui_option"
		default_hide_option = yes
		name = INTERESTING
	}
}
