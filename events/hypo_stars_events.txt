	id = hypostars.1
	title = "hypostars.1.name"
	desc = "hypostars.1.desc"
	picture = GFX_evt_star_red
	show_sound = event_ship_bridge
	location = FROM
	
	is_triggered_only = yes
	
	
	option = {
		name = "INTRIGUING"
		from = { save_event_target_as = antimatterprojectflag }
		event_target:antimatterprojectflag = {
			enable_special_project = {
				name = "hypostars_antimatter_project"
				owner = root
				location = this
			}
		}
	}
}

####PROJECT EVENTS

#Antimatter star

ship_event = {
	id = hypostars.401
	title = "hypostars.401.name"
	desc = "hypostars.401.desc"
	picture = GFX_evt_star_red_antimatter
	location = FROM
	
	is_triggered_only = yes
	
	
	option = {
		name = "EXCELLENT"
		owner = {
			if = {
				limit = {
					has_tech_option = tech_mine_antimatter
				}
				add_tech_progress = {
					tech = tech_mine_antimatter
					progress = 0.5
				}
				else = {
					add_research_option = tech_mine_antimatter
					add_tech_progress = {
						tech = tech_mine_antimatter
						progress = 0.25
					}
				}
			}
		}
	}
}
