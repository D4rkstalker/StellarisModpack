namespace = engineer_guild
country_event = {
	id = engineer_guild.0001
	hide_window = no
	is_triggered_only = yes
	title = "engineers_established"
	picture = GFX_evt_shattered_planet
	desc = "engineers_established_desc"
	option = {
		name = "engineers_established_done"
		custom_tooltip = "engineers_established_done_desc"
		add_imperial_authority = -10
		hidden_effect = {
			every_country = {
				limit = {
					is_galactic_community_member = yes
				}
				add_modifier = {
					modifier = "engineering_guild_modifier"
					days = -1
				}
			}
		}
	}
}

#Mod is added to new people# 
country_event = {
	id = engineer_guild.0003
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_global_flag = Engineers_exist
			}
			add_modifier = {
				modifier = "engineering_guild_modifier"
			}
		}
	}
}

#mod is removed from leavers
country_event = {
	id = engineer_guild.0002
	hide_window = yes
	is_triggered_only = yes
	picture = GFX_evt_galactic_senate
	immediate = {
		if = {
			limit = {
				has_modifier = engineering_guild_modifier
			}
			remove_modifier = engineering_guild_modifier
		}
	}
}
