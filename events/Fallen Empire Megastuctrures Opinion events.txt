namespace = FEMopinion

event = {
	id = FEMopinion.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_ai_personality = fallen_empire_gamers
			any_war = {
				any_defender = {
					is_same_value = PREVPREV
				}
			}
		}
	}
	
	immediate = {
		every_country = {
			limit = { has_ai_personality = fallen_empire_gamers any_war = { any_defender = { is_same_value = PREVPREV } } }
			country_event = { id = FEMopinion.2 }
		}
	}
}

country_event = { #lasting effect of declaring war on gamers
	id = FEMopinion.2
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		has_ai_personality = fallen_empire_gamers
	}
	
	immediate = {
		every_country = {
			limit = { PREV = { any_war = { any_attacker = { is_same_value = PREVPREVPREV } } } }
			PREV = {
				add_opinion_modifier = { modifier = FEM_opinion_interrupted_gaming who = PREV }
			}
		}
	}
}
