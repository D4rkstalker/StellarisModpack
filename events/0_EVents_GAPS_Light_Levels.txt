				random_system_planet = { limit = { NOT = { has_planet_flag = PFlag_GAPS_Counted } is_star = yes is_primary_star = no has_modifier = Planet_Modifier_GAPS_Light_Level_Minus_One } set_planet_flag = PFlag_GAPS_Counted }
			}
			#MISC
			if= { 
				limit = { has_global_flag = gigastructures_enabled }
				every_system_planet = { 
					limit = { is_star = no is_primary_star = no }
					if = { limit = { is_planet_class = pc_disco_moon } add_modifier = { modifier = Planet_Modifier_GAPS_Light_Level_Plus_One days = -1 } }
				}
			}
			if = { 
				limit = { any_system_planet = { NOT = { has_planet_flag = PFlag_GAPS_Counted } is_star = no is_primary_star = no has_modifier = Planet_Modifier_GAPS_Light_Level_Plus_One } }
				change_variable = {
					which = "Variable_GAPS_System_Light_Level" 
					value = 1
				}
				every_system_planet = { limit = { is_planet_class = pc_disco_moon } set_planet_flag = PFlag_GAPS_Counted }
			}
			if = { 
				limit = { any_system_planet = { has_modifier = Planet_Modifier_GAPS_Light_Level_Plus_Starbase } NOT = { has_planet_flag = PFlag_GAPS_Counted_Starbase } }
				change_variable = {
					which = "Variable_GAPS_System_Light_Level" 
					value = 1
				}
				every_system_planet = { limit = { has_modifier = Planet_Modifier_GAPS_Light_Level_Plus_Starbase } set_planet_flag = PFlag_GAPS_Counted_Starbase }
			}
		}
