namespace = cx_hive_world

### hive world start
country_event = {
	id = cx_hive_world.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_origin = origin_hive_cx_hive_world
	}

	immediate = {
		random_owned_planet ={
			limit = { 
				is_capital = yes
			}
			validate_planet_buildings_and_districts = yes
			set_planet_entity = {
				entity = "infected_planet_01_stage_3_entity"
			}
		}
	}
}


###new colonized planet into a stage 1 hive world
planet_event = {
	id = cx_hive_world.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		habitable_structure = no
		owner = {
			has_origin = origin_hive_cx_hive_world
		}
		NOR = {
			habitable_structure = yes
			has_modifier = cx_hive_stage_1
			has_modifier = cx_hive_stage_2
		}
		OR = {
			habitability = {
				who = owner.species
				value < 1
			}
			is_wet = yes
			is_dry = yes
			is_cold = yes
		}
	}

	immediate = {
		set_planet_entity = {
			entity = "infected_planet_01_stage_1_entity"
		}
		#clear_deposits = yes
		add_modifier = {
			modifier = cx_hive_stage_1
			days = 3600
		}
		## call a planetary event to make it grow of change stage
		planet_event = { id = cx_hive_world.2 days = 3601 } # Silent entity change
	}
}	
planet_event = {
	id = cx_hive_world.2
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		remove_modifier = cx_hive_stage_1
		if = {
			limit = {
				habitability = {
					who = owner.species
					value < 1
				}
			}
			set_planet_entity = {
					entity = infected_planet_01_stage_2_entity
				}
			add_modifier = {
				modifier = cx_hive_stage_2
				days = 7200
			}
			## call a planetary event to make it grow of change stage
			planet_event = { id = cx_hive_world.3 days = 7201 } # Silent entity change
		}
				
			
	}
}
planet_event = {
	id = cx_hive_world.3
	hide_window = yes

	is_triggered_only = yes
	
	immediate = {
			remove_modifier = cx_hive_stage_1
			remove_modifier = cx_hive_stage_2
		if = {
			limit = {
				habitability = {
					who = owner.species
					value < 1
				}
			}
			change_pc = pc_hive
			set_planet_entity = {
					entity = infected_planet_01_stage_3_entity
				}
		}
				
			
	}
}
