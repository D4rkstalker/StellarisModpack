event = {
	id = skalla.3
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		years_passed > 10
		OR = {
			from = {
				is_country_type = default
				used_naval_capacity_percent < 0.5
			}	
			root = {
				is_country_type = default
				used_naval_capacity_percent < 0.5
			}	
		}
	}

	immediate = {
		from = {
			if = { limit = {

event = {
	id = skalla.4
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			is_country_type = default
			is_ai = yes
		}	
	}
	
	immediate = {
		every_country = {
			limit = {
				is_country_type = default
				is_ai = yes
			}
			every_planet_within_border = {
				 limit = {
					is_colony = yes
					free_jobs < 7
					free_district_slots > 1
					NOR = {
						is_planet_class = pc_machine 
						is_planet_class = pc_relic
						is_planet_class = pc_city
						is_planet_class = pc_habitat
						is_planet_class = pc_ringworld_habitable
					}
				}
				if = { 
					limit = {
						owner = {
							resource_income_compare = { 
								resource = minerals
								value > 250
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 600
							}
						}
						free_district_slots > 1
					}
					owner = {
						add_resource = {
							minerals = -600
						}
					}
					add_district = district_industrial
				}
			}
		}
	}
}

event = {
	id = skalla.5
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			is_country_type = default
			is_ai = yes
		}	
	}

	immediate = {
		every_country = {
			limit = {
				is_country_type = default
				is_ai = yes
			}
			every_planet_within_border = {
				 limit = {
					is_colony = yes
					free_jobs < 3
					OR = {
						free_building_slots > 1
						free_district_slots > 1
					}
					NOR = {
						is_planet_class = pc_machine 
						is_planet_class = pc_relic
						is_planet_class = pc_city
						is_planet_class = pc_habitat
						is_planet_class = pc_ringworld_habitable
					}
				}
				#improve mineral production
				if = { 
					limit = {
						owner = {
							resource_income_compare = { 
								resource = minerals
								value < 100
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 300
							}
						}
						free_district_slots > 1
					}
					owner = {
						add_resource = {
							minerals = -300
						}
					}
					add_district = district_mining
				#improve energy production
				} else_if = {	
					limit = {
						owner = {
							resource_income_compare = { 
								resource = energy
								value < 100
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 300
							}
						}
						free_district_slots > 1
					}
					owner = {
						add_resource = {
							minerals = -300
						}
					}
					add_district = district_generator
				#improve food production
				} else_if = {
					limit = {
						owner = {
							country_uses_food = yes
							resource_income_compare = {
								resource = food
								value < 30
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 300
							}
						}
						free_district_slots > 1
					}
					owner = {	
						add_resource = {
							minerals = -300
						}
					}
					add_district = district_farming		
				#improve consumer goods production
				} else_if = {
					limit = {
						owner = {
							country_uses_consumer_goods = yes
							resource_income_compare = {
								resource = consumer_goods
								value < 30
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 300
							}
						}
						free_district_slots > 1
					}
					owner = {	
						add_resource = {
							minerals = -300
						}
					}
					add_district = district_industrial
				#build alloy foundry
				} else_if = {
					limit = {
						owner = {
							resource_stockpile_compare = { 
								resource = minerals
								value > 600
							}
						}
						NOT = {
							has_building = building_foundry_1
						}
						free_building_slots > 1
					}
					owner = {
						add_resource = {
							minerals = -600
						}
					}
					add_building = building_foundry_1	
				} else_if = {
					limit = {
						owner = {
							resource_stockpile_compare = { 
								resource = minerals
								value > 600
							}
						}
						NOT = {
							has_building = building_stronghold
						}
						free_building_slots > 1
					}
					owner = {
						add_resource = {
							minerals = -600
						}
					}
					add_building = building_stronghold	
				#improve alloy production
				} else_if = {
					limit = {
						owner = {
							resource_income_compare = { 
								resource = minerals
								value > 100
							}
							resource_stockpile_compare = { 
								resource = minerals
								value > 500
							}
						}
						free_district_slots > 1
					}
					owner = {					
						add_resource = {
							minerals = -500
						}
					}
					add_district = district_industrial
				#improve research
				} else_if = {
					limit = {					
						free_building_slots > 1
						owner = {
							resource_stockpile_compare = { 
								resource = minerals
								value > 500
							}
						}
					}
					owner = {		
						add_resource = {
							minerals = -500
						}
					}
					add_building = building_research_lab_1					
				} 
			}
		}
	}
}
