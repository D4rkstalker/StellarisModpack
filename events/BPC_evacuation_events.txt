namespace = evacuation

event = {
	id = evacuation.1
	hide_window = yes
    is_triggered_only = yes
	immediate = {
		every_playable_country = {
			limit = { 
				is_ai = no
				any_owned_planet = {
					has_modifier = evacuate_planet
				}
			}

			every_owned_planet = {
				limit = {
					has_modifier = evacuate_planet 
				}
				while = {
					count = 10
					random_owned_pop = {
						PREVPREV = {
							random_owned_planet = {
								limit = {
									NOT = { has_modifier = evacuate_planet }
								}
								resettle_pop = { pop = PREVPREV planet = this }
								owner = { add_resource = { energy = -150 } }
								
							}
						}
					}
					if = {
						limit = {
							count_owned_pop = {
								count < 1
							}
						}
						remove_modifier = evacuate_planet
							
					}
				}
			}
		}
	}
}
