namespace = gamestartdiplo

# Start-Up Events
event = {
	id = gamestartdiplo.1
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_global_flag = better_diplo_installed }
	}
	
	immediate = {
		set_global_flag = better_diplo_installed
	}
}

country_event = {
	id = gamestartdiplo.2
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		# Random Primitive Homeworld Names
		every_galaxy_planet = {
			limit = {
				NOT = { is_habitat = yes }
				exists = owner
				owner = { is_country_type = primitive }
			}
			planet_event = { id = gamestartdiplo.3 days =  1 }
		}
	}
}

# Random names for primitive worlds
planet_event = {
	id = gamestartdiplo.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_planet_flag = planet_earth }
			set_name = NAME_Earth
			random_moon = {
				set_name = NAME_Luna
			}
		}
		else_if = {
			limit = { owner = { NOT = { is_fear_of_the_dark_empire = yes } } }
			set_name = random
		}
		if = {
			limit = { is_planet_class = pc_gaia }
			owner = { set_origin = origin_life_seeded }
		}
	}
}
