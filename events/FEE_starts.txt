namespace = FEE_start


# keeps gaia starts as gaia except as needed for origins, effect is in common\scripted_effects\FEE_scripted_effects.txt
event = {
    id = FEE_start.1

    hide_window = yes
    is_triggered_only = yes

    immediate = {
        every_planet = {
            if = { 
                limit = { 
                    has_planet_flag = FEE_gaia 
                    owner = {
                        not = {has_origin = origin_post_apocalyptic }
                        not = {has_origin = origin_remnants }
                        not = {has_origin = origin_machine }
                    }
                }
                change_pc = "pc_gaia"
                add_modifier = { modifier = gaia_world days = -1 }         
                generate_FEE_deposits = yes
            }
            if = { 
                limit = { has_planet_flag = FEE_ring }
                change_pc = "pc_ringworld_habitable"            
                generate_FEE_deposits_ring = yes
            }
        }
    }
}
