#Spawn planet
country_event = {
	id = gigaplanet.1
	location = FROMFROM
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		FROM = { 
			is_megastructure_type = war_planet_3
		}
	}
	
	immediate = {
		FROM.planet = {
			save_event_target_as = warplanet_planet
		}
		if = {
			limit = {
				FROM.planet = {
					is_planet_class = pc_planetship
				}
			}
			every_country = {
				limit = {
					is_country = ROOT
				}
				create_fleet = {
					name = "Behemoth Planetcraft"
					settings = { 
						spawn_debris = no  
						can_change_composition = no 
						is_boss = yes
					}
					effect = {
						set_owner = PREV
						create_ship = {
							name = "Behemoth Planetcraft"
							random_existing_design = giga_planet_behemoth 
						}
						set_location = event_target:warplanet_planet
					}
				}
				event_target:warplanet_planet = {
					remove_planet = yes
				}
			}
		}
	}
}
